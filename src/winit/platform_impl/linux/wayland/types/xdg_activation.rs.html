1//! Handling of xdg activation, which is used for user attention requests.

21pub struct XdgActivationState {
22    xdg_activation: XdgActivationV1,
23}
24
25impl XdgActivationState {
26    pub fn bind(
27        globals: &GlobalList,
28        queue_handle: &QueueHandle<WinitState>,
29    ) -> Result<Self, BindError> {
30        let xdg_activation = globals.bind(queue_handle, 1..=1, GlobalData)?;
31        Ok(Self { xdg_activation })
32    }
33
34    pub fn global(&self) -> &XdgActivationV1 {
35        &self.xdg_activation
36    }
37}

39impl Dispatch<XdgActivationV1, GlobalData, WinitState> for XdgActivationState {

48    }

50
51impl Dispatch<XdgActivationTokenV1, XdgActivationTokenData, WinitState> for XdgActivationState {
52    fn event(
53        state: &mut WinitState,
54        proxy: &XdgActivationTokenV1,
55        event: <XdgActivationTokenV1 as Proxy>::Event,
56        data: &XdgActivationTokenData,
57        _: &Connection,
58        _: &QueueHandle<WinitState>,
59    ) {
60        let token = match event {
61            ActivationTokenEvent::Done { token } => token,
62            _ => return,
63        };
64
65        let global = state
66            .xdg_activation
67            .as_ref()
68            .expect("got xdg_activation event without global.")
69            .global();
70
71        match data {
72            XdgActivationTokenData::Attention((surface, fence)) => {
73                global.activate(token, surface);
74                // Mark that no request attention is in process.
75                if let Some(attention_requested) = fence.upgrade() {
76                    attention_requested.store(false, std::sync::atomic::Ordering::Relaxed);
77                }
78            },
79            XdgActivationTokenData::Obtain((window_id, serial)) => {
80                state.events_sink.push_window_event(
81                    crate::event::WindowEvent::ActivationTokenDone {
82                        serial: *serial,
83                        token: ActivationToken::from_raw(token),
84                    },
85                    *window_id,
86                );
87            },
88        }
89
90        proxy.destroy();
91    }
92}

94/// The data associated with the activation request.
95pub enum XdgActivationTokenData {
96    /// Request user attention for the given surface.
97    Attention((WlSurface, Weak<AtomicBool>)),
98    /// Get a token to be passed outside of the winit.
99    Obtain((WindowId, AsyncRequestSerial)),
100}
