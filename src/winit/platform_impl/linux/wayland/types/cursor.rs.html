8#[derive(Debug)]
9pub enum SelectedCursor {
10    Named(CursorIcon),
11    Custom(CustomCursor),
12}
13
14impl Default for SelectedCursor {
15    fn default() -> Self {
16        Self::Named(Default::default())
17    }
18}
19
20#[derive(Debug)]
21pub struct CustomCursor {
22    pub buffer: Buffer,
23    pub w: i32,
24    pub h: i32,
25    pub hotspot_x: i32,
26    pub hotspot_y: i32,
27}
28
29impl CustomCursor {
30    pub(crate) fn new(pool: &mut SlotPool, image: &CursorImage) -> Self {
31        let (buffer, canvas) = pool
32            .create_buffer(
33                image.width as i32,
34                image.height as i32,
35                4 * (image.width as i32),
36                Format::Argb8888,
37            )
38            .unwrap();
39
40        for (canvas_chunk, rgba) in canvas.chunks_exact_mut(4).zip(image.rgba.chunks_exact(4)) {
41            // Alpha in buffer is premultiplied.
42            let alpha = rgba[3] as f32 / 255.;
43            let r = (rgba[0] as f32 * alpha) as u32;
44            let g = (rgba[1] as f32 * alpha) as u32;
45            let b = (rgba[2] as f32 * alpha) as u32;
46            let color = ((rgba[3] as u32) << 24) + (r << 16) + (g << 8) + b;
47            let array: &mut [u8; 4] = canvas_chunk.try_into().unwrap();
48            *array = color.to_le_bytes();
49        }
50
51        CustomCursor {
52            buffer,
53            w: image.width as i32,
54            h: image.height as i32,
55            hotspot_x: image.hotspot_x as i32,
56            hotspot_y: image.hotspot_y as i32,
57        }
58    }
59}
