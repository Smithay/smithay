161/// A single color buffer
162#[derive(Debug, Clone)]
163pub struct SolidColorBuffer {
164    id: Id,
165    size: Size<i32, Logical>,
166    commit: CommitCounter,
167    color: Color32F,
168}
169
170impl Default for SolidColorBuffer {
171    fn default() -> Self {
172        Self {
173            id: Id::new(),
174            size: Default::default(),
175            commit: Default::default(),
176            color: Default::default(),
177        }
178    }
179}
180
181impl SolidColorBuffer {
182    /// Initialize a new solid color buffer with the specified size and color
183    pub fn new(size: impl Into<Size<i32, Logical>>, color: impl Into<Color32F>) -> Self {
184        SolidColorBuffer {
185            id: Id::new(),
186            color: color.into(),
187            commit: CommitCounter::default(),
188            size: size.into(),
189        }
190    }
191
192    /// Set the new size of this solid color buffer
193    ///
194    /// Note: If the size matches the current size this will do nothing
195    pub fn resize(&mut self, size: impl Into<Size<i32, Logical>>) {
196        let size = size.into();
197        if size != self.size {
198            self.size = size;
199            self.commit.increment();
200        }
201    }
202
203    /// Set a new color on this solid color buffer
204    ///
205    /// Note: If the color matches the current color this will do nothing
206    pub fn set_color(&mut self, color: impl Into<Color32F>) {
207        let color = color.into();
208        if color != self.color {
209            self.color = color;
210            self.commit.increment();
211        }
212    }
213
214    /// Update the size and color of this solid color buffer
215    ///
216    /// Note: If the size and color match the current size and color this will do nothing
217    pub fn update(&mut self, size: impl Into<Size<i32, Logical>>, color: impl Into<Color32F>) {
218        let color = color.into();
219        let size = size.into();
220        if size != self.size || color != self.color {
221            self.size = size;
222            self.color = color;
223            self.commit.increment();
224        }
225    }
226
227    /// Get the current color of this buffer
228    pub fn color(&self) -> Color32F {
229        self.color
230    }
231}

233/// [`Element`] to render a solid color
234#[derive(Debug, Clone)]
235pub struct SolidColorRenderElement {
236    id: Id,
237    geometry: Rectangle<i32, Physical>,
238    src: Rectangle<f64, Buffer>,
239    opaque_regions: Vec<Rectangle<i32, Physical>>,
240    commit: CommitCounter,
241    color: Color32F,
242    kind: Kind,
243}
244
245impl SolidColorRenderElement {
246    /// Create a render element from a [`SolidColorBuffer`]
247    pub fn from_buffer(
248        buffer: &SolidColorBuffer,
249        location: impl Into<Point<i32, Physical>>,
250        scale: impl Into<Scale<f64>>,
251        alpha: f32,
252        kind: Kind,
253    ) -> Self {
254        let geo = Rectangle::new(location.into(), buffer.size.to_physical_precise_round(scale));
255        let color = buffer.color * alpha;
256        Self::new(buffer.id.clone(), geo, buffer.commit, color, kind)
257    }
258
259    /// Create a new solid color render element with the specified geometry and color
260    pub fn new(
261        id: impl Into<Id>,
262        geometry: Rectangle<i32, Physical>,
263        commit: impl Into<CommitCounter>,
264        color: impl Into<Color32F>,
265        kind: Kind,
266    ) -> Self {
267        let color = color.into();
268        let src = Rectangle::from_size(geometry.size)
269            .to_f64()
270            .to_logical(1f64)
271            .to_buffer(1f64, Transform::Normal, &geometry.size.to_f64().to_logical(1f64));
272        let opaque_regions = if color.is_opaque() {
273            vec![Rectangle::from_size(geometry.size)]
274        } else {
275            vec![]
276        };
277        SolidColorRenderElement {
278            id: id.into(),
279            geometry,
280            src,
281            opaque_regions,
282            commit: commit.into(),
283            color,
284            kind,
285        }
286    }
287
288    /// Get the current color of this element
289    pub fn color(&self) -> Color32F {
290        self.color
291    }
292}

324impl<R: Renderer> RenderElement<R> for SolidColorRenderElement {
325    #[profiling::function]
326    fn draw(
327        &self,
328        frame: &mut <R as Renderer>::Frame<'_>,
329        _src: Rectangle<f64, Buffer>,
330        dst: Rectangle<i32, Physical>,
331        damage: &[Rectangle<i32, Physical>],
332        _opaque_regions: &[Rectangle<i32, Physical>],
333    ) -> Result<(), <R as Renderer>::Error> {
334        frame.draw_solid(dst, damage, self.color)
335    }
336
337    #[inline]
338    fn underlying_storage(&self, _renderer: &mut R) -> Option<super::UnderlyingStorage<'_>> {
339        None
340    }
341}
