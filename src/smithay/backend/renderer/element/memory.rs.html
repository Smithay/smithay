1//! Element to render from RGBA memory
2//!
3//! # Why use this implementation
4//!
5//! The [`MemoryRenderBuffer`] allows for easy integration of software-rendered elements
6//! in the smithay rendering pipeline. As software-rendered elements eventually have to
7//! upload to the GPU for rendering damage tracking is a crucial part. The [`MemoryRenderBuffer`]
8//! allows for efficient damage tracking by providing a [`RenderContext`] which accumulates the
9//! software-rendering damage. It automatically uploads the damaged parts to a [`RendererSuper::TextureId`](crate::backend::renderer::RendererSuper::TextureId)
10//! during rendering.
11//!
12//! # Why **not** to use this implementation
13//!
14//! As described earlier the [`MemoryRenderBuffer`] is targeted at software rendering, if you have
15//! some static content you may want to take a look at [`TextureBuffer`](super::texture::TextureBuffer)
16//! or [`TextureRenderBuffer`](super::texture::TextureRenderBuffer) for hardware accelerated rendering.
17//!
18//! # How to use it
19//!
20//! The [`MemoryRenderBuffer`] represents a buffer of your data and holds the damage as reported by the [`RenderContext`].
21//! To render the buffer you have to create a [`MemoryRenderBufferRenderElement`] in your render loop as
22//! shown in the example.
23//!
24//! ```no_run
25//! # use smithay::{
26//! #     backend::renderer::{Color32F, DebugFlags, Frame, ImportMem, Renderer, Texture, TextureFilter, sync::SyncPoint, test::{DummyRenderer, DummyFramebuffer}},
27//! #     utils::{Buffer, Physical},
28//! # };
29//! use std::time::{Duration, Instant};
30//!
31//! use smithay::{
32//!     backend::{
33//!         allocator::Fourcc,
34//!         renderer::{
35//!             damage::OutputDamageTracker,
36//!             element::{
37//!                 Kind,
38//!                 memory::{MemoryRenderBuffer, MemoryRenderBufferRenderElement},
39//!             },
40//!         },
41//!     },
42//!     utils::{Point, Rectangle, Size, Transform},
43//! };
44//!
45//! const WIDTH: i32 = 10;
46//! const HEIGHT: i32 = 10;
47//!
48//! // Initialize a empty render buffer
49//! let mut buffer = MemoryRenderBuffer::new(Fourcc::Argb8888, (WIDTH, HEIGHT), 1, Transform::Normal, None);
50//!
51//! // Create a rendering context
52//! let mut render_context = buffer.render();
53//!
54//! // Draw to the buffer
55//! render_context.draw(|buffer| {
56//!     buffer.chunks_exact_mut(4).for_each(|chunk| {
57//!         chunk.copy_from_slice(&[255, 231, 199, 255]);
58//!     });
59//!
60//!     // Return the whole buffer as damage
61//!     Result::<_, ()>::Ok(vec![Rectangle::from_size((WIDTH, HEIGHT).into())])
62//! });
63//!
64//! // Optionally update the opaque regions
65//! render_context.update_opaque_regions(Some(vec![Rectangle::from_size(
66//!     Size::from((WIDTH, HEIGHT)),
67//! )]));
68//!
69//! // We explicitly drop the context here to make the borrow checker happy
70//! std::mem::drop(render_context);
71//!
72//! // Initialize a static damage tracker
73//! let mut damage_tracker = OutputDamageTracker::new((800, 600), 1.0, Transform::Normal);
74//! # let mut renderer = DummyRenderer::default();
75//! # let mut framebuffer = DummyFramebuffer;
76//!
77//! let mut last_update = Instant::now();
78//!
79//! loop {
80//!     let now = Instant::now();
81//!     if now.duration_since(last_update) >= Duration::from_secs(3) {
82//!         let mut render_context = buffer.render();
83//!
84//!         render_context.draw(|_buffer| {
85//!             // Update the changed parts of the buffer
86//!
87//!             // Return the updated parts
88//!             Result::<_, ()>::Ok(vec![Rectangle::from_size((WIDTH, HEIGHT).into())])
89//!         });
90//!
91//!         last_update = now;
92//!     }
93//!
94//!     // Create a render element from the buffer
95//!     let location = Point::from((100.0, 100.0));
96//!     let render_element = MemoryRenderBufferRenderElement::from_buffer(&mut renderer, location, &buffer, None, None, None, Kind::Unspecified)
97//!         .expect("Failed to upload from memory to gpu");
98//!
99//!     // Render the element(s)
100//!     damage_tracker
101//!         .render_output(&mut renderer, &mut framebuffer, 0, &[&render_element], [0.8, 0.8, 0.9, 1.0])
102//!         .expect("failed to render output");
103//! }
104//! ```

127/// A buffer storing pixel data backed by system memory
128#[derive(Debug, Clone)]
129pub struct MemoryBuffer {
130    mem: Arc<Vec<u8>>,
131    format: Fourcc,
132    size: Size<i32, Buffer>,
133    stride: i32,
134}
135
136impl Default for MemoryBuffer {
137    fn default() -> Self {
138        Self {
139            mem: Default::default(),
140            format: Fourcc::Abgr8888,
141            size: Default::default(),
142            stride: Default::default(),
143        }
144    }
145}
146
147impl MemoryBuffer {
148    /// Create a new zeroed memory buffer with the specified format and size
149    pub fn new(format: Fourcc, size: impl Into<Size<i32, Buffer>>) -> Self {
150        let size = size.into();
151        let stride = size.w * (get_bpp(format).expect("Format with unknown bits per pixel") / 8) as i32;
152        let mem = vec![0; (stride * size.h) as usize];
153        Self {
154            mem: Arc::new(mem),
155            format,
156            size,
157            stride,
158        }
159    }
160
161    /// Create a new memory buffer from a slice with the specified format and size
162    pub fn from_slice(mem: &[u8], format: Fourcc, size: impl Into<Size<i32, Buffer>>) -> Self {
163        let size = size.into();
164        let stride = size.w * (get_bpp(format).expect("Format with unknown bits per pixel") / 8) as i32;
165        assert!(mem.len() >= (stride * size.h) as usize);
166        Self {
167            mem: Arc::new(mem.to_vec()),
168            format,
169            size,
170            stride,
171        }
172    }
173
174    /// Get the size of this buffer
175    pub fn size(&self) -> Size<i32, Buffer> {
176        self.size
177    }
178
179    /// Get the format of this buffer
180    pub fn format(&self) -> Fourcc {
181        self.format
182    }
183
184    /// Get the stride of this buffer
185    pub fn stride(&self) -> i32 {
186        self.stride
187    }
188
189    /// Resize this buffer to the size specified
190    pub fn resize(&mut self, size: impl Into<Size<i32, Buffer>>) -> bool {
191        self.size = size.into();
192        self.stride =
193            self.size.w * (get_bpp(self.format).expect("Format with unknown bits per pixel") / 8) as i32;
194        let mem_size = (self.stride * self.size.h) as usize;
195        if self.mem.len() != mem_size {
196            let mem = Arc::make_mut(&mut self.mem);
197            mem.resize(mem_size, 0);
198            true
199        } else {
200            false
201        }
202    }
203}

205impl std::ops::Deref for MemoryBuffer {
206    type Target = [u8];
207
208    fn deref(&self) -> &Self::Target {
209        &self.mem
210    }
211}

214    fn deref_mut(&mut self) -> &mut Self::Target {

216    }

219#[derive(Debug)]
220struct MemoryRenderBufferInner {
221    mem: MemoryBuffer,
222    scale: i32,
223    transform: Transform,
224    opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
225    damage_bag: DamageBag<i32, Buffer>,
226    textures: HashMap<ErasedContextId, Box<dyn Any + Send>>,
227    renderer_seen: HashMap<ErasedContextId, CommitCounter>,
228}
229
230impl Default for MemoryRenderBufferInner {
231    fn default() -> Self {
232        MemoryRenderBufferInner {
233            mem: Default::default(),
234            scale: 1,
235            transform: Transform::Normal,
236            opaque_regions: None,
237            damage_bag: DamageBag::default(),
238            textures: HashMap::default(),
239            renderer_seen: HashMap::default(),
240        }
241    }
242}

244impl MemoryRenderBufferInner {
245    fn new(
246        format: Fourcc,
247        size: impl Into<Size<i32, Buffer>>,
248        scale: i32,
249        transform: Transform,
250        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
251    ) -> Self {
252        MemoryRenderBufferInner {
253            mem: MemoryBuffer::new(format, size),
254            scale,
255            transform,
256            opaque_regions,
257            damage_bag: DamageBag::default(),
258            textures: HashMap::default(),
259            renderer_seen: HashMap::default(),
260        }
261    }
262
263    fn from_memory(
264        mem: MemoryBuffer,
265        scale: i32,
266        transform: Transform,
267        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
268    ) -> Self {
269        MemoryRenderBufferInner {
270            mem,
271            scale,
272            transform,
273            opaque_regions,
274            damage_bag: DamageBag::default(),
275            textures: HashMap::default(),
276            renderer_seen: HashMap::default(),
277        }
278    }
279
280    fn from_slice(
281        mem: &[u8],
282        format: Fourcc,
283        size: impl Into<Size<i32, Buffer>>,
284        scale: i32,
285        transform: Transform,
286        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
287    ) -> Self {
288        Self::from_memory(
289            MemoryBuffer::from_slice(mem, format, size),
290            scale,
291            transform,
292            opaque_regions,
293        )
294    }
295
296    fn resize(&mut self, size: impl Into<Size<i32, Buffer>>) {
297        if self.mem.resize(size) {
298            self.renderer_seen.clear();
299            self.textures.clear();
300            self.damage_bag.reset();
301            self.opaque_regions = None;
302        }
303    }
304
305    #[instrument(level = "trace", skip(renderer))]
306    #[profiling::function]
307    fn import_texture<R>(&mut self, renderer: &mut R) -> Result<R::TextureId, R::Error>
308    where
309        R: Renderer + ImportMem,
310        R::TextureId: Send + Clone + 'static,
311    {
312        let context_id = renderer.context_id().erased();
313        let current_commit = self.damage_bag.current_commit();
314        let last_commit = self.renderer_seen.get(&context_id).copied();
315        let buffer_damage = self
316            .damage_bag
317            .damage_since(last_commit)
318            .map(|d| d.into_iter().reduce(|a, b| a.merge(b)).unwrap_or_default())
319            .unwrap_or_else(|| Rectangle::from_size(self.mem.size()));
320
321        let tex = match self.textures.entry(context_id.clone()) {
322            Entry::Occupied(entry) => {
323                let tex = entry.get().downcast_ref().unwrap();
324                if !buffer_damage.is_empty() {
325                    trace!("updating memory with damage {:#?}", &buffer_damage);
326                    renderer.update_memory(tex, &self.mem, buffer_damage)?
327                }
328                tex.clone()
329            }
330            Entry::Vacant(entry) => {
331                trace!("importing memory");
332                let tex = renderer.import_memory(&self.mem, self.mem.format(), self.mem.size(), false)?;
333                entry.insert(Box::new(tex.clone()));
334                tex
335            }
336        };
337
338        self.renderer_seen.insert(context_id, current_commit);
339        Ok(tex)
340    }
341}

345pub struct MemoryRenderBuffer {

347    inner: Arc<Mutex<MemoryRenderBufferInner>>,
348}

356    }

359impl MemoryRenderBuffer {
360    /// Initialize a empty [`MemoryRenderBuffer`]
361    pub fn new(
362        format: Fourcc,
363        size: impl Into<Size<i32, Buffer>>,
364        scale: i32,
365        transform: Transform,
366        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
367    ) -> Self {
368        let inner = MemoryRenderBufferInner::new(format, size, scale, transform, opaque_regions);
369        MemoryRenderBuffer {
370            id: Id::new(),
371            inner: Arc::new(Mutex::new(inner)),
372        }
373    }
374
375    /// Initialize a [`MemoryRenderBuffer`] from an existing [`MemoryBuffer`]
376    pub fn from_memory(
377        mem: MemoryBuffer,
378        scale: i32,
379        transform: Transform,
380        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
381    ) -> Self {
382        let inner = MemoryRenderBufferInner::from_memory(mem, scale, transform, opaque_regions);
383        MemoryRenderBuffer {
384            id: Id::new(),
385            inner: Arc::new(Mutex::new(inner)),
386        }
387    }
388
389    /// Initialize a [`MemoryRenderBuffer`] from a slice
390    pub fn from_slice(
391        mem: &[u8],
392        format: Fourcc,
393        size: impl Into<Size<i32, Buffer>>,
394        scale: i32,
395        transform: Transform,
396        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
397    ) -> Self {
398        let inner = MemoryRenderBufferInner::from_slice(mem, format, size, scale, transform, opaque_regions);
399        MemoryRenderBuffer {
400            id: Id::new(),
401            inner: Arc::new(Mutex::new(inner)),
402        }
403    }
404
405    /// Render to the memory buffer
406    pub fn render(&mut self) -> RenderContext<'_> {
407        let guard = self.inner.lock().unwrap();
408        RenderContext {
409            buffer: guard,
410            damage: Vec::new(),
411            opaque_regions: None,
412        }
413    }

416/// A render context for [`MemoryRenderBuffer`]
417#[derive(Debug)]
418pub struct RenderContext<'a> {
419    buffer: MutexGuard<'a, MemoryRenderBufferInner>,
420    damage: Vec<Rectangle<i32, Buffer>>,
421    opaque_regions: Option<Option<Vec<Rectangle<i32, Buffer>>>>,
422}
423
424impl RenderContext<'_> {
425    /// Resize the buffer
426    ///
427    /// Note that this will also reset the opaque regions.
428    /// If you previously set opaque regions you should update
429    /// them with [`RenderContext::update_opaque_regions`]
430    pub fn resize(&mut self, size: impl Into<Size<i32, Buffer>>) {
431        self.buffer.resize(size);
432    }
433
434    /// Draw to the buffer
435    ///
436    /// Provided closure has to return updated regions.
437    pub fn draw<F, E>(&mut self, f: F) -> Result<(), E>
438    where
439        F: FnOnce(&mut [u8]) -> Result<Vec<Rectangle<i32, Buffer>>, E>,
440    {
441        let draw_damage = f(&mut self.buffer.mem)?;
442        self.damage.extend(draw_damage);
443        Ok(())
444    }
445
446    /// Update the opaque regions
447    pub fn update_opaque_regions(&mut self, opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>) {
448        self.opaque_regions = Some(opaque_regions);
449    }
450}
451
452impl Drop for RenderContext<'_> {
453    fn drop(&mut self) {
454        self.buffer.damage_bag.add(std::mem::take(&mut self.damage));
455        if let Some(opaque_regions) = self.opaque_regions.take() {
456            self.buffer.opaque_regions = opaque_regions;
457        }
458    }
459}

461/// A render element for [`MemoryRenderBuffer`]
462#[derive(Debug)]
463pub struct MemoryRenderBufferRenderElement<R: Renderer> {
464    id: Id,
465    location: Point<f64, Physical>,
466    buffer: MemoryBuffer,
467    alpha: f32,
468    src: Rectangle<f64, Logical>,
469    buffer_scale: i32,
470    buffer_transform: Transform,
471    size: Size<i32, Logical>,
472    damage: DamageSnapshot<i32, Buffer>,
473    opaque_regions: OpaqueRegions<i32, Buffer>,
474    texture: R::TextureId,
475    kind: Kind,
476}
477
478impl<R: Renderer> MemoryRenderBufferRenderElement<R> {
479    /// Create a new [`MemoryRenderBufferRenderElement`] for
480    /// a [`MemoryRenderBuffer`]
481    pub fn from_buffer(
482        renderer: &mut R,
483        location: impl Into<Point<f64, Physical>>,
484        buffer: &MemoryRenderBuffer,
485        alpha: Option<f32>,
486        src: Option<Rectangle<f64, Logical>>,
487        size: Option<Size<i32, Logical>>,
488        kind: Kind,
489    ) -> Result<Self, R::Error>
490    where
491        R: ImportMem,
492        R::TextureId: Send + Clone + 'static,
493    {
494        let mut inner = buffer.inner.lock().unwrap();
495        let texture = inner.import_texture(renderer)?;
496
497        let size = size
498            .or_else(|| src.map(|src| Size::from((src.size.w as i32, src.size.h as i32))))
499            .unwrap_or_else(|| inner.mem.size().to_logical(inner.scale, inner.transform));
500
501        let src = src.unwrap_or_else(|| Rectangle::from_size(size.to_f64()));
502
503        Ok(MemoryRenderBufferRenderElement {
504            id: buffer.id.clone(),
505            buffer: inner.mem.clone(),
506            location: location.into(),
507            alpha: alpha.unwrap_or(1.0),
508            src,
509            buffer_scale: inner.scale,
510            buffer_transform: inner.transform,
511            size,
512            opaque_regions: inner
513                .opaque_regions
514                .as_deref()
515                .map(OpaqueRegions::from_slice)
516                .unwrap_or_default(),
517            damage: inner.damage_bag.snapshot(),
518            texture,
519            kind,
520        })
521    }
522
523    fn physical_size(&self, scale: Scale<f64>) -> Size<i32, Physical> {
524        ((self.size.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
525            - self.location.to_i32_round())
526        .to_size()
527    }
528
529    fn scale(&self) -> Scale<f64> {
530        let src = self.src();
531        Scale::from((self.size.w as f64 / src.size.w, self.size.h as f64 / src.size.h))
532    }
533}

535impl<R: Renderer> Element for MemoryRenderBufferRenderElement<R> {
536    fn id(&self) -> &Id {
537        &self.id
538    }
539
540    fn current_commit(&self) -> CommitCounter {
541        self.damage.current_commit()
542    }
543
544    fn transform(&self) -> Transform {
545        self.buffer_transform
546    }
547
548    fn src(&self) -> Rectangle<f64, Buffer> {
549        self.src.to_buffer(
550            self.buffer_scale as f64,
551            self.buffer_transform,
552            &self.size.to_f64(),
553        )
554    }
555
556    fn geometry(&self, scale: Scale<f64>) -> Rectangle<i32, Physical> {
557        Rectangle::new(self.location.to_i32_round(), self.physical_size(scale))
558    }
559
560    fn damage_since(&self, scale: Scale<f64>, commit: Option<CommitCounter>) -> DamageSet<i32, Physical> {
561        let physical_size = self.physical_size(scale);
562        let logical_scale = self.scale();
563
564        self.damage
565            .damage_since(commit)
566            .map(|damage| {
567                damage
568                    .into_iter()
569                    .filter_map(|rect| {
570                        rect.to_f64()
571                            .to_logical(
572                                self.buffer_scale as f64,
573                                self.buffer_transform,
574                                &self.buffer.size.to_f64(),
575                            )
576                            .intersection(self.src)
577                            .map(|mut rect| {
578                                rect.loc -= self.src.loc;
579                                rect.upscale(logical_scale)
580                            })
581                            .map(|rect| {
582                                let surface_scale =
583                                    physical_size.to_f64() / self.size.to_f64().to_physical(scale);
584                                rect.to_physical_precise_up(surface_scale * scale)
585                            })
586                    })
587                    .collect::<DamageSet<_, _>>()
588            })
589            .unwrap_or_else(|| DamageSet::from_slice(&[Rectangle::from_size(physical_size)]))
590    }
591
592    fn opaque_regions(&self, scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
593        if self.alpha < 1.0 {
594            return OpaqueRegions::default();
595        }
596
597        let physical_size = self.physical_size(scale);
598        let logical_scale = self.scale();
599
600        self.opaque_regions
601            .iter()
602            .filter_map(|rect| {
603                rect.to_f64()
604                    .to_logical(
605                        self.buffer_scale as f64,
606                        self.buffer_transform,
607                        &self.buffer.size.to_f64(),
608                    )
609                    .intersection(self.src)
610                    .map(|mut rect| {
611                        rect.loc -= self.src.loc;
612                        rect.upscale(logical_scale)
613                    })
614                    .map(|rect| {
615                        let surface_scale = physical_size.to_f64() / self.size.to_f64().to_physical(scale);
616                        rect.to_physical_precise_up(surface_scale * scale)
617                    })
618            })
619            .collect::<OpaqueRegions<_, _>>()
620    }
621
622    fn alpha(&self) -> f32 {
623        self.alpha
624    }
625
626    fn kind(&self) -> Kind {
627        self.kind
628    }
629}

631impl<R> RenderElement<R> for MemoryRenderBufferRenderElement<R>
632where
633    R: Renderer + ImportMem,

637    #[profiling::function]

641        src: Rectangle<f64, Buffer>,

649            dst,
650            damage,
651            opaque_regions,
652            self.buffer_transform,

654        )
655    }

658    fn underlying_storage(&self, _renderer: &mut R) -> Option<UnderlyingStorage<'_>> {
659        Some(UnderlyingStorage::Memory(&self.buffer))
660    }
