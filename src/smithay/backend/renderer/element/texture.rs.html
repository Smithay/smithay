1//! Element to render a texture
2//!
3//! # Why use this implementation
4//!
5//! This module provides the following elements
6//!
7//! ## Static texture
8//!
9//! [`TextureBuffer`] represents a static texture which is expected to not change.
10//! It is possible to either use a [`pre-existing texture`](TextureBuffer::from_texture) or to create the texture
11//! from [`RGBA memory`](TextureBuffer::from_memory).
12//! The [`TextureBuffer`] can be used in the smithay pipeline by using [`TextureRenderElement::from_texture_buffer`].
13//!
14//! ## Hardware accelerated rendering
15//!
16//! [`TextureRenderBuffer`] provides a solution for hardware accelerated rending with
17//! proper damage tracking. It provides a [`RenderContext`] for rendering to the stored
18//! texture. The collected damage from [`RenderContext::draw`] is used to report the
19//! damaged regions in the [`TextureRenderElement`]. For rendering a [`TextureRenderBuffer`] you
20//! can create a render element with [`TextureRenderElement::from_texture_render_buffer`].
21//!
22//! # Why **not** to use this implementation
23//!
24//! For software rendering you may take a look at [`MemoryRenderBuffer`](super::memory::MemoryRenderBuffer)
25//! which provides proper damage tracking and is agnostic over the [`Renderer`]
26//!
27//! # How to use it
28//!
29//! Both, [`TextureBuffer`] and [`TextureRenderBuffer`] represent a buffer of your texture data.
30//! To render the buffer you have to create a [`TextureRenderElement`] in your render loop as
31//! shown in the examples.
32//!
33//! To integrate custom texture based buffers the [`TextureRenderElement`] provides two functions
34//! to create the render element. [`from_static_texture`](TextureRenderElement::from_static_texture) can be used to
35//! create a render element from a static texture without damage tracking.
36//! [`from_texture_with_damage`](TextureRenderElement::from_texture_with_damage) can be used to create
37//! a render element from an existing texture with custom damage tracking.
38//! In both cases you have to make sure to provide a stable [`Id`] or otherwise damage tracking will not work.
39//!
40//! ## [`TextureBuffer`]
41//!
42//! ```no_run
43//! # use smithay::{
44//! #     backend::renderer::{Color32F, DebugFlags, Frame, ImportMem, Renderer, Texture, TextureFilter, sync::SyncPoint, test::{DummyRenderer, DummyFramebuffer}},
45//! #     utils::{Buffer, Physical, Rectangle, Size},
46//! # };
47//! use smithay::{
48//!     backend::{
49//!         allocator::Fourcc,
50//!         renderer::{
51//!             damage::OutputDamageTracker,
52//!             element::{
53//!                 Kind,
54//!                 texture::{TextureBuffer, TextureRenderElement},
55//!             },
56//!         },
57//!     },
58//!     utils::{Point, Transform},
59//! };
60//!
61//! const WIDTH: i32 = 10;
62//! const HEIGHT: i32 = 10;
63//!
64//! let memory = vec![0; (WIDTH * 4 * HEIGHT) as usize];
65//! # let mut renderer = DummyRenderer::default();
66//! # let mut framebuffer = DummyFramebuffer;
67//!
68//! // Create the texture buffer from a chunk of memory
69//! let texture_buffer = TextureBuffer::from_memory(
70//!     &mut renderer,
71//!     &memory,
72//!     Fourcc::Argb8888,
73//!     (WIDTH, HEIGHT),
74//!     false,
75//!     1,
76//!     Transform::Normal,
77//!     None,
78//! )
79//! .expect("failed to import mem");
80//!
81//! let mut damage_tracker = OutputDamageTracker::new((800, 600), 1.0, Transform::Normal);
82//!
83//! loop {
84//!     // Create a render element from the buffer
85//!     let location = Point::from((100.0, 100.0));
86//!     let render_element =
87//!         TextureRenderElement::from_texture_buffer(location, &texture_buffer, None, None, None, Kind::Unspecified);
88//!
89//!     // Render the element(s)
90//!     damage_tracker
91//!         .render_output(&mut renderer, &mut framebuffer, 0, &[&render_element], [0.8, 0.8, 0.9, 1.0])
92//!         .expect("failed to render output");
93//! }
94//! ```
95//!
96//! ## [`TextureRenderBuffer`]
97//!
98//! ```no_run
99//! # use smithay::{
100//! #     backend::renderer::{Color32F, DebugFlags, Frame, ImportMem, Renderer, Texture, TextureFilter, sync::SyncPoint, test::{DummyRenderer, DummyFramebuffer}},
101//! #     utils::{Buffer, Physical},
102//! # };
103//! use std::time::{Duration, Instant};
104//!
105//! use smithay::{
106//!     backend::{
107//!         allocator::Fourcc,
108//!         renderer::{
109//!             damage::OutputDamageTracker,
110//!             element::{
111//!                 Kind,
112//!                 texture::{TextureRenderBuffer, TextureRenderElement},
113//!             },
114//!         },
115//!     },
116//!     utils::{Point, Rectangle, Size, Transform},
117//! };
118//!
119//! const WIDTH: i32 = 10;
120//! const HEIGHT: i32 = 10;
121//!
122//! let memory = vec![0; (WIDTH * 4 * HEIGHT) as usize];
123//! # let mut renderer = DummyRenderer::default();
124//! # let mut framebuffer = DummyFramebuffer;
125//!
126//! // Create the texture buffer from a chunk of memory
127//! let mut texture_render_buffer = TextureRenderBuffer::from_memory(
128//!     &mut renderer,
129//!     &memory,
130//!     Fourcc::Argb8888,
131//!     (WIDTH, HEIGHT),
132//!     false,
133//!     1,
134//!     Transform::Normal,
135//!     None,
136//! )
137//! .expect("failed to import mem");
138//!
139//! // Create the rendering context
140//! let mut render_context = texture_render_buffer.render();
141//!
142//! // Draw to the texture
143//! render_context.draw(|_texture| {
144//!     // Your draw code here...
145//!
146//!     // Return the damage areas
147//!     Result::<_, ()>::Ok(vec![Rectangle::from_size(
148//!         Size::from((WIDTH, HEIGHT)),
149//!     )])
150//! });
151//!
152//! // Optionally update the opaque regions
153//! render_context.update_opaque_regions(Some(vec![Rectangle::from_size(
154//!     Size::from((WIDTH, HEIGHT)),
155//! )]));
156//!
157//! // We explicitly drop the context here to make the borrow checker happy
158//! std::mem::drop(render_context);
159//!
160//! let mut damage_tracker = OutputDamageTracker::new((800, 600), 1.0, Transform::Normal);
161//!
162//! let mut last_update = Instant::now();
163//!
164//! loop {
165//!     let now = Instant::now();
166//!     if now.duration_since(last_update) >= Duration::from_secs(3) {
167//!         let mut render_context = texture_render_buffer.render();
168//!
169//!         render_context.draw(|_buffer| {
170//!             // Update the changed parts of the buffer
171//!
172//!             // Return the updated parts
173//!             Result::<_, ()>::Ok(vec![Rectangle::from_size((WIDTH, HEIGHT).into())])
174//!         });
175//!
176//!         last_update = now;
177//!     }
178//!
179//!     // Create a render element from the buffer
180//!     let location = Point::from((100.0, 100.0));
181//!     let render_element = TextureRenderElement::from_texture_render_buffer(
182//!         location,
183//!         &texture_render_buffer,
184//!         None,
185//!         None,
186//!         None,
187//!         Kind::Unspecified,
188//!     );
189//!
190//!     // Render the element(s)
191//!     damage_tracker
192//!         .render_output(&mut renderer, &mut framebuffer, 0, &[&render_element], [0.8, 0.8, 0.9, 1.0])
193//!         .expect("failed to render output");
194//! }
195//! ```

214/// A single texture buffer
215#[derive(Debug, Clone)]
216pub struct TextureBuffer<T: Texture> {
217    id: Id,
218    context_id: ContextId<T>,
219    texture: T,
220    scale: i32,
221    transform: Transform,
222    opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
223}
224
225impl<T: Texture> TextureBuffer<T> {
226    /// Create a [`TextureBuffer`] from an existing texture
227    pub fn from_texture<R: Renderer<TextureId = T>>(
228        renderer: &R,
229        texture: T,
230        scale: i32,
231        transform: Transform,
232        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
233    ) -> Self {
234        TextureBuffer {
235            id: Id::new(),
236            context_id: renderer.context_id(),
237            texture,
238            scale,
239            transform,
240            opaque_regions,
241        }
242    }
243
244    /// Create [`TextureBuffer`] from a chunk of memory
245    #[allow(clippy::too_many_arguments)]
246    pub fn from_memory<R: Renderer<TextureId = T> + ImportMem>(
247        renderer: &mut R,
248        data: &[u8],
249        format: Fourcc,
250        size: impl Into<Size<i32, Buffer>>,
251        flipped: bool,
252        scale: i32,
253        transform: Transform,
254        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
255    ) -> Result<Self, R::Error> {
256        let texture = renderer.import_memory(data, format, size.into(), flipped)?;
257        Ok(TextureBuffer::from_texture(
258            renderer,
259            texture,
260            scale,
261            transform,
262            opaque_regions,
263        ))
264    }
265
266    /// Format of the underlying texture
267    pub fn format(&self) -> Option<Fourcc>
268    where
269        T: Texture,
270    {
271        self.texture.format()
272    }
273}

287impl<T: Texture> TextureRenderBuffer<T> {
288    /// Create [`TextureRenderBuffer`] from an existing texture
289    pub fn from_texture<R: Renderer<TextureId = T>>(
290        renderer: &R,
291        texture: T,
292        scale: i32,
293        transform: Transform,
294        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
295    ) -> Self {
296        TextureRenderBuffer {
297            id: Id::new(),
298            context_id: renderer.context_id(),
299            texture,
300            scale,
301            transform,
302            opaque_regions,
303            damage_tracker: Arc::new(Mutex::new(DamageBag::default())),
304        }
305    }
306
307    /// Create [`TextureRenderBuffer`] from a chunk of memory
308    #[allow(clippy::too_many_arguments)]
309    pub fn from_memory<R: Renderer<TextureId = T> + ImportMem>(
310        renderer: &mut R,
311        data: &[u8],
312        format: Fourcc,
313        size: impl Into<Size<i32, Buffer>>,
314        flipped: bool,
315        scale: i32,
316        transform: Transform,
317        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
318    ) -> Result<Self, R::Error> {
319        let texture = renderer.import_memory(data, format, size.into(), flipped)?;
320        Ok(TextureRenderBuffer::from_texture(
321            renderer,
322            texture,
323            scale,
324            transform,
325            opaque_regions,
326        ))
327    }
328
329    /// Replace the stored texture
330    pub fn update_from_texture<R: Renderer<TextureId = T>>(
331        &mut self,
332        renderer: &R,
333        texture: T,
334        scale: i32,
335        transform: Transform,
336        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
337    ) {
338        assert_eq!(self.context_id, renderer.context_id());
339        self.texture = texture;
340        self.scale = scale;
341        self.transform = transform;
342        self.opaque_regions = opaque_regions;
343        self.damage_tracker.lock().unwrap().reset();
344    }
345
346    /// Update the texture from a chunk of memory
347    pub fn update_from_memory<R: Renderer<TextureId = T> + ImportMem>(
348        &mut self,
349        renderer: &mut R,
350        data: &[u8],
351        region: Rectangle<i32, Buffer>,
352        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
353    ) -> Result<(), R::Error> {
354        assert_eq!(self.context_id, renderer.context_id());
355        renderer.update_memory(&self.texture, data, region)?;
356        self.damage_tracker.lock().unwrap().add([region]);
357        self.opaque_regions = opaque_regions;
358        Ok(())
359    }
360
361    /// Render to the texture
362    pub fn render(&mut self) -> RenderContext<'_, T> {
363        RenderContext {
364            buffer: self,
365            damage: Vec::new(),
366            opaque_regions: None,
367        }
368    }
369
370    /// Format of the underlying texture
371    pub fn format(&self) -> Option<Fourcc> {
372        self.texture.format()
373    }
374}

376/// A render context for [`TextureRenderBuffer`]
377#[derive(Debug)]
378pub struct RenderContext<'a, T: Texture> {
379    buffer: &'a mut TextureRenderBuffer<T>,
380    damage: Vec<Rectangle<i32, Buffer>>,
381    opaque_regions: Option<Option<Vec<Rectangle<i32, Buffer>>>>,
382}
383
384impl<T: Texture> RenderContext<'_, T> {
385    /// Draw to the buffer
386    pub fn draw<F, E>(&mut self, f: F) -> Result<(), E>
387    where
388        F: FnOnce(&mut T) -> Result<Vec<Rectangle<i32, Buffer>>, E>,
389    {
390        let draw_damage = f(&mut self.buffer.texture)?;
391        self.damage.extend(draw_damage);
392        Ok(())
393    }
394
395    /// Update the opaque regions
396    pub fn update_opaque_regions(&mut self, opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>) {
397        self.opaque_regions = Some(opaque_regions);
398    }
399}
400
401impl<T: Texture> Drop for RenderContext<'_, T> {
402    fn drop(&mut self) {
403        self.buffer
404            .damage_tracker
405            .lock()
406            .unwrap()
407            .add(std::mem::take(&mut self.damage));
408        if let Some(opaque_regions) = self.opaque_regions.take() {
409            self.buffer.opaque_regions = opaque_regions;
410        }
411    }
412}

490impl<T: Texture> TextureRenderElement<T> {
491    /// Create a [`TextureRenderElement`] from an
492    /// existing texture and a [`DamageSnapshot`]
493    #[allow(clippy::too_many_arguments)]
494    pub fn from_texture_with_damage(
495        id: Id,
496        context_id: ContextId<T>,
497        location: impl Into<Point<f64, Physical>>,
498        texture: T,
499        scale: i32,
500        transform: Transform,
501        alpha: Option<f32>,
502        src: Option<Rectangle<f64, Logical>>,
503        size: Option<Size<i32, Logical>>,
504        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
505        snapshot: DamageSnapshot<i32, Buffer>,
506        kind: Kind,
507    ) -> Self {
508        let opaque_regions = opaque_regions.map(|regions| {
509            regions
510                .into_iter()
511                .map(|region| region.to_logical(scale, transform, &texture.size()))
512                .collect::<Vec<_>>()
513        });
514        TextureRenderElement {
515            location: location.into(),
516            id,
517            context_id,
518            texture,
519            scale,
520            transform,
521            alpha: alpha.unwrap_or(1.0),
522            src,
523            size,
524            opaque_regions,
525            snapshot,
526            kind,
527        }
528    }
529
530    /// Create a static [`TextureRenderElement`] from
531    /// an existing texture
532    #[allow(clippy::too_many_arguments)]
533    pub fn from_static_texture(
534        id: Id,
535        context_id: ContextId<T>,
536        location: impl Into<Point<f64, Physical>>,
537        texture: T,
538        scale: i32,
539        transform: Transform,
540        alpha: Option<f32>,
541        src: Option<Rectangle<f64, Logical>>,
542        size: Option<Size<i32, Logical>>,
543        opaque_regions: Option<Vec<Rectangle<i32, Buffer>>>,
544        kind: Kind,
545    ) -> Self {
546        TextureRenderElement::from_texture_with_damage(
547            id,
548            context_id,
549            location,
550            texture,
551            scale,
552            transform,
553            alpha,
554            src,
555            size,
556            opaque_regions,
557            DamageSnapshot::empty(),
558            kind,
559        )
560    }
561
562    fn logical_size(&self) -> Size<i32, Logical> {
563        self.size
564            .or_else(|| {
565                self.src
566                    .map(|src| Size::from((src.size.w as i32, src.size.h as i32)))
567            })
568            .unwrap_or_else(|| self.texture.size().to_logical(self.scale, self.transform))
569    }
570
571    fn physical_size(&self, scale: Scale<f64>) -> Size<i32, Physical> {
572        let logical_size = self.logical_size();
573        ((logical_size.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
574            - self.location.to_i32_round())
575        .to_size()
576    }
577
578    fn src(&self) -> Rectangle<f64, Logical> {
579        self.src
580            .unwrap_or_else(|| Rectangle::from_size(self.logical_size().to_f64()))
581    }
582
583    fn scale(&self) -> Scale<f64> {
584        let size = self.logical_size();
585        let src = self.src();
586        Scale::from((size.w as f64 / src.size.w, size.h as f64 / src.size.h))
587    }
588
589    fn rect_to_global<N>(&self, rect: Rectangle<N, Logical>) -> Rectangle<f64, Logical>
590    where
591        N: Coordinate,
592    {
593        let scale = self.scale();
594        let mut rect = rect.to_f64();
595        rect.loc -= self.src.map(|rect| rect.loc).unwrap_or_default();
596        rect.upscale(scale)
597    }
598}

600impl<T> Element for TextureRenderElement<T>
601where
602    T: Texture,
603{
604    fn id(&self) -> &Id {
605        &self.id
606    }
607
608    fn current_commit(&self) -> CommitCounter {
609        self.snapshot.current_commit()
610    }
611
612    fn geometry(&self, scale: Scale<f64>) -> Rectangle<i32, Physical> {
613        Rectangle::new(self.location.to_i32_round(), self.physical_size(scale))
614    }
615
616    fn transform(&self) -> Transform {
617        self.transform
618    }
619
620    fn src(&self) -> Rectangle<f64, Buffer> {
621        let size = self.logical_size();
622        self.src()
623            .to_buffer(self.scale as f64, self.transform, &size.to_f64())
624    }
625
626    fn damage_since(&self, scale: Scale<f64>, commit: Option<CommitCounter>) -> DamageSet<i32, Physical> {
627        let src = self.src();
628        let texture_size = self.texture.size();
629        let physical_size = self.physical_size(scale);
630        let logical_size = self.logical_size();
631        self.damage_since(commit)
632            .into_iter()
633            .filter_map(|rect| {
634                rect.to_f64()
635                    .to_logical(self.scale as f64, self.transform, &texture_size.to_f64())
636                    .intersection(src)
637                    .map(|rect| self.rect_to_global(rect).to_i32_up::<i32>())
638                    .map(|rect| {
639                        let surface_scale = physical_size.to_f64() / logical_size.to_f64().to_physical(scale);
640                        rect.to_physical_precise_up(surface_scale * scale)
641                    })
642            })
643            .collect::<DamageSet<_, _>>()
644    }
645
646    fn opaque_regions(&self, scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
647        if self.alpha < 1.0 {
648            return OpaqueRegions::default();
649        }
650
651        let src = self.src();
652        let physical_size = self.physical_size(scale);
653        let logical_size = self.logical_size();
654        self.opaque_regions
655            .as_ref()
656            .map(|r| {
657                r.iter()
658                    .filter_map(|rect| {
659                        rect.to_f64()
660                            .intersection(src)
661                            .map(|rect| self.rect_to_global(rect).to_i32_up::<i32>())
662                            .map(|rect| {
663                                let surface_scale =
664                                    physical_size.to_f64() / logical_size.to_f64().to_physical(scale);
665                                rect.to_physical_precise_up(surface_scale * scale)
666                            })
667                    })
668                    .collect::<OpaqueRegions<_, _>>()
669            })
670            .unwrap_or_default()
671    }
672
673    fn alpha(&self) -> f32 {
674        self.alpha
675    }
676
677    fn kind(&self) -> Kind {
678        self.kind
679    }
680}

682impl<R, T> RenderElement<R> for TextureRenderElement<T>
683where
684    R: Renderer<TextureId = T>,
685    T: Texture,
686{
687    #[instrument(level = "trace", skip(self, frame))]
688    #[profiling::function]
689    fn draw(
690        &self,
691        frame: &mut R::Frame<'_, '_>,
692        src: Rectangle<f64, Buffer>,
693        dst: Rectangle<i32, Physical>,
694        damage: &[Rectangle<i32, Physical>],
695        opaque_regions: &[Rectangle<i32, Physical>],
696    ) -> Result<(), R::Error> {
697        if frame.context_id() != self.context_id {
698            warn!("trying to render texture from different renderer context");
699            return Ok(());
700        }
701
702        frame.render_texture_from_to(
703            &self.texture,
704            src,
