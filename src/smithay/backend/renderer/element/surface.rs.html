1//! Element to render a wayland surface
2//!
3//! # Why use this implementation
4//!
5//! The [`WaylandSurfaceRenderElement`] provides an easy way to
6//! integrate a [`WlSurface`](wayland_server::protocol::wl_surface::WlSurface) in the smithay rendering pipeline
7//!
8//! # How to use it
9//!
10//! [`WaylandSurfaceRenderElement::from_surface`] allows you to obtain a [`WaylandSurfaceRenderElement`] for a single [`WlSurface`](wayland_server::protocol::wl_surface::WlSurface).
11//! To retrieve [`WaylandSurfaceRenderElement`]s for a whole surface tree you can use [`render_elements_from_surface_tree`].
12//!
13//! ```no_run
14//! # #[cfg(all(
15//! #     feature = "wayland_frontend",
16//! #     feature = "backend_egl",
17//! #     feature = "use_system_lib"
18//! # ))]
19//! # use smithay::backend::{
20//! #     egl::{self, display::EGLBufferReader},
21//! #     renderer::ImportEgl,
22//! # };
23//! # use smithay::{
24//! #     backend::allocator::{Fourcc, dmabuf::Dmabuf},
25//! #     backend::renderer::{
26//! #         Color32F, DebugFlags, Frame, ImportDma, ImportDmaWl, ImportMem, ImportMemWl, Renderer, Texture,
27//! #         TextureFilter, sync::SyncPoint, test::{DummyRenderer, DummyFramebuffer},
28//! #     },
29//! #     utils::{Buffer, Physical},
30//! #     wayland::compositor::SurfaceData,
31//! # };
32//! use smithay::{
33//!     backend::renderer::{
34//!         damage::OutputDamageTracker,
35//!         element::{
36//!             Kind,
37//!             surface::{render_elements_from_surface_tree, WaylandSurfaceRenderElement},
38//!         },
39//!     },
40//!     utils::{Point, Rectangle, Size, Transform},
41//! };
42//! # use wayland_server::{backend::ObjectId, protocol::wl_surface::WlSurface, Resource};
43//! # let display = wayland_server::Display::<()>::new().unwrap();
44//! # let dh = display.handle();
45//! # let surface = WlSurface::from_id(&dh, ObjectId::null()).unwrap();
46//!
47//! // Initialize a static damage tracked renderer
48//! let mut damage_tracker = OutputDamageTracker::new((800, 600), 1.0, Transform::Normal);
49//! # let mut renderer = DummyRenderer::default();
50//! # let mut framebuffer = DummyFramebuffer;
51//!
52//! loop {
53//!     // Create the render elements from the surface
54//!     let location = Point::from((100, 100));
55//!     let render_elements: Vec<WaylandSurfaceRenderElement<DummyRenderer>> =
56//!         render_elements_from_surface_tree(&mut renderer, &surface, location, 1.0, 1.0, Kind::Unspecified);
57//!
58//!     // Render the element(s)
59//!     damage_tracker
60//!         .render_output(&mut renderer, &mut framebuffer, 0, &*render_elements, [0.8, 0.8, 0.9, 1.0])
61//!         .expect("failed to render output");
62//! }
63//! ```

87/// Ways to evaluation the [`Kind`] of a surface.
88pub enum KindEvaluation {
89    /// Static evaluation, which will always return the same [`Kind`]
90    Static(Kind),
91    /// Dynamic evaluation, which will return a [`Kind`] based on the [`SurfaceData`]
92    Dynamic(fn(&SurfaceData) -> Kind),
93    /// Dynamic evaluation, which will return a [`Kind`] based on the [`SurfaceData`] and potentially additional captured context.
94    Closure(Box<dyn Fn(&SurfaceData) -> Kind>),
95}
96
97impl fmt::Debug for KindEvaluation {
98    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
99        match self {
100            KindEvaluation::Static(kind) => kind.fmt(f),
101            _ => f.write_str("Kind::<dynamic>"),
102        }
103    }
104}
105
106impl From<Kind> for KindEvaluation {
107    fn from(value: Kind) -> Self {
108        KindEvaluation::Static(value)
109    }
110}
111
112impl<F: Fn(&SurfaceData) -> Kind + 'static> From<F> for KindEvaluation {
113    fn from(value: F) -> Self {
114        KindEvaluation::Closure(Box::new(value))
115    }
116}

127impl KindEvaluation {
128    /// Create a `KindEvaluation` from a pure function.
129    ///
130    /// Using this instead of the `From`-implementation skips boxing
131    /// and thus can provide better performance as it is not relying on dynamic dispatching.
132    pub fn from_fn(func: fn(&SurfaceData) -> Kind) -> Self {
133        KindEvaluation::Dynamic(func)
134    }
135
136    fn eval(&self, data: &SurfaceData) -> Kind {
137        match self {
138            KindEvaluation::Static(res) => *res,
139            KindEvaluation::Dynamic(func) => func(data),
140            KindEvaluation::Closure(func) => func(data),
141        }
142    }
143}

145/// Retrieve the [`WaylandSurfaceRenderElement`]s for a surface tree
146#[instrument(level = "trace", skip(renderer, location, scale, kind))]
147#[profiling::function]
148pub fn render_elements_from_surface_tree<R, E>(
149    renderer: &mut R,
150    surface: &wl_surface::WlSurface,
151    location: impl Into<Point<i32, Physical>>,
152    scale: impl Into<Scale<f64>>,
153    alpha: f32,
154    kind: impl Into<KindEvaluation>,
155) -> Vec<E>
156where
157    R: Renderer + ImportAll,
158    R::TextureId: Clone + 'static,
159    E: From<WaylandSurfaceRenderElement<R>>,
160{
161    let location = location.into().to_f64();
162    let scale = scale.into();
163    let kind = kind.into();
164    let mut surfaces: Vec<E> = Vec::new();
165
166    compositor::with_surface_tree_downward(
167        surface,
168        location,
169        |_, states, location| {
170            let mut location = *location;
171            let data = states.data_map.get::<RendererSurfaceStateUserData>();
172
173            if let Some(data) = data {
174                if let Some(view) = data.lock().unwrap().view() {
175                    location += view.offset.to_f64().to_physical(scale);
176                    TraversalAction::DoChildren(location)
177                } else {
178                    TraversalAction::SkipChildren
179                }
180            } else {
181                TraversalAction::SkipChildren
182            }
183        },
184        |surface, states, location| {
185            let mut location = *location;
186            let kind = kind.eval(states);
187            let data = states.data_map.get::<RendererSurfaceStateUserData>();
188
189            if let Some(data) = data {
190                let has_view = if let Some(view) = data.lock().unwrap().view() {
191                    location += view.offset.to_f64().to_physical(scale);
192                    true
193                } else {
194                    false
195                };
196
197                if has_view {
198                    match WaylandSurfaceRenderElement::from_surface(
199                        renderer, surface, states, location, alpha, kind,
200                    ) {
201                        Ok(Some(surface)) => surfaces.push(surface.into()),
202                        Ok(None) => {} // surface is not mapped
203                        Err(err) => {
204                            warn!("Failed to import surface: {}", err);
205                        }
206                    };
207                }
208            }
209        },
210        |_, _, _| true,
211    );
212
213    surfaces
214}

251impl<R: Renderer + ImportAll> WaylandSurfaceRenderElement<R> {
252    /// Create a render element from a surface
253    #[profiling::function]
254    pub fn from_surface(
255        renderer: &mut R,
256        surface: &wl_surface::WlSurface,
257        states: &SurfaceData,
258        location: Point<f64, Physical>,
259        alpha: f32,
260        kind: Kind,
261    ) -> Result<Option<Self>, R::Error>
262    where
263        R::TextureId: Clone + 'static,
264    {
265        let id = Id::from_wayland_resource(surface);
266        crate::backend::renderer::utils::import_surface(renderer, states)?;
267
268        let mut alpha_modifier_state = states.cached_state.get::<AlphaModifierSurfaceCachedState>();
269        let alpha_multiplier = alpha_modifier_state.current().multiplier_f32().unwrap_or(1.0);
270
271        let Some(data_ref) = states.data_map.get::<RendererSurfaceStateUserData>() else {
272            return Ok(None);
273        };
274        Ok(Self::from_state(
275            renderer,
276            id,
277            location,
278            alpha * alpha_multiplier,
279            kind,
280            &data_ref.lock().unwrap(),
281        ))
282    }
283
284    fn from_state(
285        renderer: &mut R,
286        id: Id,
287        location: Point<f64, Physical>,
288        alpha: f32,
289        kind: Kind,
290        data: &RendererSurfaceState,
291    ) -> Option<Self>
292    where
293        R::TextureId: Clone + 'static,
294    {
295        let buffer = data.buffer()?.clone();
296
297        let texture = if let Ok(spb) = crate::wayland::single_pixel_buffer::get_single_pixel_buffer(&buffer) {
298            WaylandSurfaceTexture::SolidColor(Color32F::from(spb.rgba32f()))
299        } else {
300            WaylandSurfaceTexture::Texture(data.texture(renderer.context_id())?.clone())
301        };
302
303        Some(Self {
304            id,
305            location,
306            alpha,
307            kind,
308            view: data.view()?,
309            buffer,
310            buffer_scale: data.buffer_scale(),
311            buffer_transform: data.buffer_transform(),
312            buffer_dimensions: data.buffer_dimensions?,
313            damage: data.damage.snapshot(),
314            opaque_regions: data
315                .opaque_regions()
316                .map(OpaqueRegions::from_slice)
317                .unwrap_or_default(),
318            texture,
319        })
320    }
321
322    fn size(&self, scale: impl Into<Scale<f64>>) -> Size<i32, Physical> {
323        ((self.view.dst.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
324            - self.location.to_i32_round())
325        .to_size()
326    }
327
328    /// Get the buffer dimensions in logical coordinates
329    pub fn buffer_size(&self) -> Size<i32, Logical> {
330        self.buffer_dimensions
331            .to_logical(self.buffer_scale, self.buffer_transform)
332    }
333
334    /// Get the view into the surface
335    pub fn view(&self) -> SurfaceView {
336        self.view
337    }
338
339    /// Get the buffer texture
340    pub fn texture(&self) -> &WaylandSurfaceTexture<R> {
341        &self.texture
342    }
343}

345impl<R: Renderer + ImportAll> Element for WaylandSurfaceRenderElement<R> {
346    fn id(&self) -> &Id {
347        &self.id
348    }
349
350    fn current_commit(&self) -> CommitCounter {
351        self.damage.current_commit()
352    }
353
354    fn geometry(&self, scale: Scale<f64>) -> Rectangle<i32, Physical> {
355        Rectangle::new(self.location.to_i32_round(), self.size(scale))
356    }
357
358    fn src(&self) -> Rectangle<f64, BufferCoords> {
359        self.view.src.to_buffer(
360            self.buffer_scale as f64,
361            self.buffer_transform,
362            &self.buffer_size().to_f64(),
363        )
364    }
365
366    fn transform(&self) -> Transform {
367        self.buffer_transform
368    }
369
370    fn damage_since(&self, scale: Scale<f64>, commit: Option<CommitCounter>) -> DamageSet<i32, Physical> {
371        let dst_size = self.size(scale);
372        self.damage
373            .damage_since(commit)
374            .unwrap_or_else(|| DamageSet::from_slice(&[Rectangle::from_size(self.buffer_dimensions)]))
375            .iter()
376            .filter_map(|rect| {
377                rect.to_f64()
378                    // first bring the damage into logical space
379                    // Note: We use f64 for this as the damage could
380                    // be not dividable by the buffer scale without
381                    // a rest
382                    .to_logical(
383                        self.buffer_scale as f64,
384                        self.buffer_transform,
385                        &self.buffer_dimensions.to_f64(),
386                    )
387                    // then crop by the surface view (viewporter for example could define a src rect)
388                    .intersection(self.view.src)
389                    // move and scale the cropped rect (viewporter could define a dst size)
390                    .map(|rect| self.view.rect_to_global(rect).to_i32_up::<i32>())
391                    // now bring the damage to physical space
392                    .map(|rect| {
393                        // We calculate the scale between to rounded
394                        // surface size and the scaled surface size
395                        // and use it to scale the damage to the rounded
396                        // surface size by multiplying the output scale
397                        // with the result.
398                        let surface_scale = dst_size.to_f64() / self.view.dst.to_f64().to_physical(scale);
399                        rect.to_physical_precise_up(surface_scale * scale)
400                    })
401            })
402            .collect::<DamageSet<_, _>>()
403    }
404
405    fn opaque_regions(&self, scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
406        if self.alpha < 1.0 {
407            return OpaqueRegions::default();
408        }
409
410        self.opaque_regions
411            .iter()
412            .map(|r| {
413                let loc = r.loc.to_physical_precise_round(scale);
414                let size = ((r.size.to_f64().to_physical(scale).to_point() + self.location).to_i32_round()
415                    - self.location.to_i32_round())
416                .to_size();
417                Rectangle::new(loc, size)
418            })
419            .collect::<OpaqueRegions<_, _>>()
420    }
421
422    fn alpha(&self) -> f32 {
423        self.alpha
424    }
425
426    fn kind(&self) -> Kind {
427        self.kind
428    }
429}

431impl<R> RenderElement<R> for WaylandSurfaceRenderElement<R>
432where
433    R: Renderer + ImportAll,
434    R::TextureId: Texture + 'static,
435{
436    #[inline]
437    fn underlying_storage(&self, _renderer: &mut R) -> Option<UnderlyingStorage<'_>> {
438        Some(UnderlyingStorage::Wayland(&self.buffer))
439    }
440
441    #[instrument(level = "trace", skip(frame))]
442    #[profiling::function]
443    fn draw(
444        &self,
445        frame: &mut R::Frame<'_, '_>,
446        src: Rectangle<f64, BufferCoords>,
447        dst: Rectangle<i32, Physical>,
448        damage: &[Rectangle<i32, Physical>],
449        opaque_regions: &[Rectangle<i32, Physical>],
450    ) -> Result<(), R::Error> {
451        match self.texture {
452            WaylandSurfaceTexture::Texture(ref texture) => frame.render_texture_from_to(
453                texture,
454                src,
455                dst,
456                damage,
457                opaque_regions,
458                self.buffer_transform,
459                self.alpha,
460            ),
461            WaylandSurfaceTexture::SolidColor(color) => frame.draw_solid(dst, damage, color * self.alpha),
462        }
463    }
