12/// A element that allows to re-scale another element
13#[derive(Debug)]
14pub struct RescaleRenderElement<E> {
15    element: E,
16    origin: Point<i32, Physical>,
17    scale: Scale<f64>,
18}
19
20impl<E: Element> RescaleRenderElement<E> {
21    /// Create a new re-scale element for an existing element
22    ///
23    /// The origin can be used to scale the element geometry relative to a [`Point`].
24    /// One use case of this is to only scale the location for some elements in a group, like
25    /// a surface tree.
26    pub fn from_element(element: E, origin: Point<i32, Physical>, scale: impl Into<Scale<f64>>) -> Self {
27        RescaleRenderElement {
28            element,
29            origin,
30            scale: scale.into(),
31        }
32    }
33}
34
35impl<E: Element> Element for RescaleRenderElement<E> {
36    fn id(&self) -> &Id {
37        self.element.id()
38    }
39
40    fn current_commit(&self) -> crate::backend::renderer::utils::CommitCounter {
41        self.element.current_commit()
42    }
43
44    fn src(&self) -> crate::utils::Rectangle<f64, crate::utils::Buffer> {
45        self.element.src()
46    }
47
48    fn geometry(
49        &self,
50        scale: crate::utils::Scale<f64>,
51    ) -> crate::utils::Rectangle<i32, crate::utils::Physical> {
52        let mut element_geometry = self.element.geometry(scale);
53        // First we make the element relative to the origin
54        element_geometry.loc -= self.origin;
55        // Then we scale it by our scale
56        element_geometry = element_geometry.to_f64().upscale(self.scale).to_i32_round();
57        // At last we move it back to the origin
58        element_geometry.loc += self.origin;
59        element_geometry
60    }
61
62    fn transform(&self) -> crate::utils::Transform {
63        self.element.transform()
64    }
65
66    fn damage_since(
67        &self,
68        scale: crate::utils::Scale<f64>,
69        commit: Option<crate::backend::renderer::utils::CommitCounter>,
70    ) -> DamageSet<i32, Physical> {
71        self.element
72            .damage_since(scale, commit)
73            .into_iter()
74            .map(|rect| rect.to_f64().upscale(self.scale).to_i32_up())
75            .collect::<DamageSet<_, _>>()
76    }
77
78    fn opaque_regions(&self, scale: crate::utils::Scale<f64>) -> OpaqueRegions<i32, Physical> {
79        self.element
80            .opaque_regions(scale)
81            .into_iter()
82            .map(|rect| rect.to_f64().upscale(self.scale).to_i32_round())
83            .collect::<OpaqueRegions<_, _>>()
84    }
85
86    fn alpha(&self) -> f32 {
87        self.element.alpha()
88    }
89
90    fn kind(&self) -> Kind {
91        self.element.kind()
92    }
93}

113/// A element that allows to crop another element
114#[derive(Debug)]
115pub struct CropRenderElement<E> {
116    element: E,
117    src: Rectangle<f64, Buffer>,
118    crop_rect: Rectangle<i32, Physical>,
119}
120
121impl<E: Element> CropRenderElement<E> {
122    /// Create a cropping render element for an existing element
123    ///
124    /// The crop rect is expected to be relative to the same origin the element is relative to.
125    /// It can extend outside of the element geometry but the resulting geometry will always
126    /// be equal or smaller than the geometry of the original element.
127    ///
128    /// The scale is used to calculate the intersection between the crop rect and the
129    /// original element geometry and should therefore equal the scale that was used to
130    /// calculate the crop rect.
131    ///
132    /// Returns `None` if there is no overlap between the crop rect and the element geometry
133    pub fn from_element(
134        element: E,
135        scale: impl Into<Scale<f64>>,
136        crop_rect: Rectangle<i32, Physical>,
137    ) -> Option<Self> {
138        let scale = scale.into();
139
140        let element_geometry = element.geometry(scale);
141
142        if let Some(intersection) = element_geometry.intersection(crop_rect) {
143            // FIXME: intersection sometimes return a 0 size element
144            if intersection.is_empty() {
145                return None;
146            }
147
148            // We need to know how much we should crop from the element
149            let mut element_relative_intersection = intersection;
150            element_relative_intersection.loc -= element_geometry.loc;
151
152            // Now we calculate the scale from the original element geometry to the
153            // original element src. We need that scale to bring our intersection
154            // into buffer space.
155            // We also have to consider the buffer transform for this or otherwise
156            // the scale would be wrong
157            let element_src = element.src();
158            let transform = element.transform();
159            let physical_to_buffer_scale =
160                element_src.size / transform.invert().transform_size(element_geometry.size).to_f64();
161
162            // Ok, for the src we need to know how much we cropped from the element geometry
163            // and then bring that rectangle into buffer space. For this we have to first
164            // apply the element transform and then scale it to buffer space.
165            let mut src = element_relative_intersection.to_f64().to_logical(1.0).to_buffer(
166                physical_to_buffer_scale,
167                transform,
168                &element_geometry.size.to_f64().to_logical(1.0),
169            );
170
171            // Ensure cropping of the existing element is respected.
172            src.loc += element_src.loc;
173
174            Some(CropRenderElement {
175                element,
176                src,
177                crop_rect,
178            })
179        } else {
180            None
181        }
182    }
183
184    fn element_crop_rect(&self, scale: Scale<f64>) -> Option<Rectangle<i32, Physical>> {
185        let element_geometry = self.element.geometry(scale);
186
187        if let Some(mut intersection) = element_geometry.intersection(self.crop_rect) {
188            // FIXME: intersection sometimes return a 0 size element
189            if intersection.is_empty() {
190                return None;
191            }
192
193            intersection.loc -= element_geometry.loc;
194            Some(intersection)
195        } else {
196            None
197        }
198    }
199}

211        self.src
212    }

214    fn geometry(&self, scale: Scale<f64>) -> crate::utils::Rectangle<i32, Physical> {
215        let element_geometry = self.element.geometry(scale);
216        if let Some(intersection) = element_geometry.intersection(self.crop_rect) {
217            // FIXME: intersection sometimes return a 0 size element
218            if intersection.is_empty() {
219                return Default::default();
220            }
221
222            intersection
223        } else {
224            Default::default()
225        }
226    }
227
228    fn transform(&self) -> crate::utils::Transform {
229        self.element.transform()
230    }
231
232    fn damage_since(
233        &self,
234        scale: Scale<f64>,
235        commit: Option<crate::backend::renderer::utils::CommitCounter>,
236    ) -> DamageSet<i32, Physical> {
237        if let Some(element_crop_rect) = self.element_crop_rect(scale) {
238            self.element
239                .damage_since(scale, commit)
240                .into_iter()
241                .flat_map(|rect| {
242                    rect.intersection(element_crop_rect).map(|mut rect| {
243                        rect.loc -= element_crop_rect.loc;
244                        rect
245                    })
246                })
247                .collect::<DamageSet<_, _>>()
248        } else {
249            Default::default()
250        }
251    }
252
253    fn opaque_regions(&self, scale: Scale<f64>) -> OpaqueRegions<i32, Physical> {
254        if let Some(element_crop_rect) = self.element_crop_rect(scale) {
255            self.element
256                .opaque_regions(scale)
257                .into_iter()
258                .flat_map(|rect| {
259                    rect.intersection(element_crop_rect).map(|mut rect| {
260                        rect.loc -= element_crop_rect.loc;
261                        rect
262                    })
263                })
264                .collect::<OpaqueRegions<_, _>>()
265        } else {
266            Default::default()
267        }
268    }
269
270    fn alpha(&self) -> f32 {
271        self.element.alpha()
272    }
273
274    fn kind(&self) -> Kind {
275        self.element.kind()
276    }

297/// Defines how the location parameter should apply in [`RelocateRenderElement::from_element`]
298#[derive(Debug, Copy, Clone)]
299pub enum Relocate {
300    /// The supplied location replaces the original location
301    Absolute,
302    /// The supplied location offsets the original location
303    Relative,
304}

306/// A element that allows to offset the location of an existing element
307#[derive(Debug)]
308pub struct RelocateRenderElement<E> {
309    element: E,
310    relocate: Relocate,
311    location: Point<i32, Physical>,
312}
313
314impl<E: Element> RelocateRenderElement<E> {
315    /// Crate an re-locate element for an existing element
316    pub fn from_element(element: E, location: impl Into<Point<i32, Physical>>, relocate: Relocate) -> Self {
317        let location = location.into();
318
319        RelocateRenderElement {
320            element,
321            location,
322            relocate,
323        }
324    }
325}

353        }

399/// Defines the scale behavior for the constrain
400#[derive(Debug, Copy, Clone)]
401pub enum ConstrainScaleBehavior {
402    /// Fit the element into the size, nothing will be cropped
403    Fit,
404    /// Zoom the element into the size, crops if the aspect ratio
405    /// of the element does not match the aspect of the constrain
406    /// size
407    Zoom,
408    /// Always stretch the element to the constrain size
409    Stretch,
410    /// Do not scale, but cut off at the constrain size
411    CutOff,
412}
413
414bitflags::bitflags! {
415    /// Defines how the elements should be aligned during constrain
416    #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
417    pub struct ConstrainAlign: u8 {
418        /// Align to the top
419        const TOP = 0b000001;
420        /// Align to the left
421        const LEFT = 0b000010;
422        /// Align to the right
423        const RIGHT = 0b000100;
424        /// Align to the bottom
425        const BOTTOM = 0b001000;
426        /// Align to the top left
427        ///
428        /// Equals TOP | LEFT
429        const TOP_LEFT = Self::TOP.bits() | Self::LEFT.bits();
430        /// Align to the top right
431        ///
432        /// Equals TOP | RIGHT
433        const TOP_RIGHT = Self::TOP.bits() | Self::RIGHT.bits();
434        /// Align to the bottom left
435        ///
436        /// Equals BOTTOM | LEFt
437        const BOTTOM_LEFT = Self::BOTTOM.bits() | Self::LEFT.bits();
438        /// Align to the bottom left
439        ///
440        /// Equals BOTTOM | LEFT
441        const BOTTOM_RIGHT = Self::BOTTOM.bits() | Self::RIGHT.bits();
442        /// Align to the center
443        ///
444        /// Equals TOP | LEFT | BOTTOM | RIGHT
445        const CENTER = Self::TOP.bits() | Self::LEFT.bits() | Self::BOTTOM.bits() | Self::RIGHT.bits();
446    }
447}

449/// Convenience function to constrain something that implements [`AsRenderElements`]
450///
451/// See [`constrain_render_elements`] for more information
452#[profiling::function]
453#[allow(clippy::too_many_arguments)]
454pub fn constrain_as_render_elements<R, E, C>(
455    element: &E,
456    renderer: &mut R,
457    location: impl Into<Point<i32, Physical>>,
458    alpha: f32,
459    constrain: Rectangle<i32, Physical>,
460    reference: Rectangle<i32, Physical>,
461    behavior: ConstrainScaleBehavior,
462    align: ConstrainAlign,
463    output_scale: impl Into<Scale<f64>>,
464) -> impl Iterator<Item = C>
465where
466    R: Renderer,
467    E: AsRenderElements<R>,
468    C: From<
469        CropRenderElement<
470            RelocateRenderElement<RescaleRenderElement<<E as AsRenderElements<R>>::RenderElement>>,
471        >,
472    >,
473{
474    let location = location.into();
475    let output_scale = output_scale.into();
476    let elements: Vec<<E as AsRenderElements<R>>::RenderElement> =
477        AsRenderElements::<R>::render_elements(element, renderer, location, output_scale, alpha);
478    constrain_render_elements(
479        elements,
480        location,
481        constrain,
482        reference,
483        behavior,
484        align,
485        output_scale,
486    )
487    .map(C::from)
488}

490/// Constrain render elements on a specific location with a specific size
491///
492/// * `origin` - Defines the origin for re-scaling
493/// * `constrain` - Defines the rectangle on screen the elements should be constrain within
494/// * `reference` - Defines the reference that should be used for constraining the elements,
495///                 this is most commonly the bounding box or geometry of the elements
496/// * `behavior` - Defines the behavior for scaling the elements reference in the constrain
497/// * `align` - Defines how the scaled elements should be aligned within the constrain
498/// * `scale` - The scale that was used to create the original elements
499#[profiling::function]
500pub fn constrain_render_elements<E>(
501    elements: impl IntoIterator<Item = E>,
502    origin: impl Into<Point<i32, Physical>>,
503    constrain: Rectangle<i32, Physical>,
504    reference: Rectangle<i32, Physical>,
505    behavior: ConstrainScaleBehavior,
506    align: ConstrainAlign,
507    scale: impl Into<Scale<f64>>,
508) -> impl Iterator<Item = CropRenderElement<RelocateRenderElement<RescaleRenderElement<E>>>>
509where
510    E: Element,
511{
512    let location = origin.into();
513    let scale = scale.into();
514
515    let element_scale = match behavior {
516        ConstrainScaleBehavior::Fit => {
517            let reference = reference.to_f64();
518            let size = constrain.size.to_f64();
519            let element_scale: Scale<f64> = size / reference.size;
520            Scale::from(f64::min(element_scale.x, element_scale.y))
521        }
522        ConstrainScaleBehavior::Zoom => {
523            let reference = reference.to_f64();
524            let size = constrain.size.to_f64();
525            let element_scale: Scale<f64> = size / reference.size;
526            Scale::from(f64::max(element_scale.x, element_scale.y))
527        }
528        ConstrainScaleBehavior::Stretch => {
529            let reference = reference.to_f64();
530            let size = constrain.size.to_f64();
531            size / reference.size
532        }
533        ConstrainScaleBehavior::CutOff => Scale::from(1.0),
534    };
535
536    let scaled_reference = reference.to_f64().upscale(element_scale);
537
538    // Calculate the align offset
539    let top_offset: f64 = if align.contains(ConstrainAlign::TOP | ConstrainAlign::BOTTOM) {
540        (constrain.size.h as f64 - scaled_reference.size.h) / 2f64
541    } else if align.contains(ConstrainAlign::BOTTOM) {
542        constrain.size.h as f64 - scaled_reference.size.h
543    } else {
544        0f64
545    };
546
547    let left_offset: f64 = if align.contains(ConstrainAlign::LEFT | ConstrainAlign::RIGHT) {
548        (constrain.size.w as f64 - scaled_reference.size.w) / 2f64
549    } else if align.contains(ConstrainAlign::RIGHT) {
550        constrain.size.w as f64 - scaled_reference.size.w
551    } else {
552        0f64
553    };
554
555    let align_offset: Point<f64, Physical> = Point::from((left_offset, top_offset));
556
557    // We need to offset the elements by the reference loc or otherwise
558    // the element could be positioned outside of our constrain rect.
559    let reference_offset =
560        reference.loc.to_f64() - Point::from((scaled_reference.loc.x, scaled_reference.loc.y));
561
562    // Final offset
563    let offset = (reference_offset + align_offset).to_i32_round();
564
565    elements
566        .into_iter()
567        .map(move |e| RescaleRenderElement::from_element(e, location, element_scale))
568        .map(move |e| RelocateRenderElement::from_element(e, offset, Relocate::Relative))
569        .filter_map(move |e| CropRenderElement::from_element(e, scale, constrain))
