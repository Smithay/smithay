1//!
2//! This module aims to make multi-gpu setups easier to handle for compositors.
3//!
4//! Its main entry point is the [`GpuManager`]. Initializing this with a
5//! [`GraphicsApi`] implementation will allow you to create [`MultiRenderer`]s.
6//!
7//! smithay provides the following graphics apis:
8//! - [`gbm::GbmGlesBackend`]
9//!
10//! A [`MultiRenderer`] gets created using two [`DrmNode`]s to identify gpus.
11//! One gpu will be referred to as the render-gpu, the other as the target-gpu.
12//!
13//! Note: The render- and target-gpu may be identically to allow the multigpu
14//! module to be used on single-gpu systems as well avoiding supporting multiple code-paths.
15//! Doing so will not result in worse performance compared to rendering without the multi-gpu module.
16//!
17//! A [`MultiRenderer`] will support the [`Renderer`]-trait as well
18//! as the other corresponding traits of the [`renderer`](crate::backend::renderer)-module,
19//! if the [`GraphicsApi`] allows it.
20//!
21//! Any rendering operations will take place on the render-gpu transparently.
22//! Output will be redirected the target gpu, as such any [`Bind`]- and [`Offscreen`]-implementations
23//! will be allocated on the target-gpu.
24//!
25//! The `ImportMem`-implementation will upload the texture on the render-gpu, failing *if*
26//! the render-gpu does not support the format as normal. Be careful when mixing gpu's
27//! with different memory-formats supported, what formats you announce.
28//!
29//! In contrast the `ImportDma`-implementations will transparently create copies
30//! of client buffers, if necessary, given it is unclear if a dmabuf can be imported on any given gpu.
31//! The implementation strives for the best possible performance for a given setup,
32//! when choosing a copy-path.
33//!
34//! Any `ExportMem`-implementations will originate from the render-gpu, which again
35//! needs to support the requested format directly. No paths across other gpus are tested.
36//!
37//! *Note*: This module will not keep you from selecting sub-optimal configurations.
38//! Any heuristics for which render-gpu to use for a given set of client buffers
39//! and desired target-gpu are up to be implemented by the compositor. The module only
40//! reduces the amount of necessary setup operations.
41//!

91/// Errors generated by [`GpuManager`] and [`MultiRenderer`].
92#[derive(thiserror::Error)]
93pub enum Error<R: GraphicsApi, T: GraphicsApi>
94where
95    R::Error: 'static,
96    T::Error: 'static,
97    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
98    <<T::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
99{
100    /// The graphics api errored on device enumeration
101    #[error("The render graphics api failed enumerating devices {0:?}")]
102    RenderApiError(#[source] R::Error),
103    /// The graphics api errored on device enumeration
104    #[error("The target graphics api failed enumerating devices {0:?}")]
105    TargetApiError(#[source] T::Error),
106    /// The graphics api has found no node matching the drm node
107    #[error("The graphics api has found no node matching {0:?}")]
108    NoDevice(DrmNode),
109    /// The device requested did not match the expected
110    #[error("The devices requested {0:?} did not match the expected")]
111    MismatchedDevice(DrmNode),
112    /// The device has gone missing
113    #[error("The device has gone missing")]
114    DeviceMissing,
115    /// Error on the rendering device
116    #[error("Error on the rendering device: {0:}")]
117    Render(#[source] <<R::Device as ApiDevice>::Renderer as RendererSuper>::Error),
118    /// Error on the target device
119    #[error("Error on the target device: {0:}")]
120    Target(#[source] <<T::Device as ApiDevice>::Renderer as RendererSuper>::Error),
121    /// Failed to import buffer using the api on any device
122    #[error("Failed to import buffer")]
123    ImportFailed,
124    /// Buffer allocation failed
125    #[error("Failed to allocate buffer")]
126    AllocatorError(AnyError),
127}

129impl<R: GraphicsApi, T: GraphicsApi> Error<R, T> {
130    fn transpose(self) -> Error<T, R> {
131        match self {
132            Error::Render(err) => Error::Target(err),
133            Error::Target(err) => Error::Render(err),
134            Error::RenderApiError(err) => Error::TargetApiError(err),
135            Error::TargetApiError(err) => Error::RenderApiError(err),
136            Error::NoDevice(d) => Error::NoDevice(d),
137            Error::MismatchedDevice(d) => Error::MismatchedDevice(d),
138            Error::DeviceMissing => Error::DeviceMissing,
139            Error::ImportFailed => Error::ImportFailed,
140            Error::AllocatorError(a) => Error::AllocatorError(a),
141        }
142    }
143}

183impl<R: GraphicsApi + 'static, T: GraphicsApi + 'static> From<Error<R, T>> for SwapBuffersError
184where
185    R::Error: Into<SwapBuffersError> + Send + Sync,
186    T::Error: Into<SwapBuffersError> + Send + Sync,
187    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Error: Into<SwapBuffersError> + Send + Sync,
188    <<T::Device as ApiDevice>::Renderer as RendererSuper>::Error: Into<SwapBuffersError> + Send + Sync,
189{
190    #[inline]
191    fn from(err: Error<R, T>) -> SwapBuffersError {
192        match err {
193            x @ Error::NoDevice(_) | x @ Error::DeviceMissing | x @ Error::AllocatorError(_) => {
194                SwapBuffersError::ContextLost(Box::new(x))
195            }
196            x @ Error::MismatchedDevice(_) | x @ Error::ImportFailed => {
197                SwapBuffersError::TemporaryFailure(Box::new(x))
198            }
199            Error::RenderApiError(x) => x.into(),
200            Error::TargetApiError(x) => x.into(),
201            Error::Render(x) => x.into(),
202            Error::Target(x) => x.into(),
203        }
204    }

251    /// Create a [`MultiRenderer`] from a single device.
252    ///
253    /// This a convenience function to deal with the same types even, if you only need one device.
254    /// Because no copies are necessary in these cases, all extra arguments can be omitted.

256    #[profiling::function]
257    pub fn single_renderer<'api>(
258        &'api mut self,
259        device: &DrmNode,
260    ) -> Result<MultiRenderer<'api, 'api, A, A>, Error<A, A>> {
261        if !self.devices.iter().any(|dev| dev.node() == device) || self.api.needs_enumeration() {
262            self.api
263                .enumerate(&mut self.devices)
264                .map_err(Error::RenderApiError)?;
265        }
266
267        if !self.devices.iter().any(|dev| dev.node() == device) {
268            return Err(Error::NoDevice(*device));
269        }
270
271        let (mut render, others) = self
272            .devices
273            .iter_mut()
274            .partition::<Vec<_>, _>(|dev| dev.node() == device);
275
276        Ok(MultiRenderer {
277            render: render.remove(0),
278            target: None,
279            other_renderers: others,
280            span: tracing::Span::current(),
281        })
282    }

284    /// Create a [`MultiRenderer`].
285    ///
286    /// - `render_device` should refer to the gpu node rendering operations will take place upon.
287    /// - `target_device` should refer to the gpu node the composited buffer will end up upon
288    /// - `allocator` should refer to an `Allocator`, that works guaranteed with the `render_device`
289    ///     to do offscreen composition on. Dma copies will be used, if buffers returned by the allocator
290    ///     also work on the `target_device`.
291    /// - `copy_format` denotes the format buffers will be allocated in for offscreen rendering.
292    ///
293    /// It is valid to pass the same devices for both, but you *should* use [`GpuManager::single_renderer`] in those cases.
294    #[instrument(level = "trace", parent = &self.span, skip(self))]
295    #[profiling::function]
296    pub fn renderer<'api>(
297        &'api mut self,
298        render_device: &DrmNode,
299        target_device: &DrmNode,
300        copy_format: Fourcc,
301    ) -> Result<MultiRenderer<'api, 'api, A, A>, Error<A, A>>
302    where
303        <A::Device as ApiDevice>::Renderer: Bind<Dmabuf>,
304    {
305        if !self.devices.iter().any(|device| device.node() == render_device)
306            || !self.devices.iter().any(|device| device.node() == target_device)
307            || self.api.needs_enumeration()
308        {
309            self.api
310                .enumerate(&mut self.devices)
311                .map_err(Error::RenderApiError)?;
312        }
313
314        if !self.devices.iter().any(|device| device.node() == render_device) {
315            return Err(Error::NoDevice(*render_device));
316        }
317        if !self.devices.iter().any(|device| device.node() == target_device) {
318            return Err(Error::NoDevice(*target_device));
319        }
320
321        let (mut render, others) = self
322            .devices
323            .iter_mut()
324            .partition::<Vec<_>, _>(|device| device.node() == render_device);
325        if target_device != render_device {
326            let (mut target, others) = others
327                .into_iter()
328                .partition::<Vec<_>, _>(|device| device.node() == target_device);
329
330            Ok(MultiRenderer {
331                render: render.remove(0),
332                target: Some(TargetData {
333                    device: target.remove(0),
334                    cached_buffer: self
335                        .dmabuf_cache
336                        .entry((*render_device, *target_device))
337                        .or_default(),
338                    format: copy_format,
339                }),
340                other_renderers: others,
341                span: tracing::Span::current(),
342            })
343        } else {
344            Ok(MultiRenderer {
345                render: render.remove(0),
346                target: None,
347                other_renderers: others,
348                span: tracing::Span::current(),
349            })
350        }
351    }

413        }
414

417            .iter_mut()
418            .partition::<Vec<_>, _>(|device| device.node() == render_device);

449    /// Function for optimizing buffer imports across multiple gpus.
450    ///
451    /// If you are using [`MultiRenderer`]s do rendering of your client buffers,
452    /// you can call `early_import` on commit to start necessary copy processes early.
453    ///
454    /// - `target` refers to the gpu node, that the buffer needs to be accessible on later.
455    ///    *Note*: Usually this will be **render**ing gpu of a [`MultiRenderer`]
456    /// - `surface` is the wayland surface, whose buffer and subsurfaces buffers shall be imported
457    ///
458    /// Note: This will do nothing, if you are not using
459    /// [`crate::backend::renderer::utils::on_commit_buffer_handler`]
460    /// to let smithay handle buffer management.
461    #[cfg(feature = "wayland_frontend")]
462    #[instrument(level = "trace", parent = &self.span, skip(self))]
463    #[profiling::function]
464    pub fn early_import(&mut self, target: DrmNode, surface: &WlSurface) -> Result<(), Error<A, A>>
465    where
466        A: 'static,
467        <A::Device as ApiDevice>::Renderer: ImportMemWl + ImportDmaWl + ExportMem,
468        <<A::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping: 'static,
469    {
470        use crate::{
471            backend::renderer::utils::RendererSurfaceStateUserData,
472            wayland::compositor::{with_surface_tree_upward, TraversalAction},
473        };
474
475        if self.devices.is_empty() {
476            return Err(Error::DeviceMissing);
477        }
478
479        let mut result = Ok(());
480        with_surface_tree_upward(
481            surface,
482            (),
483            |_surface, states, _| {
484                if let Some(data) = states.data_map.get::<RendererSurfaceStateUserData>() {
485                    let mut data_ref = data.lock().unwrap();
486                    let data = &mut *data_ref;
487                    if data.textures.is_empty() {
488                        // Import a new buffer if available
489                        if let Some(buffer) = data.buffer.as_ref() {
490                            // We do an optimistic optimization here, so contrary to many much more defensive damage-tracking algorithms,
491                            // we only import the most recent set of damage here.
492                            // If we need more on rendering - which we cannot know at this point - we will call import_missing later
493                            // to receive the rest.
494                            // FIXME: We should be able to get rid of this allocation here
495                            let buffer_damage = data.damage().raw().take(1).flatten().cloned().fold(
496                                Vec::<Rectangle<i32, BufferCoords>>::new(),
497                                |damage, mut rect| {
498                                    // replace with drain_filter, when that becomes stable to reuse the original Vec's memory
499                                    let (overlapping, mut new_damage): (Vec<_>, Vec<_>) = damage
500                                        .into_iter()
501                                        .partition(|other| other.overlaps_or_touches(rect));
502
503                                    for overlap in overlapping {
504                                        rect = rect.merge(overlap);
505                                    }
506                                    new_damage.push(rect);
507                                    new_damage
508                                },
509                            );
510
511                            if let Err(err) = self.early_import_buffer(target, buffer, states, &buffer_damage)
512                            {
513                                result = Err(err);
514                            }
515                        }
516                        // Now, was the import successful?
517                        if result.is_ok() {
518                            TraversalAction::DoChildren(())
519                        } else {
520                            // we are not displayed, so our children are neither
521                            TraversalAction::SkipChildren
522                        }
523                    } else {
524                        TraversalAction::SkipChildren
525                    }
526                } else {
527                    TraversalAction::SkipChildren
528                }
529            },
530            |_, _, _| {},
531            |_, _, _| true,
532        );
533
534        result
535    }

537    #[cfg(feature = "wayland_frontend")]
538    #[profiling::function]
539    fn early_import_buffer(
540        &mut self,
541        target_node: DrmNode,
542        buffer: &wl_buffer::WlBuffer,
543        surface: &SurfaceData,
544        damage: &[Rectangle<i32, BufferCoords>],
545    ) -> Result<(), Error<A, A>>
546    where
547        A: 'static,
548        <A::Device as ApiDevice>::Renderer: ImportMemWl + ImportDmaWl + ExportMem,
549        <<A::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping: 'static,
550    {
551        match buffer_type(buffer) {
552            Some(BufferType::Dma) => {
553                let dmabuf = get_dmabuf(buffer).unwrap();
554                let mut texture = MultiTexture::from_surface(Some(surface), dmabuf.size(), dmabuf.format());
555
556                if !self.devices.iter().any(|device| target_node == *device.node()) {
557                    return Err(Error::DeviceMissing);
558                }
559
560                let mut devices = self.devices.iter_mut();
561                let first = devices.next().unwrap();
562
563                let src_node = import_on_src_node(dmabuf, Some(damage), &mut texture, first, None, devices)?;
564                if src_node != target_node {
565                    let target_id = self
566                        .devices
567                        .iter()
568                        .find_map(|dev| (*dev.node() == target_node).then(|| dev.renderer().context_id()))
569                        .unwrap();
570                    let src_id = self
571                        .devices
572                        .iter()
573                        .find_map(|dev| {
574                            (*dev.node() == src_node).then(|| dev.renderer().context_id().erased())
575                        })
576                        .unwrap();
577
578                    let texture_internal = texture.0.lock().unwrap();
579                    {
580                        let target_texture = texture_internal.textures.get(&target_id.erased());
581                        if !matches!(
582                            target_texture,
583                            Some(GpuSingleTexture::Mem {
584                                external_shadow: None,
585                                ..
586                            })
587                        ) {
588                            // if we don't have a mem texture, either
589                            // - we don't have any import, in that case lets figure this out during first import
590                            // - we have a dma shadow copy, in that case we can't copy without invalidating the shadow buffer
591                            //   if the damage doesn't match up with what we be requested during the next import.
592                            // - this also applies if it is an external copy, so we additionally need a shadow buffer before reading from memory.
593                            return Ok(());
594                        }
595                    }
596
597                    // if we do need to do a memory copy, we start with the export early
598
599                    let src_texture = match texture_internal.textures.get(&src_id).unwrap() {
600                        GpuSingleTexture::Direct(tex) => tex
601                            .downcast_ref::<<<A::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>(
602                            )
603                            .unwrap(),
604                        _ => unreachable!(),
605                    };
606
607                    let format = src_texture
608                        .format()
609                        // Check the target device supports the format, if not use 8-bit format
610                        .filter(|format| {
611                            self.devices
612                                .iter()
613                                .find(|device| target_node == *device.node())
614                                .unwrap()
615                                .renderer()
616                                .mem_formats()
617                                .any(|fmt| fmt == *format)
618                        })
619                        // The GL spec *always* supports this format.
620                        // TODO: Re-evaluate this, once we support vulkan
621                        .unwrap_or(Fourcc::Abgr8888);
622
623                    let src_device = self
624                        .devices
625                        .iter_mut()
626                        .find(|device| src_node == *device.node())
627                        .unwrap();
628
629                    let mappings = {
630                        let damage = damage
631                            .iter()
632                            .flat_map(|rect| rect.intersection(Rectangle::from_size(src_texture.size())))
633                            .fold(Vec::<Rectangle<i32, BufferCoords>>::new(), |damage, mut rect| {
634                                // replace with drain_filter, when that becomes stable to reuse the original Vec's memory
635                                let (overlapping, mut new_damage): (Vec<_>, Vec<_>) = damage
636                                    .into_iter()
637                                    .partition(|other| other.overlaps_or_touches(rect));
638
639                                for overlap in overlapping {
640                                    rect = rect.merge(overlap);
641                                }
642                                new_damage.push(rect);
643                                new_damage
644                            });
645                        damage
646                            .iter()
647                            .copied()
648                            .map(|damage| {
649                                let mapping = src_device
650                                    .renderer_mut()
651                                    .copy_texture(src_texture, damage, format)
652                                    .map_err(Error::Target)?;
653                                Ok((damage, mapping))
654                            })
655                            .collect::<Result<Vec<_>, Error<A, A>>>()?
656                    };
657
658                    std::mem::drop(texture_internal);
659                    texture.insert_mapping::<A, A, _>(
660                        src_node,
661                        &target_id,
662                        texture.size(),
663                        mappings.into_iter(),
664                    );
665                    surface.data_map.insert_if_missing_threadsafe(|| texture.0);
666                }
667
668                Ok(())
669            }
670            #[cfg(all(
671                feature = "wayland_frontend",
672                feature = "backend_egl",
673                feature = "use_system_lib"
674            ))]
675            Some(BufferType::Egl) => {
676                // we need specialization for requiring ImportEGL
677                // or require ImportAll, which will block this function for all
678                // renderers that cannot import egl buffers, so we just don't
679                // and sadly go the slow path
680                Ok(())
681            }
682            Some(BufferType::Shm) => {
683                // we just need to upload in import_shm_buffer
684                Ok(())
685            }
686            Some(BufferType::SinglePixel) => {
687                // no need to do anything
688                Ok(())
689            }
690            None => {
691                // whelp, nothing we can do
692                Ok(())
693            }
694        }
695    }
696}

698/// A graphics api, that supports enumerating graphics devices
699pub trait GraphicsApi {

701    type Device: ApiDevice;

1206                .map_err(Error::Render)?,
1207            MultiFramebufferInternal::Target(target_framebuffer) => {
1208                let (target_device, render_framebuffer, texture, format) = target_state.unwrap();
1209                target = Some(TargetFrameData {
1210                    device: target_device,
1211                    framebuffer: target_framebuffer,
1212                    texture,
1213                    format,
1214                });
1215                let mut render_framebuffer = AliasableBox::from_unique(Box::new(render_framebuffer));
1216
1217                // We extend the lifetime to 'frame, because this is self-referential.
1218                // SAFETY:
1219                //  - We drop the framebuffer before `target`, which contains the referenced dmabuf
1220                //  - We drop the frame before the framebuffer as we store both in `MultiFrame`
1221                //  - `Frame` can't store an invalid pointer into the framebuffer, as the framebuffer is moved
1222                //    to the heap and won't be moved by the compiler thanks to `AliasableBox`.
1223                let frame = unsafe {
1224                    std::mem::transmute::<
1225                        <<R::Device as ApiDevice>::Renderer as RendererSuper>::Frame<'_, '_>,
1226                        <<R::Device as ApiDevice>::Renderer as RendererSuper>::Frame<'frame, 'buffer>,
1227                    >(
1228                        self.render
1229                            .renderer_mut()
1230                            .render(&mut *render_framebuffer, size, dst_transform)
1231                            .map_err(Error::Render)?,
1232                    )
1233                };
1234                new_framebuffer = Some(render_framebuffer);
1235                frame
1236            }
1237        };
1238
1239        let span = trace_span!(
1240            parent: &self.span,
1241            "renderer_multi_frame",
1242            direct = target.as_ref().is_some_and(|t| t.texture.is_some()),
1243        )
1244        .entered();
1245
1246        Ok(MultiFrame {
1247            node,
1248            frame: Some(frame),
1249            framebuffer: new_framebuffer,
1250            render: ptr, // this is fine, as long as we have the frame, this ptr is valid
1251            target,
1252            dst_transform,
1253            size,
1254            damage: Vec::new(),
1255            span,
1256        })
1257    }

1281fn create_shared_dma_framebuffer<R, T: GraphicsApi>(
1282    buffer_size: Size<i32, BufferCoords>,
1283    src: &mut R::Device,
1284    target: &mut TargetData<'_, T>,
1285) -> Result<Dmabuf, Error<R, T>>
1286where
1287    R: GraphicsApi + 'static,
1288    R::Error: 'static,
1289    T::Error: 'static,
1290    <R::Device as ApiDevice>::Renderer: Bind<Dmabuf> + ExportMem + ImportDma + ImportMem,
1291    <T::Device as ApiDevice>::Renderer: ImportDma + ImportMem,
1292    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
1293    <<T::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
1294{
1295    if !target.device.can_do_cross_device_imports() {
1296        return Err(Error::ImportFailed);
1297    }
1298
1299    let target_formats = ImportDma::dmabuf_formats(target.device.renderer())
1300        .iter()
1301        .filter(|format| format.code == target.format)
1302        .copied()
1303        .collect::<FormatSet>();
1304    let render_formats = Bind::<Dmabuf>::supported_formats(src.renderer()).unwrap_or_default();
1305    let formats = target_formats.intersection(&render_formats);
1306    let target_modifiers = formats
1307        .map(|format| format.modifier)
1308        .filter(|modifier| *modifier != Modifier::Invalid)
1309        .collect::<Vec<_>>();
1310
1311    if target_modifiers.is_empty() {
1312        return Err(Error::ImportFailed);
1313    }
1314
1315    info!(
1316        "Found dma-copy set for {:?} <-> {:?}: {:?}@{:?}",
1317        src.node(),
1318        target.device.node(),
1319        target.format,
1320        target_modifiers,
1321    );
1322
1323    let mut dmabuf = src
1324        .allocator()
1325        .create_buffer(
1326            buffer_size.w as u32,
1327            buffer_size.h as u32,
1328            target.format,
1329            &target_modifiers,
1330        )
1331        .map_err(Error::AllocatorError)?;
1332
1333    // verify we can bind on src and import on target
1334
1335    src.renderer_mut().bind(&mut dmabuf).map_err(Error::Render)?;
1336
1337    target
1338        .device
1339        .renderer_mut()
1340        .import_dmabuf(&dmabuf, Some(&[Rectangle::from_size(buffer_size)]))
1341        .map_err(Error::Target)?;
1342
1343    Ok(dmabuf)
1344}

1356    fn flush_frame(&mut self) -> Result<(), Error<R, T>> {
1357        if self.target.is_some() {
1358            let _ = self.finish_internal()?;
1359            // now the frame is gone, lets use our unholy ptr till the end of this call:
1360            // SAFETY:
1361            // - The renderer will never be invalid because the lifetime of the frame must be shorter than the renderer.
1362            // - The pointer can't be aliased because of the following:
1363            //   - Creating a frame requires an `&mut` reference to the renderer, making the mutable borrow safe.
1364            //   - The mutable reference is used in a function which mutably borrows the frame, that either being `.finish()`
1365            //      (which takes ownership of the frame) or dropping the frame.
1366            let render = unsafe { &mut *self.render };
1367
1368            // We extend the lifetime to 'frame, because this is self-referential.
1369            // SAFETY:
1370            //  - We drop the framebuffer before `target`, which contains the referenced dmabuf
1371            //  - We drop the frame before the framebuffer as we store both in `MultiFrame`
1372            //  - `Frame` can't store an invalid pointer into the framebuffer, as the framebuffer is moved
1373            //    to the heap and won't be moved by the compiler thanks to `AliasableBox`.
1374            let frame = unsafe {
1375                std::mem::transmute::<
1376                    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Frame<'_, '_>,
1377                    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Frame<'frame, 'buffer>,
1378                >(
1379                    render
1380                        .renderer_mut()
1381                        .render(self.framebuffer.as_mut().unwrap(), self.size, self.dst_transform)
1382                        .map_err(Error::Render)?,
1383                )
1384            };
1385            self.frame = Some(frame);
1386        }
1387        Ok(())
1388    }

1390    #[instrument(level = "trace", parent = &self.span, skip(self))]
1391    #[profiling::function]
1392    fn finish_internal(&mut self) -> Result<sync::SyncPoint, Error<R, T>> {
1393        if let Some(frame) = self.frame.take() {
1394            let sync = frame.finish().map_err(Error::Render)?;
1395
1396            // now the frame is gone, lets use our unholy ptr till the end of this call:
1397            // SAFETY:
1398            // - The renderer will never be invalid because the lifetime of the frame must be shorter than the renderer.
1399            // - The pointer can't be aliased because of the following:
1400            //   - Creating a frame requires an `&mut` reference to the renderer, making the mutable borrow safe.
1401            //   - The mutable reference is used in a function which mutably borrows the frame, that either being `.finish()`
1402            //      (which takes ownership of the frame) or dropping the frame.
1403            let render = unsafe { &mut *self.render };
1404
1405            let damage_area = self.dst_transform.transform_size(self.size.to_logical(1));
1406            let mut damage = std::mem::take(&mut self.damage)
1407                .into_iter()
1408                .map(|rect| rect.to_logical(1).to_buffer(1, self.dst_transform, &damage_area))
1409                .collect::<Vec<_>>();
1410
1411            let buffer_size = self.size.to_logical(1).to_buffer(1, Transform::Normal);
1412            if let Some(target) = self.target.as_mut() {
1413                if let Some(texture) = target.texture.as_ref() {
1414                    // try gpu copy
1415                    let damage = damage
1416                        .iter()
1417                        .map(|rect| rect.to_logical(1, Transform::Normal, &buffer_size).to_physical(1))
1418                        .collect::<Vec<_>>();
1419                    let mut frame = target
1420                        .device
1421                        .renderer_mut()
1422                        .render(target.framebuffer, self.size, Transform::Normal)
1423                        .map_err(Error::Target)?;
1424                    frame.wait(&sync).map_err(Error::Target)?;
1425                    frame
1426                        .clear(Color32F::TRANSPARENT, &damage)
1427                        .map_err(Error::Target)?;
1428                    frame
1429                        .render_texture_from_to(
1430                            texture,
1431                            Rectangle::from_size(buffer_size).to_f64(),
1432                            Rectangle::from_size(self.size),
1433                            &damage,
1434                            &[Rectangle::from_size(self.size)],
1435                            Transform::Normal,
1436                            1.0,
1437                        )
1438                        .map_err(Error::Target)?;
1439                    let sync = frame.finish().map_err(Error::Target)?;
1440                    render
1441                        .renderer_mut()
1442                        .cleanup_texture_cache()
1443                        .map_err(Error::Render)?;
1444
1445                    return Ok(sync);
1446                }
1447
1448                let format = if target
1449                    .device
1450                    .renderer()
1451                    .mem_formats()
1452                    .any(|fmt| fmt == target.format)
1453                {
1454                    target.format
1455                } else {
1456                    // GL mandates this to be supported for downloading, this will truncate 10-bit data
1457                    Fourcc::Abgr8888
1458                };
1459
1460                // cpu copy
1461                damage.dedup();
1462                damage.retain(|rect| rect.overlaps_or_touches(Rectangle::from_size(buffer_size)));
1463                damage.retain(|rect| rect.size.h > 0 && rect.size.w > 0);
1464
1465                let mut copy_rects = // merge overlapping rectangles
1466                    damage.iter().cloned().fold(Vec::new(), |new_damage, mut rect| {
1467                        // replace with drain_filter, when that becomes stable to reuse the original Vec's memory
1468                        let (overlapping, mut new_damage): (Vec<_>, Vec<_>) = new_damage
1469                            .into_iter()
1470                            .partition(|other: &Rectangle<i32, BufferCoords>| other.overlaps_or_touches(rect));
1471
1472                        for overlap in overlapping {
1473                            rect = rect.merge(overlap);
1474                        }
1475                        new_damage.push(rect);
1476                        new_damage
1477                    });
1478                if copy_rects.len() > MAX_CPU_COPIES {
1479                    copy_rects = Vec::from([Rectangle::from_size(buffer_size)]);
1480                }
1481
1482                let mut mappings = Vec::new();
1483                for rect in copy_rects {
1484                    let mapping = (
1485                        ExportMem::copy_framebuffer(
1486                            render.renderer_mut(),
1487                            self.framebuffer.as_ref().unwrap(),
1488                            rect,
1489                            format,
1490                        )
1491                        .map_err(Error::Render)?,
1492                        rect,
1493                    );
1494                    mappings.push(mapping);
1495                }
1496
1497                if mappings.is_empty() {
1498                    render
1499                        .renderer_mut()
1500                        .cleanup_texture_cache()
1501                        .map_err(Error::Render)?;
1502                    return Ok(sync::SyncPoint::signaled());
1503                }
1504
1505                let textures = mappings
1506                    .into_iter()
1507                    .map(|(mapping, rect)| {
1508                        let slice = ExportMem::map_texture(render.renderer_mut(), &mapping)
1509                            .map_err(Error::Render::<R, T>)?;
1510                        let texture = target
1511                            .device
1512                            .renderer_mut()
1513                            .import_memory(slice, TextureMapping::format(&mapping), rect.size, false)
1514                            .map_err(Error::Target)?;
1515                        Ok((texture, rect))
1516                    })
1517                    .collect::<Result<Vec<_>, _>>()?;
1518
1519                let mut frame = target
1520                    .device
1521                    .renderer_mut()
1522                    .render(target.framebuffer, self.size, Transform::Normal)
1523                    .map_err(Error::Target)?;
1524                for (texture, rect) in textures {
1525                    for damage_rect in damage.iter().filter_map(|dmg_rect| dmg_rect.intersection(rect)) {
1526                        let dst = damage_rect
1527                            .to_logical(1, Transform::Normal, &buffer_size)
1528                            .to_physical(1);
1529                        let src = Rectangle::new(damage_rect.loc - rect.loc, damage_rect.size).to_f64();
1530                        let damage = &[Rectangle::from_size(dst.size)];
1531                        frame
1532                            .clear(Color32F::TRANSPARENT, &[dst])
1533                            .map_err(Error::Target)?;
1534                        frame
1535                            .render_texture_from_to(
1536                                &texture,
1537                                src,
1538                                dst,
1539                                damage,
1540                                &[Rectangle::from_size(self.size)],
1541                                Transform::Normal,
1542                                1.0,
1543                            )
1544                            .map_err(Error::Target)?;
1545                    }
1546                }
1547                let sync = frame.finish().map_err(Error::Target)?;
1548                render
1549                    .renderer_mut()
1550                    .cleanup_texture_cache()
1551                    .map_err(Error::Render)?;
1552                return Ok(sync);
1553            }
1554
1555            render
1556                .renderer_mut()
1557                .cleanup_texture_cache()
1558                .map_err(Error::Render)?;
1559            return Ok(sync);
1560        }
1561
1562        Ok(sync::SyncPoint::signaled())
1563    }

2371    S: GraphicsApi,
2372    T: GraphicsApi,
2373    <S::Device as ApiDevice>::Renderer: Renderer + ExportMem + Bind<Dmabuf>,
2374    <T::Device as ApiDevice>::Renderer: Renderer + ImportMem,
2375{
2376    let format = src_texture
2377        .format()
2378        // Check the target device supports the format, if not use 8-bit format
2379        .filter(|format| target.renderer().mem_formats().any(|fmt| fmt == *format))
2380        // The GL spec *always* supports this format.
2381        // TODO: Re-evaluate this, once we support vulkan
2382        .unwrap_or(Fourcc::Abgr8888);
2383
2384    let texture_rect = Rectangle::from_size((src_texture.width() as i32, src_texture.height() as i32).into());
2385    let damage = damage.map(|damage| {
2386        damage
2387            .iter()
2388            .flat_map(|rect| rect.intersection(texture_rect))
2389            .fold(Vec::<Rectangle<i32, BufferCoords>>::new(), |damage, mut rect| {
2390                // replace with drain_filter, when that becomes stable to reuse the original Vec's memory
2391                let (overlapping, mut new_damage): (Vec<_>, Vec<_>) = damage
2392                    .into_iter()
2393                    .partition(|other| other.overlaps_or_touches(rect));
2394
2395                for overlap in overlapping {
2396                    rect = rect.merge(overlap);
2397                }
2398                new_damage.push(rect);
2399                new_damage
2400            })
2401    });
2402
2403    if slot.is_some() {
2404        let (mapping, texture) = slot.as_mut().unwrap();
2405        let mappings = match mapping.take() {
2406            Some(mut mappings) => {
2407                mappings.retain(|(mapping, _)| TextureMapping::format(&**mapping) == format);
2408
2409                let damage_slice = [texture_rect];
2410                let new_damage = damage
2411                    .as_deref()
2412                    .unwrap_or(&damage_slice)
2413                    .iter()
2414                    .filter(|rect| !mappings.iter().any(|(_, region)| region.contains_rect(**rect)))
2415                    .copied()
2416                    .collect::<Vec<_>>();
2417
2418                if texture.is_none()
2419                    && (mappings.len() != 1
2420                        || <dyn TextureMapping>::size(&*mappings[0].0) != texture_rect.size
2421                        || !new_damage.is_empty())
2422                {
2423                    let mapping = src
2424                        .renderer_mut()
2425                        .copy_texture(src_texture, texture_rect, format)
2426                        .map_err(Error::Render)?;
2427                    trace!("Creating mapping for: {:?}", damage);
2428                    mappings = vec![(Box::new(mapping), texture_rect)];
2429                } else {
2430                    mappings.extend(
2431                        new_damage
2432                            .into_iter()
2433                            .map(|damage| {
2434                                let mapping = src
2435                                    .renderer_mut()
2436                                    .copy_texture(src_texture, damage, format)
2437                                    .map_err(Error::Render)?;
2438                                trace!("Creating mapping for: {:?}", damage);
2439                                Ok((Box::new(mapping), damage))
2440                            })
2441                            .collect::<Result<Vec<_>, Error<S, T>>>()?,
2442                    );
2443                }
2444
2445                mappings
2446            }
2447            None => {
2448                let mapping = src
2449                    .renderer_mut()
2450                    .copy_texture(src_texture, texture_rect, format)
2451                    .map_err(Error::Render)?;
2452                trace!("Creating mapping for: {:?}", damage);
2453                vec![(Box::new(mapping), texture_rect)]
2454            }
2455        };
2456
2457        for (mapping, damage) in mappings {
2458            let data = src.renderer_mut().map_texture(&mapping).map_err(Error::Render)?;
2459            if let Some(texture) = texture.as_mut() {
2460                trace!(
2461                    "Updating texture {:?} with mapping at {:?}",
2462                    texture.size(),
2463                    damage,
2464                );
2465                target
2466                    .renderer_mut()
2467                    .update_memory(texture, data, damage)
2468                    .map_err(Error::Target)?;
2469            } else {
2470                trace!("Importing mapping as full buffer {:?}", mapping.size());
2471                let target_texture = target
2472                    .renderer_mut()
2473                    .import_memory(data, format, texture_rect.size, false)
2474                    .map_err(Error::Target)?;
2475                *texture = Some(Box::new(target_texture));
2476            }
2477        }
2478    } else {
2479        let mapping = src
2480            .renderer_mut()
2481            .copy_texture(src_texture, texture_rect, format)
2482            .map_err(Error::Render)?;
2483        trace!("Importing mapping as full buffer {:?}", mapping.size());
2484        let data = src.renderer_mut().map_texture(&mapping).map_err(Error::Render)?;
2485
2486        let target_texture = target
2487            .renderer_mut()
2488            .import_memory(data, format, texture_rect.size, false)
2489            .map_err(Error::Target)?;
2490        *slot = Some((None, Some(Box::new(target_texture))));
2491    };
2492
2493    Ok(())
2494}

2496fn texture_copy<S, T>(
2497    src: &mut S::Device,
2498    target: &mut T::Device,
2499    src_texture: &<<S::Device as ApiDevice>::Renderer as RendererSuper>::TextureId,
2500    target_texture: &mut Option<GpuSingleTexture>,
2501    damage: Option<&[Rectangle<i32, BufferCoords>]>,
2502) -> Result<(), Error<S, T>>
2503where
2504    S: GraphicsApi,
2505    T: GraphicsApi,
2506    <S::Device as ApiDevice>::Renderer: Renderer + ImportDma + ExportMem + Bind<Dmabuf>,
2507    <<S::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping: 'static,
2508    <T::Device as ApiDevice>::Renderer: Renderer + ImportDma + ImportMem,
2509    <<S::Device as ApiDevice>::Renderer as RendererSuper>::TextureId: 'static,
2510    <<T::Device as ApiDevice>::Renderer as RendererSuper>::TextureId: 'static,
2511{
2512    // we need to reset in a few cases
2513    if matches!(&target_texture, Some(GpuSingleTexture::Direct(_)))
2514        || matches!(
2515            &target_texture,
2516            Some(GpuSingleTexture::Mem { mappings: Some((node, _)), .. }) if node != src.node()
2517        )
2518    {
2519        *target_texture = None;
2520    }
2521
2522    match target_texture.take() {
2523        Some(GpuSingleTexture::Direct(_)) => unreachable!(),
2524        Some(GpuSingleTexture::Mem {
2525            mut external_shadow,
2526            texture,
2527            mappings,
2528        }) => {
2529            if let Some((dmabuf, texture)) = external_shadow.take() {
2530                let mut slot = Some((dmabuf, texture, None));
2531                dma_shadow_copy::<S, S>(src_texture, damage, &mut slot, src, None, None)
2532                    .map_err(Error::generalize::<T>)?;
2533                external_shadow = slot.map(|(dmabuf, texture, sync_point)| {
2534                    if let Some(sync) = sync_point {
2535                        // ignore interrupt errors
2536                        src.renderer_mut().wait(&sync).unwrap_or_else(|_| {
2537                            let _ = sync.wait();
2538                        });
2539                    }
2540                    (dmabuf, texture)
2541                });
2542            }
2543
2544            let mut slot = Some((
2545                mappings.map(|(_, mappings)| mappings.into_iter().map(|(damage, mapping)|
2546                    (mapping.downcast::<<<S::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping>().unwrap(), damage)
2547                ).collect::<Vec<_>>()),
2548                texture.map(|texture| texture.downcast::<<<T::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>().unwrap())
2549            ));
2550
2551            let src_texture = external_shadow
2552                .as_ref()
2553                .map(|(_, texture)| {
2554                    texture
2555                        .downcast_ref::<<<S::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>()
2556                        .unwrap()
2557                })
2558                .unwrap_or(src_texture);
2559            let res = mem_copy::<S, T>(src_texture, damage, &mut slot, src, target);
2560            *target_texture = slot.map(|(mappings, texture)| GpuSingleTexture::Mem {
2561                external_shadow,
2562                texture: texture.map(|texture| texture as Box<dyn Any + 'static>),
2563                mappings: mappings.map(|mappings| {
2564                    (
2565                        *src.node(),
2566                        mappings
2567                            .into_iter()
2568                            .map(|(mapping, damage)| (damage, mapping as Box<dyn Any + 'static>))
2569                            .collect(),
2570                    )
2571                }),
2572            });
2573            res
2574        }
2575        Some(GpuSingleTexture::Dma {
2576            texture,
2577            dmabuf,
2578            sync,
2579        }) => {
2580            let mut slot = Some((dmabuf, texture, sync));
2581            let res = dma_shadow_copy::<S, T>(src_texture, damage, &mut slot, src, Some(target), None);
2582            *target_texture = slot.map(|(dmabuf, texture, sync)| GpuSingleTexture::Dma {
2583                texture,
2584                dmabuf,
2585                sync,
2586            });
2587            res
2588        }
2589        None => {
2590            let mut slot = None;
2591            match dma_shadow_copy::<S, T>(src_texture, damage, &mut slot, src, Some(target), None) {
2592                Ok(()) => {
2593                    *target_texture = slot.map(|(dmabuf, texture, sync)| GpuSingleTexture::Dma {
2594                        texture: texture as Box<dyn Any + 'static>,
2595                        dmabuf,
2596                        sync,
2597                    });
2598                    Ok(())
2599                }
2600                Err(err) => {
2601                    trace!(?err, "Dma shadow copy failed, falling back to cpu");
2602
2603                    let mut external_shadow = None;
2604                    if !ExportMem::can_read_texture(src.renderer_mut(), src_texture).map_err(Error::Render)? {
2605                        let mut slot = None;
2606                        dma_shadow_copy::<S, S>(src_texture, damage, &mut slot, src, None, None)
2607                            .map_err(Error::generalize::<T>)?;
2608                        external_shadow = slot.map(|(dmabuf, texture, sync_point)| {
2609                            if let Some(sync) = sync_point {
2610                                // ignore interrupt errors
2611                                src.renderer_mut().wait(&sync).unwrap_or_else(|_| {
2612                                    let _ = sync.wait();
2613                                });
2614                            }
2615                            (dmabuf, texture as Box<dyn Any + 'static>)
2616                        });
2617                    };
2618
2619                    let mut slot = None;
2620                    let mem_src_texture = external_shadow
2621                        .as_ref()
2622                        .map(|(_, texture)| {
2623                            texture
2624                                .downcast_ref::<<<S::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>()
2625                                .unwrap()
2626                        })
2627                        .unwrap_or(src_texture);
2628
2629                    let res = match mem_copy::<S, T>(mem_src_texture, damage, &mut slot, src, target) {
2630                        Ok(res) => Ok(res),
2631                        Err(_) => {
2632                            let mut dma_slot = None;
2633                            dma_shadow_copy::<S, S>(
2634                                src_texture,
2635                                damage,
2636                                &mut dma_slot,
2637                                src,
2638                                None,
2639                                Some(Fourcc::Abgr8888),
2640                            )
2641                            .map_err(Error::generalize::<T>)?;
2642                            external_shadow = dma_slot.map(|(dmabuf, texture, sync_point)| {
2643                                if let Some(sync) = sync_point {
2644                                    // ignore interrupt errors
2645                                    src.renderer_mut().wait(&sync).unwrap_or_else(|_| {
2646                                        let _ = sync.wait();
2647                                    });
2648                                }
2649                                (dmabuf, texture as Box<dyn Any + 'static>)
2650                            });
2651                            let src_texture = external_shadow
2652                                .as_ref()
2653                                .map(|(_, texture)| {
2654                                    texture
2655                                        .downcast_ref::<<<S::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>()
2656                                        .unwrap()
2657                                })
2658                                .unwrap_or(src_texture);
2659                            mem_copy::<S, T>(src_texture, damage, &mut slot, src, target)
2660                        }
2661                    };
2662
2663                    *target_texture = slot.map(|(mappings, texture)| GpuSingleTexture::Mem {
2664                        texture: texture.map(|texture| texture as Box<dyn Any + 'static>),
2665                        mappings: mappings.map(|mappings| {
2666                            (
2667                                *src.node(),
2668                                mappings
2669                                    .into_iter()
2670                                    .map(|(mapping, damage)| (damage, mapping as Box<dyn Any + 'static>))
2671                                    .collect(),
2672                            )
2673                        }),
2674                        external_shadow,
2675                    });
2676                    res
2677                }
2678            }
2679        }
2680    }
2681}

2683impl<R: GraphicsApi, T: GraphicsApi> MultiRenderer<'_, '_, R, T>
2684where
2685    <R::Device as ApiDevice>::Renderer: ImportDma + ImportMem,
2686    <T::Device as ApiDevice>::Renderer: Bind<Dmabuf> + ExportMem,
2687    <<R::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping: 'static,
2688    <<T::Device as ApiDevice>::Renderer as ExportMem>::TextureMapping: 'static,
2689    T: 'static,
2690    // We need this because the Renderer-impl does and ImportDma requires Renderer
2691    R: 'static,
2692    <R::Device as ApiDevice>::Renderer: Bind<Dmabuf> + ExportMem + ImportDma + ImportMem,
2693    <T::Device as ApiDevice>::Renderer: ImportDma + ImportMem,
2694    <<R::Device as ApiDevice>::Renderer as RendererSuper>::TextureId: Clone + Send,
2695    <<R::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
2696    <<T::Device as ApiDevice>::Renderer as RendererSuper>::Error: 'static,
2697{
2698    #[profiling::function]
2699    fn import_dmabuf_internal(
2700        &mut self,
2701        dmabuf: &Dmabuf,
2702        mut texture: MultiTexture,
2703        damage: Option<&[Rectangle<i32, BufferCoords>]>,
2704    ) -> Result<<Self as RendererSuper>::TextureId, <Self as RendererSuper>::Error> {
2705        let src_node = import_on_src_node::<R, T>(
2706            dmabuf,
2707            damage,
2708            &mut texture,
2709            self.render,
2710            self.target.as_mut().map(|target| &mut *target.device),
2711            self.other_renderers.iter_mut().map(|d| &mut **d),
2712        )?;
2713
2714        if src_node == *self.render.node() {
2715            // when we are on the same node, we are done
2716            Ok(texture)
2717        } else {
2718            // else we need to copy
2719            let mut texture_internal = texture.0.lock().unwrap();
2720            let target_id = self.render.renderer().context_id().erased();
2721            let mut target_texture = texture_internal.textures.remove(&target_id);
2722
2723            let res = if let Some(target) = self
2724                .target
2725                .as_mut()
2726                .filter(|target| src_node == *target.device.node())
2727            {
2728                let src_id = target.device.renderer().context_id().erased();
2729                let src_texture = match texture_internal.textures.get(&src_id).unwrap() {
2730                    GpuSingleTexture::Direct(tex) => tex
2731                        .downcast_ref::<<<T::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>()
2732                        .unwrap(),
2733                    _ => unreachable!(),
2734                };
2735
2736                texture_copy::<T, R>(
2737                    target.device,
2738                    self.render,
2739                    src_texture,
2740                    &mut target_texture,
2741                    damage,
2742                )
2743                .map_err(Error::transpose)
2744            } else if let Some(other) = self
2745                .other_renderers
2746                .iter_mut()
2747                .find(|other| src_node == *other.node())
2748            {
2749                let src_id = other.renderer().context_id().erased();
2750                let src_texture = match texture_internal.textures.get(&src_id).unwrap() {
2751                    GpuSingleTexture::Direct(tex) => tex
2752                        .downcast_ref::<<<R::Device as ApiDevice>::Renderer as RendererSuper>::TextureId>()
2753                        .unwrap(),
2754                    _ => unreachable!(),
2755                };
2756
2757                texture_copy::<R, R>(other, self.render, src_texture, &mut target_texture, damage)
2758                    .map_err(Error::generalize::<T>)
2759            } else {
2760                Err(Error::DeviceMissing)
2761            };
2762
2763            if let Some(target_texture) = target_texture.filter(|_| res.is_ok()) {
2764                texture_internal.textures.insert(target_id, target_texture);
2765            }
2766
2767            std::mem::drop(texture_internal);
2768            res.map(|_| texture)
2769        }
2770    }
2771}
