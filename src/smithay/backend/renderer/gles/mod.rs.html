207impl GlesTargetInternal<'_> {
208    fn format(&self) -> Option<(ffi::types::GLenum, bool)> {
209        match self {
210            GlesTargetInternal::Image { dmabuf, .. } => {
211                let format = crate::backend::allocator::Buffer::format(*dmabuf).code;
212                let has_alpha = has_alpha(format);
213                let (format, _, _) = fourcc_to_gl_formats(format)?;
214
215                Some((format, has_alpha))
216            }
217            GlesTargetInternal::Surface { surface, .. } => {
218                let format = surface.pixel_format();
219                let format = match (format.color_bits, format.alpha_bits) {
220                    (24, 8) => ffi::RGB8,
221                    (30, 2) => ffi::RGB10_A2,
222                    (48, 16) => ffi::RGB16F,
223                    _ => return None,
224                };
225
226                Some((format, true))
227            }
228            GlesTargetInternal::Texture { texture, .. } => Some((texture.0.format?, texture.0.has_alpha)),
229            GlesTargetInternal::Renderbuffer { buf, .. } => Some((buf.0.format, buf.0.has_alpha)),
230        }
231    }
232
233    #[profiling::function]
234    fn make_current(&self, gl: &ffi::Gles2, egl: &EGLContext) -> Result<(), MakeCurrentError> {
235        unsafe {
236            if let GlesTargetInternal::Surface { surface, .. } = self {
237                egl.make_current_with_surface(surface)?;
238                gl.BindFramebuffer(ffi::FRAMEBUFFER, 0);
239            } else {
240                egl.make_current()?;
241                match self {
242                    GlesTargetInternal::Image { ref buf, .. } => {
243                        gl.BindFramebuffer(ffi::FRAMEBUFFER, buf.0.fbo)
244                    }
245                    GlesTargetInternal::Texture { ref fbo, .. } => gl.BindFramebuffer(ffi::FRAMEBUFFER, *fbo),
246                    GlesTargetInternal::Renderbuffer { ref fbo, .. } => {
247                        gl.BindFramebuffer(ffi::FRAMEBUFFER, *fbo)
248                    }
249                    _ => unreachable!(),
250                }
251            }
252            Ok(())
253        }
254    }

297/// GL resources need to be destroyed with a context active on the current thread,
298/// so `Drop` impls don't directly destroy objects, and instead send messages
299/// to free resources on periodic calls to `cleanup`.
300///
301/// Stored as `EGLContext` user data, so shared contexts can share cleanup. And
302/// destroying one renderer will not leak resources, but allow another renderer
303/// with a shared context to clean them up.
304struct GlesCleanup {
305    receiver: Mutex<mpsc::Receiver<CleanupResource>>,
306    sender: mpsc::Sender<CleanupResource>,
307}
308
309impl Default for GlesCleanup {
310    fn default() -> Self {
311        let (sender, receiver) = mpsc::channel();
312        Self {
313            receiver: Mutex::new(receiver),
314            sender,
315        }
316    }
317}
318
319impl GlesCleanup {
320    fn cleanup(&self, egl: &EGLContext, gl: &ffi::Gles2) {
321        let receiver = match self.receiver.try_lock() {
322            Ok(receiver) => receiver,
323            Err(TryLockError::Poisoned(err)) => {
324                self.receiver.clear_poison();
325                err.into_inner()
326            }
327            Err(TryLockError::WouldBlock) => {
328                // Another thread is running `cleanup`, so just return
329                return;
330            }
331        };
332        for resource in receiver.try_iter() {
333            match resource {
334                CleanupResource::Texture(texture) => unsafe {
335                    gl.DeleteTextures(1, &texture);
336                },
337                CleanupResource::EGLImage(image) => unsafe {
338                    ffi_egl::DestroyImageKHR(**egl.display().get_display_handle(), image);
339                },
340                CleanupResource::FramebufferObject(fbo) => unsafe {
341                    gl.DeleteFramebuffers(1, &fbo);
342                },
343                CleanupResource::RenderbufferObject(rbo) => unsafe {
344                    gl.DeleteRenderbuffers(1, &rbo);
345                },
346                CleanupResource::Mapping(pbo, mapping) => unsafe {
347                    if !mapping.is_null() {
348                        gl.BindBuffer(ffi::PIXEL_PACK_BUFFER, pbo);
349                        gl.UnmapBuffer(ffi::PIXEL_PACK_BUFFER);
350                        gl.BindBuffer(ffi::PIXEL_PACK_BUFFER, 0);
351                    }
352                    gl.DeleteBuffers(1, &pbo);
353                },
354                CleanupResource::Program(program) => unsafe {
355                    gl.DeleteProgram(program);
356                },
357                CleanupResource::Sync(sync) => unsafe {
358                    gl.DeleteSync(sync);
359                },
360            }
361        }
362    }
363}

480impl GlesRenderer {
481    /// Get the supported [`Capabilities`](Capability) of the renderer
482    ///
483    /// # Safety
484    ///
485    /// This operation will cause undefined behavior if the given EGLContext is active in another thread.
486    pub unsafe fn supported_capabilities(context: &EGLContext) -> Result<Vec<Capability>, GlesError> {
487        context.make_current()?;
488
489        let gl = ffi::Gles2::load_with(|s| crate::backend::egl::get_proc_address(s) as *const _);
490        let ext_ptr = gl.GetString(ffi::EXTENSIONS) as *const c_char;
491        if ext_ptr.is_null() {
492            return Err(GlesError::GLFunctionLoaderError);
493        }
494
495        let exts = {
496            let p = CStr::from_ptr(ext_ptr);
497            let list = String::from_utf8(p.to_bytes().to_vec()).unwrap_or_else(|_| String::new());
498            list.split(' ').map(|e| e.to_string()).collect::<Vec<_>>()
499        };
500
501        let gl_version = version::GlVersion::try_from(&gl).unwrap_or_else(|_| {
502            warn!("Failed to detect GLES version, defaulting to 2.0");
503            version::GLES_2_0
504        });
505
506        let mut capabilities = Vec::new();
507        // required for more optimized rendering, otherwise we render in batches
508        if gl_version >= version::GLES_3_0
509            || (exts.iter().any(|ext| ext == "GL_EXT_instanced_arrays")
510                && exts.iter().any(|ext| ext == "GL_EXT_draw_instanced"))
511        {
512            capabilities.push(Capability::Instancing);
513            debug!("Instancing is supported");
514        }
515        // required to use 8-bit color formats in renderbuffers, we don't deal with anything lower as a render target
516        if gl_version >= version::GLES_3_0 || exts.iter().any(|ext| ext == "GL_OES_rgb8_rgba8") {
517            capabilities.push(Capability::Renderbuffer);
518            debug!("Rgba8 Renderbuffers are supported");
519        }
520        // required for blit operations
521        if gl_version >= version::GLES_3_0 {
522            capabilities.push(Capability::Blit);
523            debug!("Blitting is supported");
524            capabilities.push(Capability::_10Bit);
525            debug!("10-bit formats are supported");
526            capabilities.push(Capability::Fencing);
527            debug!("Fencing is supported");
528        }
529
530        if exts.iter().any(|ext| ext == "GL_OES_EGL_sync") {
531            debug!("EGL Fencing is supported");
532            capabilities.push(Capability::ExportFence);
533        }
534
535        if exts.iter().any(|ext| ext == "GL_KHR_debug") {
536            capabilities.push(Capability::Debug);
537            debug!("GL Debug is supported");
538        }
539
540        Ok(capabilities)
541    }

556    /// Creates a new OpenGL ES renderer from a given [`EGLContext`]
557    /// with the specified [`Capabilities`](Capability). If a requested [`Capability`] is not supported an
558    /// error will be returned.
559    ///
560    /// # Safety
561    ///
562    /// This operation will cause undefined behavior if the given EGLContext is active in another thread.
563    ///
564    /// # Implementation details
565    ///
566    /// - Texture handles created by the resulting renderer are valid for every rendered created with an
567    ///   `EGLContext` shared with the given one (see `EGLContext::new_shared`) and can be used on
568    ///   any of these renderers.
569    /// - This renderer has no default framebuffer, use `Bind::bind` before rendering.
570    /// - Shm buffers can be released after a successful import, without the texture handle becoming invalid.
571    /// - Texture filtering starts with Linear-downscaling and Linear-upscaling.
572    /// - If OpenGL ES 3.0 is not available and the underlying [`EGLContext`] is shared, memory textures
573    ///   will insert `glFinish`-calls into the pipeline. Consider not sharing contexts, if OpenGL ES 3 isn't available.
574    pub unsafe fn with_capabilities(
575        context: EGLContext,
576        capabilities: impl IntoIterator<Item = Capability>,
577    ) -> Result<GlesRenderer, GlesError> {
578        let span = info_span!(parent: &context.span, "renderer_gles2");
579        let _guard = span.enter();
580
581        context.make_current()?;
582
583        // TODO: check for angle as well?
584        let is_software = EGLDevice::device_for_display(context.display())
585            .map(|dev| dev.is_software())
586            .unwrap_or(false);
587
588        let supported_capabilities = Self::supported_capabilities(&context)?;
589        let requested_capabilities = capabilities.into_iter().collect::<Vec<_>>();
590
591        let unsupported_capabilities = requested_capabilities
592            .iter()
593            .copied()
594            .filter(|c| !supported_capabilities.contains(c))
595            .collect::<Vec<_>>();
596
597        if let Some(missing_capability) = unsupported_capabilities.first() {
598            let err = match missing_capability {
599                Capability::Instancing => {
600                    GlesError::GLExtensionNotSupported(&["GL_EXT_instanced_arrays", "GL_EXT_draw_instanced"])
601                }
602                Capability::Blit | Capability::_10Bit | Capability::Fencing => {
603                    GlesError::GLVersionNotSupported(version::GLES_3_0)
604                }
605                Capability::Renderbuffer => GlesError::GLExtensionNotSupported(&["GL_OES_rgb8_rgba8"]),
606                Capability::ExportFence => GlesError::GLExtensionNotSupported(&["GL_OES_EGL_sync"]),
607                Capability::Debug => GlesError::GLExtensionNotSupported(&["GL_KHR_debug"]),
608            };
609            return Err(err);
610        };
611
612        let (gl, gl_version, exts, capabilities, gl_debug_span) = {
613            let gl = ffi::Gles2::load_with(|s| crate::backend::egl::get_proc_address(s) as *const _);
614            let ext_ptr = gl.GetString(ffi::EXTENSIONS) as *const c_char;
615            if ext_ptr.is_null() {
616                return Err(GlesError::GLFunctionLoaderError);
617            }
618
619            let exts = {
620                let p = CStr::from_ptr(ext_ptr);
621                let list = String::from_utf8(p.to_bytes().to_vec()).unwrap_or_else(|_| String::new());
622                list.split(' ').map(|e| e.to_string()).collect::<Vec<_>>()
623            };
624
625            info!("Initializing OpenGL ES Renderer");
626            info!(
627                "GL Version: {:?}",
628                CStr::from_ptr(gl.GetString(ffi::VERSION) as *const c_char)
629            );
630            info!(
631                "GL Vendor: {:?}",
632                CStr::from_ptr(gl.GetString(ffi::VENDOR) as *const c_char)
633            );
634            info!(
635                "GL Renderer: {:?}",
636                CStr::from_ptr(gl.GetString(ffi::RENDERER) as *const c_char)
637            );
638            info!("Supported GL Extensions: {:?}", exts);
639
640            let gl_version = version::GlVersion::try_from(&gl).unwrap_or_else(|_| {
641                warn!("Failed to detect GLES version, defaulting to 2.0");
642                version::GLES_2_0
643            });
644
645            // required for the manditory wl_shm formats
646            if !exts.iter().any(|ext| ext == "GL_EXT_texture_format_BGRA8888") {
647                return Err(GlesError::GLExtensionNotSupported(&[
648                    "GL_EXT_texture_format_BGRA8888",
649                ]));
650            }
651
652            // required for buffers without linear memory layout
653            if gl_version < version::GLES_3_0 && !exts.iter().any(|ext| ext == "GL_EXT_unpack_subimage") {
654                return Err(GlesError::GLExtensionNotSupported(&["GL_EXT_unpack_subimage"]));
655            }
656
657            let gl_debug_span = if requested_capabilities.contains(&Capability::Debug) {
658                gl.Enable(ffi::DEBUG_OUTPUT);
659                gl.Enable(ffi::DEBUG_OUTPUT_SYNCHRONOUS);
660                let span = Box::into_raw(Box::new(span.clone()));
661                gl.DebugMessageCallback(Some(gl_debug_log), span as *mut _);
662                Some(span)
663            } else {
664                None
665            };
666
667            (gl, gl_version, exts, requested_capabilities, gl_debug_span)
668        };
669
670        let gles_cleanup = context.user_data().get_or_insert_threadsafe(GlesCleanup::default);
671
672        let tex_program = texture_program(&gl, shaders::FRAGMENT_SHADER, &[], gles_cleanup.sender.clone())?;
673        let solid_program = solid_program(&gl)?;
674
675        // Initialize vertices based on drawing methodology.
676        let vertices: &[ffi::types::GLfloat] = if capabilities.contains(&Capability::Instancing) {
677            &INSTANCED_VERTS
678        } else {
679            &TRIANGLE_VERTS
680        };
681
682        let mut vbos = [0; 2];
683        gl.GenBuffers(vbos.len() as i32, vbos.as_mut_ptr());
684        gl.BindBuffer(ffi::ARRAY_BUFFER, vbos[0]);
685        gl.BufferData(
686            ffi::ARRAY_BUFFER,
687            std::mem::size_of_val(vertices) as isize,
688            vertices.as_ptr() as *const _,
689            ffi::STATIC_DRAW,
690        );
691        gl.BindBuffer(ffi::ARRAY_BUFFER, vbos[1]);
692        gl.BufferData(
693            ffi::ARRAY_BUFFER,
694            (std::mem::size_of::<ffi::types::GLfloat>() * OUTPUT_VERTS.len()) as isize,
695            OUTPUT_VERTS.as_ptr() as *const _,
696            ffi::STATIC_DRAW,
697        );
698        gl.BindBuffer(ffi::ARRAY_BUFFER, 0);
699
700        context
701            .user_data()
702            .insert_if_missing_threadsafe(ContextId::<GlesTexture>::new);
703
704        drop(_guard);
705
706        let renderer = GlesRenderer {
707            gl,
708            egl: context,
709            #[cfg(all(feature = "wayland_frontend", feature = "use_system_lib"))]
710            egl_reader: None,
711
712            extensions: exts,
713            is_software,
714            gl_version,
715            capabilities,
716
717            tex_program,
718            solid_program,
719            vbos,
720            min_filter: TextureFilter::Linear,
721            max_filter: TextureFilter::Linear,
722
723            buffers: Vec::new(),
724            dmabuf_cache: std::collections::HashMap::new(),
725            vertices: Vec::with_capacity(6 * 16),
726            non_opaque_damage: Vec::with_capacity(16),
727            opaque_damage: Vec::with_capacity(16),
728
729            debug_flags: DebugFlags::empty(),
730            _not_send: PhantomData,
731            span,
732            gl_debug_span,
733        };
734        renderer.egl.unbind()?;
735        Ok(renderer)
736    }

738    fn bind_texture<'a>(&mut self, texture: &'a GlesTexture) -> Result<GlesTarget<'a>, GlesError> {
739        unsafe {
740            self.egl.make_current()?;
741        }
742
743        let bind = || {
744            let mut sync_lock = texture.0.sync.write().unwrap();
745            let mut fbo = 0;
746            unsafe {
747                sync_lock.wait_for_all(&self.gl);
748                self.gl.GenFramebuffers(1, &mut fbo as *mut _);
749                self.gl.BindFramebuffer(ffi::FRAMEBUFFER, fbo);
750                self.gl.FramebufferTexture2D(
751                    ffi::FRAMEBUFFER,
752                    ffi::COLOR_ATTACHMENT0,
753                    ffi::TEXTURE_2D,
754                    texture.0.texture,
755                    0,
756                );
757                let status = self.gl.CheckFramebufferStatus(ffi::FRAMEBUFFER);
758                self.gl.BindFramebuffer(ffi::FRAMEBUFFER, 0);
759
760                if status != ffi::FRAMEBUFFER_COMPLETE {
761                    self.gl.DeleteFramebuffers(1, &mut fbo as *mut _);
762                    return Err(GlesError::FramebufferBindingError);
763                }
764            }
765
766            Ok(GlesTarget(GlesTargetInternal::Texture {
767                texture: texture.clone(),
768                sync_lock,
769                destruction_callback_sender: self.gles_cleanup().sender.clone(),
770                fbo,
771            }))
772        };
773
774        bind().inspect_err(|_| {
775            if let Err(err) = self.unbind() {
776                self.span.in_scope(|| warn!(?err, "Failed to unbind on err"));
777            }
778        })
779    }

783        unsafe {
784            self.egl.make_current()?;
785        }
786        unsafe { self.gl.BindFramebuffer(ffi::FRAMEBUFFER, 0) };
787        self.cleanup();
788        self.egl.unbind()?;
789        Ok(())
790    }

793        self.egl_context().user_data().get().unwrap()
794    }

798        self.dmabuf_cache.retain(|entry, _tex| !entry.is_gone());
799        self.buffers.retain(|buffer| !buffer.0.dmabuf.is_gone());
800        self.gles_cleanup().cleanup(&self.egl, &self.gl);
801    }

827impl ImportMemWl for GlesRenderer {
828    #[instrument(level = "trace", parent = &self.span, skip(self))]
829    #[profiling::function]
830    fn import_shm_buffer(
831        &mut self,
832        buffer: &wl_buffer::WlBuffer,
833        surface: Option<&crate::wayland::compositor::SurfaceData>,
834        damage: &[Rectangle<i32, BufferCoord>],
835    ) -> Result<GlesTexture, GlesError> {
836        use crate::wayland::shm::with_buffer_contents;
837
838        // why not store a `GlesTexture`? because the user might do so.
839        // this is guaranteed a non-public internal type, so we are good.
840        type CacheMap = HashMap<ContextId<GlesTexture>, Arc<GlesTextureInternal>>;
841
842        let mut surface_lock = surface.as_ref().map(|surface_data| {
843            surface_data
844                .data_map
845                .get_or_insert_threadsafe(|| Arc::new(Mutex::new(CacheMap::new())))
846                .lock()
847                .unwrap()
848        });
849
850        with_buffer_contents(buffer, |ptr, len, data| {
851            let offset = data.offset;
852            let width = data.width;
853            let height = data.height;
854            let stride = data.stride;
855            let fourcc =
856                shm_format_to_fourcc(data.format).ok_or(GlesError::UnsupportedWlPixelFormat(data.format))?;
857
858            if self.gl_version.major >= 3 {
859                if !SUPPORTED_MEM_FORMATS_3.contains(&fourcc) {
860                    return Err(GlesError::UnsupportedWlPixelFormat(data.format));
861                }
862            } else if !SUPPORTED_MEM_FORMATS_2.contains(&fourcc) {
863                return Err(GlesError::UnsupportedWlPixelFormat(data.format));
864            }
865
866            let has_alpha = has_alpha(fourcc);
867            let (mut internal_format, read_format, type_) =
868                fourcc_to_gl_formats(fourcc).ok_or(GlesError::UnsupportedWlPixelFormat(data.format))?;
869            if self.gl_version.major == 2 {
870                // es 2.0 doesn't define sized variants
871                internal_format = match internal_format {
872                    ffi::BGRA_EXT => ffi::BGRA_EXT,
873                    ffi::RGBA8 => ffi::RGBA,
874                    ffi::RGB8 => ffi::RGB,
875                    _ => unreachable!(),
876                };
877            }
878
879            // number of bytes per pixel
880            let pixelsize = gl_bpp(read_format, type_).expect("We check the format before") / 8;
881            // ensure consistency, the SHM handler of smithay should ensure this
882            assert!((offset + (height - 1) * stride + width * pixelsize as i32) as usize <= len);
883
884            let mut upload_full = false;
885
886            unsafe {
887                self.egl.make_current()?;
888            }
889
890            let id = self.context_id();
891            let texture = GlesTexture(
892                surface_lock
893                    .as_ref()
894                    .and_then(|cache| cache.get(&id).cloned())
895                    .filter(|texture| texture.size == (width, height).into())
896                    .unwrap_or_else(|| {
897                        let mut tex = 0;
898                        unsafe { self.gl.GenTextures(1, &mut tex) };
899                        // new texture, upload in full
900                        upload_full = true;
901                        let new = Arc::new(GlesTextureInternal {
902                            texture: tex,
903                            sync: RwLock::default(),
904                            format: Some(internal_format),
905                            has_alpha,
906                            is_external: false,
907                            y_inverted: false,
908                            size: (width, height).into(),
909                            egl_images: None,
910                            destruction_callback_sender: self.gles_cleanup().sender.clone(),
911                        });
912                        if let Some(cache) = surface_lock.as_mut() {
913                            cache.insert(id, new.clone());
914                        }
915                        new
916                    }),
917            );
918
919            let mut sync_lock = texture.0.sync.write().unwrap();
920            unsafe {
921                sync_lock.wait_for_all(&self.gl);
922                self.gl.BindTexture(ffi::TEXTURE_2D, texture.0.texture);
923                self.gl
924                    .TexParameteri(ffi::TEXTURE_2D, ffi::TEXTURE_WRAP_S, ffi::CLAMP_TO_EDGE as i32);
925                self.gl
926                    .TexParameteri(ffi::TEXTURE_2D, ffi::TEXTURE_WRAP_T, ffi::CLAMP_TO_EDGE as i32);
927                self.gl
928                    .PixelStorei(ffi::UNPACK_ROW_LENGTH, stride / pixelsize as i32);
929
930                if upload_full || damage.is_empty() {
931                    trace!("Uploading shm texture");
932                    self.gl.TexImage2D(
933                        ffi::TEXTURE_2D,
934                        0,
935                        internal_format as i32,
936                        width,
937                        height,
938                        0,
939                        read_format,
940                        type_,
941                        ptr.offset(offset as isize) as *const _,
942                    );
943                } else {
944                    for region in damage.iter() {
945                        trace!("Uploading partial shm texture");
946                        self.gl.PixelStorei(ffi::UNPACK_SKIP_PIXELS, region.loc.x);
947                        self.gl.PixelStorei(ffi::UNPACK_SKIP_ROWS, region.loc.y);
948                        self.gl.TexSubImage2D(
949                            ffi::TEXTURE_2D,
950                            0,
951                            region.loc.x,
952                            region.loc.y,
953                            region.size.w,
954                            region.size.h,
955                            read_format,
956                            type_,
957                            ptr.offset(offset as isize) as *const _,
958                        );
959                        self.gl.PixelStorei(ffi::UNPACK_SKIP_PIXELS, 0);
960                        self.gl.PixelStorei(ffi::UNPACK_SKIP_ROWS, 0);
961                    }
962                }
963
964                self.gl.PixelStorei(ffi::UNPACK_ROW_LENGTH, 0);
965                self.gl.BindTexture(ffi::TEXTURE_2D, 0);
966
967                if self.capabilities.contains(&Capability::Fencing) {
968                    sync_lock.update_write(&self.gl);
969                } else if self.egl.is_shared() {
970                    self.gl.Finish();
971                }
972            }
973            std::mem::drop(sync_lock);
974
975            Ok(texture)
976        })
977        .map_err(GlesError::BufferAccessError)?
978    }
979}

1056                );

1089        &mut self,

1091        data: &[u8],
1092        region: Rectangle<i32, BufferCoord>,
1093    ) -> Result<(), Self::Error> {

1098            return Err(GlesError::UnsupportedPixelLayout);
1099        }
1100        let (read_format, type_) = gl_read_for_internal(texture.0.format.expect("We check that before"))
1101            .ok_or(GlesError::UnknownPixelFormat)?;
1102

1104            < (region.size.w * region.size.h) as usize

1108        }
1109
1110        let mut sync_lock = texture.0.sync.write().unwrap();
1111        unsafe {
1112            self.egl.make_current()?;
1113            sync_lock.wait_for_all(&self.gl);
1114            self.gl.BindTexture(ffi::TEXTURE_2D, texture.0.texture);
1115            self.gl
1116                .TexParameteri(ffi::TEXTURE_2D, ffi::TEXTURE_WRAP_S, ffi::CLAMP_TO_EDGE as i32);
1117            self.gl
1118                .TexParameteri(ffi::TEXTURE_2D, ffi::TEXTURE_WRAP_T, ffi::CLAMP_TO_EDGE as i32);
1119            self.gl.PixelStorei(ffi::UNPACK_ROW_LENGTH, texture.0.size.w);
1120            self.gl.PixelStorei(ffi::UNPACK_SKIP_PIXELS, region.loc.x);
1121            self.gl.PixelStorei(ffi::UNPACK_SKIP_ROWS, region.loc.y);
1122            self.gl.TexSubImage2D(
1123                ffi::TEXTURE_2D,
1124                0,
1125                region.loc.x,
1126                region.loc.y,
1127                region.size.w,
1128                region.size.h,
1129                read_format,
1130                type_,
1131                data.as_ptr() as *const _,
1132            );
1133            self.gl.PixelStorei(ffi::UNPACK_ROW_LENGTH, 0);
1134            self.gl.PixelStorei(ffi::UNPACK_SKIP_PIXELS, 0);
1135            self.gl.PixelStorei(ffi::UNPACK_SKIP_ROWS, 0);
1136            self.gl.BindTexture(ffi::TEXTURE_2D, 0);
1137
1138            if self.capabilities.contains(&Capability::Fencing) {
1139                sync_lock.update_write(&self.gl);
1140            } else if self.egl.is_shared() {
1141                self.gl.Finish();
1142            }
1143        }
1144
1145        Ok(())
1146    }

1736    #[instrument(level = "trace", parent = &self.span, skip(self, src_target, dst_target))]
1737    #[profiling::function]
1738    fn blit(
1739        &mut self,
1740        src_target: &GlesTarget<'_>,
1741        dst_target: &mut GlesTarget<'_>,
1742        src: Rectangle<i32, Physical>,
1743        dst: Rectangle<i32, Physical>,
1744        filter: TextureFilter,
1745    ) -> Result<SyncPoint, GlesError> {
1746        // glBlitFramebuffer is sadly only available for GLES 3.0 and higher
1747        if self.gl_version < version::GLES_3_0 {
1748            return Err(GlesError::GLVersionNotSupported(version::GLES_3_0));
1749        }
1750
1751        match (&src_target.0, &dst_target.0) {
1752            (
1753                GlesTargetInternal::Surface { surface: src, .. },
1754                GlesTargetInternal::Surface { surface: dst, .. },
1755            ) => unsafe {
1756                self.egl.make_current_with_draw_and_read_surface(dst, src)?;
1757            },
1758            (GlesTargetInternal::Surface { surface: src, .. }, _) => unsafe {
1759                self.egl.make_current_with_surface(src)?;
1760            },
1761            (_, GlesTargetInternal::Surface { surface: dst, .. }) => unsafe {
1762                self.egl.make_current_with_surface(dst)?;
1763            },
1764            (_, _) => unsafe {
1765                self.egl.make_current()?;
1766            },
1767        }
1768
1769        match &src_target.0 {
1770            GlesTargetInternal::Image { ref buf, .. } => unsafe {
1771                self.gl.BindFramebuffer(ffi::READ_FRAMEBUFFER, buf.0.fbo)
1772            },
1773            GlesTargetInternal::Texture { ref fbo, .. } => unsafe {
1774                self.gl.BindFramebuffer(ffi::READ_FRAMEBUFFER, *fbo)
1775            },
1776            GlesTargetInternal::Renderbuffer { ref fbo, .. } => unsafe {
1777                self.gl.BindFramebuffer(ffi::READ_FRAMEBUFFER, *fbo)
1778            },
1779            _ => {} // Note: The only target missing is `Surface` and handled above
1780        }
1781        match &dst_target.0 {
1782            GlesTargetInternal::Image { ref buf, .. } => unsafe {
1783                self.gl.BindFramebuffer(ffi::DRAW_FRAMEBUFFER, buf.0.fbo)
1784            },
1785            GlesTargetInternal::Texture { ref fbo, .. } => unsafe {
1786                self.gl.BindFramebuffer(ffi::DRAW_FRAMEBUFFER, *fbo)
1787            },
1788            GlesTargetInternal::Renderbuffer { ref fbo, .. } => unsafe {
1789                self.gl.BindFramebuffer(ffi::DRAW_FRAMEBUFFER, *fbo)
1790            },
1791            _ => {} // Note: The only target missing is `Surface` and handled above
1792        }
1793
1794        let status = unsafe { self.gl.CheckFramebufferStatus(ffi::FRAMEBUFFER) };
1795        if status != ffi::FRAMEBUFFER_COMPLETE {
1796            let _ = self.unbind();
1797            return Err(GlesError::FramebufferBindingError);
1798        }
1799
1800        let errno = unsafe {
1801            while self.gl.GetError() != ffi::NO_ERROR {} // clear flag before
1802            self.gl.BlitFramebuffer(
1803                src.loc.x,
1804                src.loc.y,
1805                src.loc.x + src.size.w,
1806                src.loc.y + src.size.h,
1807                dst.loc.x,
1808                dst.loc.y,
1809                dst.loc.x + dst.size.w,
1810                dst.loc.y + dst.size.h,
1811                ffi::COLOR_BUFFER_BIT,
1812                match filter {
1813                    TextureFilter::Linear => ffi::LINEAR,
1814                    TextureFilter::Nearest => ffi::NEAREST,
1815                },
1816            );
1817            self.gl.GetError()
1818        };
1819
1820        if errno == ffi::INVALID_OPERATION {
1821            Err(GlesError::BlitError)
1822        } else {
1823            if let Some(sync_point) = self.export_sync_point() {
1824                return Ok(sync_point);
1825            }
1826
1827            unsafe {
1828                self.gl.Finish();
1829            }
1830            Ok(SyncPoint::signaled())
1831        }
1832    }

2802                    TextureFilter::Nearest => ffi::NEAREST as i32,

3079fn build_texture_mat(
3080    src: Rectangle<f64, BufferCoord>,
3081    dest: Rectangle<i32, Physical>,
3082    texture: Size<i32, BufferCoord>,
3083    transform: Transform,
3084) -> Matrix3<f32> {
3085    let dst_src_size = transform.transform_size(src.size);
3086    let scale = dst_src_size.to_f64() / dest.size.to_f64();
3087
3088    let mut tex_mat = Matrix3::<f32>::identity();
3089
3090    // first bring the damage into src scale
3091    tex_mat = Matrix3::from_nonuniform_scale(scale.x as f32, scale.y as f32) * tex_mat;
3092
3093    // then compensate for the texture transform
3094    let transform_mat = transform.matrix();
3095    let translation = match transform {
3096        Transform::Normal => Matrix3::identity(),
3097        Transform::_90 => Matrix3::from_translation(Vector2::new(0f32, dst_src_size.w as f32)),
3098        Transform::_180 => {
3099            Matrix3::from_translation(Vector2::new(dst_src_size.w as f32, dst_src_size.h as f32))
3100        }
3101        Transform::_270 => Matrix3::from_translation(Vector2::new(dst_src_size.h as f32, 0f32)),
3102        Transform::Flipped => Matrix3::from_translation(Vector2::new(dst_src_size.w as f32, 0f32)),
3103        Transform::Flipped90 => Matrix3::identity(),
3104        Transform::Flipped180 => Matrix3::from_translation(Vector2::new(0f32, dst_src_size.h as f32)),
3105        Transform::Flipped270 => {
3106            Matrix3::from_translation(Vector2::new(dst_src_size.h as f32, dst_src_size.w as f32))
3107        }
3108    };
3109    tex_mat = transform_mat * tex_mat;
3110    tex_mat = translation * tex_mat;
3111
3112    // now we can add the src crop loc, the size already done implicit by the src size
3113    tex_mat = Matrix3::from_translation(Vector2::new(src.loc.x as f32, src.loc.y as f32)) * tex_mat;
3114
3115    // at last we have to normalize the values for UV space
3116    tex_mat = Matrix3::from_nonuniform_scale(
3117        (1.0f64 / texture.w as f64) as f32,
3118        (1.0f64 / texture.h as f64) as f32,
3119    ) * tex_mat;
3120
3121    tex_mat
3122}
