19/// Describes the physical type of tool. The physical type of tool generally defines its base usage.
20#[derive(Debug, Clone, Copy, Hash, Eq, PartialEq)]
21pub enum TabletToolType {
22    /// A generic pen.
23    Pen,
24    /// Eraser
25    Eraser,
26    /// A paintbrush-like tool.
27    Brush,
28    /// Physical drawing tool, e.g. Wacom Inking Pen
29    Pencil,
30    /// An airbrush-like tool.
31    Airbrush,
32    /// A mouse bound to the tablet.
33    Mouse,
34    /// A mouse tool with a lens.
35    Lens,
36    /// A rotary device with positional and rotation data.  
37    Totem,
38    /// Type of the device is not known or does not match any known ones
39    Unknown,
40}

63/// Tablet tool event
64///
65/// The [AbsolutePositionEvent] implementation for tablet tool events produces (untransformed)
66/// coordinates in mm from the top left corner of the tablet in its current logical orientation.
67pub trait TabletToolEvent<B: InputBackend>: AbsolutePositionEvent<B> {
68    /// Get tablet tool that caused this event
69    fn tool(&self) -> TabletToolDescriptor;
70
71    /// Delta between the last and new pointer device position interpreted as pixel movement
72    fn delta(&self) -> Point<f64, Logical> {
73        (self.delta_x(), self.delta_y()).into()
74    }
75
76    /// Returns the current tilt along the (X,Y) axis of the tablet's current logical
77    /// orientation, in degrees off the tablet's z axis.
78    ///
79    /// That is, if the tool is perfectly orthogonal to the tablet, the tilt angle is 0.
80    /// When the top tilts towards the logical top/left of the tablet, the x/y tilt
81    /// angles are negative, if the top tilts towards the logical bottom/right of the
82    /// tablet, the x/y tilt angles are positive.
83    ///
84    /// If this axis does not exist on the current tool, this function returns (0,0).
85    fn tilt(&self) -> (f64, f64) {
86        (self.tilt_x(), self.tilt_y())
87    }
88
89    /// Check if the tilt was updated in this event.
90    fn tilt_has_changed(&self) -> bool {
91        self.tilt_x_has_changed() || self.tilt_y_has_changed()
92    }
93
94    /// Delta on the x axis between the last and new pointer device position interpreted as pixel movement
95    fn delta_x(&self) -> f64;
96    /// Delta on the y axis between the last and new pointer device position interpreted as pixel movement
97    fn delta_y(&self) -> f64;
98
99    /// If this axis does not exist on the current tool, this function returns 0.
100    fn distance(&self) -> f64;
101
102    /// Check if the distance axis was updated in this event.
103    fn distance_has_changed(&self) -> bool;
104
105    /// Returns the current pressure being applied on the tool in use, normalized to the range [0, 1].
106    ///
107    /// If this axis does not exist on the current tool, this function returns 0.
108    fn pressure(&self) -> f64;
109
110    /// Check if the pressure axis was updated in this event.
111    fn pressure_has_changed(&self) -> bool;
112
113    /// Returns the current position of the slider on the tool, normalized to the range
114    /// [-1, 1].
115    ///
116    /// The logical zero is the neutral position of the slider, or the logical center of
117    /// the axis. This axis is available on e.g. the Wacom Airbrush.
118    ///
119    /// If this axis does not exist on the current tool, this function returns 0.
120    fn slider_position(&self) -> f64;
121
122    /// Check if the slider axis was updated in this event.
123    fn slider_has_changed(&self) -> bool;
124
125    /// Returns the current tilt along the X axis of the tablet's current logical
126    /// orientation, in degrees off the tablet's z axis.
127    ///
128    /// That is, if the tool is perfectly orthogonal to the tablet, the tilt angle is 0.
129    /// When the top tilts towards the logical top/left of the tablet, the x/y tilt
130    /// angles are negative, if the top tilts towards the logical bottom/right of the
131    /// tablet, the x/y tilt angles are positive.
132    ///
133    /// If this axis does not exist on the current tool, this function returns 0.
134    fn tilt_x(&self) -> f64;
135
136    /// Check if the tilt x axis was updated in this event.
137    fn tilt_x_has_changed(&self) -> bool;
138
139    /// Returns the current tilt along the Y axis of the tablet's current logical
140    /// orientation, in degrees off the tablet's z axis.
141    ///
142    /// That is, if the tool is perfectly orthogonal to the tablet, the tilt angle is 0.
143    /// When the top tilts towards the logical top/left of the tablet, the x/y tilt
144    /// angles are negative, if the top tilts towards the logical bottom/right of the
145    /// tablet, the x/y tilt angles are positive.
146    ///
147    /// If this axis does not exist on the current tool, this function returns 0.
148    fn tilt_y(&self) -> f64;
149
150    /// Check if the tilt y axis was updated in this event.
151    fn tilt_y_has_changed(&self) -> bool;
152
153    /// Returns the current z rotation of the tool in degrees, clockwise from the tool's logical neutral position.
154    ///
155    /// For tools of type Mouse and Lens the logical neutral position is pointing to the current logical north of the tablet.
156    /// For tools of type Brush, the logical neutral position is with the buttons pointing up.
157    ///
158    /// If this axis does not exist on the current tool, this function returns 0.
159    fn rotation(&self) -> f64;
160
161    /// Check if the z-rotation axis was updated in this event.
162    fn rotation_has_changed(&self) -> bool;
163
164    /// Return the delta for the wheel in degrees.
165    fn wheel_delta(&self) -> f64;
166    /// Return the delta for the wheel in discrete steps (e.g. wheel clicks).
167    fn wheel_delta_discrete(&self) -> i32;
168    /// Check if the wheel axis was updated in this event.
169    fn wheel_has_changed(&self) -> bool;
170}

234/// The state of proximity for a tool on a device.
235///
236/// The proximity of a tool is a binary state signalling whether the tool is within a
237/// detectable distance of the tablet device. A tool that is out of proximity cannot
238/// generate events.
239///
240/// On some hardware a tool goes out of proximity when it ceases to touch the surface. On
241/// other hardware, the tool is still detectable within a short distance (a few cm) off
242/// the surface.
243#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
244pub enum ProximityState {
245    /// Out of proximity
246    Out,
247    /// In proximity
248    In,
249}

273#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]

276    Up,
277    /// Touching the surface
278    Down,

281/// Signals that a tool has come in contact with the surface of a device with the
282/// `DeviceCapability::TabletTool` capability.
283///
284/// On devices without distance proximity detection, the `TabletToolTipEvent` is sent
285/// immediately after `TabletToolProximityEvent` for the tip down event, and
286/// immediately before for the tip up event.
287///
288/// The decision when a tip touches the surface is device-dependent and may be
289/// derived from pressure data or other means. If the tip state is changed by axes
290/// changing state, the `TabletToolTipEvent` includes the changed axes and no
291/// additional axis event is sent for this state change. In other words, a caller
292/// must look at both `TabletToolAxisEvent` and `TabletToolTipEvent` events to know
293/// the current state of the axes.
294///
295/// If a button state change occurs at the same time as a tip state change, the order
296/// of events is device-dependent.
297pub trait TabletToolTipEvent<B: InputBackend>: TabletToolEvent<B> + Event<B> {
298    /// Returns the new tip state of a tool from a tip event.
299    ///
300    /// Used to check whether or not a tool came in contact with the tablet surface or
301    /// left contact with the tablet surface during an `TabletToolTipEvent`.
302    fn tip_state(&self) -> TabletToolTipState;
303}

308    }
