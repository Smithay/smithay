5/// The extension macro.
6///
7/// This macro generates a struct which checks for the presence of some X11 extensions and stores
8/// the version supplied by the X server.
9///
10/// ```ignore(cannot-test-this-because-non-exported-macro)
11/// extensions! {
12///     // The extension to check for. This should correspond to the name of the extension inside x11rb's `x11rb::protocol::xproto::<name>` module path.
13///     xfixes {
14///         // The function used to query the available version of the extension. This will be inside the module path as explained above
15///         xfixes_query_version,
16///         // The minimum version of the extension that will be accepted.
17///         minimum: (4, 0),
18///         // The version of the extension to request.
19///         request: (4, 0),
20///     },
21/// }
22///
23/// // The extensions may be checked then using the generated `Extensions` struct using the `check_extensions` function.
24/// ```
25macro_rules! extensions {
26    (
27        $(
28            $extension:ident { // Extension name for path lookup
29                $extension_fn:ident, // Function used to look up the version of the extension
30                required: $required:expr,
31                minimum: ($min_major:expr, $min_minor:expr),
32                request: ($req_major:expr, $req_minor:expr),
33            },
34        )*
35    ) => {
36        #[allow(dead_code)]
37        #[derive(Debug, Copy, Clone)]
38        pub struct Extensions {
39            $(
40                #[doc = concat!(" The version of the `", stringify!($extension), "` extension.")]
41                pub $extension: Option<(u32, u32)>,
42            )*
43        }
44
45        impl Extensions {
46            pub fn check_extensions<C: x11rb::connection::Connection>(connection: &C) -> Result<Extensions, X11Error> {
47                $(
48                    let $extension = {
49                        use x11rb::protocol::$extension::{ConnectionExt as _, X11_EXTENSION_NAME};
50
51                        if connection.extension_information(X11_EXTENSION_NAME)?.is_some() {
52                            let version = connection.$extension_fn($req_major, $req_minor)?.reply()?;
53                            let major_version = u32::from(version.major_version);
54                            let minor_version = u32::from(version.minor_version);
55
56                            #[allow(unused_comparisons)] // Macro comparisons
57                            if major_version >= $req_major
58                                || (major_version == $req_major && minor_version >= $req_minor)
59                            {
60                                info!(
61                                    "Loaded extension {} version {}.{}",
62                                    X11_EXTENSION_NAME,
63                                    major_version,
64                                    minor_version,
65                                );
66
67                                Some((major_version, minor_version))
68                            } else {
69                                if $required {
70                                    error!(
71                                        "required extension {} version is too low (have {}.{}, expected {}.{})",
72                                        X11_EXTENSION_NAME,
73                                        major_version,
74                                        minor_version,
75                                        $req_major,
76                                        $req_minor,
77                                    );
78
79                                    return Err(MissingExtensionError::WrongVersion {
80                                        name: X11_EXTENSION_NAME,
81                                        required_major: $req_major,
82                                        required_minor: $req_minor,
83                                        available_major: major_version,
84                                        available_minor: minor_version,
85                                    }.into());
86                                } else {
87                                    None
88                                }
89                            }
90                        } else {
91                            if $required {
92                                error!("required extension {} not found", X11_EXTENSION_NAME);
93
94                                return Err(MissingExtensionError::NotFound {
95                                    name: X11_EXTENSION_NAME,
96                                    major: $min_major,
97                                    minor: $min_minor,
98                                }
99                                .into());
100                            } else {
101                                None
102                            }
103                        }
104                    };
105                )*
106
107                Ok(Extensions {
108                    $(
109                        $extension,
110                    )*
111                })
112            }
113        }
114    };
115}
