23/// An error that may occur when presenting.
24#[derive(Debug, thiserror::Error)]
25pub enum PresentError {
26    /// The dmabuf being presented has too many planes.
27    #[error("The Dmabuf had too many planes")]
28    TooManyPlanes,
29
30    /// Duplicating the dmabuf handles failed.
31    #[error("Duplicating the file descriptors for the dmabuf handles failed")]
32    DupFailed(String),
33
34    /// The format dmabuf presented does not match the format of the window.
35    #[error("Buffer had incorrect format, expected: {0}")]
36    IncorrectFormat(DrmFourcc),
37}

39/// An X11 surface which uses GBM to allocate and present buffers.
40#[derive(Debug)]
41pub struct X11Surface {
42    pub(crate) connection: Weak<RustConnection>,
43    pub(crate) window: Weak<WindowInner>,
44    pub(crate) resize: Receiver<Size<u16, Logical>>,
45    pub(crate) swapchain: Swapchain<Box<dyn Allocator<Buffer = Dmabuf, Error = AnyError> + 'static>>,
46    pub(crate) format: DrmFourcc,
47    pub(crate) width: u16,
48    pub(crate) height: u16,
49    pub(crate) buffer: Option<Slot<Dmabuf>>,
50    pub(crate) span: tracing::Span,
51}
52
53impl X11Surface {
54    /// Returns the window the surface presents to.
55    ///
56    /// This will return [`None`] if the window has been destroyed.
57    pub fn window(&self) -> Option<impl AsRef<Window> + '_> {
58        self.window.upgrade().map(Window).map(WindowTemporary)
59    }
60
61    /// Returns the format of the buffers the surface accepts.
62    pub fn format(&self) -> DrmFourcc {
63        self.format
64    }
65
66    /// Returns the next buffer that will be presented to the Window and its age.
67    ///
68    /// You may bind this buffer to a renderer to render.
69    /// This function will return the same buffer until [`submit`](Self::submit) is called
70    /// or [`reset_buffers`](Self::reset_buffers) is used to reset the buffers.
71    #[instrument(level = "trace", parent = &self.span, skip(self))]
72    #[profiling::function]
73    pub fn buffer(&mut self) -> Result<(Dmabuf, u8), AllocateBuffersError> {
74        if let Some(new_size) = self.resize.try_iter().last() {
75            self.resize(new_size);
76        }
77
78        if self.buffer.is_none() {
79            self.buffer = Some(
80                self.swapchain
81                    .acquire()
82                    .map_err(Into::<AllocateBuffersError>::into)?
83                    .ok_or(AllocateBuffersError::NoFreeSlots)?,
84            );
85        }
86
87        let slot = self.buffer.as_ref().unwrap();
88        let age = slot.age();
89        Ok(((*slot).clone(), age))
90    }
91
92    /// Consume and submit the buffer to the window.
93    #[instrument(level = "trace", parent = &self.span, skip(self))]
94    #[profiling::function]
95    pub fn submit(&mut self) -> Result<(), X11Error> {
96        if let Some(connection) = self.connection.upgrade() {
97            // Get a new buffer
98            let mut next = self
99                .swapchain
100                .acquire()
101                .map_err(Into::<AllocateBuffersError>::into)?
102                .ok_or(AllocateBuffersError::NoFreeSlots)?;
103
104            // Swap the buffers
105            if let Some(current) = self.buffer.as_mut() {
106                mem::swap(&mut next, current);
107            }
108
109            {
110                let window = self.window().ok_or(AllocateBuffersError::WindowDestroyed)?;
111                let pixmap = PixmapWrapper::with_dmabuf(&*connection, window.as_ref(), &next)?;
112
113                // Now present the current buffer
114                let _ = pixmap.present(&*connection, window.as_ref())?;
115            }
116            self.swapchain.submitted(&next);
117
118            // Flush the connection after presenting to the window to ensure we don't run out of buffer space in the X11 connection.
119            let _ = connection.flush();
120        }
121        Ok(())
122    }
123
124    /// Resets the internal buffers.
125    #[instrument(level = "trace", parent = &self.span, skip(self))]
126    pub fn reset_buffers(&mut self) {
127        self.swapchain.reset_buffers();
128        self.buffer = None;
129    }
130
131    /// Reset the age for all internal buffers.
132    ///
133    /// This can be used to clear the damage history.
134    #[instrument(level = "trace", parent = &self.span, skip(self))]
135    pub fn reset_buffer_ages(&mut self) {
136        self.swapchain.reset_buffer_ages();
137    }
138
139    fn resize(&mut self, size: Size<u16, Logical>) {
140        self.swapchain.resize(size.w as u32, size.h as u32);
141        self.buffer = None;
142
143        self.width = size.w;
144        self.height = size.h;
145    }
146}
