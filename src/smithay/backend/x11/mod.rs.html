1//! Implementation of the backend types using X11.
2//!
3//! This backend provides the appropriate backend implementations to run a Wayland compositor as an
4//! X11 client.
5//!
6//! The backend is initialized using [`X11Backend::new`](self::X11Backend::new).
7//!
8//! ## Example usage
9//!
10//! ```rust,no_run
11//! # use std::{sync::{Arc, Mutex}, error::Error};
12//! # use smithay::backend::x11::{X11Backend, X11Surface, WindowBuilder};
13//! use smithay::backend::allocator::dmabuf::DmabufAllocator;
14//! use smithay::backend::allocator::gbm::{GbmAllocator, GbmDevice, GbmBufferFlags};
15//! use smithay::backend::egl::{EGLDisplay, EGLContext};
16//! use smithay::utils::DeviceFd;
17//! use std::collections::HashSet;
18//!
19//! # struct CompositorState;
20//! fn init_x11_backend(
21//!    handle: calloop::LoopHandle<CompositorState>,
22//! ) -> Result<(), Box<dyn Error>> {
23//!     // Create the backend
24//!     let backend = X11Backend::new()?;
25//!
26//!     // Get a handle from the backend to interface with the X server
27//!     let x_handle = backend.handle();
28//!     // Create a window
29//!     let window = WindowBuilder::new()
30//!         .title("Wayland inside X11")
31//!         .build(&x_handle)
32//!         .expect("Could not create window");
33//!
34//!     // To render to a window, we need to create an X11 surface.
35//!
36//!     // Get the DRM node used by the X server for direct rendering.
37//!     let (_drm_node, fd) = x_handle.drm_node()?;
38//!     // Create the gbm device for allocating buffers
39//!     let device = GbmDevice::new(DeviceFd::from(fd))?;
40//!     // Initialize EGL to retrieve the support modifier list
41//!     let egl = unsafe { EGLDisplay::new(device.clone()).expect("Failed to create EGLDisplay") };
42//!     let context = EGLContext::new(&egl).expect("Failed to create EGLContext");
43//!     let modifiers = context.dmabuf_render_formats().iter().map(|format| format.modifier).collect::<HashSet<_>>();
44//!
45//!     // Finally create the X11 surface, you will use this to obtain buffers that will be presented to the
46//!     // window.
47//!     let surface = x_handle.create_surface(&window, DmabufAllocator(GbmAllocator::new(device, GbmBufferFlags::RENDERING)), modifiers.into_iter());
48//!
49//!     // Insert the backend into the event loop to receive events.
50//!     handle.insert_source(backend, |event, _window, state| {
51//!         // Process events from the X server that apply to the window.
52//!     })?;
53//!
54//!     Ok(())
55//! }
56//! ```
57//!
58//! ## EGL
59//!
60//! When using [`EGL`](crate::backend::egl), an [`X11Surface`] may be used to create an [`EGLDisplay`].

130/// An event emitted by the X11 backend.
131#[derive(Debug)]
132pub enum X11Event {
133    /// The X server has required the compositor to redraw the contents of window.
134    Refresh {
135        /// XID of the window
136        window_id: u32,
137    },
138
139    /// The focus state of the window changed.
140    Focus {
141        /// The new focus state
142        focused: bool,
143        /// The XID of the window
144        window_id: u32,
145    },
146
147    /// An input event occurred.
148    Input {
149        /// The received event.
150        event: InputEvent<X11Input>,
151        /// The XID of the window.
152        ///
153        /// If the XID is None, this means that no specific window was targeted for this, and all
154        /// of them should react to this change.
155        window_id: Option<u32>,
156    },
157
158    /// The window was resized.
159    Resized {
160        /// The new size of the window
161        new_size: Size<u16, Logical>,
162        /// XID of the window
163        window_id: u32,
164    },
165
166    /// The last buffer presented to the window has been displayed.
167    ///
168    /// When this event is scheduled, the next frame may be rendered.
169    PresentCompleted {
170        /// XID of the window
171        window_id: u32,
172    },
173
174    /// The window has received a request to be closed.
175    CloseRequested {
176        /// XID of the window
177        window_id: u32,
178    },
179}

181/// Represents an active connection to the X to manage events on the Window provided by the backend.
182#[derive(Debug)]
183pub struct X11Backend {
184    connection: Arc<RustConnection>,
185    source: X11Source,
186    inner: Arc<Mutex<X11Inner>>,
187    span: tracing::Span,
188}
189
190impl X11Backend {
191    /// Initializes the X11 backend by connecting to the X server.
192    pub fn new() -> Result<X11Backend, X11Error> {
193        let span = debug_span!("backend_x11");
194        let _guard = span.enter();
195
196        info!("Connecting to the X server");
197
198        let (connection, screen_number) = RustConnection::connect(None)?;
199        let connection = Arc::new(connection);
200        info!(screen = screen_number, "Connected");
201        span.record("screen", screen_number);
202
203        let extensions = Extensions::check_extensions(&*connection)?;
204
205        let screen = &connection.setup().roots[screen_number];
206
207        let depth = screen
208            .allowed_depths
209            .iter()
210            .find(|depth| depth.depth == 32) // Prefer 32-bit color
211            .or_else(|| screen.allowed_depths.iter().find(|depth| depth.depth == 24)) // 24-bit fallback for Xrgb8888
212            .cloned()
213            .ok_or(CreateWindowError::NoDepth)?;
214
215        // Next find a visual using the supported depth
216        let visual_id = depth
217            .visuals
218            .iter()
219            // Ensure the visual is little endian to comply with the format needed with X/ARGB8888
220            .filter(|visual| visual.red_mask == 0xff0000)
221            .find(|visual| visual.class == VisualClass::TRUE_COLOR)
222            .ok_or(CreateWindowError::NoVisual)?
223            .visual_id;
224
225        let format = match depth.depth {
226            24 => DrmFourcc::Xrgb8888,
227            32 => DrmFourcc::Argb8888,
228            _ => unreachable!(),
229        };
230        info!(?depth, visual_id, %format, "Window parameters selected");
231
232        // Make a colormap
233        let colormap = connection.generate_id()?;
234        connection.create_colormap(ColormapAlloc::NONE, colormap, screen.root, visual_id)?;
235
236        let atoms = Atoms::new(&*connection)?.reply()?;
237
238        // We need to give the X11Source a window we have created, we cannot send the close event to the root
239        // window (0). To handle this, we will create a window we never map or provide to users to the backend
240        // can be sent a message for shutdown.
241
242        let close_window = WindowWrapper::create_window(
243            &*connection,
244            x11rb::COPY_DEPTH_FROM_PARENT,
245            screen.root,
246            0,
247            0,
248            1,
249            1,
250            0,
251            WindowClass::INPUT_OUTPUT,
252            x11rb::COPY_FROM_PARENT,
253            &CreateWindowAux::new(),
254        )?
255        .into_window();
256
257        let source = X11Source::new(connection.clone(), close_window, atoms._SMITHAY_X11_BACKEND_CLOSE);
258
259        let inner = X11Inner {
260            connection: connection.clone(),
261            screen_number,
262            windows: HashMap::new(),
263            key_counter: Arc::new(AtomicU32::new(0)),
264            window_format: format,
265            extensions,
266            colormap,
267            atoms,
268            depth,
269            visual_id,
270            devices: false,
271        };
272
273        drop(_guard);
274        Ok(X11Backend {
275            connection,
276            source,
277            inner: Arc::new(Mutex::new(inner)),
278            span,
279        })
280    }
281
282    /// Returns a handle to the X11 backend.
283    pub fn handle(&self) -> X11Handle {
284        X11Handle {
285            connection: self.connection.clone(),
286            inner: self.inner.clone(),
287            span: self.span.clone(),
288        }
289    }

310impl X11Handle {
311    /// Returns the default screen number of the X server.
312    pub fn screen(&self) -> usize {
313        self.inner.lock().unwrap().screen_number
314    }
315
316    /// Returns the underlying connection to the X server.
317    pub fn connection(&self) -> Arc<RustConnection> {
318        self.connection.clone()
319    }
320
321    /// Returns the format of the window.
322    pub fn format(&self) -> DrmFourcc {
323        self.inner.lock().unwrap().window_format
324    }
325
326    /// Returns the DRM node the X server uses for direct rendering.
327    ///
328    /// The DRM node may be used to create a [`gbm::Device`] to allocate buffers.
329    #[instrument(parent = &self.span, skip(self), ret, err)]
330    pub fn drm_node(&self) -> Result<(DrmNode, OwnedFd), X11Error> {
331        // Kernel documentation explains why we should prefer the node to be a render node:
332        // https://kernel.readthedocs.io/en/latest/gpu/drm-uapi.html
333        //
334        // > Render nodes solely serve render clients, that is, no modesetting or privileged ioctls
335        // > can be issued on render nodes. Only non-global rendering commands are allowed. If a
336        // > driver supports render nodes, it must advertise it via the DRIVER_RENDER DRM driver
337        // > capability. If not supported, the primary node must be used for render clients together
338        // > with the legacy drmAuth authentication procedure.
339        //
340        // Since giving the X11 backend the ability to do modesetting is a big nono, we try to only
341        // ever create a gbm device from a render node.
342        //
343        // Of course if the DRM device does not support render nodes, no DRIVER_RENDER capability, then
344        // fall back to the primary node.
345
346        // We cannot fallback on the egl_init method, because there is no way for us to authenticate a primary node.
347        // dri3 does not work for closed-source drivers, but *may* give us a authenticated fd as a fallback.
348        // As a result we try to use egl for a cleaner, better supported approach at first and only if that fails use dri3.
349        let inner = self.inner.lock().unwrap();
350
351        egl_init(&inner).or_else(|err| {
352            warn!(
353                "Failed to init X11 surface via egl, falling back to dri3: {}",
354                err
355            );
356            dri3_init(&inner)
357        })
358    }
359
360    /// Creates a surface that allocates and presents buffers to the window.
361    ///
362    /// This will fail if the window has already been used to create a surface.
363    #[instrument(parent = &self.span, skip(self, allocator, modifiers))]
364    pub fn create_surface<A: Allocator<Buffer = Dmabuf, Error = AnyError> + 'static>(
365        &self,
366        window: &Window,
367        allocator: A,
368        modifiers: impl Iterator<Item = DrmModifier>,
369    ) -> Result<X11Surface, X11Error> {
370        let has_resize = { window.0.resize.lock().unwrap().is_some() };
371
372        if has_resize {
373            return Err(X11Error::SurfaceExists);
374        }
375
376        let inner = self.inner.clone();
377        let inner_guard = inner.lock().unwrap();
378
379        // Fail if the window is not managed by this backend or is destroyed
380        if !inner_guard.windows.contains_key(&window.id()) {
381            return Err(X11Error::InvalidWindow);
382        }
383
384        let mut modifiers = modifiers.collect::<Vec<_>>();
385        // older dri3 versions do only support buffers with one plane.
386        // we need to make sure, we don't accidently allocate buffers with more.
387        if window.0.extensions.dri3 < Some((1, 2)) {
388            modifiers.retain(|modi| modi == &DrmModifier::Invalid || modi == &DrmModifier::Linear);
389        }
390
391        let format = window.0.format;
392        let size = window.size();
393        let swapchain = Swapchain::new(
394            Box::new(allocator) as Box<dyn Allocator<Buffer = Dmabuf, Error = AnyError> + 'static>,
395            size.w as u32,
396            size.h as u32,
397            format,
398            modifiers,
399        );
400
401        let (sender, recv) = mpsc::channel();
402
403        {
404            let mut resize = window.0.resize.lock().unwrap();
405            *resize = Some(sender);
406        }
407
408        Ok(X11Surface {
409            connection: Arc::downgrade(&inner_guard.connection),
410            window: Arc::downgrade(&window.0),
411            swapchain,
412            format,
413            width: size.w,
414            height: size.h,
415            buffer: None,
416            resize: recv,
417            span: self.span.clone(),
418        })
419    }
420
421    /// Get a temporary reference to a window by its XID
422    pub fn window_ref_from_id(&self, id: u32) -> Option<impl AsRef<Window> + '_> {
423        X11Inner::window_ref_from_id(&self.inner, &id)
424            .and_then(|w| w.upgrade())
425            .map(Window)
426            .map(WindowTemporary)
427    }

430/// Builder used to construct a window.
431#[derive(Debug)]
432pub struct WindowBuilder<'a> {
433    name: Option<&'a str>,
434    size: Option<Size<u16, Logical>>,
435}
436
437impl<'a> WindowBuilder<'a> {
438    #[allow(clippy::new_without_default)]
439    /// Returns a new builder.
440    pub fn new() -> WindowBuilder<'a> {
441        WindowBuilder {
442            name: None,
443            size: None,
444        }
445    }
446
447    /// Sets the title of the window that will be created by the builder.
448    pub fn title(self, name: &'a str) -> Self {
449        Self {
450            name: Some(name),
451            ..self
452        }
453    }
454
455    /// Sets the size of the window that will be created.
456    ///
457    /// There is no guarantee the size specified here will be the actual size of the window when it is
458    /// presented.
459    pub fn size(self, size: Size<u16, Logical>) -> Self {
460        Self {
461            size: Some(size),
462            ..self
463        }
464    }
465
466    /// Creates a window using the options specified in the builder.
467    pub fn build(self, handle: &X11Handle) -> Result<Window, X11Error> {
468        let _guard = handle.span.enter();
469        let connection = handle.connection();
470
471        let inner = &mut *handle.inner.lock().unwrap();
472
473        let window = Arc::new(WindowInner::new(
474            Arc::downgrade(&connection),
475            &connection.setup().roots[inner.screen_number],
476            self.size.unwrap_or_else(|| (1280, 800).into()),
477            self.name.unwrap_or("Smithay"),
478            inner.window_format,
479            inner.atoms,
480            inner.depth.clone(),
481            inner.visual_id,
482            inner.colormap,
483            inner.extensions,
484        )?);
485
486        let downgrade = Arc::downgrade(&window);
487        inner.windows.insert(window.id, downgrade);
488
489        Ok(Window(window))
490    }

561impl EventSource for X11Backend {
562    type Event = X11Event;
563    type Metadata = ();
564    type Ret = ();
565    type Error = X11Error;
566
567    #[profiling::function]
568    fn process_events<F>(
569        &mut self,
570        readiness: Readiness,
571        token: Token,
572        mut callback: F,
573    ) -> Result<PostAction, X11Error>
574    where
575        F: FnMut(Self::Event, &mut Self::Metadata) -> Self::Ret,
576    {
577        let connection = self.connection.clone();
578        let inner = self.inner.clone();
579        let _guard = self.span.enter();
580
581        let post_action = self
582            .source
583            .process_events(readiness, token, |event, _| match event {
584                calloop::channel::Event::Msg(event) => {
585                    X11Inner::process_event(&inner, event, &mut callback);
586                }
587                calloop::channel::Event::Closed => {}
588            })
589            .map_err(|_| X11Error::ConnectionLost)?;
590
591        // Flush the connection so changes to the window state during callbacks can be emitted.
592        let _ = connection.flush();
593
594        Ok(post_action)
595    }
596
597    fn register(&mut self, poll: &mut Poll, token_factory: &mut TokenFactory) -> calloop::Result<()> {
598        self.source.register(poll, token_factory)
599    }
600
601    fn reregister(&mut self, poll: &mut Poll, token_factory: &mut TokenFactory) -> calloop::Result<()> {
602        self.source.reregister(poll, token_factory)
603    }
604
605    fn unregister(&mut self, poll: &mut Poll) -> calloop::Result<()> {
606        self.source.unregister(poll)
607    }
608}

620#[derive(Debug)]
621pub(crate) struct X11Inner {
622    connection: Arc<RustConnection>,
623    screen_number: usize,
624    windows: HashMap<u32, Weak<WindowInner>>,
625    key_counter: Arc<AtomicU32>,
626    window_format: DrmFourcc,
627    extensions: Extensions,
628    colormap: u32,
629    atoms: Atoms,
630    depth: x11::xproto::Depth,
631    visual_id: u32,
632    devices: bool,
633}
634
635impl X11Inner {
636    fn window_ref_from_id(inner: &Arc<Mutex<X11Inner>>, id: &u32) -> Option<Weak<WindowInner>> {
637        let mut inner = inner.lock().unwrap();
638        inner.windows.retain(|_, weak| weak.strong_count() != 0);
639        inner.windows.get(id).cloned()
640    }
641
642    #[profiling::function]
643    fn process_event<F>(inner: &Arc<Mutex<X11Inner>>, event: x11::Event, callback: &mut F)
644    where
645        F: FnMut(X11Event, &mut ()),
646    {
647        {
648            let mut inner = inner.lock().unwrap();
649            if !inner.windows.is_empty() && !inner.devices {
650                callback(
651                    Input {
652                        event: InputEvent::DeviceAdded {
653                            device: X11VirtualDevice,
654                        },
655                        window_id: None,
656                    },
657                    &mut (),
658                );
659                inner.devices = true;
660            } else if inner.windows.is_empty() && inner.devices {
661                callback(
662                    Input {
663                        event: InputEvent::DeviceRemoved {
664                            device: X11VirtualDevice,
665                        },
666                        window_id: None,
667                    },
668                    &mut (),
669                );
670                inner.devices = false;
671            }
672        }
673
674        use self::X11Event::{Focus, Input};
675
676        // If X11 is deadlocking somewhere here, make sure you drop your mutex guards.
677
678        match event {
679            x11::Event::XinputFocusIn(focus_in) => {
680                callback(
681                    Focus {
682                        focused: true,
683                        window_id: focus_in.event,
684                    },
685                    &mut (),
686                );
687            }
688
689            x11::Event::XinputFocusOut(focus_out) => {
690                callback(
691                    Focus {
692                        focused: false,
693                        window_id: focus_out.event,
694                    },
695                    &mut (),
696                );
697            }
698
699            x11::Event::XinputButtonPress(button_press) => {
700                if let Some(window) = X11Inner::window_ref_from_id(inner, &button_press.event) {
701                    // X11 decided to associate scroll wheel with a button, 4, 5, 6 and 7 for
702                    // up, down, right and left. For scrolling, a press event is emitted and a
703                    // release is them immediately followed for scrolling. This means we can
704                    // ignore release for scrolling.
705
706                    // Ideally we would use `ButtonIndex` from XCB, however it does not cover 6 and 7
707                    // for horizontal scroll and does not work nicely in match statements, so we
708                    // use magic constants here:
709                    //
710                    // 1 => MouseButton::Left
711                    // 2 => MouseButton::Middle
712                    // 3 => MouseButton::Right
713                    // 4 => Axis::Vertical +1.0
714                    // 5 => Axis::Vertical -1.0
715                    // 6 => Axis::Horizontal -1.0
716                    // 7 => Axis::Horizontal +1.0
717                    // Others => ??
718
719                    // Scrolling
720                    if button_press.detail >= 4 && button_press.detail <= 7 {
721                        callback(
722                            Input {
723                                event: InputEvent::PointerAxis {
724                                    event: X11MouseWheelEvent {
725                                        time: button_press.time,
726                                        axis: match button_press.detail {
727                                            // Up | Down
728                                            4 | 5 => Axis::Vertical,
729
730                                            // Right | Left
731                                            6 | 7 => Axis::Horizontal,
732
733                                            _ => unreachable!(),
734                                        },
735                                        amount: match button_press.detail {
736                                            // Up | Right
737                                            4 | 7 => 1.0,
738
739                                            // Down | Left
740                                            5 | 6 => -1.0,
741
742                                            _ => unreachable!(),
743                                        },
744                                        window,
745                                    },
746                                },
747                                window_id: Some(button_press.event),
748                            },
749                            &mut (),
750                        )
751                    } else {
752                        callback(
753                            Input {
754                                event: InputEvent::PointerButton {
755                                    event: X11MouseInputEvent {
756                                        time: button_press.time,
757                                        raw: button_press.detail,
758                                        state: ButtonState::Pressed,
759                                        window,
760                                    },
761                                },
762                                window_id: Some(button_press.event),
763                            },
764                            &mut (),
765                        )
766                    }
767                }
768            }
769
770            x11::Event::XinputButtonRelease(button_release) => {
771                // Ignore release tick because this event is always sent immediately after the press
772                // tick for scrolling and the backend will dispatch release event automatically during
773                // the press event.
774                if button_release.detail >= 4 && button_release.detail <= 7 {
775                    return;
776                }
777
778                if let Some(window) = X11Inner::window_ref_from_id(inner, &button_release.event) {
779                    callback(
780                        Input {
781                            event: InputEvent::PointerButton {
782                                event: X11MouseInputEvent {
783                                    time: button_release.time,
784                                    raw: button_release.detail,
785                                    state: ButtonState::Released,
786                                    window,
787                                },
788                            },
789                            window_id: Some(button_release.event),
790                        },
791                        &mut (),
792                    );
793                }
794            }
795
796            x11::Event::XinputKeyPress(key_press) => {
797                if key_press.flags.contains(xinput::KeyEventFlags::KEY_REPEAT) {
798                    return;
799                }
800
801                if let Some(window) = X11Inner::window_ref_from_id(inner, &key_press.event) {
802                    // Do not hold the lock.
803                    let count = { inner.lock().unwrap().key_counter.fetch_add(1, Ordering::SeqCst) + 1 };
804
805                    callback(
806                        Input {
807                            event: InputEvent::Keyboard {
808                                event: X11KeyboardInputEvent {
809                                    time: key_press.time,
810                                    key: Keycode::from(key_press.detail),
811                                    count,
812                                    state: KeyState::Pressed,
813                                    window,
814                                },
815                            },
816                            window_id: Some(key_press.event),
817                        },
818                        &mut (),
819                    )
820                }
821            }
822
823            x11::Event::XinputKeyRelease(key_release) => {
824                if let Some(window) = X11Inner::window_ref_from_id(inner, &key_release.event) {
825                    let count = {
826                        let key_counter = inner.lock().unwrap().key_counter.clone();
827
828                        // atomic u32 has no checked_sub, so load and store to do the same.
829                        let mut count = key_counter.load(Ordering::SeqCst);
830                        count = count.saturating_sub(1);
831                        key_counter.store(count, Ordering::SeqCst);
832
833                        count
834                    };
835
836                    callback(
837                        Input {
838                            event: InputEvent::Keyboard {
839                                event: X11KeyboardInputEvent {
840                                    time: key_release.time,
841                                    key: Keycode::from(key_release.detail),
842                                    count,
843                                    state: KeyState::Released,
844                                    window,
845                                },
846                            },
847                            window_id: Some(key_release.event),
848                        },
849                        &mut (),
850                    );
851                }
852            }
853
854            x11::Event::XinputMotion(motion_notify) => {
855                if let Some(window) =
856                    X11Inner::window_ref_from_id(inner, &motion_notify.event).and_then(|w| w.upgrade())
857                {
858                    // Use event_x/y since those are relative the the window receiving events.
859                    let x = fixed_point_to_float(motion_notify.event_x);
860                    let y = fixed_point_to_float(motion_notify.event_y);
861
862                    let window_size = { *window.size.lock().unwrap() };
863
864                    callback(
865                        Input {
866                            event: InputEvent::PointerMotionAbsolute {
867                                event: X11MouseMovedEvent {
868                                    time: motion_notify.time,
869                                    x,
870                                    y,
871                                    size: window_size,
872                                    window: Arc::downgrade(&window),
873                                },
874                            },
875                            window_id: Some(motion_notify.event),
876                        },
877                        &mut (),
878                    )
879                }
880            }
881
882            x11::Event::ConfigureNotify(configure_notify) => {
883                if let Some(window) =
884                    X11Inner::window_ref_from_id(inner, &configure_notify.window).and_then(|w| w.upgrade())
885                {
886                    let previous_size = { *window.size.lock().unwrap() };
887
888                    // Did the size of the window change?
889                    let configure_notify_size: Size<u16, Logical> =
890                        (configure_notify.width, configure_notify.height).into();
891
892                    if configure_notify_size != previous_size {
893                        // Intentionally drop the lock on the size mutex incase a user
894                        // requests a resize or does something which causes a resize
895                        // inside the callback.
896                        {
897                            let mut resize_guard = window.size.lock().unwrap();
898                            *resize_guard = configure_notify_size;
899                        }
900
901                        (callback)(
902                            X11Event::Resized {
903                                new_size: configure_notify_size,
904                                window_id: configure_notify.window,
905                            },
906                            &mut (),
907                        );
908
909                        if let Some(resize_sender) = window.resize.lock().unwrap().as_ref() {
910                            let _ = resize_sender.send(configure_notify_size);
911                        }
912                    }
913                }
914            }
915
916            x11::Event::XinputEnter(enter_notify) => {
917                if let Some(window) =
918                    X11Inner::window_ref_from_id(inner, &enter_notify.event).and_then(|w| w.upgrade())
919                {
920                    window.cursor_enter();
921                }
922            }
923
924            x11::Event::XinputLeave(leave_notify) => {
925                if let Some(window) =
926                    X11Inner::window_ref_from_id(inner, &leave_notify.event).and_then(|w| w.upgrade())
927                {
928                    window.cursor_leave();
929                }
930            }
931
932            x11::Event::ClientMessage(client_message) => {
933                if let Some(window) =
934                    X11Inner::window_ref_from_id(inner, &client_message.window).and_then(|w| w.upgrade())
935                {
936                    if client_message.data.as_data32()[0] == window.atoms.WM_DELETE_WINDOW
937                    // Destroy the window?
938                    {
939                        (callback)(
940                            X11Event::CloseRequested {
941                                window_id: client_message.window,
942                            },
943                            &mut (),
944                        );
945                    }
946                }
947            }
948
949            x11::Event::Expose(expose) => {
950                if expose.count == 0 {
951                    (callback)(
952                        X11Event::Refresh {
953                            window_id: expose.window,
954                        },
955                        &mut (),
956                    );
957                }
958            }
959
960            x11::Event::PresentCompleteNotify(complete_notify) => {
961                if let Some(window) =
962                    X11Inner::window_ref_from_id(inner, &complete_notify.window).and_then(|w| w.upgrade())
963                {
964                    window.last_msc.store(complete_notify.msc, Ordering::SeqCst);
965
966                    (callback)(
967                        X11Event::PresentCompleted {
968                            window_id: complete_notify.window,
969                        },
970                        &mut (),
971                    );
972                }
973            }
974
975            x11::Event::PresentIdleNotify(_) => {
976                // Pixmap is reference counted in the X server, so we do not need to take and drop.
977            }
978
979            x11::Event::Error(e) => {
980                error!("X11 protocol error: {:?}", e);
981            }
982
983            _ => (),
984        }
985    }
986}

988fn fixed_point_to_float(value: i32) -> f64 {
989    let int = value >> 16;
990    let frac = value & 0xffff;
991    int as f64 + frac as f64 / u16::MAX as f64
992}

994fn egl_init(_: &X11Inner) -> Result<(DrmNode, OwnedFd), EGLInitError> {
995    let display = unsafe { EGLDisplay::new(X11DefaultDisplay)? };
996    let device = EGLDevice::device_for_display(&display)?;
997    let path = path_to_type(device.drm_device_path()?, NodeType::Render)?;
998    let node = DrmNode::from_path(&path)
999        .map_err(|err| match err {
1000            CreateDrmNodeError::Io(err) => err,
1001            _ => unreachable!(),
1002        })
1003        .map_err(EGLInitError::IO)?;
1004    let fd = rustix::fs::open(&path, OFlags::RDWR | OFlags::CLOEXEC, Mode::empty())
1005        .map_err(Into::<io::Error>::into)
1006        .map_err(EGLInitError::IO)?;
1007    Ok((node, fd))
1008}

1010fn dri3_init(x11: &X11Inner) -> Result<(DrmNode, OwnedFd), X11Error> {
1011    let connection = &x11.connection;
1012
1013    // Determine which drm-device the Display is using.
1014    let screen = &connection.setup().roots[x11.screen_number];
1015    // provider being NONE tells the X server to use the RandR provider.
1016    let dri3 = match connection.dri3_open(screen.root, x11rb::NONE)?.reply() {
1017        Ok(reply) => reply,
1018        Err(err) => {
1019            return Err(if let ReplyError::X11Error(ref protocol_error) = err {
1020                match protocol_error.error_kind {
1021                    // Implementation is risen when the renderer is not capable of X server is not capable
1022                    // of rendering at all.
1023                    ErrorKind::Implementation => X11Error::CannotDirectRender,
1024                    // Match may occur when the node cannot be authenticated for the client.
1025                    ErrorKind::Match => X11Error::CannotDirectRender,
1026                    _ => err.into(),
1027                }
1028            } else {
1029                err.into()
1030            });
1031        }
1032    };
1033    let device_fd = dri3.device_fd;
1034
1035    let dri_node = DrmNode::from_file(&device_fd).map_err(Into::<AllocateBuffersError>::into)?;
1036    if dri_node.ty() != NodeType::Render {
1037        // Try to get the render node.
1038        match dri_node.node_with_type(NodeType::Render) {
1039            Some(Ok(node)) => {
1040                match node
1041                    .dev_path()
1042                    .map(|path| rustix::fs::open(path, OFlags::RDWR | OFlags::CLOEXEC, Mode::empty()))
1043                {
1044                    Some(Ok(fd)) => return Ok((node, fd)),
1045                    Some(Err(err)) => {
1046                        warn!("Could not create render node from existing DRM node ({:?}): {}, falling back to primary node", dri_node.dev_path().as_ref().map(|x| x.display()), err);
1047                    }
1048                    None => {
1049                        warn!("Could not create render node from existing DRM node ({:?}), falling back to primary node", dri_node.dev_path().as_ref().map(|x| x.display()));
1050                    }
1051                }
1052            }
1053            Some(Err(err)) => {
1054                warn!("Could not create render node from existing DRM node ({:?}): {}, falling back to primary node", dri_node.dev_path().as_ref().map(|x| x.display()), err);
1055            }
1056            None => {
1057                warn!(
1058                    "No render node available for DRM node ({:?}), falling back to primary node",
1059                    dri_node.dev_path().as_ref().map(|x| x.display())
1060                );
1061            }
1062        };
1063    }
1064
1065    let fd_flags = rustix::io::fcntl_getfd(&device_fd).map_err(AllocateBuffersError::from)?;
1066
1067    // Enable the close-on-exec flag.
1068    rustix::io::fcntl_setfd(&device_fd, fd_flags | rustix::io::FdFlags::CLOEXEC)
1069        .map_err(AllocateBuffersError::from)?;
1070
1071    Ok((dri_node, device_fd))
1072}
