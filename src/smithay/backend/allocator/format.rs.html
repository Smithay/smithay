1//! Format info tables for DRM formats.
2//!
3//! This module provides three functions, [`get_opaque`], [`has_alpha`] and [`get_bpp`].
4//!
5//! [`get_opaque`] returns the opaque alternative of a DRM format with an alpha channel.
6//!
7//! ```
8//! # use smithay::backend::allocator::Fourcc;
9//! # use smithay::backend::allocator::format::get_opaque;
10//! assert_eq!(Some(Fourcc::Xrgb8888), get_opaque(Fourcc::Argb8888));
11//! ```
12//!
13//! [`has_alpha`] returns true if the format has an alpha channel.
14//!
15//! ```
16//! # use smithay::backend::allocator::Fourcc;
17//! # use smithay::backend::allocator::format::has_alpha;
18//! assert!(has_alpha(Fourcc::Argb8888));
19//! assert!(!has_alpha(Fourcc::Xrgb8888));
20//! ```
21//!
22//! [`get_bpp`] returns the number of bits per pixel of a format.
23//!
24//! ```
25//! # use smithay::backend::allocator::Fourcc;
26//! # use smithay::backend::allocator::format::get_bpp;
27//! assert_eq!(get_bpp(Fourcc::Argb8888), Some(32));
28//! ```
29//!
30//! [`get_depth`] returns the number of used bits per pixel of a format
31//! (excluding padding or non-alpha "X" parts of the format).
32//!
33//! ```
34//! # use smithay::backend::allocator::Fourcc;
35//! # use smithay::backend::allocator::format::get_depth;
36//! assert_eq!(get_depth(Fourcc::Argb8888), Some(32));
37//! assert_eq!(get_depth(Fourcc::Xrgb8888), Some(24));
38//! ```

45/// Macro to generate table lookup functions for formats.
46///
47/// See the module documentation for usage details.
48macro_rules! format_tables {
49    (
50        $($fourcc: ident {
51            $(opaque: $opaque: ident,)?
52            alpha: $alpha: expr,
53            bpp: $bpp: expr,
54            depth: $depth: expr $(,)?
55        }),*
56    ) => {
57        /// Returns the opaque alternative of the specified format.
58        ///
59        /// If the format has an alpha channel, this may return the corresponding opaque format.
60        ///
61        /// Unknown formats will always return [`None`].
62        pub const fn get_opaque(
63            fourcc: $crate::backend::allocator::Fourcc,
64        ) -> Option<$crate::backend::allocator::Fourcc> {
65            match fourcc {
66                $($(
67                    $crate::backend::allocator::Fourcc::$fourcc
68                        => Some($crate::backend::allocator::Fourcc::$opaque),
69                )?)*
70                _ => None,
71            }
72        }
73
74        /// Returns the transparent alternative of the specified format.
75        ///
76        /// If the format has an unused alpha channel, this may return the corresponding non-opaque format.
77        ///
78        /// Unknown formats will always return [`None`].
79        pub const fn get_transparent(
80            fourcc: $crate::backend::allocator::Fourcc,
81        ) -> Option<$crate::backend::allocator::Fourcc> {
82            match fourcc {
83                $($(
84                    $crate::backend::allocator::Fourcc::$opaque
85                        => Some($crate::backend::allocator::Fourcc::$fourcc),
86                )?)*
87                _ => None,
88            }
89        }
90
91        /// Returns true if the format has an alpha channel.
92        ///
93        /// This function may be useful to know if the alpha channel may need to be swizzled when rendering
94        /// with some graphics apis.
95        ///
96        /// Unknown formats will always return `false`.
97        pub const fn has_alpha(fourcc: $crate::backend::allocator::Fourcc) -> bool {
98            match fourcc {
99                $(
100                    $crate::backend::allocator::Fourcc::$fourcc => $alpha,
101                )*
102                _ => false,
103            }
104        }
105
106        /// Returns the bits per pixel of the specified format.
107        ///
108        /// Unknown formats will always return [`None`].
109        pub const fn get_bpp(
110            fourcc: $crate::backend::allocator::Fourcc,
111        ) -> Option<usize> {
112            match fourcc {
113                $($crate::backend::allocator::Fourcc::$fourcc => Some($bpp),)*
114                _ => None,
115            }
116        }
117
118        /// Returns the depth of the specified format.
119        ///
120        /// Unknown formats will always return [`None`].
121        pub const fn get_depth(
122            fourcc: $crate::backend::allocator::Fourcc,
123        ) -> Option<usize> {
124            match fourcc {
125                $($crate::backend::allocator::Fourcc::$fourcc => Some($depth),)*
126                _ => None,
127            }
128        }
129
130        fn _impl_formats() -> &'static [$crate::backend::allocator::Fourcc] {
131            &[
132                $(
133                    $crate::backend::allocator::Fourcc::$fourcc,
134                )*
135            ]
136        }
137    };
138}

371/// A set of [`Format`]s
372#[derive(Debug, Default, Clone)]
373pub struct FormatSet {
374    formats: Arc<IndexSet<Format>>,
375}
376
377impl FormatSet {
378    #[cfg(any(feature = "backend_egl", feature = "backend_drm"))]
379    pub(crate) fn from_formats(formats: IndexSet<Format>) -> Self {
380        FormatSet {
381            formats: Arc::new(formats),
382        }
383    }
384}
385
386impl FormatSet {
387    /// Return an iterator over the values of the set, in their order
388    pub fn iter(&self) -> FormatSetIter<'_> {
389        FormatSetIter {
390            inner: self.formats.iter(),
391        }
392    }
393
394    /// Return `true` if an equivalent to `value` exists in the set.
395    pub fn contains(&self, format: &Format) -> bool {
396        self.formats.contains(format)
397    }
398
399    /// Return an iterator over the values that are in both `self` and `other`.
400    pub fn intersection<'a>(&'a self, other: &'a FormatSet) -> FormatSetIntersection<'a> {
401        FormatSetIntersection {
402            inner: self.formats.intersection(&other.formats),
403        }
404    }
405
406    /// Get access to the underlying storage.
407    pub fn indexset(&self) -> &IndexSet<Format> {
408        &self.formats
409    }
410}
411
412/// A lazy iterator producing elements in the intersection of [`FormatSet`]s.
413#[derive(Debug)]
414pub struct FormatSetIntersection<'a> {
415    inner: indexmap::set::Intersection<'a, Format, std::collections::hash_map::RandomState>,
416}
417
418impl<'a> Iterator for FormatSetIntersection<'a> {
419    type Item = &'a Format;
420
421    #[inline]
422    fn next(&mut self) -> Option<Self::Item> {
423        self.inner.next()
424    }
425
426    #[inline]
427    fn size_hint(&self) -> (usize, Option<usize>) {
428        self.inner.size_hint()
429    }
430}
431
432impl IntoIterator for FormatSet {
433    type Item = Format;
434
435    type IntoIter = FormatSetIntoIter;
436
437    #[inline]
438    fn into_iter(self) -> Self::IntoIter {
439        FormatSetIntoIter {
440            inner: (*self.formats).clone().into_iter(),
441        }
442    }
443}
444
445impl FromIterator<Format> for FormatSet {
446    #[inline]
447    fn from_iter<T: IntoIterator<Item = Format>>(iter: T) -> Self {
448        Self {
449            formats: Arc::new(IndexSet::from_iter(iter)),
450        }
451    }
452}

494#[cfg(test)]
495mod tests {
496    use super::{_impl_formats, get_bpp, get_depth, get_opaque, get_transparent, has_alpha};
497
498    /// Tests that opaque alternatives are not the same as the variant with alpha.
499    #[test]
500    fn opaque_neq() {
501        for &format in _impl_formats() {
502            if let Some(opaque) = get_opaque(format) {
503                assert_ne!(
504                    format, opaque,
505                    "{}'s opaque alternative is the same format",
506                    format
507                );
508            }
509        }
510    }
511
512    /// Tests that opaque alternatives are cleanly converting back with get_transparent.
513    #[test]
514    fn opaque_inverse() {
515        for &format in _impl_formats() {
516            if let Some(opaque) = get_opaque(format) {
517                let transparent = get_transparent(opaque);
518                assert_eq!(
519                    Some(format),
520                    transparent,
521                    "{}'s opaque alternative {} doesn't cleanly convert back, got: {:?}",
522                    format,
523                    opaque,
524                    transparent
525                );
526            }
527        }
528    }
529
530    /// Tests that opaque alternatives for formats do not have opaque alternatives themselves.
531    ///
532    /// For example, `Argb8888` should have an opaque alternative of `Xrgb8888` and `Xrgb8888` should have no
533    /// opaque alternative.
534    #[test]
535    fn opaque_alternatives() {
536        for &format in _impl_formats() {
537            if let Some(opaque) = get_opaque(format) {
538                // If a format is considered to be an opaque alternative, the format should not have an opaque
539                // alternative.
540                let result = get_opaque(opaque);
541
542                assert!(
543                    result.is_none(),
544                    "Format {format} has an opaque alternative, {opaque}. However {opaque} reports an opaque alternative {opaque_opaque:?} which is incorrect",
545                    format = format,
546                    opaque = opaque,
547                    opaque_opaque = result,
548                );
549            }
550        }
551    }
552
553    /// Tests that a format and it's opaque alternative have the same number of bits per pixel.
554    #[test]
555    fn opaque_has_same_bpp() {
556        for &format in _impl_formats() {
557            if let Some(opaque) = get_opaque(format) {
558                let format_bpp = get_bpp(format);
559                let opaque_bpp = get_bpp(opaque);
560
561                assert_eq!(
562                    format_bpp,
563                    opaque_bpp,
564                    "Format {format} has a bpp of {format_bpp:?}. However the opaque alternative {opaque} has a different bpp of {opaque_bpp:?}",
565                    format_bpp = get_bpp(format),
566                    opaque = opaque,
567                    opaque_bpp = get_bpp(opaque),
568                );
569            }
570        }
571    }
572
573    /// A format with an opaque alternative should have alpha.
574    ///
575    /// The opaque alternative should not have alpha.
576    #[test]
577    fn format_with_opaque_has_alpha() {
578        for &format in _impl_formats() {
579            if let Some(opaque) = get_opaque(format) {
580                // Since the format has an opaque alternative, verify the opaque alternative does not have alpha.
581                assert!(
582                    has_alpha(format),
583                    "{} has an opaque alternative but does not state it has an alpha component",
584                    format
585                );
586
587                // The opaque alternative should not have alpha.
588                assert!(
589                    !has_alpha(opaque),
590                    "opaque alternative to {} ({}) has an alpha channel",
591                    format,
592                    opaque
593                );
594            }
595        }
596    }
597
598    // A format's depth should always be equal or small to it's bits-per-pixel
599    #[test]
600    fn format_bpp_greater_or_equal_than_depth() {
601        for &format in _impl_formats() {
602            assert!(
603                get_depth(format) <= get_bpp(format),
604                "{} has a depth higher than its bpp",
605                format
606            );
607        }
608    }
