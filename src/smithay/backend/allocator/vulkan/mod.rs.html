1//! Module for Buffers created using Vulkan.
2//!
3//! The [`VulkanAllocator`] type implements the [`Allocator`] trait and [`VulkanImage`] implements [`Buffer`].
4//! A [`VulkanImage`] may be exported as a [dmabuf](super::dmabuf).
5//!
6//! The Vulkan allocator supports up to Vulkan 1.3.
7//!
8//! The Vulkan allocator requires the following device extensions (and their dependencies):
9//! - `VK_EXT_image_drm_format_modifier`
10//! - `VK_EXT_external_memory_dmabuf`
11//! - `VK_KHR_external_memory_fd`
12//!
13//! Additionally the Vulkan allocator may enable the following extensions if available:
14//! - `VK_EXT_4444_formats`
15//!
16//! To get the required extensions a device must support to use the Vulkan allocator, use
17//! [`VulkanAllocator::required_extensions`].

50bitflags! {
51    /// Flags to indicate the intended usage for a buffer.
52    ///
53    /// The usage flags may influence whether a buffer with a specific format and modifier combination may
54    /// successfully be imported by a graphics api when it is exported as a [`Dmabuf`].
55    ///
56    /// For example, if a [`Buffer`] is used for scan-out, it is only necessary to specify the color
57    /// attachment usage. However, if the exported buffer is used by a client and intended to be rendered to,
58    /// then it is possible that the import will fail because the buffer was not allocated with right usage
59    /// for the format and modifier combination.
60    ///
61    /// The default usage set when creating a [`VulkanAllocator`] guarantees that an exported buffer may be
62    /// imported successfully with the same usage.
63    ///
64    /// If you need to allocate buffers with different usages dynamically, then you may use
65    /// [`VulkanAllocator::create_buffer_with_usage`].
66    ///
67    /// [`VulkanAllocator::is_format_supported`] can check if the combination of format, modifier and usage
68    /// flags are supported.
69    #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
70    pub struct ImageUsageFlags: vk::Flags {
71        /// The image may be the source of a transfer command.
72        ///
73        /// This allows the content of the exported buffer to be downloaded from the GPU.
74        const TRANSFER_SRC = vk::ImageUsageFlags::TRANSFER_SRC.as_raw();
75
76        /// The image may be the destination of a transfer command.
77        ///
78        /// This allows the content of the exported buffer to be modified by a memory upload to the GPU.
79        const TRANSFER_DST = vk::ImageUsageFlags::TRANSFER_DST.as_raw();
80
81        /// Image may be sampled in a shader.
82        ///
83        /// This should be used if the exported buffer will be used as a texture.
84        const SAMPLED = vk::ImageUsageFlags::SAMPLED.as_raw();
85
86        /// The image may be used in a color attachment.
87        ///
88        /// This should be used if the exported buffer will be rendered to.
89        const COLOR_ATTACHMENT = vk::ImageUsageFlags::COLOR_ATTACHMENT.as_raw();
90    }
91}

93/// Error type for [`VulkanAllocator`].
94#[derive(Debug, thiserror::Error)]
95pub enum Error {
96    /// The allocator could not be created.
97    ///
98    /// This can occur for a few reasons:
99    /// - No suitable queue family was found.
100    #[error("could not create allocator")]
101    Setup,
102
103    /// The size specified to create the buffer was too small.
104    ///
105    /// The size must be greater than `0 x 0` pixels.
106    #[error("invalid buffer size")]
107    InvalidSize,
108
109    /// The specified format is not supported.
110    ///
111    /// This error may occur for a few reasons:
112    /// 1. There is no equivalent Vulkan format for the specified format code.
113    /// 2. The driver does not support any of the specified modifiers for the format code.
114    /// 3. The driver does support some of the format, but the size of the buffer requested is too large.
115    /// 4. The usage is empty
116    #[error("format is not supported")]
117    UnsupportedFormat,
118
119    /// Some error from the Vulkan driver.
120    #[error(transparent)]
121    Vk(#[from] vk::Result),
122}

153impl VulkanAllocator {
154    /// Maximum supported version instance version that may be used with the allocator.
155    pub const MAX_INSTANCE_VERSION: Version = Version::VERSION_1_3;
156
157    /// Returns the list of device extensions required by the Vulkan allocator.
158    ///
159    /// This function may return a different list for each [`PhysicalDevice`], meaning each device should be
160    /// filtered using it's own call to this function.
161    pub fn required_extensions(phd: &PhysicalDevice) -> Vec<&'static CStr> {
162        // Always required extensions
163        let mut extensions = vec![
164            ext::image_drm_format_modifier::NAME,
165            ext::external_memory_dma_buf::NAME,
166            khr::external_memory_fd::NAME,
167        ];
168
169        if phd.api_version() < Version::VERSION_1_2 {
170            // VK_EXT_image_drm_format_modifier requires VK_KHR_image_format_list.
171            // VK_KHR_image_format_list is part of the core API in Vulkan 1.2
172            extensions.push(khr::image_format_list::NAME);
173        }
174
175        // Optional extensions:
176
177        // VK_EXT_4444_formats is part of the core API in Vulkan 1.3. Although not always supported
178        // (see the 1.3 features to enable)
179        if phd.api_version() < Version::VERSION_1_3 {
180            // In 1.2 and below, the device must support the extension to use it.
181            if phd.has_device_extension(ext::_4444_formats::NAME) {
182                extensions.push(ext::_4444_formats::NAME);
183            }
184        }
185
186        extensions
187    }
188
189    /// Creates a [`VulkanAllocator`].
190    ///
191    /// # Panics
192    ///
193    /// - If the version of instance which created the [`PhysicalDevice`] is higher than [`VulkanAllocator::MAX_INSTANCE_VERSION`].
194    /// - If the default [`ImageUsageFlags`] are empty.
195    #[instrument(err, skip(phd), fields(physical_device = phd.name()))]
196    pub fn new(phd: &PhysicalDevice, default_usage: ImageUsageFlags) -> Result<VulkanAllocator, Error> {
197        // Panic if the instance version is too high
198        if phd.instance().api_version() > Self::MAX_INSTANCE_VERSION {
199            panic!("Exceeded maximum instance api version for VulkanAllocator (1.3 max)")
200        }
201
202        // VUID-VkPhysicalDeviceImageFormatInfo2-usage-requiredbitmask
203        // At least one image usage flag must be specified.
204        if default_usage.is_empty() {
205            panic!("Default usage flags for allocator are empty")
206        }
207
208        // Get required extensions
209        let extensions = Self::required_extensions(phd);
210        let extension_pointers = extensions.iter().copied().map(CStr::as_ptr).collect::<Vec<_>>();
211
212        // We don't actually submit any commands to the queue, but Vulkan requires that we create devices with
213        // at least one queue (VUID-VkDeviceCreateInfo-queueCreateInfoCount-arraylength).
214        let queue_families = unsafe {
215            phd.instance()
216                .handle()
217                .get_physical_device_queue_family_properties(phd.handle())
218        };
219
220        let queue_family_index = queue_families
221            .iter()
222            // Find a queue with transfer
223            .position(|properties| properties.queue_flags.contains(vk::QueueFlags::TRANSFER))
224            // If there is no transfer queue family, then try graphics (which must support transfer)
225            .or_else(|| {
226                queue_families
227                    .iter()
228                    .position(|properties| properties.queue_flags.contains(vk::QueueFlags::GRAPHICS))
229            })
230            .ok_or(Error::Setup)?;
231
232        // TODO: Enable 4444 formats features
233        let queue_create_info = [vk::DeviceQueueCreateInfo::default()
234            .queue_family_index(queue_family_index as u32)
235            .queue_priorities(&[0.0])];
236        let create_info = vk::DeviceCreateInfo::default()
237            .enabled_extension_names(&extension_pointers)
238            .queue_create_infos(&queue_create_info);
239
240        let instance = phd.instance().handle();
241        let device = unsafe { instance.create_device(phd.handle(), &create_info, None) }?;
242
243        // Load extension functions
244        let extension_fns = ExtensionFns {
245            ext_image_format_modifier: ext::image_drm_format_modifier::Device::new(instance, &device),
246            khr_external_memory_fd: khr::external_memory_fd::Device::new(instance, &device),
247        };
248
249        let (dropped_sender, dropped_recv) = mpsc::channel();
250
251        #[cfg(feature = "backend_drm")]
252        let node = phd
253            .render_node()
254            .ok()
255            .flatten()
256            .or_else(|| phd.primary_node().ok().flatten());
257
258        let mut allocator = VulkanAllocator {
259            formats: Vec::new(),
260            images: Vec::new(),
261            default_usage,
262            remaining_allocations: phd.limits().max_memory_allocation_count,
263            extension_fns,
264            dropped_recv,
265            dropped_sender,
266            phd: phd.clone(),
267            #[cfg(feature = "backend_drm")]
268            node,
269            device: Arc::new(device),
270        };
271
272        allocator.init_formats();
273
274        Ok(allocator)
275    }

277    /// Returns whether this allocator supports the specified format with the usage flags.
278    pub fn is_format_supported(&self, format: DrmFormat, usage: ImageUsageFlags) -> bool {
279        // VUID-VkPhysicalDeviceImageFormatInfo2-usage-requiredbitmask
280        // At least one image usage flag must be specified.
281        if usage.is_empty() {
282            return false;
283        }
284
285        // TODO: Check if the extents are also valid?
286        // Vulkan states a maximum extent size for images.
287        // This may also be useful as a function on Allocator.
288        unsafe { self.get_format_info(format, vk::ImageUsageFlags::from_raw(usage.bits())) }
289            .ok()
290            .is_some()
291    }

293    /// Try to create a buffer with the given dimensions, pixel format and usage flags.
294    ///
295    /// This may return [`Err`] for one of the following reasons:
296    /// - The `usage` is empty.
297    /// - The `fourcc` format is not supported.
298    /// - All of the allowed `modifiers` are not supported.
299    /// - The size of the buffer is too large for the `usage`, `fourcc` format or `modifiers`.
300    /// - The `fourcc` format and `modifiers` do not support the specified usage.
301    #[instrument(level = "trace", err)]
302    #[profiling::function]
303    pub fn create_buffer_with_usage(
304        &mut self,
305        width: u32,
306        height: u32,
307        fourcc: DrmFourcc,
308        modifiers: &[DrmModifier],
309        usage: ImageUsageFlags,
310    ) -> Result<VulkanImage, Error> {
311        self.cleanup();
312
313        let vk_format = format::get_vk_format(fourcc).ok_or(Error::UnsupportedFormat)?;
314        let vk_usage = vk::ImageUsageFlags::from_raw(usage.bits());
315
316        // VUID-VkImageCreateInfo-extent-00944, VUID-VkImageCreateInfo-extent-00945
317        if width == 0 || height == 0 {
318            return Err(Error::InvalidSize);
319        }
320
321        // VUID-VkPhysicalDeviceImageFormatInfo2-usage-requiredbitmask
322        // At least one image usage flag must be specified.
323        if usage.is_empty() {
324            return Err(Error::UnsupportedFormat);
325        }
326
327        // VUID-VkImageCreateInfo-usage-00964, VUID-VkImageCreateInfo-usage-00965
328        if usage.contains(ImageUsageFlags::COLOR_ATTACHMENT) {
329            let limits = self.phd.limits();
330
331            if width > limits.max_framebuffer_width || height > limits.max_framebuffer_height {
332                return Err(Error::InvalidSize);
333            }
334        }
335
336        // Filter out any format + modifier combinations that are not supported
337        let modifiers = self.filter_modifiers(width, height, vk_usage, fourcc, modifiers);
338
339        // VUID-VkImageDrmFormatModifierListCreateInfoEXT-drmFormatModifierCount-arraylength
340        if modifiers.is_empty() {
341            return Err(Error::UnsupportedFormat);
342        }
343
344        Ok(unsafe { self.create_image(width, height, vk_format, vk_usage, fourcc, &modifiers[..]) }?)
345    }

349        &self.phd
350    }

427impl AsDmabuf for VulkanImage {
428    type Error = ExportError;
429
430    #[profiling::function]
431    fn export(&self) -> Result<Dmabuf, Self::Error> {
432        let device = self.device.upgrade().ok_or(ExportError::AllocatorDestroyed)?;
433
434        // Implementation may be broken if the plane count is wrong.
435        if self.format_plane_count == 0 {
436            return Err(ExportError::Failed);
437        }
438
439        assert!(
440            self.format_plane_count as usize <= MAX_PLANES,
441            "Vulkan implementation reported too many planes"
442        );
443
444        let create_info = vk::MemoryGetFdInfoKHR::default()
445            .handle_type(vk::ExternalMemoryHandleTypeFlags::DMA_BUF_EXT)
446            // VUID-VkMemoryGetFdInfoKHR-handleType-00671: Memory was allocated with DMA_BUF_EXT
447            .memory(self.inner.memory);
448
449        let fd = unsafe { self.khr_external_memory_fd.get_memory_fd(&create_info) }?;
450        // SAFETY: `vkGetMemoryFdKHR` creates a new file descriptor owned by the caller.
451        let fd = unsafe { OwnedFd::from_raw_fd(fd) };
452        let mut builder = Dmabuf::builder(
453            self.size(),
454            self.format().code,
455            self.format().modifier,
456            DmabufFlags::empty(),
457        );
458
459        for idx in 0..self.format_plane_count {
460            // get_image_subresource_layout only gets the layout of one memory plane. This mask specifies
461            // which plane should the layout be obtained for.
462            let aspect_mask = match idx {
463                0 => vk::ImageAspectFlags::MEMORY_PLANE_0_EXT,
464                1 => vk::ImageAspectFlags::MEMORY_PLANE_1_EXT,
465                2 => vk::ImageAspectFlags::MEMORY_PLANE_2_EXT,
466                3 => vk::ImageAspectFlags::MEMORY_PLANE_3_EXT,
467                _ => unreachable!(),
468            };
469
470            // VUID-vkGetImageSubresourceLayout-image-02270: All allocate images are created with drm tiling
471            let subresource = vk::ImageSubresource::default().aspect_mask(aspect_mask);
472            let layout = unsafe { device.get_image_subresource_layout(self.inner.image, subresource) };
473            builder.add_plane(
474                fd.try_clone().or(Err(ExportError::Failed))?,
475                idx,
476                layout.offset as u32,
477                layout.row_pitch as u32,
478            );
479        }
480
481        #[cfg(feature = "backend_drm")]
482        if let Some(node) = self.node {
483            builder.set_node(node);
484        }
485
486        Ok(builder.build().unwrap())
487    }
488}

539impl VulkanAllocator {
540    fn init_formats(&mut self) {
541        for &fourcc in format::known_formats() {
542            let vk_format = format::get_vk_format(fourcc).unwrap();
543            let modifier_properties = self
544                .phd
545                .get_format_modifier_properties(vk_format)
546                .expect("The Vulkan allocator requires VK_EXT_image_drm_format_modifier");
547
548            for modifier_properties in modifier_properties {
549                self.formats.push(FormatEntry {
550                    format: DrmFormat {
551                        code: fourcc,
552                        modifier: DrmModifier::from(modifier_properties.drm_format_modifier),
553                    },
554                    modifier_properties,
555                });
556            }
557        }
558    }
559
560    /// Returns whether the format + modifier combination and the usage flags are supported.
561    unsafe fn get_format_info(
562        &self,
563        format: DrmFormat,
564        usage: vk::ImageUsageFlags,
565    ) -> Result<Option<vk::ImageFormatProperties>, vk::Result> {
566        let vk_format = format::get_vk_format(format.code);
567
568        match vk_format {
569            Some(vk_format) => {
570                let mut image_drm_format_info = vk::PhysicalDeviceImageDrmFormatModifierInfoEXT::default()
571                    .drm_format_modifier(format.modifier.into())
572                    .sharing_mode(vk::SharingMode::EXCLUSIVE);
573                let format_info = vk::PhysicalDeviceImageFormatInfo2::default()
574                    .format(vk_format)
575                    .ty(vk::ImageType::TYPE_2D)
576                    .tiling(vk::ImageTiling::DRM_FORMAT_MODIFIER_EXT)
577                    .usage(usage)
578                    .flags(vk::ImageCreateFlags::empty())
579                    // VUID-VkPhysicalDeviceImageFormatInfo2-tiling-02249
580                    .push_next(&mut image_drm_format_info);
581                let mut image_format_properties = vk::ImageFormatProperties2::default();
582
583                // VUID-vkGetPhysicalDeviceImageFormatProperties-tiling-02248: Must use vkGetPhysicalDeviceImageFormatProperties2
584                let result = unsafe {
585                    self.phd
586                        .instance()
587                        .handle()
588                        .get_physical_device_image_format_properties2(
589                            self.phd.handle(),
590                            &format_info,
591                            &mut image_format_properties,
592                        )
593                };
594
595                result
596                    .map(|_| Some(image_format_properties.image_format_properties))
597                    .or_else(|result| {
598                        // Unsupported format + usage combination
599                        if result == vk::Result::ERROR_FORMAT_NOT_SUPPORTED {
600                            Ok(None)
601                        } else {
602                            Err(result)
603                        }
604                    })
605            }
606
607            None => Ok(None),
608        }
609    }
610
611    fn filter_modifiers(
612        &self,
613        width: u32,
614        height: u32,
615        vk_usage: vk::ImageUsageFlags,
616        fourcc: DrmFourcc,
617        modifiers: &[DrmModifier],
618    ) -> Vec<u64> {
619        modifiers
620            .iter()
621            .copied()
622            .filter_map(move |modifier| {
623                let info = unsafe {
624                    self.get_format_info(
625                        DrmFormat {
626                            code: fourcc,
627                            modifier,
628                        },
629                        vk_usage,
630                    )
631                }
632                .ok()
633                .flatten()?;
634
635                Some((modifier, info))
636            })
637            // Filter modifiers where the required image creation limits are not met
638            .filter(move |(_, properties)| {
639                let max_extent = properties.max_extent;
640
641                // VUID-VkImageCreateInfo-extent-02252
642                max_extent.width >= width
643                // VUID-VkImageCreateInfo-extent-02253
644                && max_extent.height >= height
645                // VUID-VkImageCreateInfo-extent-02254
646                // VUID-VkImageCreateInfo-extent-00946
647                // VUID-VkImageCreateInfo-imageType-00957
648                && max_extent.depth >= 1
649                // VUID-VkImageCreateInfo-samples-02258
650                && properties.sample_counts.contains(vk::SampleCountFlags::TYPE_1)
651            })
652            .map(|(modifier, _)| modifier)
653            .map(Into::<u64>::into)
654            // TODO: Could use a smallvec or tinyvec to reduce number allocations
655            .collect::<Vec<_>>()
656    }
657
658    /// # Safety
659    ///
660    /// * The list of modifiers must be supported for the given format and image usage flags.
661    /// * The extent of the image must be within the maximum extents Vulkan tells.
662    unsafe fn create_image(
663        &mut self,
664        width: u32,
665        height: u32,
666        vk_format: vk::Format,
667        vk_usage: vk::ImageUsageFlags,
668        fourcc: DrmFourcc,
669        modifiers: &[u64],
670    ) -> Result<VulkanImage, vk::Result> {
671        assert!(width > 0);
672        assert!(height > 0);
673
674        // Ensure maximum allocations are not exceeded.
675        if self.remaining_allocations == 0 {
676            todo!()
677        }
678
679        // Now that the list of valid modifiers is known, create an image using one of the modifiers.
680        let mut modifier_list =
681            vk::ImageDrmFormatModifierListCreateInfoEXT::default().drm_format_modifiers(modifiers);
682        let mut external_memory_image_create_info = vk::ExternalMemoryImageCreateInfo::default()
683            .handle_types(vk::ExternalMemoryHandleTypeFlags::DMA_BUF_EXT);
684
685        let image_create_info = vk::ImageCreateInfo::default()
686            .image_type(vk::ImageType::TYPE_2D)
687            .format(vk_format)
688            .extent(vk::Extent3D {
689                width,
690                height,
691                // VUID-VkImageCreateInfo-extent-00946
692                // VUID-VkImageCreateInfo-imageType-00957
693                depth: 1,
694            })
695            // VUID-VkImageCreateInfo-samples-parameter
696            .samples(vk::SampleCountFlags::TYPE_1)
697            // VUID-VkImageCreateInfo-mipLevels-00947
698            .mip_levels(1)
699            // VUID-VkImageCreateInfo-arrayLayers-00948
700            .array_layers(1)
701            // VUID-VkImageCreateInfo-pNext-02262
702            .tiling(vk::ImageTiling::DRM_FORMAT_MODIFIER_EXT)
703            // VUID-VkImageCreateInfo-usage-requiredbitmask
704            // FIXME: We don't assert any usage flags are set
705            .usage(vk_usage)
706            // VUID-VkImageCreateInfo-initialLayout-00993
707            .initial_layout(vk::ImageLayout::UNDEFINED)
708            // VUID-VkImageCreateInfo-tiling-02261
709            .push_next(&mut modifier_list)
710            // TODO: VUID-VkImageCreateInfo-pNext-00990
711            .push_next(&mut external_memory_image_create_info);
712
713        // The image is placed in a scope guard to safely handle future allocation failures.
714        let mut guard = scopeguard::guard(
715            ImageInner {
716                // This is the only spot where ? may be used to detect and error since no previous handles have been created.
717                image: unsafe { self.device.create_image(&image_create_info, None) }?,
718                memory: vk::DeviceMemory::null(),
719            },
720            |inner| unsafe { self.device.destroy_image(inner.image, None) },
721        );
722
723        // Get the modifier Vulkan created the image using.
724        let format = {
725            let mut image_modifier_properties = vk::ImageDrmFormatModifierPropertiesEXT::default();
726
727            unsafe {
728                self.extension_fns
729                    .ext_image_format_modifier
730                    .get_image_drm_format_modifier_properties(guard.image, &mut image_modifier_properties)
731            }?;
732
733            DrmFormat {
734                code: fourcc,
735                modifier: DrmModifier::from(image_modifier_properties.drm_format_modifier),
736            }
737        };
738
739        // Now that we know the plane count, get the number of planes for the format + modifier
740        let format_plane_count = self
741            .formats
742            .iter()
743            .find(|entry| entry.format == format)
744            .unwrap()
745            .modifier_properties
746            .drm_format_modifier_plane_count;
747
748        // Allocate image memory
749        let memory_reqs = unsafe { self.device.get_image_memory_requirements(guard.image) };
750        // TODO: Memory type index
751        let mut export_memory_allocate_info = vk::ExportMemoryAllocateInfo::default()
752            .handle_types(vk::ExternalMemoryHandleTypeFlags::DMA_BUF_EXT);
753        let alloc_create_info = vk::MemoryAllocateInfo::default()
754            .allocation_size(memory_reqs.size)
755            .push_next(&mut export_memory_allocate_info);
756
757        unsafe {
758            // Allocate memory for the image.
759            guard.memory = self.device.allocate_memory(&alloc_create_info, None)?;
760            // Finally bind the memory to the image
761            self.device.bind_image_memory(guard.image, guard.memory, 0)?;
762        }
763
764        // Initialization is complete, prevent the scope guard from running it's dropfn.
765        let inner = scopeguard::ScopeGuard::into_inner(guard);
766
767        // Track the image for destruction.
768        self.images.push(inner);
769
770        self.remaining_allocations -= 1;
771
772        Ok(VulkanImage {
773            inner,
774            width,
775            height,
776            format,
777            format_plane_count,
778            khr_external_memory_fd: self.extension_fns.khr_external_memory_fd.clone(),
779            dropped_sender: self.dropped_sender.clone(),
780            device: Arc::downgrade(&self.device),
781            #[cfg(feature = "backend_drm")]
782            node: self.node,
783        })
784    }
785
786    fn cleanup(&mut self) {
787        let dropped = self.dropped_recv.try_iter().collect::<Vec<_>>();
788
789        self.images.retain(|image| {
790            // Only drop if the
791            let drop = dropped.contains(image);
792
793            if drop {
794                // Destroy the underlying image resource
795                unsafe {
796                    self.device.destroy_image(image.image, None);
797                    self.device.free_memory(image.memory, None);
798                }
799
800                self.remaining_allocations = self
801                    .remaining_allocations
802                    .checked_add(1)
803                    .expect("Remaining allocations overflowed");
804                debug_assert!(
805                    self.phd.limits().max_memory_allocation_count >= self.remaining_allocations,
806                    "Too many allocations released",
807                );
808            }
809
810            // If the image was dropped, return false
811            !drop
812        })
813    }
