19/// Swapchain handling a fixed set of re-usable buffers e.g. for scan-out.
20///
21/// ## How am I supposed to use this?
22///
23/// To do proper buffer management, most compositors do so called double-buffering.
24/// Which means you use two buffers, one that is currently presented (the front buffer)
25/// and one that is currently rendered to (the back buffer). After each rendering operation
26/// you swap the buffers around, the old front buffer becomes the new back buffer, while
27/// the new front buffer is displayed to the user. This avoids showing the user rendering
28/// artifacts doing rendering.
29///
30/// There are also reasons to do triple-buffering, e.g. if you swap operation takes a
31/// unspecified amount of time. In that case you have one buffer, that is currently
32/// displayed, one that is done drawing and about to be swapped in and another one,
33/// which you can use to render currently.
34///
35/// Re-using and managing these buffers becomes increasingly complex the more buffers you
36/// introduce, which is where `Swapchain` comes into play.
37///
38/// `Swapchain` allocates buffers for you and transparently re-created them, e.g. when resizing.
39/// All you tell the swapchain is: *"Give me the next free buffer"* (by calling [`acquire`](Swapchain::acquire)).
40/// You then hold on to the returned buffer during rendering and swapping and free it once it is displayed.
41/// Efficient re-use of the buffers is done by the swapchain.
42///
43/// If you have associated resources for each buffer that can be reused (e.g. framebuffer `Handle`s for a `DrmDevice`),
44/// you can store then in the `Slot`s userdata field. If a buffer is re-used, its userdata is preserved for the next time
45/// it is returned by `acquire()`.
46pub struct Swapchain<A: Allocator> {
47    /// Allocator used by the swapchain
48    pub allocator: A,
49
50    width: u32,
51    height: u32,
52    fourcc: Fourcc,
53    modifiers: Vec<Modifier>,
54
55    slots: [Arc<InternalSlot<A::Buffer>>; SLOT_CAP],
56}
57
58impl<A: Allocator> fmt::Debug for Swapchain<A> {

69/// Slot of a swapchain containing an allocated buffer and its userdata.
70///
71/// The buffer is marked for re-use once all copies are dropped.
72/// Holding on to this struct will block the buffer in the swapchain.
73#[derive(Debug)]
74pub struct Slot<B: Buffer>(Arc<InternalSlot<B>>);
75
76#[derive(Debug)]
77struct InternalSlot<B: Buffer> {
78    buffer: Option<B>,
79    acquired: AtomicBool,
80    age: AtomicU8,
81    userdata: UserDataMap,
82}
83
84impl<B: Buffer> Slot<B> {
85    /// Retrieve userdata for this slot.
86    pub fn userdata(&self) -> &UserDataMap {
87        &self.0.userdata
88    }
89
90    /// Retrieve the age of the buffer
91    pub fn age(&self) -> u8 {
92        self.0.age.load(Ordering::SeqCst)
93    }
94}
95
96impl<B: Buffer> Default for InternalSlot<B> {
97    fn default() -> Self {
98        InternalSlot {
99            buffer: None,
100            acquired: AtomicBool::new(false),
101            age: AtomicU8::new(0),
102            userdata: UserDataMap::new(),
103        }
104    }
105}

107impl<B: Buffer> Deref for Slot<B> {
108    type Target = B;
109    fn deref(&self) -> &B {
110        Option::as_ref(&self.0.buffer).unwrap()
111    }
112}

114impl<B: Buffer + AsDmabuf> AsDmabuf for Slot<B> {
115    type Error = <B as AsDmabuf>::Error;
116
117    fn export(&self) -> Result<super::dmabuf::Dmabuf, Self::Error> {
118        let maybe_dmabuf = self.userdata().get::<Dmabuf>();
119        if maybe_dmabuf.is_none() {
120            let dmabuf = (**self).export()?;
121            self.userdata().insert_if_missing(|| dmabuf);
122        }
123
124        Ok(self.userdata().get::<Dmabuf>().cloned().unwrap())
125    }
126}

128impl<B: Buffer> Drop for Slot<B> {
129    fn drop(&mut self) {
130        self.0.acquired.store(false, Ordering::SeqCst);
131    }
132}

134impl<A> Swapchain<A>
135where
136    A: Allocator,
137{
138    /// Create a new swapchain with the desired allocator, dimensions and pixel format for the created buffers.
139    pub fn new(
140        allocator: A,
141        width: u32,
142        height: u32,
143        fourcc: Fourcc,
144        modifiers: Vec<Modifier>,
145    ) -> Swapchain<A> {
146        Swapchain {
147            allocator,
148            width,
149            height,
150            fourcc,
151            modifiers,
152            slots: Default::default(),
153        }
154    }
155
156    /// Acquire a new slot from the swapchain, if one is still free.
157    ///
158    /// The swapchain has an internal maximum of four re-usable buffers.
159    /// This function returns the first free one.
160    #[instrument(level = "trace", skip_all, err)]
161    #[profiling::function]
162    pub fn acquire(&mut self) -> Result<Option<Slot<A::Buffer>>, A::Error> {
163        if let Some(free_slot) = self
164            .slots
165            .iter_mut()
166            .find(|s| !s.acquired.swap(true, Ordering::SeqCst))
167        {
168            if free_slot.buffer.is_none() {
169                let free_slot = Arc::get_mut(free_slot).expect("Acquired was false, but Arc is not unique?");
170                match self
171                    .allocator
172                    .create_buffer(self.width, self.height, self.fourcc, &self.modifiers)
173                {
174                    Ok(buffer) => free_slot.buffer = Some(buffer),
175                    Err(err) => {
176                        free_slot.acquired.store(false, Ordering::SeqCst);
177                        return Err(err);
178                    }
179                }
180            }
181            assert!(free_slot.buffer.is_some());
182            return Ok(Some(Slot(free_slot.clone())));
183        }
184
185        // no free slots
186        Ok(None)
187    }
188
189    /// Mark a given buffer as submitted.
190    ///
191    /// This might effect internal data (e.g. buffer age) and may only be called,
192    /// the buffer may not be used for rendering anymore.
193    /// You may hold on to it, if you require keeping it alive.
194    ///
195    /// Buffers can always just be safely discarded by dropping them, but not
196    /// calling this function before may affect performance characteristics
197    /// (e.g. by not tracking the buffer age).
198    pub fn submitted(&mut self, slot: &Slot<A::Buffer>) {
199        // don't mess up the state, if the user submitted and old buffer, after e.g. a resize
200        if !self.slots.iter().any(|other| Arc::ptr_eq(&slot.0, other)) {
201            return;
202        }
203
204        slot.0.age.store(1, Ordering::SeqCst);
205        for other_slot in &mut self.slots {
206            if !Arc::ptr_eq(other_slot, &slot.0) && other_slot.buffer.is_some() {
207                let res = other_slot
208                    .age
209                    .fetch_update(Ordering::SeqCst, Ordering::SeqCst, |age| {
210                        if age > 0 {
211                            age.checked_add(1)
212                        } else {
213                            Some(0)
214                        }
215                    });
216                // If the age overflows the slot was not used for a long time. Lets clear it
217                if res.is_err() {
218                    *other_slot = Default::default();
219                }
220            }
221        }
222    }
223
224    /// Change the dimensions of newly returned buffers.
225    ///
226    /// Already obtained buffers are unaffected and will be cleaned up on drop.
227    pub fn resize(&mut self, width: u32, height: u32) {
228        if self.width == width && self.height == height {
229            return;
230        }
231
232        self.width = width;
233        self.height = height;
234        self.slots = Default::default();
235    }
236
237    /// Remove all internally cached buffers.
238    pub fn reset_buffers(&mut self) {
239        for slot in &mut self.slots {
240            *slot = Default::default();
241        }
242    }
243
244    /// Reset the age for each buffer.
245    ///
246    /// Resetting the buffer age will discard all damage information and force a
247    /// full redraw for the next frame.
248    pub fn reset_buffer_ages(&mut self) {
249        for slot in &mut self.slots {
250            match Arc::get_mut(slot) {
251                Some(slot) => slot.age = AtomicU8::new(0),
252                None => *slot = Default::default(),
253            }
254        }
255    }
256
257    /// Get set format
258    pub fn format(&self) -> Fourcc {
259        self.fourcc
260    }
261
262    /// Get allowed modifiers
263    pub fn modifiers(&self) -> &[Modifier] {
264        &self.modifiers
265    }
