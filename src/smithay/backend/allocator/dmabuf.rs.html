1//! Module for [dmabuf](https://docs.kernel.org/driver-api/dma-buf.html) buffers.
2//!
3//! `Dmabuf`s act alike to smart pointers and can be freely cloned and passed around.
4//! Once the last `Dmabuf` reference is dropped, its file descriptor is closed and
5//! underlying resources are freed.
6//!
7//! If you want to hold on to a potentially alive dmabuf without blocking the free up
8//! of the underlying resources, you may `downgrade` a `Dmabuf` reference to a `WeakDmabuf`.
9//!
10//! This can be especially useful in resources where other parts of the stack should decide upon
11//! the lifetime of the buffer. E.g. when you are only caching associated resources for a dmabuf.
12

152/// Builder for Dmabufs
153#[derive(Debug)]
154pub struct DmabufBuilder {
155    internal: DmabufInternal,
156}

158impl DmabufBuilder {
159    /// Add a plane to the constructed Dmabuf
160    ///
161    /// *Note*: Each Dmabuf needs at least one plane.
162    /// MAX_PLANES notes the maximum amount of planes any format may use with this implementation.
163    pub fn add_plane(&mut self, fd: OwnedFd, idx: u32, offset: u32, stride: u32) -> bool {
164        if self.internal.planes.len() == MAX_PLANES {
165            return false;
166        }
167        self.internal.planes.push(Plane {
168            fd,
169            plane_idx: idx,
170            offset,
171            stride,
172        });
173
174        true
175    }
176
177    /// Sets a known good node to import the dmabuf with.
178    ///
179    /// While this is only a strong hint and no guarantee, implementations
180    /// should avoid setting this at all, if they can't be reasonably certain.
181    #[cfg(feature = "backend_drm")]
182    pub fn set_node(&mut self, node: DrmNode) {
183        self.internal.node = Mutex::new(Some(node));
184    }
185
186    /// Build a `Dmabuf` out of the provided parameters and planes
187    ///
188    /// Returns `None` if the builder has no planes attached.
189    pub fn build(mut self) -> Option<Dmabuf> {
190        if self.internal.planes.is_empty() {
191            return None;
192        }
193
194        self.internal.planes.sort_by_key(|plane| plane.plane_idx);
195        Some(Dmabuf(Arc::new(self.internal)))
196    }
197}

295    /// Map the plane at specified index with the specified mode
296    ///
297    /// Returns `Err` if the plane with the specified index does not exist or
298    /// mmap failed
299    pub fn map_plane(
300        &self,
301        idx: usize,
302        mode: DmabufMappingMode,
303    ) -> Result<DmabufMapping, DmabufMappingFailed> {
304        let plane = self
305            .0
306            .planes
307            .get(idx)
308            .ok_or(DmabufMappingFailed::PlaneIndexOutOfBound)?;
309
310        let size = rustix::fs::seek(&plane.fd, rustix::fs::SeekFrom::End(0)).map_err(std::io::Error::from)?;
311        rustix::fs::seek(&plane.fd, rustix::fs::SeekFrom::Start(0)).map_err(std::io::Error::from)?;
312
313        let len = (size - plane.offset as u64) as usize;
314        let ptr = unsafe {
315            rustix::mm::mmap(
316                std::ptr::null_mut(),
317                len,
318                mode.into(),
319                rustix::mm::MapFlags::SHARED,
320                &plane.fd,
321                plane.offset as u64,
322            )
323        }
324        .map_err(std::io::Error::from)?;
325        Ok(DmabufMapping { len, ptr })
326    }
327
328    /// Synchronize access for the plane at the specified index
329    ///
330    /// Returns `Err` if the plane with the specified index does not exist or
331    /// the dmabuf_sync ioctl failed
332    pub fn sync_plane(&self, idx: usize, flags: DmabufSyncFlags) -> Result<(), DmabufSyncFailed> {
333        let plane = self
334            .0
335            .planes
336            .get(idx)
337            .ok_or(DmabufSyncFailed::PlaneIndexOutOfBound)?;
338        unsafe { rustix::ioctl::ioctl(&plane.fd, Setter::<DMA_BUF_SYNC, _>::new(dma_buf_sync { flags })) }
339            .map_err(std::io::Error::from)?;
340        Ok(())
341    }
342}

345    /// Modes of mapping a dmabuf plane
346    #[derive(Clone, Copy, Debug, PartialEq, Eq, Hash)]
347    pub struct DmabufMappingMode: u32 {
348        /// Map the dmabuf readable
349        const READ = 0b00000001;
350        /// Map the dmabuf writable
351        const WRITE = 0b00000010;
352    }
353}
354
355impl From<DmabufMappingMode> for rustix::mm::ProtFlags {
356    #[inline]
357    fn from(mode: DmabufMappingMode) -> Self {
358        let mut flags = rustix::mm::ProtFlags::empty();
359
360        if mode.contains(DmabufMappingMode::READ) {
361            flags |= rustix::mm::ProtFlags::READ;
362        }
363
364        if mode.contains(DmabufMappingMode::WRITE) {
365            flags |= rustix::mm::ProtFlags::WRITE;
366        }
367
368        flags
369    }
370}

551#[derive(Debug)]
552enum Subsource {
553    Active(Generic<PlaneRef, std::io::Error>),
554    Done(Generic<PlaneRef, std::io::Error>),
555    Empty,
556}
557
558impl Subsource {
559    fn done(&mut self) {
560        let mut this = Subsource::Empty;
561        std::mem::swap(self, &mut this);
562        match this {
563            Subsource::Done(source) | Subsource::Active(source) => {
564                *self = Subsource::Done(source);
565            }
566            _ => {}
567        }
568    }

571/// [`Dmabuf`]-based event source. Can be used to monitor implicit fences of a dmabuf.
572#[derive(Debug)]
573pub struct DmabufSource {
574    dmabuf: Dmabuf,
575    signal: Arc<AtomicBool>,
576    sources: [Subsource; 4],
577}
578
579#[derive(thiserror::Error, Debug, Clone, Copy, PartialEq, Eq, Hash)]
580#[error("Dmabuf is already ready for the given interest")]
581/// Dmabuf is already ready for the given interest
582pub struct AlreadyReady;
583
584impl DmabufSource {
585    /// Creates a new [`DmabufSource`] from a [`Dmabuf`] and interest.
586    ///
587    /// This event source will monitor the implicit fences of the given dmabuf.
588    /// Monitoring for READ-access will monitor the state of the most recent write or exclusive fence.
589    /// Monitoring for WRITE-access, will monitor state of all attached fences, shared and exclusive ones.
590    ///
591    /// The event source is a one shot event source and will remove itself from the event loop after being triggered once.
592    /// To monitor for new fences added at a later time a new DmabufSource needs to be created.
593    ///
594    /// Returns `AlreadyReady` if all corresponding fences are already signalled or if `interest` is empty.
595    #[profiling::function]
596    pub fn new(dmabuf: Dmabuf, interest: Interest) -> Result<Self, AlreadyReady> {
597        if !interest.readable && !interest.writable {
598            return Err(AlreadyReady);
599        }
600
601        let mut sources = [
602            Subsource::Empty,
603            Subsource::Empty,
604            Subsource::Empty,
605            Subsource::Empty,
606        ];
607        for (idx, handle) in dmabuf.handles().enumerate() {
608            if matches!(
609                rustix::event::poll(
610                    &mut [rustix::event::PollFd::new(
611                        &handle,
612                        if interest.writable {
613                            rustix::event::PollFlags::OUT
614                        } else {
615                            rustix::event::PollFlags::IN
616                        },
617                    )],
618                    Some(&rustix::time::Timespec {
619                        tv_sec: 0,
620                        tv_nsec: 0
621                    })
622                ),
623                Ok(1)
624            ) {
625                continue;
626            }
627            let fd = PlaneRef {
628                dmabuf: dmabuf.clone(),
629                idx,
630            };
631            sources[idx] = Subsource::Active(Generic::new(fd, interest, Mode::OneShot));
632        }
633        if sources
634            .iter()
635            .all(|x| matches!(x, Subsource::Done(_) | Subsource::Empty))
636        {
637            Err(AlreadyReady)
638        } else {
639            Ok(DmabufSource {
640                dmabuf,
641                sources,
642                signal: Arc::new(AtomicBool::new(false)),
643            })
644        }
645    }
646}

648impl EventSource for DmabufSource {
649    type Event = ();
650    type Metadata = Dmabuf;
651    type Ret = Result<(), std::io::Error>;
652
653    type Error = std::io::Error;
654
655    #[profiling::function]
656    fn process_events<F>(
657        &mut self,
658        readiness: calloop::Readiness,
659        token: calloop::Token,
660        mut callback: F,
661    ) -> Result<PostAction, Self::Error>
662    where
663        F: FnMut(Self::Event, &mut Self::Metadata) -> Self::Ret,
664    {
665        for i in 0..4 {
666            if let Ok(PostAction::Remove) = if let Subsource::Active(ref mut source) = &mut self.sources[i] {
667                // luckily Generic skips events for other tokens
668                source.process_events(readiness, token, |_, _| Ok(PostAction::Remove))
669            } else {
670                Ok(PostAction::Continue)
671            } {
672                self.sources[i].done();
673            }
674        }
675
676        if self
677            .sources
678            .iter()
679            .all(|x| matches!(x, Subsource::Done(_) | Subsource::Empty))
680        {
681            self.signal.store(true, Ordering::SeqCst);
682            callback((), &mut self.dmabuf)?;
683            Ok(PostAction::Remove)
684        } else {
685            Ok(PostAction::Reregister)
686        }
687    }
688
689    fn register(
690        &mut self,
691        poll: &mut calloop::Poll,
692        token_factory: &mut calloop::TokenFactory,
693    ) -> calloop::Result<()> {
694        for source in self.sources.iter_mut().filter_map(|source| match source {
695            Subsource::Active(ref mut source) => Some(source),
696            _ => None,
697        }) {
698            source.register(poll, token_factory)?;
699        }
700        Ok(())
701    }
702
703    fn reregister(
704        &mut self,
705        poll: &mut calloop::Poll,
706        token_factory: &mut calloop::TokenFactory,
707    ) -> calloop::Result<()> {
708        for source in self.sources.iter_mut() {
709            match source {
710                Subsource::Active(ref mut source) => source.reregister(poll, token_factory)?,
711                Subsource::Done(ref mut source) => {
712                    let _ = source.unregister(poll);
713                }
714                _ => {}
715            }
716        }
717        Ok(())
718    }
719
720    fn unregister(&mut self, poll: &mut calloop::Poll) -> calloop::Result<()> {
721        for source in self.sources.iter_mut() {
722            match source {
723                Subsource::Active(ref mut source) => source.unregister(poll)?,
724                Subsource::Done(ref mut source) => {
725                    let _ = source.unregister(poll);
726                }
727                _ => {}
728            }
729        }
730        Ok(())
731    }
