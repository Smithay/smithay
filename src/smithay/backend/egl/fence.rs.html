11#[derive(Debug)]
12struct InnerEGLFence {
13    display_handle: Arc<EGLDisplayHandle>,
14    handle: ffi::egl::types::EGLSync,
15    native: bool,
16}
17unsafe impl Send for InnerEGLFence {}
18unsafe impl Sync for InnerEGLFence {}
19
20impl Drop for InnerEGLFence {
21    fn drop(&mut self) {
22        unsafe {
23            // Not much we can do about when this fails
24            ffi::egl::DestroySync(**self.display_handle, self.handle);
25        }
26    }
27}
28
29/// A EGL fence
30#[derive(Debug, Clone)]
31pub struct EGLFence(Arc<InnerEGLFence>);
32
33impl EGLFence {
34    /// Returns whether the display supports creating fences from native fences
35    pub fn supports_importing(display: &EGLDisplay) -> bool {
36        display.extensions.has_native_fences
37    }
38
39    /// Import a native fence fd
40    #[profiling::function]
41    pub fn import(display: &EGLDisplay, native: OwnedFd) -> Result<Self, Error> {
42        // SAFETY: we do not have to test for EGL_KHR_fence_sync as EGL_ANDROID_native_fence_sync
43        // requires it already
44        if !display.extensions.has_native_fences {
45            return Err(Error::EglExtensionNotSupported(&[
46                "EGL_ANDROID_native_fence_sync",
47            ]));
48        }
49
50        let display_handle = display.get_display_handle();
51        let fd = native.into_raw_fd();
52        let attributes: [ffi::egl::types::EGLAttrib; 3] = [
53            ffi::egl::SYNC_NATIVE_FENCE_FD_ANDROID as ffi::egl::types::EGLAttrib,
54            fd as ffi::egl::types::EGLAttrib,
55            ffi::egl::NONE as ffi::egl::types::EGLAttrib,
56        ];
57
58        let res = wrap_egl_call_ptr(|| unsafe {
59            ffi::egl::CreateSync(
60                **display_handle,
61                ffi::egl::SYNC_NATIVE_FENCE_ANDROID,
62                attributes.as_ptr(),
63            )
64        })
65        .map_err(Error::CreationFailed);
66
67        // In case of an error we have to close the fd to not leak it
68        if res.is_err() {
69            unsafe {
70                rustix::io::close(fd);
71            }
72        }
73
74        let handle = res?;
75
76        Ok(Self(Arc::new(InnerEGLFence {
77            display_handle,
78            handle,
79            native: true,
80        })))
81    }
82
83    /// Create a new egl fence
84    ///
85    /// This will create a `EGL_SYNC_NATIVE_FENCE_ANDROID` or a `EGL_SYNC_FENCE_KHR` depending
86    /// on the availability of the `EGL_ANDROID_native_fence_sync` extension.
87    ///
88    /// Returns `Err` if the `EGL_KHR_fence_sync` is not present or creating the fence failed.
89    ///
90    /// Note: It is the callers responsibility to make sure the current bound client API supports
91    /// fences. For OpenglES the `GL_OES_EGL_sync` extension indicates support for fences.
92    #[profiling::function]
93    pub fn create(display: &EGLDisplay) -> Result<Self, Error> {
94        if !display.extensions.has_fences {
95            return Err(Error::EglExtensionNotSupported(&["EGL_KHR_fence_sync"]));
96        }
97
98        let (type_, native) = if display.extensions.has_native_fences {
99            (ffi::egl::SYNC_NATIVE_FENCE_ANDROID, true)
100        } else {
101            (ffi::egl::SYNC_FENCE, false)
102        };
103
104        let display_handle = display.get_display_handle();
105        let handle =
106            wrap_egl_call_ptr(|| unsafe { ffi::egl::CreateSync(**display_handle, type_, std::ptr::null()) })
107                .map_err(Error::CreationFailed)?;
108
109        Ok(Self(Arc::new(InnerEGLFence {
110            display_handle,
111            handle,
112            native,
113        })))
114    }
115
116    /// Indicates if this fence represents a native fence
117    pub fn is_native(&self) -> bool {
118        self.0.native
119    }
120
121    /// Export this [`EGLFence`] as a native fence fd
122    #[profiling::function]
123    pub fn export(&self) -> Result<OwnedFd, Error> {
124        if !self.0.native {
125            return Err(Error::EglExtensionNotSupported(&[
126                "EGL_ANDROID_native_fence_sync",
127            ]));
128        }
129
130        let fd = wrap_egl_call(
131            || unsafe { ffi::egl::DupNativeFenceFDANDROID(**self.0.display_handle, self.0.handle) },
132            ffi::egl::NO_NATIVE_FENCE_FD_ANDROID,
133        )
134        .map_err(Error::CreationFailed)?;
135        // SAFETY: eglDupNativeFenceFDANDROID generates EGL_BAD_PARAMETER in case of an error, so fd should always
136        // contain a valid fd
137        Ok(unsafe { OwnedFd::from_raw_fd(fd) })
138    }
139
140    /// Tries to insert the fence in the current bound client context
141    ///
142    /// Returns `Err` if the supplied display does not match the display
143    /// used at creation time of the fence.
144    #[profiling::function]
145    pub fn wait(&self, display: &EGLDisplay) -> Result<(), Error> {
146        if **display.get_display_handle() != **self.0.display_handle {
147            return Err(Error::DisplayNotSupported);
148        }
149
150        wrap_egl_call_bool(|| unsafe { ffi::egl::WaitSync(**self.0.display_handle, self.0.handle, 0) })
151            .map_err(Error::CreationFailed)?;
152        Ok(())
153    }
154
155    /// Blocks the current thread until the fence is signaled or the supplied
156    /// timeout is reached.
157    ///
158    /// If the timeout is reached `false` is returned
159    #[profiling::function]
160    pub fn client_wait(&self, timeout: Option<Duration>, flush: bool) -> Result<bool, Error> {
161        let timeout = timeout
162            .map(|t| t.as_nanos() as ffi::egl::types::EGLuint64KHR)
163            .unwrap_or(ffi::egl::FOREVER);
164        let flags = if flush {
165            ffi::egl::SYNC_FLUSH_COMMANDS_BIT as ffi::egl::types::EGLint
166        } else {
167            0
168        };
169        // SAFETY: eglClientWaitSyncKHR only defines two errors
170        // 1. If <sync> is not a valid sync object for <dpy>, EGL_FALSE is returned and an EGL_BAD_PARAMETER error is generated.
171        // 2. If <dpy> does not match the EGLDisplay passed to eglCreateSyncKHR when <sync> was created, the behaviour is undefined.
172        // Both should not be possible as we own both, the handle and the display the handle was created on
173        let status = wrap_egl_call(
174            || unsafe { ffi::egl::ClientWaitSync(**self.0.display_handle, self.0.handle, flags, timeout) },
175            ffi::egl::FALSE as ffi::egl::types::EGLint,
176        )
177        .map_err(Error::WaitFailed)?;
178
179        Ok(status == ffi::egl::CONDITION_SATISFIED as ffi::EGLint)
180    }
