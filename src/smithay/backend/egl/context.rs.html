35/// Defines the priority for an [`EGLContext`]
36///
37/// see: <https://registry.khronos.org/EGL/extensions/IMG/EGL_IMG_context_priority.txt>
38#[derive(Debug, Clone, Copy, PartialEq, Eq, Default)]
39pub enum ContextPriority {
40    /// High priority
41    ///
42    /// Note: This might require special system privileges like `CAP_SYS_NICE` to succeed.
43    High,
44    /// Medium priority
45    ///
46    /// Default if not specified
47    #[default]
48    Medium,
49    /// Low priority
50    Low,
51}
52
53impl TryFrom<ffi::egl::types::EGLenum> for ContextPriority {
54    type Error = EGLError;
55
56    #[inline]
57    fn try_from(value: ffi::egl::types::EGLenum) -> Result<Self, Self::Error> {
58        let priority = match value {
59            ffi::egl::CONTEXT_PRIORITY_HIGH_IMG => ContextPriority::High,
60            ffi::egl::CONTEXT_PRIORITY_MEDIUM_IMG => ContextPriority::Medium,
61            ffi::egl::CONTEXT_PRIORITY_LOW_IMG => ContextPriority::Low,
62            _ => return Err(EGLError::BadParameter),
63        };
64        Ok(priority)
65    }
66}
67
68impl From<ContextPriority> for ffi::egl::types::EGLenum {
69    #[inline]
70    fn from(value: ContextPriority) -> Self {
71        match value {
72            ContextPriority::High => ffi::egl::CONTEXT_PRIORITY_HIGH_IMG,
73            ContextPriority::Medium => ffi::egl::CONTEXT_PRIORITY_MEDIUM_IMG,
74            ContextPriority::Low => ffi::egl::CONTEXT_PRIORITY_LOW_IMG,
75        }
76    }
77}

79// SAFETY: A context can be sent to another thread when this context is not current.
80//
81// If the context is made current, the safety requirements for calling `EGLDisplay::make_current` must be
82// upheld. This means a context which is made current must be unbound (using `unbind`) in order to send this
83// context to another thread.

108            context,
109            display,
110            config_id,
111            pixel_format: Some(pixel_format),
112            user_data: Arc::new(UserDataMap::default()),
113            externally_managed: true,
114            span,
115        })

198    fn new_internal(
199        display: &EGLDisplay,
200        shared: Option<&EGLContext>,
201        config: Option<(GlAttributes, PixelFormatRequirements)>,
202        priority: Option<ContextPriority>,
203    ) -> Result<EGLContext, Error> {
204        let span = info_span!(parent: &display.span, "egl_context", ptr = tracing::field::Empty, shared = tracing::field::Empty);
205        let _guard = span.enter();
206
207        if let Some(shared) = shared {
208            span.record("shared", shared.context as usize);
209        }
210
211        let (pixel_format, config_id) = match config {
212            Some((attributes, reqs)) => {
213                let (format, config_id) = display.choose_config(attributes, reqs)?;
214                (Some(format), config_id)
215            }
216            None => {
217                if !display
218                    .extensions()
219                    .iter()
220                    .any(|x| x == "EGL_KHR_no_config_context")
221                    && !display
222                        .extensions()
223                        .iter()
224                        .any(|x| x == "EGL_MESA_configless_context")
225                    && !display
226                        .extensions()
227                        .iter()
228                        .any(|x| x == "EGL_KHR_surfaceless_context")
229                {
230                    return Err(Error::EglExtensionNotSupported(&[
231                        "EGL_KHR_no_config_context",
232                        "EGL_MESA_configless_context",
233                        "EGL_KHR_surfaceless_context",
234                    ]));
235                }
236                (None, ffi::egl::NO_CONFIG_KHR)
237            }
238        };
239
240        let mut context_attributes = Vec::with_capacity(12);
241
242        if let Some((attributes, _)) = config {
243            let version = attributes.version;
244
245            if display.get_egl_version() >= (1, 5)
246                || display.extensions().iter().any(|s| s == "EGL_KHR_create_context")
247            {
248                trace!("Setting CONTEXT_MAJOR_VERSION to {}", version.0);
249                context_attributes.push(ffi::egl::CONTEXT_MAJOR_VERSION as i32);
250                context_attributes.push(version.0 as i32);
251                trace!("Setting CONTEXT_MINOR_VERSION to {}", version.1);
252                context_attributes.push(ffi::egl::CONTEXT_MINOR_VERSION as i32);
253                context_attributes.push(version.1 as i32);
254
255                if attributes.debug && display.get_egl_version() >= (1, 5) {
256                    trace!("Setting CONTEXT_OPENGL_DEBUG to TRUE");
257                    context_attributes.push(ffi::egl::CONTEXT_OPENGL_DEBUG as i32);
258                    context_attributes.push(ffi::egl::TRUE as i32);
259                }
260
261                context_attributes.push(ffi::egl::CONTEXT_FLAGS_KHR as i32);
262                context_attributes.push(0);
263            } else if display.get_egl_version() >= (1, 3) {
264                trace!("Setting CONTEXT_CLIENT_VERSION to {}", version.0);
265                context_attributes.push(ffi::egl::CONTEXT_CLIENT_VERSION as i32);
266                context_attributes.push(version.0 as i32);
267            }
268        } else {
269            trace!("Setting CONTEXT_CLIENT_VERSION to 2");
270            context_attributes.push(ffi::egl::CONTEXT_CLIENT_VERSION as i32);
271            context_attributes.push(2);
272        }
273
274        let has_context_priority = display
275            .extensions()
276            .iter()
277            .any(|x| x == "EGL_IMG_context_priority");
278        if let Some(priority) = priority {
279            if !has_context_priority {
280                warn!(
281                    ?priority,
282                    "ignoring requested context priority, EGL_IMG_context_priority not supported"
283                );
284            } else {
285                context_attributes.push(ffi::egl::CONTEXT_PRIORITY_LEVEL_IMG as i32);
286                context_attributes.push(Into::<ffi::egl::types::EGLenum>::into(priority) as i32);
287            }
288        }
289
290        context_attributes.push(ffi::egl::NONE as i32);
291
292        trace!("Creating EGL context...");
293        let context = wrap_egl_call_ptr(|| unsafe {
294            ffi::egl::CreateContext(
295                **display.get_display_handle(),
296                config_id,
297                shared
298                    .map(|context| context.context)
299                    .unwrap_or(ffi::egl::NO_CONTEXT),
300                context_attributes.as_ptr(),
301            )
302        })
303        .map_err(Error::CreationFailed)?;
304        span.record("ptr", context as usize);
305
306        let context_priority = if has_context_priority {
307            let mut context_priority = 0;
308            let res = wrap_egl_call_bool(|| unsafe {
309                ffi::egl::QueryContext(
310                    **display.get_display_handle(),
311                    context,
312                    ffi::egl::CONTEXT_PRIORITY_LEVEL_IMG as ffi::egl::types::EGLint,
313                    &mut context_priority,
314                )
315            });
316
317            if res.is_ok() {
318                match ContextPriority::try_from(context_priority as ffi::egl::types::EGLenum) {
319                    Ok(context_priority) => Some(context_priority),
320                    Err(_) => {
321                        warn!(context_priority, "failed to parse context priority");
322                        None
323                    }
324                }
325            } else {
326                warn!("failed to query context priority");
327                None
328            }
329        } else {
330            None
331        };
332
333        if priority.is_some() && has_context_priority && priority != context_priority {
334            warn!(requested = ?priority, got = ?context_priority, "failed to set context priority");
335        }
336
337        info!(priority = ?context_priority, "EGL context created");
338
339        drop(_guard);
340        Ok(EGLContext {
341            context,
342            display: display.clone(),
343            config_id,
344            pixel_format,
345            user_data: if let Some(shared) = shared {
346                shared.user_data.clone()
347            } else {
348                Arc::new(UserDataMap::default())
349            },
350            externally_managed: false,
351            span,
352        })
353    }

355    /// Makes the OpenGL context the current context in the current thread with no surface bound.
356    ///
357    /// # Safety
358    ///
359    /// This function is marked unsafe, because the context cannot be made current on another thread without
360    /// being unbound again (see [`EGLContext::unbind`]).
361    #[instrument(level = "trace", skip_all, parent = &self.span, err)]
362    #[profiling::function]
363    pub unsafe fn make_current(&self) -> Result<(), MakeCurrentError> {
364        wrap_egl_call_bool(|| {
365            ffi::egl::MakeCurrent(
366                **self.display.get_display_handle(),
367                ffi::egl::NO_SURFACE,
368                ffi::egl::NO_SURFACE,
369                self.context,
370            )
371        })
372        .map(|_| ())
373        .map_err(Into::into)
374    }

388    /// Makes the OpenGL context the current context in the current thread with surfaces to
389    /// read/draw to.
390    ///
391    /// # Safety
392    ///
393    /// This function is marked unsafe, because the context cannot be made current on another thread without

395    #[instrument(level = "trace", skip_all, parent = &self.span, err)]
396    #[profiling::function]
397    pub unsafe fn make_current_with_draw_and_read_surface(
398        &self,
399        draw_surface: &EGLSurface,
400        read_surface: &EGLSurface,
401    ) -> Result<(), MakeCurrentError> {
402        let draw_surface_ptr = draw_surface.surface.load(Ordering::SeqCst);
403        let read_surface_ptr = read_surface.surface.load(Ordering::SeqCst);
404        wrap_egl_call_bool(|| {
405            ffi::egl::MakeCurrent(
406                **self.display.get_display_handle(),
407                draw_surface_ptr,
408                read_surface_ptr,
409                self.context,
410            )
411        })
412        .map(|_| ())
413        .map_err(Into::into)
414    }

439    /// Unbinds this context from the current thread, if set.
440    ///
441    /// This does nothing if this context is not the current context.
442    #[instrument(level = "trace", skip_all, parent = &self.span, err)]
443    #[profiling::function]
444    pub fn unbind(&self) -> Result<(), MakeCurrentError> {
445        if self.is_current() {
446            wrap_egl_call_bool(|| unsafe {
447                ffi::egl::MakeCurrent(
448                    **self.display.get_display_handle(),
449                    ffi::egl::NO_SURFACE,
450                    ffi::egl::NO_SURFACE,
451                    ffi::egl::NO_CONTEXT,
452                )
453            })?;
454        }
455        Ok(())
456    }

489impl Drop for EGLContext {
490    fn drop(&mut self) {
491        if !self.externally_managed {
492            let _guard = self.span.enter();
493            unsafe {
494                // We need to ensure the context is unbound, otherwise it egl stalls the destroy call
495                // ignore failures at this point
496                let _ = self.unbind();
497                ffi::egl::DestroyContext(**self.display.get_display_handle(), self.context);
498            }
499        }
500    }

531/// Describes how the backend should choose a pixel format.
532#[derive(Debug, Clone, Copy, PartialEq, Eq)]
533pub struct PixelFormatRequirements {
534    /// If `true`, only hardware-accelerated formats will be considered. If `false`, only software renderers.
535    /// `None` means "don't care". Default is `None`.
536    pub hardware_accelerated: Option<bool>,
537    /// Minimum number of bits for the color buffer, excluding alpha. `None` means "don't care". The default is `None`.
538    pub color_bits: Option<u8>,
539    /// If `true`, the color buffer must be in a floating point format. Default is `false`.
540    ///
541    /// Using floating points allows you to write values outside the `[0.0, 1.0]` range.
542    pub float_color_buffer: bool,
543    /// Minimum number of bits for the alpha in the color buffer. `None` means "don't care". The default is `None`.
544    pub alpha_bits: Option<u8>,
545    /// Minimum number of bits for the depth buffer. `None` means "don't care". The default value is `None`.
546    pub depth_bits: Option<u8>,
547    /// Minimum number of bits for the depth buffer. `None` means "don't care". The default value is `None`.
548    pub stencil_bits: Option<u8>,
549    /// Contains the minimum number of samples per pixel in the color, depth and stencil buffers.
550    /// `None` means "don't care". Default is `None`. A value of `Some(0)` indicates that multisampling must not be enabled.
551    pub multisampling: Option<u16>,
552}

595impl PixelFormatRequirements {
596    /// Append the requirements to the given attribute list
597    pub fn create_attributes(&self, out: &mut Vec<c_int>) {
598        if let Some(hardware_accelerated) = self.hardware_accelerated {
599            out.push(ffi::egl::CONFIG_CAVEAT as c_int);
600            out.push(if hardware_accelerated {
601                trace!("Setting CONFIG_CAVEAT to NONE");
602                ffi::egl::NONE as c_int
603            } else {
604                trace!("Setting CONFIG_CAVEAT to SLOW_CONFIG");
605                ffi::egl::SLOW_CONFIG as c_int
606            });
607        }
608
609        if let Some(color) = self.color_bits {
610            trace!("Setting RED_SIZE to {}", color / 3);
611            out.push(ffi::egl::RED_SIZE as c_int);
612            out.push((color / 3) as c_int);
613            trace!("Setting GREEN_SIZE to {}", color / 3 + u8::from(color % 3 != 0));
614            out.push(ffi::egl::GREEN_SIZE as c_int);
615            out.push((color / 3 + u8::from(color % 3 != 0)) as c_int);
616            trace!("Setting BLUE_SIZE to {}", color / 3 + u8::from(color % 3 == 2));
617            out.push(ffi::egl::BLUE_SIZE as c_int);
618            out.push((color / 3 + u8::from(color % 3 == 2)) as c_int);
619        }
620
621        if self.float_color_buffer {
622            out.push(ffi::egl::COLOR_COMPONENT_TYPE_EXT as c_int);
623            out.push(ffi::egl::COLOR_COMPONENT_TYPE_FLOAT_EXT as c_int);
624        }
625
626        if let Some(alpha) = self.alpha_bits {
627            trace!("Setting ALPHA_SIZE to {}", alpha);
628            out.push(ffi::egl::ALPHA_SIZE as c_int);
629            out.push(alpha as c_int);
630        }
631
632        if let Some(depth) = self.depth_bits {
633            trace!("Setting DEPTH_SIZE to {}", depth);
634            out.push(ffi::egl::DEPTH_SIZE as c_int);
635            out.push(depth as c_int);
636        }
637
638        if let Some(stencil) = self.stencil_bits {
639            trace!("Setting STENCIL_SIZE to {}", stencil);
640            out.push(ffi::egl::STENCIL_SIZE as c_int);
641            out.push(stencil as c_int);
642        }
643
644        if let Some(multisampling) = self.multisampling {
645            trace!("Setting SAMPLES to {}", multisampling);
646            out.push(ffi::egl::SAMPLES as c_int);
647            out.push(multisampling as c_int);
648        }
649    }
