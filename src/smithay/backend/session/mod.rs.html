1//! Abstraction of different session APIs.
2//!
3//! Sessions provide a way for multiple graphical systems to run in parallel by providing
4//! mechanisms to switch between and handle device access and permissions for every running
5//! instance. They are crucial to allow unprivileged processes to use graphical or input
6//! devices.
7//!
8//! The functions of the provided interfaces lies in two main components: the ability to
9//! open privileged devices, and be notified about when the session is paused and resumed
10//! (for example when the user switches to an other TTY, or is the computer goes to sleep).
11//!
12//! ## General use
13//!
14//! Session handling in Smithay is done through a pair of types that each session provider implements.
15//!
16//! The first is a handle implementing the [`Session`] trait, which allows you to request the opening
17//! of devices, a VT change, or information about the session state.
18//!
19//! The second is a notifier which informs you when the session is enabled or disabled by the system.
20//! This notifier takes the form of a [`calloop`] event source to deliver pause and activation events.
21//!
22//! ## Available providers
23//!
24//! This module provides just one session implementation, through [libseat](https://sr.ht/~kennylevinsen/seatd/),
25//! gated by the `backend_session_libseat` cargo feature.
26//!
27//! Other implementations can be provided out-of-tree.
28

38/// General session interface.
39///
40/// Provides a way to open and close devices and change the active vt.
41pub trait Session {
42    /// Error type of the implementation
43    type Error: AsErrno;
44
45    /// Opens a device at the given `path` with the given flags.
46    ///
47    /// Returns a raw file descriptor
48    fn open(&mut self, path: &Path, flags: OFlags) -> Result<OwnedFd, Self::Error>;
49    /// Close a previously opened file descriptor
50    fn close(&mut self, fd: OwnedFd) -> Result<(), Self::Error>;
51
52    /// Change the currently active virtual terminal
53    fn change_vt(&mut self, vt: i32) -> Result<(), Self::Error>;
54
55    /// Check if this session is currently active
56    fn is_active(&self) -> bool;
57    /// Which seat this session is on
58    fn seat(&self) -> String;
59}
60
61/// Events that can be generated by a session
62#[derive(Copy, Clone, Debug)]
63pub enum Event {
64    /// The whole session has been paused
65    ///
66    /// All devices should be considered as paused
67    PauseSession,
68    /// The whole session has been activated
69    ActivateSession,
70}

72impl Session for () {
73    type Error = ();
74
75    fn open(&mut self, _path: &Path, _flags: OFlags) -> Result<OwnedFd, Self::Error> {
76        Err(())
77    }
78    fn close(&mut self, _fd: OwnedFd) -> Result<(), Self::Error> {
79        Err(())
80    }
81
82    fn change_vt(&mut self, _vt: i32) -> Result<(), Self::Error> {
83        Err(())
84    }
85
86    fn is_active(&self) -> bool {
87        false
88    }
89    fn seat(&self) -> String {
90        String::from("seat0")
91    }
92}

94impl<S: Session> Session for Rc<RefCell<S>> {
95    type Error = S::Error;
96
97    fn open(&mut self, path: &Path, flags: OFlags) -> Result<OwnedFd, Self::Error> {
98        self.borrow_mut().open(path, flags)
99    }
100
101    fn close(&mut self, fd: OwnedFd) -> Result<(), Self::Error> {
102        self.borrow_mut().close(fd)
103    }
104
105    fn change_vt(&mut self, vt: i32) -> Result<(), Self::Error> {
106        self.borrow_mut().change_vt(vt)
107    }
108
109    fn is_active(&self) -> bool {
110        self.borrow().is_active()
111    }
112
113    fn seat(&self) -> String {
114        self.borrow().seat()
115    }
116}

118impl<S: Session> Session for Arc<Mutex<S>> {

120

122        self.lock().unwrap().open(path, flags)
123    }
124
125    fn close(&mut self, fd: OwnedFd) -> Result<(), Self::Error> {

133    fn is_active(&self) -> bool {
