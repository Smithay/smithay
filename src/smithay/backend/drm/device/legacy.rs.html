15#[derive(Debug)]
16pub struct LegacyDrmDevice {
17    pub(crate) fd: DrmDeviceFd,
18    pub(crate) active: Arc<AtomicBool>,
19    old_state: HashMap<crtc::Handle, (crtc::Info, Vec<connector::Handle>)>,
20    pub(super) span: tracing::Span,
21}
22
23impl LegacyDrmDevice {
24    pub fn new(fd: DrmDeviceFd, active: Arc<AtomicBool>, disable_connectors: bool) -> Result<Self, Error> {
25        let span = info_span!("drm_legacy");
26        let mut dev = LegacyDrmDevice {
27            fd,
28            active,
29            old_state: HashMap::new(),
30            span,
31        };
32        let _guard = dev.span.enter();
33
34        // Enumerate (and save) the current device state.
35        // We need to keep the previous device configuration to restore the state later,
36        // so we query everything, that we can set.
37        let res_handles = dev.fd.resource_handles().map_err(|source| {
38            Error::Access(AccessError {
39                errmsg: "Error loading drm resources",
40                dev: dev.fd.dev_path(),
41                source,
42            })
43        })?;
44        for &con in res_handles.connectors() {
45            let con_info = dev.fd.get_connector(con, false).map_err(|source| {
46                Error::Access(AccessError {
47                    errmsg: "Error loading connector info",
48                    dev: dev.fd.dev_path(),
49                    source,
50                })
51            })?;
52            if let Some(enc) = con_info.current_encoder() {
53                let enc_info = dev.fd.get_encoder(enc).map_err(|source| {
54                    Error::Access(AccessError {
55                        errmsg: "Error loading encoder info",
56                        dev: dev.fd.dev_path(),
57                        source,
58                    })
59                })?;
60                if let Some(crtc) = enc_info.crtc() {
61                    let info = dev.fd.get_crtc(crtc).map_err(|source| {
62                        Error::Access(AccessError {
63                            errmsg: "Error loading crtc info",
64                            dev: dev.fd.dev_path(),
65                            source,
66                        })
67                    })?;
68                    dev.old_state
69                        .entry(crtc)
70                        .or_insert((info, Vec::new()))
71                        .1
72                        .push(con);
73                }
74            }
75        }
76
77        // If the user does not explicitly requests us to skip this,
78        // we clear out the complete connector<->crtc mapping on device creation.
79        //
80        // The reason is, that certain operations may be racy otherwise, as surfaces can
81        // exist on different threads. As a result, we cannot enumerate the current state
82        // on surface creation (it might be changed on another thread during the enumeration).
83        // An easy workaround is to set a known state on device creation.
84        if disable_connectors {
85            debug!("Resetting drm device to known state");
86            dev.reset_state()?;
87        }
88
89        drop(_guard);
90        Ok(dev)
91    }
92
93    pub(super) fn reset_state(&self) -> Result<(), Error> {
94        let res_handles = self.fd.resource_handles().map_err(|source| {
95            Error::Access(AccessError {
96                errmsg: "Failed to query resource handles",
97                dev: self.fd.dev_path(),
98                source,
99            })
100        })?;
101        set_connector_state(&self.fd, res_handles.connectors().iter().copied(), false)?;
102
103        for crtc in res_handles.crtcs() {
104            #[allow(deprecated)]
105            let _ = self
106                .fd
107                .set_cursor(*crtc, Option::<&drm::control::dumbbuffer::DumbBuffer>::None);
108            // null commit (necessary to trigger removal on the kernel side with the legacy api.)
109            self.fd
110                .set_crtc(*crtc, None, (0, 0), &[], None)
111                .map_err(|source| {
112                    Error::Access(AccessError {
113                        errmsg: "Error setting crtc",
114                        dev: self.fd.dev_path(),
115                        source,
116                    })
117                })?;
118        }
119
120        Ok(())
121    }
122}

124impl Drop for LegacyDrmDevice {
125    fn drop(&mut self) {
126        if self.active.load(Ordering::SeqCst) {
127            let _guard = self.span.enter();
128
129            // Here we restore the tty to it's previous state.
130            // In case e.g. getty was running on the tty sets the correct framebuffer again,
131            // so that getty will be visible.
132            // We do exit correctly, if this fails, but the user will be presented with
133            // a black screen, if no display handler takes control again.
134
135            debug!("Device still active, trying to restore previous state");
136            for (handle, (info, connectors)) in self.old_state.drain() {
137                trace!(
138                    framebuffer = ?info.framebuffer(),
139                    offset = ?info.position(),
140                    ?connectors,
141                    mode = ?info.mode(),
142                    "Resetting crtc {:?}",
143                    handle,
144                );
145                if let Err(err) = self.fd.set_crtc(
146                    handle,
147                    info.framebuffer(),
148                    info.position(),
149                    &connectors,
150                    info.mode(),
151                ) {
152                    error!("Failed to reset crtc ({:?}). Error: {}", handle, err);
153                }
154            }
155        }
156    }
157}

159pub fn set_connector_state<D>(
160    dev: &D,
161    connectors: impl Iterator<Item = connector::Handle>,
162    enabled: bool,
163) -> Result<(), Error>
164where
165    D: DevPath + ControlDevice,
166{
167    // for every connector...
168    for conn in connectors {
169        let info = dev.get_connector(conn, false).map_err(|source| {
170            Error::Access(AccessError {
171                errmsg: "Failed to get connector infos",
172                dev: dev.dev_path(),
173                source,
174            })
175        })?;
176        // that is currently connected ...
177        if info.state() == connector::State::Connected {
178            // get a list of it's properties.
179            let props = dev.get_properties(conn).map_err(|source| {
180                Error::Access(AccessError {
181                    errmsg: "Failed to get properties for connector",
182                    dev: dev.dev_path(),
183                    source,
184                })
185            })?;
186            let (handles, _) = props.as_props_and_values();
187            // for every handle ...
188            for handle in handles {
189                // get information of that property
190                let info = dev.get_property(*handle).map_err(|source| {
191                    Error::Access(AccessError {
192                        errmsg: "Failed to get property of connector",
193                        dev: dev.dev_path(),
194                        source,
195                    })
196                })?;
197                // to find out, if we got the handle of the "DPMS" property ...
198                if info.name().to_str().map(|x| x == "DPMS").unwrap_or(false) {
199                    // so we can use that to turn on / off the connector
200                    trace!(connector = ?conn, "Setting DPMS {}", enabled);
201                    dev.set_property(
202                        conn,
203                        *handle,
204                        if enabled {
205                            drm_ffi::DRM_MODE_DPMS_ON.into()
206                        } else {
207                            drm_ffi::DRM_MODE_DPMS_OFF.into()
208                        },
209                    )
210                    .map_err(|source| {
211                        Error::Access(AccessError {
212                            errmsg: "Failed to set property of connector",
213                            dev: dev.dev_path(),
214                            source,
215                        })
216                    })?;
217                }
218            }
219        }
220    }
221    Ok(())
