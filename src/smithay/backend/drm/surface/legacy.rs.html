25impl State {
26    fn current_state<A: DevPath + ControlDevice>(fd: &A, crtc: crtc::Handle) -> Result<Self, Error> {
27        // Try to enumarate the current state to set the initial state variable correctly.
28        // We need an accurate state to handle `commit_pending`.
29        let crtc_info = fd.get_crtc(crtc).map_err(|source| {
30            Error::Access(AccessError {
31                errmsg: "Error loading crtc info",
32                dev: fd.dev_path(),
33                source,
34            })
35        })?;
36
37        let current_mode = crtc_info.mode();
38
39        let mut current_connectors = HashSet::new();
40        let res_handles = fd.resource_handles().map_err(|source| {
41            Error::Access(AccessError {
42                errmsg: "Error loading drm resources",
43                dev: fd.dev_path(),
44                source,
45            })
46        })?;
47        for &con in res_handles.connectors() {
48            let con_info = fd.get_connector(con, false).map_err(|source| {
49                Error::Access(AccessError {
50                    errmsg: "Error loading connector info",
51                    dev: fd.dev_path(),
52                    source,
53                })
54            })?;
55            if let Some(enc) = con_info.current_encoder() {
56                let enc_info = fd.get_encoder(enc).map_err(|source| {
57                    Error::Access(AccessError {
58                        errmsg: "Error loading encoder info",
59                        dev: fd.dev_path(),
60                        source,
61                    })
62                })?;
63                if let Some(current_crtc) = enc_info.crtc() {
64                    if crtc == current_crtc {
65                        current_connectors.insert(con);
66                    }
67                }
68            }
69        }
70
71        // If we have no current mode, we create a fake one, which will not match (and thus gets overriden on the commit below).
72        // A better fix would probably be making mode an `Option`, but that would mean
73        // we need to be sure, we require a mode to always be set without relying on the compiler.
74        // So we cheat, because it works and is easier to handle later.
75        Ok(State {
76            mode: current_mode.unwrap_or_else(|| unsafe { std::mem::zeroed() }),
77            connectors: current_connectors,
78        })
79    }
80}

94    pub fn new(
95        fd: Arc<DrmDeviceInternal>,
96        active: Arc<AtomicBool>,
97        crtc: crtc::Handle,
98        mode: Mode,
99        connectors: &[connector::Handle],
100    ) -> Result<Self, Error> {
101        let span = info_span!("drm_legacy", crtc = ?crtc);
102        let _guard = span.enter();
103        info!(?mode, ?connectors, ?crtc, "Initializing drm surface",);
104
105        let state = State::current_state(&*fd, crtc)?;
106        let pending = State {
107            mode,
108            connectors: connectors.iter().copied().collect(),
109        };
110
111        drop(_guard);
112        let surface = LegacyDrmSurface {
113            fd,
114            active,
115            crtc,
116            state: RwLock::new(state),
117            pending: RwLock::new(pending),
118            dpms: Mutex::new(true),
119            span,
120        };
121
122        Ok(surface)
123    }

142    pub fn add_connector(&self, conn: connector::Handle) -> Result<(), Error> {
143        if !self.active.load(Ordering::SeqCst) {
144            return Err(Error::DeviceInactive);
145        }
146
147        let mut pending = self.pending.write().unwrap();
148
149        if self.check_connector(conn, &pending.mode)? {
150            pending.connectors.insert(conn);
151        }
152
153        Ok(())
154    }

156    #[instrument(parent = &self.span, skip(self))]
157    pub fn remove_connector(&self, connector: connector::Handle) -> Result<(), Error> {
158        let mut pending = self.pending.write().unwrap();
159
160        if pending.connectors.contains(&connector) && pending.connectors.len() == 1 {
161            return Err(Error::SurfaceWithoutConnectors(self.crtc));
162        }
163
164        pending.connectors.remove(&connector);
165        Ok(())
166    }

168    #[instrument(parent = &self.span, skip(self))]
169    pub fn set_connectors(&self, connectors: &[connector::Handle]) -> Result<(), Error> {
170        if connectors.is_empty() {
171            return Err(Error::SurfaceWithoutConnectors(self.crtc));
172        }
173
174        if !self.active.load(Ordering::SeqCst) {
175            return Err(Error::DeviceInactive);
176        }
177
178        let mut pending = self.pending.write().unwrap();
179
180        if connectors
181            .iter()
182            .map(|conn| self.check_connector(*conn, &pending.mode))
183            .collect::<Result<Vec<bool>, _>>()?
184            .iter()
185            .all(|v| *v)
186        {
187            pending.connectors = connectors.iter().cloned().collect();
188        }
189
190        Ok(())
191    }

193    #[instrument(level = "debug", parent = &self.span, skip(self))]
194    pub fn use_mode(&self, mode: Mode) -> Result<(), Error> {
195        if !self.active.load(Ordering::SeqCst) {
196            return Err(Error::DeviceInactive);
197        }
198
199        let mut pending = self.pending.write().unwrap();
200
201        // check the connectors to see if this mode is supported
202        for connector in &pending.connectors {
203            if !self
204                .fd
205                .get_connector(*connector, false)
206                .map_err(|source| {
207                    Error::Access(AccessError {
208                        errmsg: "Error loading connector info",
209                        dev: self.fd.dev_path(),
210                        source,
211                    })
212                })?
213                .modes()
214                .contains(&mode)
215            {
216                return Err(Error::ModeNotSuitable(mode));
217            }
218        }
219
220        pending.mode = mode;
221
222        Ok(())
223    }

229    #[instrument(level = "trace", parent = &self.span, skip(self))]
230    #[profiling::function]
231    pub fn commit(&self, framebuffer: framebuffer::Handle, event: bool) -> Result<(), Error> {
232        if !self.active.load(Ordering::SeqCst) {
233            return Err(Error::DeviceInactive);
234        }
235
236        let mut current = self.state.write().unwrap();
237        let pending = self.pending.read().unwrap();
238        let mut dpms = self.dpms.lock().unwrap();
239
240        if !*dpms {
241            let connectors = current.connectors.intersection(&pending.connectors);
242            set_connector_state(&*self.fd, connectors.copied(), true)?;
243            *dpms = true;
244        }
245
246        {
247            let removed = current.connectors.difference(&pending.connectors);
248            let added = pending.connectors.difference(&current.connectors);
249
250            let mut conn_removed = false;
251            for conn in removed.clone() {
252                if let Ok(info) = self.fd.get_connector(*conn, false) {
253                    info!("Removing connector: {:?}", info.interface());
254                } else {
255                    info!("Removing unknown connector");
256                }
257                // if the connector was mapped to our crtc, we need to ack the disconnect.
258                // the graphics pipeline will not be freed otherwise
259                conn_removed = true;
260            }
261            set_connector_state(&*self.fd, removed.copied(), false)?;
262
263            if conn_removed {
264                // null commit (necessary to trigger removal on the kernel side with the legacy api.)
265                self.fd
266                    .set_crtc(self.crtc, None, (0, 0), &[], None)
267                    .map_err(|source| {
268                        Error::Access(AccessError {
269                            errmsg: "Error setting crtc",
270                            dev: self.fd.dev_path(),
271                            source,
272                        })
273                    })?;
274            }
275
276            for conn in added.clone() {
277                if let Ok(info) = self.fd.get_connector(*conn, false) {
278                    info!("Adding connector: {:?}", info.interface());
279                } else {
280                    info!("Adding unknown connector");
281                }
282            }
283            set_connector_state(&*self.fd, added.copied(), true)?;
284
285            if current.mode != pending.mode {
286                info!("Setting new mode: {:?}", pending.mode.name());
287            }
288        }
289
290        debug!("Setting screen");
291        // do a modeset and attach the given framebuffer
292        self.fd
293            .set_crtc(
294                self.crtc,
295                Some(framebuffer),
296                (0, 0),
297                &pending
298                    .connectors
299                    .iter()
300                    .copied()
301                    .collect::<Vec<connector::Handle>>(),
302                Some(pending.mode),
303            )
304            .map_err(|source| {
305                Error::Access(AccessError {
306                    errmsg: "Error setting crtc",
307                    dev: self.fd.dev_path(),
308                    source,
309                })
310            })?;
311
312        *current = pending.clone();
313
314        if event {
315            // set crtc does not trigger page_flip events, so we immediately queue a flip
316            // with the same framebuffer.
317            // this will result in wasting a frame, because this flip will need to wait
318            // for `set_crtc`, but is necessary to drive the event loop and thus provide
319            // a more consistent api.
320            ControlDevice::page_flip(&*self.fd, self.crtc, framebuffer, PageFlipFlags::EVENT, None).map_err(
321                |source| {
322                    Error::Access(AccessError {
323                        errmsg: "Failed to queue page flip",
324                        dev: self.fd.dev_path(),
325                        source,
326                    })
327                },
328            )?;
329        }
330
331        Ok(())
332    }

334    #[instrument(level = "trace", parent = &self.span, skip(self))]
335    #[profiling::function]
336    pub fn page_flip(&self, framebuffer: framebuffer::Handle, event: bool) -> Result<(), Error> {
337        trace!("Queueing Page flip");
338
339        if !self.active.load(Ordering::SeqCst) {
340            return Err(Error::DeviceInactive);
341        }
342
343        let current = self.state.read().unwrap();
344        let mut dpms = self.dpms.lock().unwrap();
345        if !*dpms {
346            set_connector_state(&*self.fd, current.connectors.iter().copied(), true)?;
347            *dpms = true;
348        }
349
350        ControlDevice::page_flip(
351            &*self.fd,
352            self.crtc,
353            framebuffer,
354            if event {
355                PageFlipFlags::EVENT
356            } else {
357                PageFlipFlags::empty()
358            },
359            None,
360        )
361        .map_err(|source| {
362            Error::Access(AccessError {
363                errmsg: "Failed to page flip",
364                dev: self.fd.dev_path(),
365                source,
366            })
367        })
368    }

401    // we use this function to verify, if a certain connector/mode combination
402    // is valid on our crtc. We do this with the most basic information we have:
403    // - is there a matching encoder
404    // - does the connector support the provided Mode.
405    //
406    // Better would be some kind of test commit to ask the driver,
407    // but that only exists for the atomic api.
408    fn check_connector(&self, conn: connector::Handle, mode: &Mode) -> Result<bool, Error> {
409        let info = self.fd.get_connector(conn, false).map_err(|source| {
410            Error::Access(AccessError {
411                errmsg: "Error loading connector info",
412                dev: self.fd.dev_path(),
413                source,
414            })
415        })?;
416
417        // check if the connector can handle the current mode
418        if info.modes().contains(mode) {
419            // check if there is a valid encoder
420            let encoders = info
421                .encoders()
422                .iter()
423                .map(|encoder| {
424                    self.fd.get_encoder(*encoder).map_err(|source| {
425                        Error::Access(AccessError {
426                            errmsg: "Error loading encoder info",
427                            dev: self.fd.dev_path(),
428                            source,
429                        })
430                    })
431                })
432                .collect::<Result<Vec<encoder::Info>, _>>()?;
433
434            // and if any encoder supports the selected crtc
435            let resource_handles = self.fd.resource_handles().map_err(|source| {
436                Error::Access(AccessError {
437                    errmsg: "Error loading resources",
438                    dev: self.fd.dev_path(),
439                    source,
440                })
441            })?;
442            if !encoders
443                .iter()
444                .map(|encoder| encoder.possible_crtcs())
445                .all(|crtc_list| resource_handles.filter_crtcs(crtc_list).contains(&self.crtc))
446            {
447                Ok(false)
448            } else {
449                Ok(true)
450            }
451        } else {
452            Ok(false)
453        }
454    }

472    pub fn clear(&self) -> Result<(), Error> {
473        let current = self.state.read().unwrap();
474        let mut dpms = self.dpms.lock().unwrap();

483impl Drop for LegacyDrmSurface {
484    fn drop(&mut self) {
485        let _guard = self.span.enter();
486        // ignore failure at this point
487
488        if !self.active.load(Ordering::SeqCst) {
489            // the device is gone or we are on another tty
490            // old state has been restored, we shouldn't touch it.
491            // if we are on another tty the connectors will get disabled
492            // by the device, when switching back
493            return;
494        }
495
496        // disable connectors again
497        let current = self.state.read().unwrap();
498        if set_connector_state(&*self.fd, current.connectors.iter().copied(), false).is_ok() {
499            // null commit
500            let _ = self.fd.set_crtc(self.crtc, None, (0, 0), &[], None);
501        }
502    }
