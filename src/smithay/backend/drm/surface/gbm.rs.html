31/// Simplified abstraction of a swapchain for gbm-buffers displayed on a [`DrmSurface`].
32#[derive(Debug)]
33pub struct GbmBufferedSurface<A: Allocator<Buffer = GbmBuffer> + 'static, U> {
34    current_fb: Slot<GbmBuffer>,
35    pending_fb: Option<(Slot<GbmBuffer>, U)>,

41    supports_fencing: bool,
42    span: tracing::Span,
43}

50    /// Create a new `GbmBufferedSurface` from a given compatible combination
51    /// of a surface, an allocator and renderer formats.
52    ///
53    /// The provided color_formats are tested in order until a working configuration is found.
54    ///
55    /// To successfully call this function, you need to have a renderer,
56    /// which can render into a Dmabuf, and a gbm allocator that can produce
57    /// buffers of a supported format for rendering.
58    pub fn new(
59        drm: DrmSurface,
60        mut allocator: A,
61        color_formats: &[Fourcc],
62        renderer_formats: impl IntoIterator<Item = Format>,
63    ) -> Result<GbmBufferedSurface<A, U>, Error<A::Error>> {
64        let span = info_span!(parent: drm.span(), "drm_gbm");
65        let _guard = span.enter();
66
67        let mut error = None;
68        let drm = Arc::new(drm);
69        let renderer_formats = renderer_formats.into_iter().collect::<Vec<_>>();
70
71        for format in color_formats {
72            debug!("Testing color format: {}", format);
73            match Self::new_internal(drm.clone(), allocator, renderer_formats.clone(), *format) {
74                Ok((current_fb, swapchain, is_opaque)) => {
75                    drop(_guard);
76                    let supports_fencing = !drm.is_legacy()
77                        && drm
78                            .get_driver_capability(DriverCapability::SyncObj)
79                            .map(|val| val != 0)
80                            .map_err(|err| {
81                                Error::DrmError(DrmError::Access(AccessError {
82                                    errmsg: "Failed to query driver capability",
83                                    dev: drm.dev_path(),
84                                    source: err,
85                                }))
86                            })?
87                        && plane_has_property(&*drm, drm.plane(), "IN_FENCE_FD")?;
88
89                    return Ok(GbmBufferedSurface {
90                        current_fb,
91                        pending_fb: None,
92                        queued_fb: None,
93                        next_fb: None,
94                        swapchain,
95                        drm,
96                        is_opaque,
97                        supports_fencing,
98                        span,
99                    });
100                }
101                Err((alloc, err)) => {
102                    warn!("Preferred format {} not available: {:?}", format, err);
103                    allocator = alloc;
104                    error = Some(err);
105                }
106            }
107        }
108        Err(error.unwrap())
109    }

111    #[allow(clippy::type_complexity)]
112    fn new_internal(
113        drm: Arc<DrmSurface>,
114        allocator: A,
115        mut renderer_formats: Vec<Format>,
116        code: Fourcc,
117    ) -> Result<(Slot<GbmBuffer>, Swapchain<A>, bool), (A, Error<A::Error>)> {
118        // select a format
119        let mut plane_formats = drm.plane_info().formats.iter().copied().collect::<IndexSet<_>>();
120        let opaque_code = get_opaque(code).unwrap_or(code);
121        if !plane_formats
122            .iter()
123            .any(|fmt| fmt.code == code || fmt.code == opaque_code)
124        {
125            return Err((allocator, Error::NoSupportedPlaneFormat));
126        }
127        plane_formats.retain(|fmt| fmt.code == code || fmt.code == opaque_code);
128        renderer_formats.retain(|fmt| fmt.code == code);
129
130        let plane_modifiers = plane_formats
131            .iter()
132            .map(|fmt| fmt.modifier)
133            .collect::<IndexSet<_>>();
134        let renderer_modifiers = renderer_formats
135            .iter()
136            .map(|fmt| fmt.modifier)
137            .collect::<IndexSet<_>>();
138
139        trace!("Plane formats: {:?}", plane_formats);
140        trace!("Renderer formats: {:?}", renderer_formats);
141        debug!(
142            "Remaining intersected modifiers: {:?}",
143            plane_modifiers
144                .intersection(&renderer_modifiers)
145                .collect::<IndexSet<_>>()
146        );
147
148        if plane_formats.is_empty() {
149            return Err((allocator, Error::NoSupportedPlaneFormat));
150        } else if renderer_formats.is_empty() {
151            return Err((allocator, Error::NoSupportedRendererFormat));
152        }
153
154        let formats = {
155            // Special case: if a format supports explicit LINEAR (but no implicit Modifiers)
156            // and the other doesn't support any modifier, force Implicit.
157            // This should at least result in a working pipeline possibly with a linear buffer,
158            // but we cannot be sure.
159            if (plane_formats.len() == 1
160                && plane_formats.iter().next().unwrap().modifier == Modifier::Invalid
161                && renderer_formats.iter().all(|x| x.modifier != Modifier::Invalid)
162                && renderer_formats.iter().any(|x| x.modifier == Modifier::Linear))
163                || (renderer_formats.len() == 1
164                    && renderer_formats.first().unwrap().modifier == Modifier::Invalid
165                    && plane_formats.iter().all(|x| x.modifier != Modifier::Invalid)
166                    && plane_formats.iter().any(|x| x.modifier == Modifier::Linear))
167            {
168                vec![Format {
169                    code,
170                    modifier: Modifier::Invalid,
171                }]
172            } else {
173                plane_modifiers
174                    .intersection(&renderer_modifiers)
175                    .cloned()
176                    .map(|modifier| Format { code, modifier })
177                    .collect::<Vec<_>>()
178            }
179        };
180        debug!("Testing Formats: {:?}", formats);
181
182        let modifiers = formats.iter().map(|x| x.modifier).collect::<Vec<_>>();
183        let mode = drm.pending_mode();
184
185        let mut swapchain: Swapchain<A> = Swapchain::new(
186            allocator,
187            mode.size().0 as u32,
188            mode.size().1 as u32,
189            code,
190            modifiers,
191        );
192
193        // Test format
194        let buffer = match swapchain.acquire() {
195            Ok(buffer) => buffer.unwrap(),
196            Err(err) => return Err((swapchain.allocator, Error::GbmError(err))),
197        };
198        let format = Format {
199            code,
200            modifier: buffer.modifier(), // no guarantee
201                                         // that this is stable across allocations, but
202                                         // we want to print that here for debugging proposes.
203                                         // It has no further use.
204        };
205
206        let use_opaque = !plane_formats.iter().any(|f| f.code == code);
207        let fb = match framebuffer_from_bo(drm.device_fd(), &buffer, use_opaque) {
208            Ok(fb) => fb,
209            Err(err) => return Err((swapchain.allocator, Error::DrmError(err.into()))),
210        };
211        match buffer.export() {
212            Ok(dmabuf) => dmabuf,
213            Err(err) => return Err((swapchain.allocator, err.into())),
214        };
215        buffer.userdata().insert_if_missing(|| fb);
216
217        let handle = buffer.userdata().get::<GbmFramebuffer>().unwrap();
218
219        let plane_state = PlaneState {
220            handle: drm.plane(),
221            config: Some(PlaneConfig {
222                src: Rectangle::from_size((mode.size().0 as i32, mode.size().1 as i32).into()).to_f64(),
223                dst: Rectangle::from_size((mode.size().0 as i32, mode.size().1 as i32).into()),
224                alpha: 1.0,
225                transform: Transform::Normal,
226                damage_clips: None,
227                fb: *handle.as_ref(),
228                fence: None,
229            }),
230        };
231
232        match drm.test_state([plane_state], true) {
233            Ok(_) => {
234                debug!("Choosen format: {:?}", format);
235                Ok((buffer, swapchain, use_opaque))
236            }
237            Err(err) => {
238                warn!(
239                    "Mode-setting failed with automatically selected buffer format {:?}: {}",
240                    format, err
241                );
242                Err((swapchain.allocator, err.into()))
243            }
244        }
245    }

247    /// Retrieves the next buffer to be rendered into and it's age.
248    ///
249    /// *Note*: This function can be called multiple times and
250    /// will return the same buffer until it is queued (see [`GbmBufferedSurface::queue_buffer`]).
251    #[instrument(level = "trace", skip_all, parent = &self.span, err)]
252    #[profiling::function]
253    pub fn next_buffer(&mut self) -> Result<(Dmabuf, u8), Error<A::Error>> {
254        if !self.drm.is_active() {
255            return Err(Error::<A::Error>::DrmError(DrmError::DeviceInactive));
256        }
257
258        if self.next_fb.is_none() {
259            let slot = self
260                .swapchain
261                .acquire()
262                .map_err(Error::GbmError)?
263                .ok_or(Error::NoFreeSlotsError)?;
264
265            let maybe_buffer = slot.userdata().get::<GbmFramebuffer>();
266            if maybe_buffer.is_none() {
267                let fb = framebuffer_from_bo(self.drm.device_fd(), &slot, self.is_opaque)
268                    .map_err(|err| Error::DrmError(err.into()))?;
269                slot.userdata().insert_if_missing(|| fb);
270            }
271
272            self.next_fb = Some(slot);
273        }
274
275        let slot = self.next_fb.as_ref().unwrap();
276        Ok((slot.export()?, slot.age()))
277    }

279    /// Queues the current buffer for rendering.
280    ///
281    /// Returns [`Error::NoBuffer`] in case [`GbmBufferedSurface::next_buffer`] has not been called
282    /// prior to this function.
283    ///
284    /// *Note*: This function needs to be followed up with [`GbmBufferedSurface::frame_submitted`]
285    /// when a vblank event is received, that denotes successful scanout of the buffer.
286    /// Otherwise the underlying swapchain will eventually run out of buffers.
287    ///
288    /// `user_data` can be used to attach some data to a specific buffer and later retrieved with [`GbmBufferedSurface::frame_submitted`]
289    #[profiling::function]
290    pub fn queue_buffer(
291        &mut self,
292        sync: Option<SyncPoint>,
293        damage: Option<Vec<Rectangle<i32, Physical>>>,
294        user_data: U,
295    ) -> Result<(), Error<A::Error>> {
296        if !self.drm.is_active() {
297            return Err(Error::<A::Error>::DrmError(DrmError::DeviceInactive));
298        }
299
300        let next_fb = self.next_fb.take().ok_or(Error::<A::Error>::NoBuffer)?;
301
302        self.swapchain.submitted(&next_fb);
303
304        self.queued_fb = Some(QueuedFb {
305            slot: next_fb,
306            sync,
307            damage,
308            user_data,
309        });
310        if self.pending_fb.is_none() {
311            self.submit()?;
312        }
313        Ok(())
314    }

316    /// Marks the current frame as submitted.
317    ///
318    /// *Note*: Needs to be called, after the vblank event of the matching [`DrmDevice`](super::super::DrmDevice)
319    /// was received after calling [`GbmBufferedSurface::queue_buffer`] on this surface.
320    /// Otherwise the underlying swapchain will run out of buffers eventually.
321    ///
322    /// Returns the user data that was stored with [`GbmBufferedSurface::queue_buffer`] if a buffer was pending, otherwise
323    /// `None` is returned.
324    #[profiling::function]
325    pub fn frame_submitted(&mut self) -> Result<Option<U>, Error<A::Error>> {
326        if let Some((mut pending, user_data)) = self.pending_fb.take() {
327            std::mem::swap(&mut pending, &mut self.current_fb);
328            if self.queued_fb.is_some() {
329                self.submit()?;
330            }
331            Ok(Some(user_data))
332        } else {
333            Ok(None)
334        }
335    }

337    #[profiling::function]
338    fn submit(&mut self) -> Result<(), Error<A::Error>> {
339        // yes it does not look like it, but both of these lines should be safe in all cases.
340        let QueuedFb {
341            slot,
342            sync,
343            damage,
344            user_data,
345        } = self.queued_fb.take().unwrap();
346        let handle = slot.userdata().get::<GbmFramebuffer>().unwrap();
347        let mode = self.drm.pending_mode();
348        let src = Rectangle::from_size((mode.size().0 as i32, mode.size().1 as i32).into()).to_f64();
349        let dst = Rectangle::from_size((mode.size().0 as i32, mode.size().1 as i32).into());
350
351        let damage_clips = damage.and_then(|damage| {
352            PlaneDamageClips::from_damage(self.drm.device_fd(), src, dst, damage)
353                .ok()
354                .flatten()
355        });
356
357        // Try to extract a native fence out of the supplied sync point if any
358        // If the sync point has no native fence or the surface does not support
359        // fencing force a wait
360        let fence = if let Some(sync) = sync {
361            if !self.supports_fencing {
362                // we probably don't want to fail to submit in this case, lets hope on implicit sync
363                let _ = sync.wait();
364                None
365            } else {
366                let fence = sync.export();
367
368                if fence.is_none() {
369                    let _ = sync.wait();
370                }
371
372                fence
373            }
374        } else {
375            None
376        };
377
378        let plane_state = PlaneState {
379            handle: self.plane(),
380            config: Some(PlaneConfig {
381                src,
382                dst,
383                transform: Transform::Normal,
384                alpha: 1.0,
385                damage_clips: damage_clips.as_ref().map(|d| d.blob()),
386                fb: *handle.as_ref(),
387                fence: fence.as_ref().map(|fence| fence.as_fd()),
388            }),
389        };
390
391        let flip = if self.drm.commit_pending() {
392            self.drm.commit([plane_state], true)
393        } else {
394            self.drm.page_flip([plane_state], true)
395        };
396        if flip.is_ok() {
397            self.pending_fb = Some((slot, user_data));
398        }
399        flip.map_err(Error::DrmError)
400    }

527/// Errors thrown by a [`GbmBufferedSurface`]
528#[derive(Debug, thiserror::Error)]
529pub enum Error<E: std::error::Error + Send + Sync + 'static> {
530    /// No supported pixel format for the given plane could be determined
531    #[error("No supported plane buffer format found")]
532    NoSupportedPlaneFormat,
533    /// No supported pixel format for the given renderer could be determined
534    #[error("No supported renderer buffer format found")]
535    NoSupportedRendererFormat,
536    /// The supported pixel formats of the renderer and plane are incompatible
537    #[error("Supported plane and renderer buffer formats are incompatible")]
538    FormatsNotCompatible,
539    /// The swapchain is exhausted, you need to call `frame_submitted`
540    #[error("Failed to allocate a new buffer")]
541    NoFreeSlotsError,
542    /// Failed to renderer using the given renderer
543    #[error("Failed to render test frame")]
544    InitialRenderingError,
545    /// Error accessing the drm device
546    #[error("The underlying drm surface encounted an error: {0}")]
547    DrmError(#[from] DrmError),
548    /// Error importing the rendered buffer to libgbm for scan-out
549    #[error("The underlying gbm device encounted an error: {0}")]
550    GbmError(#[source] E),
551    /// Error exporting as Dmabuf
552    #[error("The allocated buffer could not be exported as a dmabuf: {0}")]
553    AsDmabufError(#[from] GbmConvertError),
554    /// No buffer to queue
555    #[error("No buffer has been acquired to get queued")]
556    NoBuffer,
557}
558
559impl<E: std::error::Error + Send + Sync + 'static> From<Error<E>> for SwapBuffersError {
560    #[inline]
561    fn from(err: Error<E>) -> SwapBuffersError {
562        match err {
563            x @ Error::NoSupportedPlaneFormat
564            | x @ Error::NoSupportedRendererFormat
565            | x @ Error::FormatsNotCompatible
566            | x @ Error::InitialRenderingError => SwapBuffersError::ContextLost(Box::new(x)),
567            x @ Error::NoFreeSlotsError | x @ Error::NoBuffer => {
568                SwapBuffersError::TemporaryFailure(Box::new(x))
569            }
570            Error::DrmError(err) => err.into(),
571            Error::GbmError(err) => SwapBuffersError::ContextLost(Box::new(err)),
572            Error::AsDmabufError(err) => SwapBuffersError::ContextLost(Box::new(err)),
573        }
574    }
