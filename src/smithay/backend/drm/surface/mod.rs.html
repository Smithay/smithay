43impl Drop for PlaneDamageInner {
44    fn drop(&mut self) {
45        // There is nothing we can do if that fails

47            let _ = self.drm.destroy_property_blob(id);
48        }
49    }

65impl PlaneDamageClips {
66    /// Initialize damage clips for a a plane
67    #[profiling::function]
68    pub fn from_damage(
69        device: &DrmDeviceFd,
70        src: Rectangle<f64, Buffer>,
71        dst: Rectangle<i32, Physical>,
72        damage: impl IntoIterator<Item = Rectangle<i32, Physical>>,
73    ) -> io::Result<Option<Self>> {
74        let scale = src.size / dst.size.to_logical(1).to_buffer(1, Transform::Normal).to_f64();
75
76        let mut rects = damage
77            .into_iter()
78            .map(|rect| {
79                let mut rect = rect
80                    .to_f64()
81                    .to_logical(1f64)
82                    .to_buffer(
83                        1f64,
84                        Transform::Normal,
85                        &src.size.to_logical(1f64, Transform::Normal),
86                    )
87                    .upscale(scale);
88                rect.loc += src.loc;
89                let rect = rect.to_i32_up();
90
91                drm_ffi::drm_mode_rect {
92                    x1: rect.loc.x,
93                    y1: rect.loc.y,
94                    x2: rect.loc.x.saturating_add(rect.size.w),
95                    y2: rect.loc.y.saturating_add(rect.size.h),
96                }
97            })
98            .collect::<Vec<_>>();
99
100        if rects.is_empty() {
101            return Ok(None);
102        }
103
104        let data = unsafe {
105            std::slice::from_raw_parts_mut(
106                rects.as_mut_ptr() as *mut u8,
107                std::mem::size_of::<drm_ffi::drm_mode_rect>() * rects.len(),
108            )
109        };
110
111        let blob = drm_ffi::mode::create_property_blob(device.as_fd(), data)?;
112
113        Ok(Some(PlaneDamageClips {
114            inner: Arc::new(PlaneDamageInner {
115                drm: device.clone(),
116                blob: Some(drm::control::property::Value::Blob(blob.blob_id as u64)),
117            }),
118        }))
119    }
120}

368    /// Test a state given a set of framebuffers.
369    ///
370    /// *Note*: This will always return `Ok` for legacy devices if `allow_modeset = false`.
371    /// The legacy drm api has no way to test a buffer without triggering a modeset.
372    #[profiling::function]
373    pub fn test_state<'a>(
374        &self,
375        planes: impl IntoIterator<Item = PlaneState<'a>>,
376        allow_modeset: bool,
377    ) -> Result<(), Error> {
378        match &*self.internal {
379            DrmSurfaceInternal::Atomic(surf) => surf.test_state(planes, allow_modeset),
380            DrmSurfaceInternal::Legacy(surf) => {
381                let fb = ensure_legacy_planes(self, planes)?;
382
383                if allow_modeset {
384                    surf.test_buffer(fb, &self.pending_mode())
385                } else {
386                    // Legacy can not test a buffer without triggering a modeset, so we can
387                    // only assume it works and hope for the best. A later call to commit or
388                    // page_flip will show the correct result
389                    Ok(())
390                }
391            }
392        }
393    }

395    /// Commit the pending state rendering a given set of framebuffers.
396    ///
397    /// *Note*: This will trigger a full modeset on the underlying device,
398    /// potentially causing some flickering. Check before performing this
399    /// operation if a commit really is necessary using [`commit_pending`](DrmSurface::commit_pending).
400    ///
401    /// This operation is not necessarily blocking until the crtc is in the desired state,
402    /// but will trigger a `vblank` event once done.
403    /// Make sure to have the device registered in your event loop prior to invoking this, to not miss
404    /// any generated event.

407        &self,
408        planes: impl IntoIterator<Item = PlaneState<'a>>,
409        event: bool,
410    ) -> Result<(), Error> {
411        match &*self.internal {
412            DrmSurfaceInternal::Atomic(surf) => surf.commit(planes, event),
413            DrmSurfaceInternal::Legacy(surf) => {
414                let fb = ensure_legacy_planes(self, planes)?;
415                surf.commit(fb, event)
416            }
417        }
418    }

447    /// Claim a plane so that it won't be used by a different crtc
448    ///  
449    /// Returns `None` if the plane could not be claimed
450    pub fn claim_plane(&self, plane: plane::Handle) -> Option<PlaneClaim> {
451        // Validate that we are called with an plane that belongs to us
452        if self.planes.primary.iter().any(|p| p.handle == plane)
453            || self.planes.cursor.iter().any(|p| p.handle == plane)
454            || self.planes.overlay.iter().any(|p| p.handle == plane)
455        {
456            self.plane_claim_storage.claim(plane, self.crtc)
457        } else {
458            None
459        }
460    }

505fn ensure_legacy_planes<'a>(
506    dev: &(impl ControlDevice + DevPath),
507    planes: impl IntoIterator<Item = PlaneState<'a>>,
508) -> Result<framebuffer::Handle, Error> {
509    let state = planes.into_iter().next().ok_or(Error::NoPlane)?;
510
511    if plane_type(dev, state.handle)? != PlaneType::Primary {
512        return Err(Error::NonPrimaryPlane(state.handle));
513    }
514
515    let Some(config) = state.config else {
516        // we need a config on the primary plane
517        return Err(Error::NoFramebuffer(state.handle));
518    };
519
520    if config.dst.loc != Point::default() {
521        // legacy does not support crtc position (technically we could do it,
522        // but the position can only be changed by commit, not by page-flip,
523        // so we just not allow it)
524        return Err(Error::UnsupportedPlaneConfiguration(state.handle));
525    }
526
527    if config.src.loc != Point::default()
528        || config
529            .src
530            .size
531            .to_logical(1.0, Transform::Normal)
532            .to_physical(1.0)
533            != config.dst.size.to_f64()
534    {
535        // legacy does not support crop nor scale
536        return Err(Error::UnsupportedPlaneConfiguration(state.handle));
537    }
538
539    if config.transform != Transform::Normal {
540        // legacy does not support transform
541        return Err(Error::UnsupportedPlaneConfiguration(state.handle));
542    }
543
544    Ok(config.fb)
