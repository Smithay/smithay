8/// DRM access error
9#[derive(Debug, thiserror::Error)]
10#[error("DRM access error: {errmsg} on device `{dev:?}` ({source:})")]
11pub struct AccessError {
12    /// Error message associated to the access error
13    pub(crate) errmsg: &'static str,
14    /// Device on which the error was generated
15    pub(crate) dev: Option<PathBuf>,
16    /// Underlying device error
17    #[source]
18    pub source: io::Error,
19}

21/// Errors thrown by the [`DrmDevice`](crate::backend::drm::DrmDevice)
22/// and the [`DrmSurface`](crate::backend::drm::DrmSurface).
23#[derive(thiserror::Error, Debug)]
24pub enum Error {
25    /// Unable to acquire DRM master
26    #[error("Failed to aquire DRM master")]
27    DrmMasterFailed,
28    /// The `DrmDevice` encountered an access error
29    #[error(transparent)]
30    Access(#[from] AccessError),
31    /// Unable to determine device id of drm device
32    #[error("Unable to determine device id of drm device")]
33    UnableToGetDeviceId(#[source] rustix::io::Errno),
34    /// Device is currently paused
35    #[error("Device is currently paused, operation rejected")]
36    DeviceInactive,
37    /// Mode is not compatible with all given connectors
38    #[error("Mode `{0:?}` is not compatible with all given connectors")]
39    ModeNotSuitable(Mode),
40    /// The given crtc is already in use by another surface
41    #[error("Crtc `{0:?}` is already in use by another surface")]
42    CrtcAlreadyInUse(crtc::Handle),
43    /// This operation would result in a surface without connectors.
44    #[error("Surface of crtc `{0:?}` would have no connectors, which is not accepted")]
45    SurfaceWithoutConnectors(crtc::Handle),
46    /// The given plane cannot be used with the given crtc
47    #[error("Plane `{1:?}` is not compatible for use with crtc `{0:?}`")]
48    PlaneNotCompatible(crtc::Handle, plane::Handle),
49    /// The given configuration does not specify a plane which is not supported by the underlying implementation
50    #[error("No Plane has been specified which is not supported by the underlying implementation")]
51    NoPlane,
52    /// The given plane is not a primary plane and therefor not supported by the underlying implementation
53    #[error("Non-Primary Planes (provided was `{0:?}`) are not available for use with legacy devices")]
54    NonPrimaryPlane(plane::Handle),
55    /// The given plane does not allow to clear the framebuffer
56    #[error("Clearing the framebuffer on plane `{0:?}` is not supported")]
57    NoFramebuffer(plane::Handle),
58    /// The configuration is not supported on the given plane
59    #[error("The configuration is not supported on plane `{0:?}`")]
60    UnsupportedPlaneConfiguration(plane::Handle),
61    /// No encoder was found for a given connector on the set crtc
62    #[error("No encoder found for the given connector '{connector:?}' on crtc `{crtc:?}`")]
63    NoSuitableEncoder {
64        /// Connector
65        connector: connector::Handle,
66        /// CRTC
67        crtc: crtc::Handle,
68    },
69    /// Unknown connector handle
70    #[error("The connector ({0:?}) is unknown")]
71    UnknownConnector(connector::Handle),
72    /// Unknown crtc handle
73    #[error("The crtc ({0:?}) is unknown")]
74    UnknownCrtc(crtc::Handle),
75    /// Unknown crtc handle
76    #[error("The plane ({0:?}) is unknown")]
77    UnknownPlane(plane::Handle),
78    /// The DrmDevice is missing a required property
79    #[error("The DrmDevice is missing a required property '{name}' for handle ({handle:?})")]
80    UnknownProperty {
81        /// Property handle
82        handle: RawResourceHandle,
83        /// Property name
84        name: &'static str,
85    },
86    /// Atomic Test failed for new properties
87    #[error("Atomic Test failed for new properties on crtc ({0:?})")]
88    TestFailed(crtc::Handle),
89}

91impl From<Error> for SwapBuffersError {
92    #[inline]
93    fn from(err: Error) -> SwapBuffersError {
94        // FIXME: replace the special handling for EBUSY with ErrorKind::ResourceBusy once
95        // we reach MSRV >= 1.83
96        match err {
97            x @ Error::DeviceInactive => SwapBuffersError::TemporaryFailure(Box::new(x)),
98            Error::Access(AccessError {
99                errmsg, dev, source, ..
100            }) if matches!(
101                source.kind(),
102                ErrorKind::PermissionDenied | ErrorKind::WouldBlock | ErrorKind::Interrupted
103            ) || rustix::io::Errno::from_io_error(&source) == Some(rustix::io::Errno::BUSY) =>
104            {
105                SwapBuffersError::TemporaryFailure(Box::new(Error::Access(AccessError {
106                    errmsg,
107                    dev,
108                    source,
109                })))
110            }
111            x => SwapBuffersError::ContextLost(Box::new(x)),
112        }
113    }
114}
