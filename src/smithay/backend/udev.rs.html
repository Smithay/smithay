1//! `udev` related functionality for automated device scanning
2//!
3//! This module mainly provides the [`UdevBackend`], which monitors available DRM devices and acts as
4//! an event source to be inserted in [`calloop`], generating events whenever these devices change.
5//!
6//! *Note:* Once inserted into the event loop, the [`UdevBackend`] will only notify you about *changes*
7//! in the device list. To get an initial snapshot of the state during your initialization, you need to
8//! call its `device_list` method.
9//!
10//! ```no_run
11//! use smithay::backend::udev::{UdevBackend, UdevEvent};
12//!
13//! let udev = UdevBackend::new("seat0").expect("Failed to monitor udev.");
14//!
15//! for (dev_id, node_path) in udev.device_list() {
16//!     // process the initial list of devices
17//! }
18//!
19//! # let event_loop = smithay::reexports::calloop::EventLoop::<()>::try_new().unwrap();
20//! # let loop_handle = event_loop.handle();
21//! // setup the event source for long-term monitoring
22//! loop_handle.insert_source(udev, |event, _, _dispatch_data| match event {
23//!     UdevEvent::Added { device_id, path } => {
24//!         // a new device has been added
25//!     },
26//!     UdevEvent::Changed { device_id } => {
27//!         // a device has been changed
28//!     },
29//!     UdevEvent::Removed { device_id } => {
30//!         // a device has been removed
31//!     }
32//! }).expect("Failed to insert the udev source into the event loop");
33//! ```
34//!
35//! Additionally this contains some utility functions related to scanning.
36//!
37//! See also `anvil/src/udev.rs` for pure hardware backed example of a compositor utilizing this
38//! backend.

55/// Backend to monitor available drm devices.
56///
57/// Provides a way to automatically scan for available gpus and notifies the
58/// given handler of any changes. Can be used to provide hot-plug functionality for gpus and
59/// attached monitors.
60pub struct UdevBackend {
61    devices: HashMap<dev_t, PathBuf>,
62    monitor: MonitorSocket,
63    token: Option<Token>,
64    span: tracing::Span,
65}
66
67// MonitorSocket does not implement debug, so we have to impl Debug manually
68impl fmt::Debug for UdevBackend {
69    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
70        use udev::AsRaw;
71        f.debug_struct("UdevBackend")
72            .field("devices", &self.devices)
73            .field("monitor", &format!("MonitorSocket ({:?})", self.monitor.as_raw()))
74            .finish()
75    }
76}

78impl AsFd for UdevBackend {
79    fn as_fd(&self) -> BorrowedFd<'_> {
80        self.monitor.as_fd()
81    }
82}

84impl UdevBackend {
85    /// Creates a new [`UdevBackend`]
86    ///
87    /// ## Arguments
88    /// `seat`    - system seat which should be bound
89    pub fn new<S: AsRef<str>>(seat: S) -> io::Result<UdevBackend> {
90        let seat = seat.as_ref();
91        let span = debug_span!("backend_udev", seat = seat.to_string());
92        let _guard = span.enter();
93
94        let devices = all_gpus(seat)?
95            .into_iter()
96            // Create devices
97            .flat_map(|path| match stat(&path) {
98                Ok(stat) => Some((stat.st_rdev, path)),
99                Err(err) => {
100                    warn!("Unable to get id of {:?}, Error: {:?}. Skipping", path, err);
101                    None
102                }
103            })
104            .collect();
105
106        let monitor = MonitorBuilder::new()?.match_subsystem("drm")?.listen()?;
107
108        drop(_guard);
109        Ok(UdevBackend {
110            devices,
111            monitor,
112            token: None,
113            span,
114        })
115    }
116
117    /// Get a list of DRM devices currently known to the backend
118    ///
119    /// You should call this once before inserting the event source into your
120    /// event loop, to get an initial snapshot of the device state.
121    pub fn device_list(&self) -> impl Iterator<Item = (dev_t, &Path)> {
122        self.devices.iter().map(|(&id, path)| (id, path.as_ref()))
123    }
124}

126impl EventSource for UdevBackend {
127    type Event = UdevEvent;
128    type Metadata = ();
129    type Ret = ();
130    type Error = io::Error;
131
132    #[profiling::function]
133    fn process_events<F>(
134        &mut self,
135        _: Readiness,
136        token: Token,
137        mut callback: F,
138    ) -> std::io::Result<PostAction>
139    where
140        F: FnMut(UdevEvent, &mut ()),
141    {
142        if Some(token) != self.token {
143            return Ok(PostAction::Continue);
144        }
145
146        let _guard = self.span.enter();
147        for event in self.monitor.iter() {
148            debug!(
149                "Udev event: type={}, devnum={:?} devnode={:?}",
150                event.event_type(),
151                event.devnum(),
152                event.devnode()
153            );
154            match event.event_type() {
155                // New device
156                EventType::Add => {
157                    if let (Some(path), Some(devnum)) = (event.devnode(), event.devnum()) {
158                        info!("New device: #{} at {}", devnum, path.display());
159                        if self.devices.insert(devnum, path.to_path_buf()).is_none() {
160                            callback(
161                                UdevEvent::Added {
162                                    device_id: devnum,
163                                    path: path.to_path_buf(),
164                                },
165                                &mut (),
166                            );
167                        }
168                    }
169                }
170                // Device removed
171                EventType::Remove => {
172                    if let Some(devnum) = event.devnum() {
173                        info!("Device removed: #{}", devnum);
174                        if self.devices.remove(&devnum).is_some() {
175                            callback(UdevEvent::Removed { device_id: devnum }, &mut ());
176                        }
177                    }
178                }
179                // New connector
180                EventType::Change => {
181                    if let Some(devnum) = event.devnum() {
182                        info!("Device changed: #{}", devnum);
183                        if self.devices.contains_key(&devnum) {
184                            callback(UdevEvent::Changed { device_id: devnum }, &mut ());
185                        }
186                    }
187                }
188                _ => {}
189            }
190        }
191        Ok(PostAction::Continue)
192    }
193
194    fn register(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
195        self.token = Some(factory.token());
196        // Safety: the fd is owned by the UdevBackend and cannot be closed before it is removed from the event loop
197        unsafe { poll.register(self.as_fd(), Interest::READ, Mode::Level, self.token.unwrap()) }
198    }
199
200    fn reregister(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
201        self.token = Some(factory.token());
202        poll.reregister(self.as_fd(), Interest::READ, Mode::Level, self.token.unwrap())
203    }
204
205    fn unregister(&mut self, poll: &mut Poll) -> calloop::Result<()> {
206        self.token = None;
207        poll.unregister(self.as_fd())
208    }
209}

211/// Events generated by the [`UdevBackend`], notifying you of changes in system devices
212#[derive(Debug)]
213pub enum UdevEvent {
214    /// A new device has been detected
215    Added {
216        /// ID of the new device
217        device_id: dev_t,
218        /// Path of the new device
219        path: PathBuf,
220    },
221    /// A device has changed
222    Changed {
223        /// ID of the changed device
224        device_id: dev_t,
225    },
226    /// A device has been removed
227    Removed {
228        /// ID of the removed device
229        device_id: dev_t,
230    },
231}

233/// Returns the path of the primary GPU device if any
234///
235/// Might be used for filtering of [`UdevEvent::Added`] or for manual
236/// [`DrmDevice`](crate::backend::drm::DrmDevice) initialization.
237pub fn primary_gpu<S: AsRef<str>>(seat: S) -> io::Result<Option<PathBuf>> {
238    let mut enumerator = Enumerator::new()?;
239    enumerator.match_subsystem("drm")?;
240    enumerator.match_sysname("card[0-9]*")?;
241
242    if let Some(path) = enumerator
243        .scan_devices()?
244        .filter(|device| {
245            let seat_name = device
246                .property_value("ID_SEAT")
247                .map(|x| x.to_os_string())
248                .unwrap_or_else(|| OsString::from("seat0"));
249            if seat_name == *seat.as_ref() {
250                if let Ok(Some(pci)) = device.parent_with_subsystem(Path::new("pci")) {
251                    if let Some(id) = pci.attribute_value("boot_vga") {
252                        return id == "1";
253                    }
254                }
255            }
256            false
257        })
258        .flat_map(|device| device.devnode().map(PathBuf::from))
259        .next()
260    {
261        Ok(Some(path))
262    } else {
263        all_gpus(seat).map(|all| all.into_iter().next())
264    }
265}

267/// Returns the paths of all available GPU devices
268///
269/// Might be used for manual  [`DrmDevice`](crate::backend::drm::DrmDevice)
270/// initialization.
271pub fn all_gpus<S: AsRef<str>>(seat: S) -> io::Result<Vec<PathBuf>> {
272    let mut enumerator = Enumerator::new()?;
273    enumerator.match_subsystem("drm")?;
274    enumerator.match_sysname("card[0-9]*")?;
275    let mut gpus = enumerator
276        .scan_devices()?
277        .filter(|device| {
278            device
279                .property_value("ID_SEAT")
280                .map(|x| x.to_os_string())
281                .unwrap_or_else(|| OsString::from("seat0"))
282                == *seat.as_ref()
283        })
284        .flat_map(|device| device.devnode().map(PathBuf::from))
285        .collect::<Vec<_>>();
286    gpus.sort();
287    Ok(gpus)
288}

290/// Returns the loaded driver for a device named by it's [`dev_t`].
291pub fn driver(dev: dev_t) -> io::Result<Option<OsString>> {
292    let mut enumerator = Enumerator::new()?;
293    enumerator.match_subsystem("drm")?;
294    enumerator.match_sysname("card[0-9]*")?;
295    Ok(enumerator
296        .scan_devices()?
297        .filter(|device| device.devnum() == Some(dev))
298        .flat_map(|dev| {
299            let mut device = Some(dev);
300            while let Some(dev) = device {
301                if dev.driver().is_some() {
302                    return dev.driver().map(std::ffi::OsStr::to_os_string);
303                }
304                device = dev.parent();
305            }
306            None
307        })
308        .next())
