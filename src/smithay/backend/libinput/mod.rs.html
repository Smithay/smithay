25/// Libinput based [`InputBackend`].
26///
27/// Tracks input of all devices given manually or via a udev seat to a provided libinput
28/// context.
29#[derive(Debug)]
30pub struct LibinputInputBackend {
31    context: libinput::Libinput,
32    token: Option<Token>,
33    span: tracing::Span,
34}
35
36impl LibinputInputBackend {
37    /// Initialize a new [`LibinputInputBackend`] from a given already initialized
38    /// [libinput context](libinput::Libinput).
39    pub fn new(context: libinput::Libinput) -> Self {
40        let span = debug_span!("backend_libinput");
41        let _guard = span.enter();
42
43        info!("Initializing a libinput backend");
44
45        drop(_guard);
46        LibinputInputBackend {
47            context,
48            token: None,
49            span,
50        }
51    }
52
53    /// Returns a reference to the underlying libinput context
54    pub fn context(&self) -> &libinput::Libinput {
55        &self.context
56    }
57}

158/// Generic pointer scroll event from libinput
159#[derive(Debug, PartialEq, Eq, Hash)]
160pub enum PointerScrollAxis {
161    /// Scroll event from pointer wheel
162    Wheel(event::pointer::PointerScrollWheelEvent),
163    /// Scroll event from pointer finger
164    Finger(event::pointer::PointerScrollFingerEvent),
165    /// Continuous scroll event
166    Continuous(event::pointer::PointerScrollContinuousEvent),
167}
168
169impl PointerScrollAxis {
170    fn has_axis(&self, axis: event::pointer::Axis) -> bool {
171        use input::event::pointer::PointerScrollEvent;
172        match self {
173            Self::Wheel(evt) => evt.has_axis(axis),
174            Self::Finger(evt) => evt.has_axis(axis),
175            Self::Continuous(evt) => evt.has_axis(axis),
176        }
177    }
178}
179
180impl backend::Event<LibinputInputBackend> for PointerScrollAxis {
181    fn time(&self) -> u64 {
182        match self {
183            Self::Wheel(evt) => event::pointer::PointerEventTrait::time_usec(evt),
184            Self::Finger(evt) => event::pointer::PointerEventTrait::time_usec(evt),
185            Self::Continuous(evt) => event::pointer::PointerEventTrait::time_usec(evt),
186        }
187    }
188
189    fn device(&self) -> libinput::Device {
190        match self {
191            Self::Wheel(evt) => event::EventTrait::device(evt),
192            Self::Finger(evt) => event::EventTrait::device(evt),
193            Self::Continuous(evt) => event::EventTrait::device(evt),
194        }
195    }
196}

198impl backend::PointerAxisEvent<LibinputInputBackend> for PointerScrollAxis {
199    fn amount(&self, axis: Axis) -> Option<f64> {
200        use input::event::pointer::PointerScrollEvent;
201        let axis = axis.into();
202        if self.has_axis(axis) {
203            Some(match self {
204                Self::Wheel(evt) => evt.scroll_value(axis),
205                Self::Finger(evt) => evt.scroll_value(axis),
206                Self::Continuous(evt) => evt.scroll_value(axis),
207            })
208        } else {
209            None
210        }
211    }
212
213    fn amount_v120(&self, axis: Axis) -> Option<f64> {
214        let axis = axis.into();
215        if self.has_axis(axis) {
216            match self {
217                Self::Wheel(evt) => Some(evt.scroll_value_v120(axis)),
218                Self::Finger(_evt) => None,
219                Self::Continuous(_evt) => None,
220            }
221        } else {
222            None
223        }
224    }
225
226    fn source(&self) -> AxisSource {
227        match self {
228            Self::Wheel(_) => AxisSource::Wheel,
229            Self::Finger(_) => AxisSource::Finger,
230            Self::Continuous(_) => AxisSource::Continuous,
231        }
232    }
233
234    fn relative_direction(&self, _axis: Axis) -> backend::AxisRelativeDirection {
235        let device = backend::Event::<LibinputInputBackend>::device(self);
236        if device.config_scroll_natural_scroll_enabled() {
237            AxisRelativeDirection::Inverted
238        } else {
239            AxisRelativeDirection::Identical
240        }
241    }
242}

656    #[inline]
657    fn from(value: crate::input::keyboard::LedState) -> Self {
658        let mut leds = libinput::Led::empty();
659        if value.num.unwrap_or_default() {
660            leds |= libinput::Led::NUMLOCK;
661        }
662        if value.caps.unwrap_or_default() {
663            leds |= libinput::Led::CAPSLOCK;
664        }
665        if value.scroll.unwrap_or_default() {
666            leds |= libinput::Led::SCROLLLOCK;
667        }
668        leds
669    }

672/// Wrapper for types implementing the [`Session`] trait to provide
673/// a [`libinput::LibinputInterface`] implementation.
674#[cfg(feature = "backend_session")]
675#[derive(Debug)]
676pub struct LibinputSessionInterface<S: Session>(S);
677
678#[cfg(feature = "backend_session")]
679impl<S: Session> From<S> for LibinputSessionInterface<S> {
680    #[inline]
681    fn from(session: S) -> LibinputSessionInterface<S> {
682        LibinputSessionInterface(session)
683    }
684}
685
686#[cfg(feature = "backend_session")]
687impl<S: Session> libinput::LibinputInterface for LibinputSessionInterface<S> {
688    fn open_restricted(&mut self, path: &Path, flags: i32) -> Result<OwnedFd, i32> {
689        use rustix::fs::OFlags;
690        self.0
691            .open(path, OFlags::from_bits_truncate(flags as u32))
692            .map_err(|err| err.as_errno().unwrap_or(1 /*Use EPERM by default*/))
693    }
694
695    fn close_restricted(&mut self, fd: OwnedFd) {
696        let _ = self.0.close(fd);
697    }
698}

707impl EventSource for LibinputInputBackend {
708    type Event = InputEvent<LibinputInputBackend>;
709    type Metadata = ();
710    type Ret = ();
711    type Error = io::Error;
712
713    #[profiling::function]
714    fn process_events<F>(&mut self, _: Readiness, token: Token, mut callback: F) -> io::Result<PostAction>
715    where
716        F: FnMut(Self::Event, &mut ()) -> Self::Ret,
717    {
718        if Some(token) == self.token {
719            let _guard = self.span.enter();
720            self.context.dispatch()?;
721
722            for event in &mut self.context {
723                match event {
724                    libinput::Event::Device(device_event) => match device_event {
725                        event::DeviceEvent::Added(device_added_event) => {
726                            let added = event::EventTrait::device(&device_added_event);
727
728                            info!("New device {:?}", added.sysname(),);
729
730                            callback(InputEvent::DeviceAdded { device: added }, &mut ());
731                        }
732                        event::DeviceEvent::Removed(device_removed_event) => {
733                            let removed = event::EventTrait::device(&device_removed_event);
734
735                            info!("Removed device {:?}", removed.sysname(),);
736
737                            callback(InputEvent::DeviceRemoved { device: removed }, &mut ());
738                        }
739                        _ => {
740                            trace!("Unknown libinput device event");
741                        }
742                    },
743                    libinput::Event::Touch(touch_event) => match touch_event {
744                        event::TouchEvent::Down(down_event) => {
745                            callback(InputEvent::TouchDown { event: down_event }, &mut ());
746                        }
747                        event::TouchEvent::Motion(motion_event) => {
748                            callback(InputEvent::TouchMotion { event: motion_event }, &mut ());
749                        }
750                        event::TouchEvent::Up(up_event) => {
751                            callback(InputEvent::TouchUp { event: up_event }, &mut ());
752                        }
753                        event::TouchEvent::Cancel(cancel_event) => {
754                            callback(InputEvent::TouchCancel { event: cancel_event }, &mut ());
755                        }
756                        event::TouchEvent::Frame(frame_event) => {
757                            callback(InputEvent::TouchFrame { event: frame_event }, &mut ());
758                        }
759                        _ => {
760                            trace!("Unknown libinput touch event");
761                        }
762                    },
763                    libinput::Event::Keyboard(keyboard_event) => match keyboard_event {
764                        event::KeyboardEvent::Key(key_event) => {
765                            callback(InputEvent::Keyboard { event: key_event }, &mut ());
766                        }
767                        _ => {
768                            trace!("Unknown libinput keyboard event");
769                        }
770                    },
771                    libinput::Event::Pointer(pointer_event) => match pointer_event {
772                        event::PointerEvent::Motion(motion_event) => {
773                            callback(InputEvent::PointerMotion { event: motion_event }, &mut ());
774                        }
775                        event::PointerEvent::MotionAbsolute(motion_abs_event) => {
776                            callback(
777                                InputEvent::PointerMotionAbsolute {
778                                    event: motion_abs_event,
779                                },
780                                &mut (),
781                            );
782                        }
783                        event::PointerEvent::ScrollWheel(scroll_event) => {
784                            callback(
785                                InputEvent::PointerAxis {
786                                    event: PointerScrollAxis::Wheel(scroll_event),
787                                },
788                                &mut (),
789                            );
790                        }
791                        event::PointerEvent::ScrollFinger(scroll_event) => {
792                            callback(
793                                InputEvent::PointerAxis {
794                                    event: PointerScrollAxis::Finger(scroll_event),
795                                },
796                                &mut (),
797                            );
798                        }
799                        event::PointerEvent::ScrollContinuous(scroll_event) => {
800                            callback(
801                                InputEvent::PointerAxis {
802                                    event: PointerScrollAxis::Continuous(scroll_event),
803                                },
804                                &mut (),
805                            );
806                        }
807                        event::PointerEvent::Button(button_event) => {
808                            callback(InputEvent::PointerButton { event: button_event }, &mut ());
809                        }
810                        _ => {
811                            trace!("Unknown libinput pointer event");
812                        }
813                    },
814                    libinput::Event::Gesture(gesture_event) => match gesture_event {
815                        event::GestureEvent::Swipe(event::gesture::GestureSwipeEvent::Begin(event)) => {
816                            callback(InputEvent::GestureSwipeBegin { event }, &mut ());
817                        }
818                        event::GestureEvent::Swipe(event::gesture::GestureSwipeEvent::Update(event)) => {
819                            callback(InputEvent::GestureSwipeUpdate { event }, &mut ());
820                        }
821                        event::GestureEvent::Swipe(event::gesture::GestureSwipeEvent::End(event)) => {
822                            callback(InputEvent::GestureSwipeEnd { event }, &mut ());
823                        }
824                        event::GestureEvent::Pinch(event::gesture::GesturePinchEvent::Begin(event)) => {
825                            callback(InputEvent::GesturePinchBegin { event }, &mut ());
826                        }
827                        event::GestureEvent::Pinch(event::gesture::GesturePinchEvent::Update(event)) => {
828                            callback(InputEvent::GesturePinchUpdate { event }, &mut ());
829                        }
830                        event::GestureEvent::Pinch(event::gesture::GesturePinchEvent::End(event)) => {
831                            callback(InputEvent::GesturePinchEnd { event }, &mut ());
832                        }
833                        event::GestureEvent::Hold(event::gesture::GestureHoldEvent::Begin(event)) => {
834                            callback(InputEvent::GestureHoldBegin { event }, &mut ());
835                        }
836                        event::GestureEvent::Hold(event::gesture::GestureHoldEvent::End(event)) => {
837                            callback(InputEvent::GestureHoldEnd { event }, &mut ());
838                        }
839                        _ => {
840                            trace!("Unknown libinput gesture event");
841                        }
842                    },
843                    libinput::Event::Tablet(tablet_event) => match tablet_event {
844                        event::TabletToolEvent::Axis(event) => {
845                            callback(InputEvent::TabletToolAxis { event }, &mut ());
846                        }
847                        event::TabletToolEvent::Proximity(event) => {
848                            callback(InputEvent::TabletToolProximity { event }, &mut ());
849                        }
850                        event::TabletToolEvent::Tip(event) => {
851                            callback(InputEvent::TabletToolTip { event }, &mut ());
852                        }
853                        event::TabletToolEvent::Button(event) => {
854                            callback(InputEvent::TabletToolButton { event }, &mut ());
855                        }
856                        _ => {
857                            trace!("Unknown libinput tablet event");
858                        }
859                    },
860                    libinput::Event::Switch(switch_event) => match switch_event {
861                        event::SwitchEvent::Toggle(event) => {
862                            callback(InputEvent::SwitchToggle { event }, &mut ());
863                        }
864                        _ => {
865                            trace!("Unknown libinput switch event");
866                        }
867                    },
868                    _ => {} //FIXME: What to do with the rest.
869                }
870            }
871        }
872
873        Ok(PostAction::Continue)
874    }
875
876    fn register(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
877        self.token = Some(factory.token());
878        // Safety: the FD cannot be closed without removing the LibinputInputBackend from the event loop
879        unsafe { poll.register(self.as_fd(), Interest::READ, Mode::Level, self.token.unwrap()) }
880    }
881
882    fn reregister(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
883        self.token = Some(factory.token());
884        poll.reregister(self.as_fd(), Interest::READ, Mode::Level, self.token.unwrap())
885    }
886
887    fn unregister(&mut self, poll: &mut Poll) -> calloop::Result<()> {
888        self.token = None;
889        poll.unregister(self.as_fd())
890    }
