10impl super::PhysicalDevice {
11    /// # Safety:
12    ///
13    /// The physical device must belong to the specified instance.
14    pub(super) unsafe fn from_phd(
15        instance: &super::Instance,
16        phd: vk::PhysicalDevice,
17    ) -> VkResult<Option<super::PhysicalDevice>> {
18        let instance = instance.clone();
19
20        let extensions = unsafe { instance.handle().enumerate_device_extension_properties(phd) }?;
21        let extensions = extensions
22            .iter()
23            .map(|extension| {
24                // SAFETY: Vulkan guarantees the device name is valid UTF-8 with a null terminator.
25                unsafe { CStr::from_ptr(&extension.extension_name as *const _) }.to_owned()
26            })
27            .collect::<Vec<_>>();
28
29        if let Some(info) =
30            unsafe { PhdInfo::from_phd(instance.handle(), instance.api_version(), phd, &extensions) }
31        {
32            let span = info_span!(parent: &instance.0.span, "backend_vulkan_device", name = info.name);
33            Ok(Some(Self {
34                phd,
35                info,
36                extensions,
37                instance,
38                span,
39            }))
40        } else {
41            Ok(None)
42        }
43    }
44}

46impl super::PhdInfo {
47    /// Returns [`None`] if the physical device does not support Vulkan 1.1
48    ///
49    /// # Panics
50    ///
51    /// - If the instance version is not at least Vulkan 1.1
52    ///
53    /// # Safety
54    ///
55    /// - The instance version must be the same version the instance was created with.
56    /// - The physical device must belong to the specified instance.
57    unsafe fn from_phd(
58        instance: &ash::Instance,
59        instance_version: Version,
60        phd: vk::PhysicalDevice,
61        supported_extensions: &[CString],
62    ) -> Option<Self> {
63        assert!(instance_version >= Version::VERSION_1_1);
64
65        let properties = unsafe { instance.get_physical_device_properties(phd) };
66
67        // Pick the lower of the instance version and device version to get the actual version of Vulkan that
68        // can be used with the device.
69        let api_version = Version::from_raw(u32::min(properties.api_version, instance_version.to_raw()));
70
71        if api_version < Version::VERSION_1_1 {
72            // Device does not support Vulkan 1.1, so ignore it.
73            return None;
74        }
75
76        // SAFETY: Vulkan guarantees the device name is valid UTF-8 with a null terminator.
77        let name = unsafe { CStr::from_ptr(&properties.device_name as *const _) }
78            .to_str()
79            .unwrap()
80            .to_string();
81
82        // Initialize the type with the api_version.
83        let mut info = PhdInfo {
84            api_version,
85            name,
86            ..Default::default()
87        };
88
89        let mut properties = vk::PhysicalDeviceProperties2::default();
90
91        // Maintenance3 and IDProperties are both Core in Vulkan 1.1
92        //
93        // SAFETY: Maintenance3 extension is supported since Vulkan 1.1
94        properties = properties
95            .push_next(&mut info.maintenance_3)
96            .push_next(&mut info.id);
97
98        // VK_EXT_physical_device_drm
99        if supported_extensions
100            .iter()
101            .any(|name| name.as_c_str() == ext::physical_device_drm::NAME)
102        {
103            // SAFETY: The caller has garunteed the physical device supports VK_EXT_physical_device_drm
104            let next = info
105                .properties_drm
106                .insert(vk::PhysicalDeviceDrmPropertiesEXT::default());
107            properties = properties.push_next(next);
108        }
109
110        // VK_KHR_driver_properties or Vulkan 1.2
111        if api_version >= Version::VERSION_1_2
112            || supported_extensions
113                .iter()
114                .any(|name| name.as_c_str() == khr::driver_properties::NAME)
115        {
116            // SAFETY: VK_KHR_driver_properties is supported
117            let next = info
118                .properties_driver
119                .insert(vk::PhysicalDeviceDriverProperties::default());
120            properties = properties.push_next(next);
121        }
122
123        unsafe { instance.get_physical_device_properties2(phd, &mut properties) };
124
125        info.properties = properties.properties;
126        // Initialize the driver info
127        info.driver = info.properties_driver.map(DriverInfo::from_driver_properties);
128
129        Some(info)
130    }
131
132    #[cfg_attr(not(feature = "backend_drm"), allow(dead_code))]
133    pub(super) fn get_drm_properties(
134        &self,
135    ) -> Result<vk::PhysicalDeviceDrmPropertiesEXT<'_>, UnsupportedProperty> {
136        const EXTENSIONS: &[&CStr] = &[ext::physical_device_drm::NAME];
137        self.properties_drm
138            .ok_or(UnsupportedProperty::Extensions(EXTENSIONS))
139    }
140}

142impl super::DriverInfo {
143    fn from_driver_properties(properties: vk::PhysicalDeviceDriverProperties<'_>) -> DriverInfo {
144        // SAFETY: Vulkan guarantees the driver name is valid UTF-8 with a null terminator.
145        let name = unsafe { CStr::from_ptr(&properties.driver_name as *const _) }
146            .to_str()
147            .unwrap()
148            .to_string();
149
150        // SAFETY: Vulkan guarantees the driver info is valid UTF-8 with a null terminator.
151        let info = unsafe { CStr::from_ptr(&properties.driver_info as *const _) }
152            .to_str()
153            .unwrap()
154            .to_string();
155
156        DriverInfo {
157            id: properties.driver_id,
158            name,
159            info,
160            conformance: properties.conformance_version,
161        }
162    }
