65impl WlPointerHandle {
66    pub(super) fn new_pointer(&self, pointer: WlPointer) {
67        self.known_pointers.lock().unwrap().push(pointer.downgrade());
68    }
69
70    fn enter<D: SeatHandler + 'static>(&self, surface: &WlSurface, event: &MotionEvent) {
71        *self.last_enter.lock().unwrap() = Some(event.serial);
72
73        self.for_each_focused_pointer(surface, |ptr| {
74            let client_scale = ptr
75                .data::<PointerUserData<D>>()
76                .unwrap()
77                .client_scale
78                .load(Ordering::Acquire);
79            let location = event.location.to_client(client_scale);
80            ptr.enter(event.serial.into(), surface, location.x, location.y);
81        })
82    }
83
84    fn leave(&self, surface: &WlSurface, serial: Serial, _time: u32) {
85        self.for_each_focused_pointer(surface, |ptr| {
86            ptr.leave(serial.into(), surface);
87            if ptr.version() >= 5 {
88                ptr.frame();
89            }
90        });
91
92        *self.last_enter.lock().unwrap() = None;
93    }
94
95    fn motion<D: SeatHandler + 'static>(&self, surface: &WlSurface, event: &MotionEvent) {
96        self.for_each_focused_pointer(surface, |ptr| {
97            let client_scale = ptr
98                .data::<PointerUserData<D>>()
99                .unwrap()
100                .client_scale
101                .load(Ordering::Acquire);
102            let location = event.location.to_client(client_scale);
103            ptr.motion(event.time, location.x, location.y);
104        })
105    }
106
107    fn button(&self, surface: &WlSurface, event: &ButtonEvent) {
108        self.for_each_focused_pointer(surface, |ptr| {
109            ptr.button(event.serial.into(), event.time, event.button, event.state.into());
110        })
111    }
112
113    fn axis<D: SeatHandler + 'static>(&self, surface: &WlSurface, details: AxisFrame) {
114        self.for_each_focused_pointer(surface, |ptr| {
115            if ptr.version() >= 5 {
116                // axis source
117                if let Some(source) = details.source {
118                    // WheelTilt was not supported before version 6
119                    // The best we can do is replace it with Wheel
120                    let source = if ptr.version() < 6 {
121                        match source {
122                            AxisSource::WheelTilt => AxisSource::Wheel,
123                            other => other,
124                        }
125                    } else {
126                        source
127                    }
128                    .into();
129                    ptr.axis_source(source);
130                }
131                // axis discrete
132                if let Some((x, y)) = details.v120 {
133                    if ptr.version() >= 8 {
134                        if x != 0 {
135                            ptr.axis_value120(WlAxis::HorizontalScroll, x);
136                        }
137                        if y != 0 {
138                            ptr.axis_value120(WlAxis::VerticalScroll, y);
139                        }
140                    } else {
141                        compositor::with_states(surface, |states| {
142                            let mut data = states
143                                .data_map
144                                .get_or_insert_threadsafe(Mutex::<V120UserData>::default)
145                                .lock()
146                                .unwrap();
147
148                            data.x += x;
149                            if data.x.abs() >= 120 {
150                                ptr.axis_discrete(WlAxis::HorizontalScroll, data.x / 120);
151                                data.x %= 120;
152                            }
153
154                            data.y += y;
155                            if data.y.abs() >= 120 {
156                                ptr.axis_discrete(WlAxis::VerticalScroll, data.y / 120);
157                                data.y %= 120;
158                            }
159                        });
160                    }
161                }
162                // stop
163                if details.stop.0 {
164                    ptr.axis_stop(details.time, WlAxis::HorizontalScroll);
165
166                    compositor::with_states(surface, |states| {
167                        if let Some(data) = states.data_map.get::<Mutex<V120UserData>>() {
168                            data.lock().unwrap().x = 0;
169                        }
170                    });
171                }
172                if details.stop.1 {
173                    ptr.axis_stop(details.time, WlAxis::VerticalScroll);
174
175                    compositor::with_states(surface, |states| {
176                        if let Some(data) = states.data_map.get::<Mutex<V120UserData>>() {
177                            data.lock().unwrap().y = 0;
178                        }
179                    });
180                }
181            }
182            // axis
183            let client_scale = ptr
184                .data::<PointerUserData<D>>()
185                .unwrap()
186                .client_scale
187                .load(Ordering::Acquire);
188            if details.axis.0 != 0.0 {
189                if ptr.version() >= 9 {
190                    ptr.axis_relative_direction(
191                        WlAxis::HorizontalScroll,
192                        details.relative_direction.0.into(),
193                    );
194                }
195                ptr.axis(
196                    details.time,
197                    WlAxis::HorizontalScroll,
198                    details.axis.0 * client_scale,
199                );
200            }
201            if details.axis.1 != 0.0 {
202                if ptr.version() >= 9 {
203                    ptr.axis_relative_direction(WlAxis::VerticalScroll, details.relative_direction.1.into());
204                }
205                ptr.axis(
206                    details.time,
207                    WlAxis::VerticalScroll,
208                    details.axis.1 * client_scale,
209                );
210            }
211        })
212    }
213
214    fn frame(&self, surface: &WlSurface) {
215        self.for_each_focused_pointer(surface, |ptr| {
216            if ptr.version() >= 5 {
217                ptr.frame();
218            }
219        });
220    }
221
222    fn for_each_focused_pointer(&self, surface: &WlSurface, mut f: impl FnMut(WlPointer)) {
223        let inner = self.known_pointers.lock().unwrap();
224        for ptr in &*inner {
225            let Ok(ptr) = ptr.upgrade() else {
226                continue;
227            };
228
229            if ptr.id().same_client_as(&surface.id()) {
230                f(ptr.clone())
231            }
232        }
233    }
234}

345#[derive(Debug)]

347    pub(crate) handle: Option<PointerHandle<D>>,
348    pub(crate) client_scale: Arc<AtomicF64>,
349}

351impl<D> Dispatch<WlPointer, PointerUserData<D>, D> for SeatState<D>
352where
353    D: Dispatch<WlPointer, PointerUserData<D>>,
354    D: SeatHandler,
355    <D as SeatHandler>::PointerFocus: WaylandFocus,
356    D: 'static,
357{
358    fn request(
359        state: &mut D,
360        _client: &wayland_server::Client,
361        pointer: &WlPointer,
362        request: wl_pointer::Request,
363        data: &PointerUserData<D>,
364        _dh: &DisplayHandle,
365        _data_init: &mut wayland_server::DataInit<'_, D>,
366    ) {
367        match request {
368            Request::SetCursor {
369                serial,
370                surface,
371                hotspot_x,
372                hotspot_y,
373            } => {
374                let handle = match &data.handle {
375                    Some(handle) => handle,
376                    None => return,
377                };
378
379                if !allow_setting_cursor(handle, Serial(serial), &pointer.id()) {
380                    return;
381                }
382
383                let cursor_image = match surface {
384                    Some(surface) => {
385                        // tolerate re-using the same surface
386                        if compositor::give_role(&surface, CURSOR_IMAGE_ROLE).is_err()
387                            && compositor::get_role(&surface) != Some(CURSOR_IMAGE_ROLE)
388                        {
389                            pointer.post_error(wl_pointer::Error::Role, "Given wl_surface has another role.");
390                            return;
391                        }
392
393                        compositor::with_states(&surface, |states| {
394                            states.data_map.insert_if_missing_threadsafe(|| {
395                                Mutex::new(CursorImageAttributes {
396                                    hotspot: (0, 0).into(),
397                                })
398                            });
399                            let client_scale = pointer
400                                .data::<PointerUserData<D>>()
401                                .unwrap()
402                                .client_scale
403                                .load(Ordering::Acquire);
404                            let hotspot = Point::<i32, ClientCoords>::from((hotspot_x, hotspot_y))
405                                .to_f64()
406                                .to_logical(client_scale)
407                                .to_i32_round();
408                            states
409                                .data_map
410                                .get::<Mutex<CursorImageAttributes>>()
411                                .unwrap()
412                                .lock()
413                                .unwrap()
414                                .hotspot = hotspot;
415                        });
416
417                        CursorImageStatus::Surface(surface)
418                    }
419                    None => CursorImageStatus::Hidden,
420                };
421
422                let seat = state
423                    .seat_state()
424                    .seats
425                    .iter()
426                    .find(|seat| seat.get_pointer().map(|h| h == *handle).unwrap_or(false))
427                    .cloned();
428
429                if let Some(seat) = seat {
430                    state.cursor_image(&seat, cursor_image)
431                }
432            }
433            Request::Release => {
434                // Our destructors already handle it
435            }
436            _ => unreachable!(),
437        };
438    }
439
440    fn destroyed(_state: &mut D, _: ClientId, pointer: &WlPointer, data: &PointerUserData<D>) {
441        if let Some(ref handle) = data.handle {
442            handle
443                .wl_pointer
444                .known_pointers
445                .lock()
446                .unwrap()
447                .retain(|p| p.id() != pointer.id());
448        }
449    }
450}

516pub(crate) fn allow_setting_cursor<D>(handle: &PointerHandle<D>, serial: Serial, object_id: &ObjectId) -> bool
517where
518    D: SeatHandler + 'static,
519    <D as SeatHandler>::PointerFocus: WaylandFocus,
520{
521    // Allow client if there is a pointer grab for that client. Like drag and drop.
522    if handle
523        .grab_start_data()
524        .and_then(|data| data.focus)
525        .is_some_and(|focus| focus.0.same_client_as(object_id))
526    {
527        return true;
528    }
529
530    if !handle
531        .wl_pointer
532        .last_enter
533        .lock()
534        .unwrap()
535        .as_ref()
536        .is_some_and(|last_serial| *last_serial == serial)
537    {
538        return false; // Ignore mismatches in serial
539    }
540
541    // Only allow setting the cursor icon if the current pointer focus is of the same
542    // client.
543    handle
544        .inner
545        .lock()
546        .unwrap()
547        .focus
548        .as_ref()
549        .is_some_and(|(focus, _)| focus.same_client_as(object_id))
550}
