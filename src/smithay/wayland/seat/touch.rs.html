19impl<D: SeatHandler> TouchHandle<D> {
20    pub(crate) fn new_touch(&self, touch: WlTouch) {
21        let mut guard = self.known_instances.lock().unwrap();
22        guard.push((touch.downgrade(), None));
23    }

26impl<D: SeatHandler + 'static> TouchHandle<D> {

36fn for_each_focused_touch<D: SeatHandler + 'static>(
37    seat: &Seat<D>,
38    surface: &WlSurface,
39    seq: Serial,
40    mut f: impl FnMut(WlTouch),
41) {
42    if let Some(touch) = seat.get_touch() {
43        let mut inner = touch.known_instances.lock().unwrap();
44        for (ptr, last_seq) in &mut *inner {
45            let Ok(ptr) = ptr.upgrade() else {
46                continue;
47            };
48
49            if ptr.id().same_client_as(&surface.id()) && last_seq.map(|last| last < seq).unwrap_or(true) {
50                *last_seq = Some(seq);
51                f(ptr.clone());
52            }
53        }
54    }
55}

57#[cfg(feature = "wayland_frontend")]
58impl<D> TouchTarget<D> for WlSurface
59where
60    D: SeatHandler + 'static,
61{
62    fn down(&self, seat: &Seat<D>, _data: &mut D, event: &DownEvent, seq: Serial) {
63        let serial = event.serial;
64        let slot = event.slot;
65        for_each_focused_touch(seat, self, seq, |touch| {
66            let client_scale = touch
67                .data::<TouchUserData<D>>()
68                .unwrap()
69                .client_scale
70                .load(Ordering::Acquire);
71            let location = event.location.to_client(client_scale);
72            touch.down(
73                serial.into(),
74                event.time,
75                self,
76                slot.into(),
77                location.x,
78                location.y,
79            );
80        })
81    }
82
83    fn up(&self, seat: &Seat<D>, _data: &mut D, event: &UpEvent, seq: Serial) {
84        let serial = event.serial;
85        let slot = event.slot;
86        for_each_focused_touch(seat, self, seq, |touch| {
87            touch.up(serial.into(), event.time, slot.into());
88        })
89    }
90
91    fn motion(&self, seat: &Seat<D>, _data: &mut D, event: &MotionEvent, seq: Serial) {
92        let slot = event.slot;
93        for_each_focused_touch(seat, self, seq, |touch| {
94            let client_scale = touch
95                .data::<TouchUserData<D>>()
96                .unwrap()
97                .client_scale
98                .load(Ordering::Acquire);
99            let location = event.location.to_client(client_scale);
100            touch.motion(event.time, slot.into(), location.x, location.y);
101        })
102    }
103
104    fn frame(&self, seat: &Seat<D>, _data: &mut D, seq: Serial) {
105        for_each_focused_touch(seat, self, seq, |touch| {
106            touch.frame();
107        })
108    }
109
110    fn cancel(&self, seat: &Seat<D>, _data: &mut D, seq: Serial) {
111        for_each_focused_touch(seat, self, seq, |touch| {
112            touch.cancel();
113        })
114    }
115
116    fn shape(&self, seat: &Seat<D>, _data: &mut D, event: &ShapeEvent, seq: Serial) {
117        let slot = event.slot;
118        for_each_focused_touch(seat, self, seq, |touch| {
119            touch.shape(slot.into(), event.major, event.minor);
120        })
121    }
122
123    fn orientation(&self, seat: &Seat<D>, _data: &mut D, event: &OrientationEvent, seq: Serial) {
124        let slot = event.slot;
125        for_each_focused_touch(seat, self, seq, |touch| {
126            touch.orientation(slot.into(), event.orientation);
127        })
128    }
129}

157            handle
158                .known_instances

160                .unwrap()
