3/// A simplified version of [`wayland_server::Dispatch`]
4///
5/// A future version of `wayland-server` will replace `Dispatch` with this.
6pub trait Dispatch2<I: Resource, State> {
7    /// Called when a request from a client is processed.
8    fn request(
9        &self,
10        state: &mut State,
11        client: &Client,
12        resource: &I,
13        request: I::Request,
14        dhandle: &DisplayHandle,
15        data_init: &mut DataInit<'_, State>,
16    );
17
18    /// Called when the object this user data is associated with has been destroyed.
19    fn destroyed(&self, _state: &mut State, _client: ClientId, _resource: &I) {}
20}
21
22/// A simplified version of [`wayland_server::GlobalDispatch`]
23///
24/// A future version of `wayland-server` will replace `GlobalDispatch` with this.
25pub trait GlobalDispatch2<I: Resource, State> {
26    /// Called when a client has bound this global.
27    fn bind(
28        &self,
29        state: &mut State,
30        handle: &DisplayHandle,
31        client: &Client,
32        resource: New<I>,
33        data_init: &mut DataInit<'_, State>,
34    );
35
36    /// Checks if the global should be advertised to some client.
37    fn can_view(&self, _client: &Client) -> bool {
38        true
39    }
40}

42/// Implement `Dispatch` and `GlobalDispatch` for every implementation of [`Dispatch2`] and
43/// [`GlobalDispatch2`].
44#[macro_export]
45macro_rules! delegate_dispatch2 {
46    ($(@< $( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+ >)? $ty:ty) => {
47        impl<$( $( $lt $( : $clt $(+ $dlt )* )? ),+, )? I, UserData> $crate::reexports::wayland_server::Dispatch<I, UserData> for $ty
48            where
49                I: $crate::reexports::wayland_server::Resource,
50                UserData: $crate::wayland::Dispatch2<I, $ty> {
51            fn request(
52                state: &mut Self,
53                client: &$crate::reexports::wayland_server::Client,
54                resource: &I,
55                request: <I as $crate::reexports::wayland_server::Resource>::Request,
56                data: &UserData,
57                dhandle: &$crate::reexports::wayland_server::DisplayHandle,
58                data_init: &mut $crate::reexports::wayland_server::DataInit<'_, Self>,
59            ) {
60                data.request(state, client, resource, request, dhandle, data_init);
61            }
62
63            fn destroyed(state: &mut Self, client: $crate::reexports::wayland_server::backend::ClientId, resource: &I, data: &UserData) {
64                data.destroyed(state, client, resource);
65            }
66        }
67
68        impl<$( $( $lt $( : $clt $(+ $dlt )* )? ),+, )? I, UserData> $crate::reexports::wayland_server::GlobalDispatch<I, UserData> for $ty
69            where
70                I: $crate::reexports::wayland_server::Resource,
71                UserData: $crate::wayland::GlobalDispatch2<I, $ty> {
72            fn bind(
73                state: &mut Self,
74                dhandle: &$crate::reexports::wayland_server::DisplayHandle,
75                client: &$crate::reexports::wayland_server::Client,
76                resource: $crate::reexports::wayland_server::New<I>,
77                data: &UserData,
78                data_init: &mut $crate::reexports::wayland_server::DataInit<'_, Self>,
79            ) {
80                data.bind(state, dhandle, client, resource, data_init);
81            }
82
83            fn can_view(
84                client: $crate::reexports::wayland_server::Client,
85                data: &UserData
86            ) -> bool {
87                data.can_view(&client)
88            }
89        }
90    };
