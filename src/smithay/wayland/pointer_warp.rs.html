1//! Wp Pointer Warp
2//!
3//! This global interface allows applications to request the pointer to be moved to a position
4//! relative to a wl_surface.
5//!
6//! In order to advertise pointer warp global call [PointerWarpManager::new].
7//!
8//! ```
9//! use smithay::wayland::pointer_warp::{PointerWarpManager, PointerWarpHandler};
10//! use wayland_server::protocol::wl_surface::WlSurface;
11//! use wayland_server::protocol::wl_pointer::WlPointer;
12//! use smithay::utils::{Serial, Logical, Point};
13//!
14//! # struct State;
15//! # let mut display = wayland_server::Display::<State>::new().unwrap();
16//! #
17//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
18//! # use smithay::reexports::wayland_server::Client;
19//! # impl CompositorHandler for State {
20//! #     fn compositor_state(&mut self) -> &mut CompositorState { unimplemented!() }
21//! #     fn client_compositor_state<'a>(&self, client: &'a Client) -> &'a CompositorClientState { unimplemented!() }
22//! #     fn commit(&mut self, surface: &WlSurface) {}
23//! # }
24//! #
25//! # impl smithay::input::SeatHandler for State {
26//! #     type KeyboardFocus = WlSurface;
27//! #     type PointerFocus = WlSurface;
28//! #     type TouchFocus = WlSurface;
29//! #     fn seat_state(&mut self) -> &mut smithay::input::SeatState<Self> {
30//! #         todo!()
31//! #     }
32//! # }
33//!
34//! PointerWarpManager::new::<State>(
35//!     &display.handle(),
36//! );
37//!
38//! impl PointerWarpHandler for State {
39//!     fn warp_pointer(&mut self, surface: WlSurface, pointer: WlPointer, pos: Point<f64, Logical>, serial: Serial) {
40//!         // Pointer warp was requested by the client
41//!     }
42//! }
43//!
44//! smithay::delegate_dispatch2!(State);
45//! ```

62/// Handler trait for pointer warp events.
63pub trait PointerWarpHandler: SeatHandler + 'static {
64    /// Request the compositor to move the pointer to a surface-local position.
65    /// Whether or not the compositor honors the request is implementation defined, but it should
66    /// - honor it if the surface has pointer focus, including when it has an implicit pointer grab
67    /// - reject it if the enter serial is incorrect
68    /// - reject it if the requested position is outside of the surface
69    ///
70    /// Note that the enter serial is valid for any surface of the client, and does not have to be from the surface the pointer is warped to.
71    ///
72    /// * `serial` - serial number of the surface enter event
73    #[allow(unused)]
74    fn warp_pointer(
75        &mut self,
76        surface: WlSurface,
77        pointer: WlPointer,
78        pos: Point<f64, Logical>,
79        serial: Serial,
80    ) {
81    }
82}

84/// Delegate type for handling pointer warp events.
85#[derive(Debug)]
86pub struct PointerWarpManager {
87    global: GlobalId,
88}
89
90impl PointerWarpManager {
91    /// Creates a new delegate type for handling [WpPointerWarpV1] events.
92    pub fn new<D>(display: &DisplayHandle) -> Self
93    where
94        D: PointerWarpHandler + GlobalDispatch<WpPointerWarpV1, GlobalData>,
95    {
96        let global = display.create_global::<D, WpPointerWarpV1, _>(1, GlobalData);
97        Self { global }
98    }
99
100    /// Returns the [WpPointerWarpV1] global id.
101    pub fn global(&self) -> GlobalId {
102        self.global.clone()
103    }
104}

106impl<D: PointerWarpHandler> GlobalDispatch2<WpPointerWarpV1, D> for GlobalData
107where
108    D: Dispatch<WpPointerWarpV1, GlobalData>,
109{
110    fn bind(
111        &self,
112        _state: &mut D,
113        _handle: &DisplayHandle,
114        _client: &Client,
115        resource: New<WpPointerWarpV1>,
116        data_init: &mut DataInit<'_, D>,
117    ) {
118        data_init.init(resource, GlobalData);
119    }
120}
121
122impl<D: PointerWarpHandler> Dispatch2<WpPointerWarpV1, D> for GlobalData {
123    fn request(
124        &self,
125        state: &mut D,
126        _client: &Client,
127        _resource: &WpPointerWarpV1,
128        request: wp_pointer_warp_v1::Request,
129        _dhandle: &DisplayHandle,
130        _data_init: &mut DataInit<'_, D>,
131    ) {
132        use wp_pointer_warp_v1::Request;
133
134        match request {
135            Request::WarpPointer {
136                surface,
137                pointer,
138                x,
139                y,
140                serial,
141            } => {
142                let client_scale = pointer
143                    .data::<PointerUserData<D>>()
144                    .unwrap()
145                    .client_scale
146                    .load(Ordering::Acquire);
147                let location: Point<f64, ClientCords> = Point::new(x, y);
148                let location = location.to_logical(client_scale);
149
150                state.warp_pointer(surface, pointer, location, Serial::from(serial));
151            }
152            Request::Destroy => {}
153            _ => unreachable!(),
154        }
155    }
156}
