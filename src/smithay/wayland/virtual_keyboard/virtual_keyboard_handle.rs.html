74impl<D> Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>, D> for VirtualKeyboardManagerState
75where
76    D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
77    D: SeatHandler + 'static,
78    <D as SeatHandler>::KeyboardFocus: WaylandFocus,
79{
80    fn request(
81        user_data: &mut D,
82        _client: &Client,
83        virtual_keyboard: &ZwpVirtualKeyboardV1,
84        request: zwp_virtual_keyboard_v1::Request,
85        data: &VirtualKeyboardUserData<D>,
86        _dh: &DisplayHandle,
87        _data_init: &mut DataInit<'_, D>,
88    ) {
89        match request {
90            zwp_virtual_keyboard_v1::Request::Keymap { format, fd, size } => {
91                update_keymap(data, format, fd, size as usize);
92            }
93            zwp_virtual_keyboard_v1::Request::Key { time, key, state } => {
94                // Ensure keymap was initialized.
95                let mut virtual_data = data.handle.inner.lock().unwrap();
96                let vk_state = match virtual_data.state.as_mut() {
97                    Some(vk_state) => vk_state,
98                    None => {
99                        virtual_keyboard.post_error(NoKeymap, "`key` sent before keymap.");
100                        return;
101                    }
102                };
103
104                // Ensure virtual keyboard's keymap is active.
105                let keyboard_handle = data.seat.get_keyboard().unwrap();
106                let mut internal = keyboard_handle.arc.internal.lock().unwrap();
107                let focus = internal.focus.as_mut().map(|(focus, _)| focus);
108                keyboard_handle.send_keymap(user_data, &focus, &vk_state.keymap, vk_state.mods);
109
110                if let Some(wl_surface) = focus.and_then(|f| f.wl_surface()) {
111                    for_each_focused_kbds(&data.seat, &wl_surface, |kbd| {
112                        // This should be wl_keyboard::KeyState, but the protocol does not state
113                        // the parameter is an enum.
114                        let key_state = if state == 1 {
115                            KeyState::Pressed
116                        } else {
117                            KeyState::Released
118                        };
119
120                        kbd.key(SERIAL_COUNTER.next_serial().0, time, key, key_state);
121                    });
122                }
123            }
124            zwp_virtual_keyboard_v1::Request::Modifiers {
125                mods_depressed,
126                mods_latched,
127                mods_locked,
128                group,
129            } => {
130                // Ensure keymap was initialized.
131                let mut virtual_data = data.handle.inner.lock().unwrap();
132                let state = match virtual_data.state.as_mut() {
133                    Some(state) => state,
134                    None => {
135                        virtual_keyboard.post_error(NoKeymap, "`modifiers` sent before keymap.");
136                        return;
137                    }
138                };
139
140                // Update virtual keyboard's modifier state.
141                state
142                    .state
143                    .update_mask(mods_depressed, mods_latched, mods_locked, 0, 0, group);
144                state.mods.update_with(&state.state);
145
146                // Ensure virtual keyboard's keymap is active.
147                let keyboard_handle = data.seat.get_keyboard().unwrap();
148                let mut internal = keyboard_handle.arc.internal.lock().unwrap();
149                let focus = internal.focus.as_mut().map(|(focus, _)| focus);
150                let keymap_changed =
151                    keyboard_handle.send_keymap(user_data, &focus, &state.keymap, state.mods);
152
153                // Report modifiers change to all keyboards.
154                if !keymap_changed {
155                    if let Some(focus) = focus {
156                        focus.modifiers(&data.seat, user_data, state.mods, SERIAL_COUNTER.next_serial());
157                    }
158                }
159            }
160            zwp_virtual_keyboard_v1::Request::Destroy => {
161                // Nothing to do
162            }
163            _ => unreachable!(),
164        }
165    }
166
167    fn destroyed(
168        _state: &mut D,
169        _client: ClientId,
170        _virtual_keyboard: &ZwpVirtualKeyboardV1,
171        _data: &VirtualKeyboardUserData<D>,
172    ) {
173    }
174}

176/// Handle the zwp_virtual_keyboard_v1::keymap request.
177///
178/// The `true` returns when keymap was properly loaded.
179fn update_keymap<D>(data: &VirtualKeyboardUserData<D>, format: u32, fd: OwnedFd, size: usize)
180where
181    D: SeatHandler + 'static,
182{
183    // Only libxkbcommon compatible keymaps are supported.
184    if format != KeymapFormat::XkbV1 as u32 {
185        debug!("Unsupported keymap format: {format:?}");
186        return;
187    }
188
189    let context = xkb::Context::new(xkb::CONTEXT_NO_FLAGS);
190    // SAFETY: we can map the keymap into the memory.
191    let new_keymap = match unsafe {
192        xkb::Keymap::new_from_fd(
193            &context,
194            fd,
195            size,
196            xkb::KEYMAP_FORMAT_TEXT_V1,
197            xkb::KEYMAP_COMPILE_NO_FLAGS,
198        )
199    } {
200        Ok(Some(new_keymap)) => new_keymap,
201        Ok(None) => {
202            debug!("Invalid libxkbcommon keymap");
203            return;
204        }
205        Err(err) => {
206            debug!("Could not map the keymap: {err:?}");
207            return;
208        }
209    };
210
211    // Store active virtual keyboard map.
212    let mut inner = data.handle.inner.lock().unwrap();
213    let mods = inner.state.take().map(|state| state.mods).unwrap_or_default();
214    inner.state = Some(VirtualKeyboardState {
215        mods,
216        keymap: KeymapFile::new(&new_keymap),
217        state: xkb::State::new(&new_keymap),
218    });
