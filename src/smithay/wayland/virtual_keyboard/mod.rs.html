1//! Utilities for virtual keyboard support
2//!
3//! This module provides you with utilities to handle virtual keyboard instances.
4//! It can be used standalone to implement virtual keyboards or together with
5//! an input method to pass through keys from the keyboard.
6//!
7//! ```
8//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
9//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
10//! use smithay::wayland::virtual_keyboard::VirtualKeyboardManagerState;
11//! use smithay::reexports::wayland_server::{Display, protocol::wl_surface::WlSurface};
12//! # use smithay::reexports::wayland_server::Client;
13//!
14//! # struct State { seat_state: SeatState<Self> };
15//!
16//! smithay::delegate_dispatch2!(State);
17//!
18//! # let mut display = Display::<State>::new().unwrap();
19//! # let display_handle = display.handle();
20//!
21//! let seat_state = SeatState::<State>::new();
22//!
23//! // implement the required traits
24//! impl SeatHandler for State {
25//!     type KeyboardFocus = WlSurface;
26//!     type PointerFocus = WlSurface;
27//!     type TouchFocus = WlSurface;
28//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
29//!         &mut self.seat_state
30//!     }
31//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
32//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
33//! }
34//!
35//! // Add the seat state to your state, create manager global and add client filter
36//! // to avoid untrusted clients requesting a new keyboard
37//! VirtualKeyboardManagerState::new::<State, _>(&display_handle, |_client| true);
38//!
39//! # impl CompositorHandler for State {
40//! #     fn compositor_state(&mut self) -> &mut CompositorState { unimplemented!() }
41//! #     fn client_compositor_state<'a>(&self, client: &'a Client) -> &'a CompositorClientState { unimplemented!() }
42//! #     fn commit(&mut self, surface: &WlSurface) {}
43//! # }
44//! ```
45//!

66/// State of wp misc virtual keyboard protocol

72/// Data associated with a VirtualKeyboardManager global.
73#[allow(missing_debug_implementations)]
74pub struct VirtualKeyboardManagerGlobalData {
75    filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
76}
77
78fn create_global_with_filter<D, F>(display: &DisplayHandle, filter: F) -> GlobalId
79where
80    D: GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData> + 'static,
81    F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
82{
83    let data = VirtualKeyboardManagerGlobalData {
84        filter: Box::new(filter),
85    };
86
87    display.create_global::<D, ZwpVirtualKeyboardManagerV1, _>(MANAGER_VERSION, data)
88}
89
90impl VirtualKeyboardManagerState {
91    /// Initialize a virtual keyboard manager global.
92    pub fn new<D, F>(display: &DisplayHandle, filter: F) -> Self
93    where
94        D: GlobalDispatch<ZwpVirtualKeyboardManagerV1, VirtualKeyboardManagerGlobalData>,
95        D: Dispatch<ZwpVirtualKeyboardManagerV1, GlobalData>,
96        D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
97        D: SeatHandler,
98        D: 'static,
99        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
100    {
101        let global = create_global_with_filter::<D, F>(display, filter);
102
103        Self { global }
104    }
105
106    /// Get the id of ZwpVirtualKeyboardManagerV1 global
107    pub fn global(&self) -> GlobalId {
108        self.global.clone()
109    }
110}

112impl<D> GlobalDispatch2<ZwpVirtualKeyboardManagerV1, D> for VirtualKeyboardManagerGlobalData
113where
114    D: Dispatch<ZwpVirtualKeyboardManagerV1, GlobalData>,
115    D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
116    D: SeatHandler,
117    D: 'static,
118{
119    fn bind(
120        &self,
121        _: &mut D,
122        _: &DisplayHandle,
123        _: &Client,
124        resource: New<ZwpVirtualKeyboardManagerV1>,
125        data_init: &mut DataInit<'_, D>,
126    ) {
127        data_init.init(resource, GlobalData);
128    }
129
130    fn can_view(&self, client: &Client) -> bool {
131        (self.filter)(client)
132    }
133}
134
135impl<D> Dispatch2<ZwpVirtualKeyboardManagerV1, D> for GlobalData
136where
137    D: Dispatch<ZwpVirtualKeyboardV1, VirtualKeyboardUserData<D>>,
138    D: SeatHandler,
139    D: 'static,
140{
141    fn request(
142        &self,
143        _state: &mut D,
144        _client: &Client,
145        _resource: &ZwpVirtualKeyboardManagerV1,
146        request: zwp_virtual_keyboard_manager_v1::Request,
147        _handle: &DisplayHandle,
148        data_init: &mut DataInit<'_, D>,
149    ) {
150        match request {
151            zwp_virtual_keyboard_manager_v1::Request::CreateVirtualKeyboard { seat, id } => {
152                let seat = Seat::<D>::from_resource(&seat).unwrap();
153                let virtual_keyboard_handle = VirtualKeyboardHandle::default();
154                data_init.init(
155                    id,
156                    VirtualKeyboardUserData {
157                        handle: virtual_keyboard_handle,
158                        seat: seat.clone(),
159                    },
160                );
161            }
162            _ => unreachable!(),
163        }
164    }
