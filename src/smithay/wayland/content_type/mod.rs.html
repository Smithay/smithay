1//! Implementation of wp_content_type protocol
2//!
3//! ### Example
4//!
5//! ```no_run
6//! # extern crate wayland_server;
7//! #
8//! use wayland_server::{protocol::wl_surface::WlSurface, DisplayHandle};
9//! use smithay::{
10//!     delegate_content_type, delegate_compositor,
11//!     wayland::compositor::{self, CompositorState, CompositorClientState, CompositorHandler},
12//!     wayland::content_type::{ContentTypeSurfaceCachedState, ContentTypeState},
13//! };
14//!
15//! pub struct State {
16//!     compositor_state: CompositorState,
17//! };
18//! struct ClientState { compositor_state: CompositorClientState }
19//! impl wayland_server::backend::ClientData for ClientState {}
20//!
21//! delegate_content_type!(State);
22//! delegate_compositor!(State);
23//!
24//! impl CompositorHandler for State {
25//!    fn compositor_state(&mut self) -> &mut CompositorState {
26//!        &mut self.compositor_state
27//!    }
28//!
29//!    fn client_compositor_state<'a>(&self, client: &'a wayland_server::Client) -> &'a CompositorClientState {
30//!        &client.get_data::<ClientState>().unwrap().compositor_state    
31//!    }
32//!
33//!    fn commit(&mut self, surface: &WlSurface) {
34//!        compositor::with_states(&surface, |states| {
35//!            let mut guard = states.cached_state.get::<ContentTypeSurfaceCachedState>();
36//!            let current = guard.current();
37//!            dbg!(current.content_type());
38//!        });
39//!    }
40//! }
41//!
42//! let mut display = wayland_server::Display::<State>::new().unwrap();
43//!
44//! let compositor_state = CompositorState::new::<State>(&display.handle());
45//! ContentTypeState::new::<State>(&display.handle());
46//!

49//! };

70/// Data associated with WlSurface
71/// Represents the client pending state
72///
73/// ```no_run
74/// use smithay::wayland::compositor;
75/// use smithay::wayland::content_type::ContentTypeSurfaceCachedState;
76///
77/// # let wl_surface = todo!();
78/// compositor::with_states(&wl_surface, |states| {
79///     let mut guard = states.cached_state.get::<ContentTypeSurfaceCachedState>();
80///     let current = guard.current();
81///     dbg!(current.content_type());
82/// });
83/// ```
84#[derive(Debug, Clone, Copy)]
85pub struct ContentTypeSurfaceCachedState {
86    content_type: wp_content_type_v1::Type,
87}
88
89impl ContentTypeSurfaceCachedState {
90    /// This informs the compositor that the client believes it is displaying buffers matching this content type.
91    pub fn content_type(&self) -> &wp_content_type_v1::Type {
92        &self.content_type
93    }
94}
95
96impl Default for ContentTypeSurfaceCachedState {
97    fn default() -> Self {
98        Self {
99            content_type: wp_content_type_v1::Type::None,
100        }
101    }
102}
103
104impl Cacheable for ContentTypeSurfaceCachedState {
105    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
106        *self
107    }
108
109    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
110        *into = self;
111    }
112}

114#[derive(Debug)]
115struct ContentTypeSurfaceData {
116    is_resource_attached: AtomicBool,
117}
118
119impl ContentTypeSurfaceData {
120    fn new() -> Self {
121        Self {
122            is_resource_attached: AtomicBool::new(false),
123        }
124    }
125
126    fn set_is_resource_attached(&self, is_attached: bool) {
127        self.is_resource_attached
128            .store(is_attached, atomic::Ordering::Release)
129    }
130
131    fn is_resource_attached(&self) -> bool {
132        self.is_resource_attached.load(atomic::Ordering::Acquire)
133    }
134}

136/// User data of `WpContentTypeV1` object
137#[derive(Debug)]
138pub struct ContentTypeUserData(Mutex<Weak<WlSurface>>);
139
140impl ContentTypeUserData {
141    fn new(surface: WlSurface) -> Self {
142        Self(Mutex::new(surface.downgrade()))
143    }
144
145    #[inline]
146    fn wl_surface(&self) -> Option<WlSurface> {
147        self.0.lock().unwrap().upgrade().ok()
148    }
149}

151/// Delegate type for [WpContentTypeManagerV1] global.
152#[derive(Debug)]
153pub struct ContentTypeState {
154    global: GlobalId,
155}
156
157impl ContentTypeState {
158    /// Regiseter new [WpContentTypeManagerV1] global
159    pub fn new<D>(display: &DisplayHandle) -> ContentTypeState
160    where
161        D: GlobalDispatch<WpContentTypeManagerV1, ()>
162            + Dispatch<WpContentTypeManagerV1, ()>
163            + Dispatch<WpContentTypeV1, ContentTypeUserData>
164            + 'static,
165    {
166        let global = display.create_global::<D, WpContentTypeManagerV1, _>(1, ());
167
168        ContentTypeState { global }
169    }
170
171    /// Returns the WpContentTypeManagerV1 global id
172    pub fn global(&self) -> GlobalId {
173        self.global.clone()
174    }
175}

177/// Macro to delegate implementation of the wp content type protocol
178#[macro_export]
179macro_rules! delegate_content_type {
180    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
181        type __WpContentTypeManagerV1 =
182            $crate::reexports::wayland_protocols::wp::content_type::v1::server::wp_content_type_manager_v1::WpContentTypeManagerV1;
183        type __WpContentTypeV1 =
184            $crate::reexports::wayland_protocols::wp::content_type::v1::server::wp_content_type_v1::WpContentTypeV1;
185
186        $crate::reexports::wayland_server::delegate_global_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty:
187            [
188                __WpContentTypeManagerV1: ()
189            ] => $crate::wayland::content_type::ContentTypeState
190        );
191
192        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty:
193            [
194                __WpContentTypeManagerV1: ()
195            ] => $crate::wayland::content_type::ContentTypeState
196        );
197
198        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty:
199            [
200                __WpContentTypeV1: $crate::wayland::content_type::ContentTypeUserData
201            ] => $crate::wayland::content_type::ContentTypeState
202        );
203    };
