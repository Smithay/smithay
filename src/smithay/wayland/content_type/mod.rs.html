1//! Implementation of wp_content_type protocol
2//!
3//! ### Example
4//!
5//! ```no_run
6//! # extern crate wayland_server;
7//! #
8//! use wayland_server::{protocol::wl_surface::WlSurface, DisplayHandle};
9//! use smithay::{
10//!     wayland::compositor::{self, CompositorState, CompositorClientState, CompositorHandler},
11//!     wayland::content_type::{ContentTypeSurfaceCachedState, ContentTypeState},
12//! };
13//!
14//! pub struct State {
15//!     compositor_state: CompositorState,
16//! };
17//! struct ClientState { compositor_state: CompositorClientState }
18//! impl wayland_server::backend::ClientData for ClientState {}
19//!
20//! smithay::delegate_dispatch2!(State);
21//!
22//! impl CompositorHandler for State {
23//!    fn compositor_state(&mut self) -> &mut CompositorState {
24//!        &mut self.compositor_state
25//!    }
26//!
27//!    fn client_compositor_state<'a>(&self, client: &'a wayland_server::Client) -> &'a CompositorClientState {
28//!        &client.get_data::<ClientState>().unwrap().compositor_state    
29//!    }
30//!
31//!    fn commit(&mut self, surface: &WlSurface) {
32//!        compositor::with_states(&surface, |states| {
33//!            let mut guard = states.cached_state.get::<ContentTypeSurfaceCachedState>();
34//!            let current = guard.current();
35//!            dbg!(current.content_type());
36//!        });
37//!    }
38//! }
39//!
40//! let mut display = wayland_server::Display::<State>::new().unwrap();
41//!
42//! let compositor_state = CompositorState::new::<State>(&display.handle());
43//! ContentTypeState::new::<State>(&display.handle());
44//!
45//! let state = State {
46//!     compositor_state,
47//! };
48//! ```

68/// Data associated with WlSurface
69/// Represents the client pending state
70///
71/// ```no_run
72/// use smithay::wayland::compositor;
73/// use smithay::wayland::content_type::ContentTypeSurfaceCachedState;
74///
75/// # let wl_surface = todo!();
76/// compositor::with_states(&wl_surface, |states| {
77///     let mut guard = states.cached_state.get::<ContentTypeSurfaceCachedState>();
78///     let current = guard.current();
79///     dbg!(current.content_type());
80/// });
81/// ```
82#[derive(Debug, Clone, Copy)]
83pub struct ContentTypeSurfaceCachedState {
84    content_type: wp_content_type_v1::Type,
85}
86
87impl ContentTypeSurfaceCachedState {
88    /// This informs the compositor that the client believes it is displaying buffers matching this content type.
89    pub fn content_type(&self) -> &wp_content_type_v1::Type {
90        &self.content_type
91    }
92}
93
94impl Default for ContentTypeSurfaceCachedState {
95    fn default() -> Self {
96        Self {
97            content_type: wp_content_type_v1::Type::None,
98        }
99    }
100}
101
102impl Cacheable for ContentTypeSurfaceCachedState {
103    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
104        *self
105    }
106
107    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
108        *into = self;
109    }
110}

112#[derive(Debug)]
113struct ContentTypeSurfaceData {
114    is_resource_attached: AtomicBool,
115}
116
117impl ContentTypeSurfaceData {
118    fn new() -> Self {
119        Self {
120            is_resource_attached: AtomicBool::new(false),
121        }
122    }
123
124    fn set_is_resource_attached(&self, is_attached: bool) {
125        self.is_resource_attached
126            .store(is_attached, atomic::Ordering::Release)
127    }
128
129    fn is_resource_attached(&self) -> bool {
130        self.is_resource_attached.load(atomic::Ordering::Acquire)
131    }
132}

134/// User data of `WpContentTypeV1` object
135#[derive(Debug)]
136pub struct ContentTypeUserData(Mutex<Weak<WlSurface>>);
137
138impl ContentTypeUserData {
139    fn new(surface: WlSurface) -> Self {
140        Self(Mutex::new(surface.downgrade()))
141    }
142
143    #[inline]
144    fn wl_surface(&self) -> Option<WlSurface> {
145        self.0.lock().unwrap().upgrade().ok()
146    }
147}

149/// Delegate type for [WpContentTypeManagerV1] global.
150#[derive(Debug)]
151pub struct ContentTypeState {
152    global: GlobalId,
153}
154
155impl ContentTypeState {
156    /// Regiseter new [WpContentTypeManagerV1] global
157    pub fn new<D>(display: &DisplayHandle) -> ContentTypeState
158    where
159        D: GlobalDispatch<WpContentTypeManagerV1, GlobalData> + 'static,
160    {
161        let global = display.create_global::<D, WpContentTypeManagerV1, _>(1, GlobalData);
162
163        ContentTypeState { global }
164    }
165
166    /// Returns the WpContentTypeManagerV1 global id
167    pub fn global(&self) -> GlobalId {
168        self.global.clone()
169    }
