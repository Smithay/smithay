1//! Automatic handling of the `wlr_data_control` protocol
2//!
3//! ## Initialization
4//!
5//! To initialize this implementation, create [`DataControlState`], store it in your `State`
6//! struct, and implement the required trait, as shown in the example:
7//!
8//! ```
9//! # extern crate wayland_server;
10//! # #[macro_use] extern crate smithay;
11//! use smithay::wayland::selection::SelectionHandler;
12//! use smithay::wayland::selection::wlr_data_control::{DataControlState, DataControlHandler};
13//! # use smithay::input::{Seat, SeatHandler, SeatState, pointer::CursorImageStatus};
14//! # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
15//!
16//! # struct State { data_control_state: DataControlState }
17//! # let mut display = wayland_server::Display::<State>::new().unwrap();
18//! // Create the data_control state
19//! let data_control_state = DataControlState::new::<State, _>(
20//!     &display.handle(), None, |_| true
21//! );
22//!
23//! // insert the DataControlState into your state
24//! // ..
25//!
26//! // implement the necessary traits
27//! # impl SeatHandler for State {
28//! #     type KeyboardFocus = WlSurface;
29//! #     type PointerFocus = WlSurface;
30//! #     type TouchFocus = WlSurface;
31//! #     fn seat_state(&mut self) -> &mut SeatState<Self> { unimplemented!() }
32//! #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
33//! #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
34//! # }
35//! impl SelectionHandler for State {
36//!     type SelectionUserData = ();
37//! }
38//! impl DataControlHandler for State {
39//!     fn data_control_state(&mut self) -> &mut DataControlState { &mut self.data_control_state }
40//!     // ... override default implementations here to customize handling ...
41//! }
42//! delegate_data_control!(State);
43//!
44//! // You're now ready to go!
45//! ```
46//!
47//! Be aware that data control clients rely on other selection providers to be implemneted, like
48//! wl_data_device or zwp_primary_selection.

68/// Access the data control state.
69pub trait DataControlHandler: Sized + SelectionHandler {
70    /// [`DataControlState`] getter.
71    fn data_control_state(&mut self) -> &mut DataControlState;
72}
73
74/// State of the data control.
75#[derive(Debug)]
76pub struct DataControlState {
77    manager_global: GlobalId,
78    /// Used sources.
79    ///
80    /// Protocol states that each source can only be used once. We
81    /// also use it during destruction to get seat data.
82    pub(crate) used_sources: HashMap<ZwlrDataControlSourceV1, WlSeat>,
83}
84
85impl DataControlState {
86    /// Register new [ZwlrDataControlManagerV1] global.
87    ///
88    /// Passing `primary_selection` will enable support for primary selection as well.
89    pub fn new<D, F>(
90        display: &DisplayHandle,
91        primary_selection: Option<&PrimarySelectionState>,
92        filter: F,
93    ) -> Self
94    where
95        D: GlobalDispatch<ZwlrDataControlManagerV1, DataControlManagerGlobalData> + 'static,
96        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
97    {
98        let data = DataControlManagerGlobalData {
99            primary_selection_filter: primary_selection
100                .map(|x| Arc::clone(&x.filter))
101                .unwrap_or_else(|| Arc::new(Box::new(|_| false))),
102            filter: Box::new(filter),
103        };
104        let manager_global = display.create_global::<D, ZwlrDataControlManagerV1, _>(2, data);
105        Self {
106            manager_global,
107            used_sources: Default::default(),
108        }
109    }
110
111    /// [ZwlrDataControlManagerV1]  GlobalId getter.
112    pub fn global(&self) -> GlobalId {
113        self.manager_global.clone()
114    }
115}

158    impl<D> GlobalDispatch<ZwlrDataControlManagerV1, DataControlManagerGlobalData, D> for DataControlState
159    where
160        D: GlobalDispatch<ZwlrDataControlManagerV1, DataControlManagerGlobalData>,
161        D: Dispatch<ZwlrDataControlManagerV1, DataControlManagerUserData>,
162        D: Dispatch<ZwlrDataControlDeviceV1, DataControlDeviceUserData>,
163        D: Dispatch<ZwlrDataControlSourceV1, DataControlSourceUserData>,
164        D: DataControlHandler,
165        D: 'static,
166    {
167        fn bind(
168            _state: &mut D,
169            _handle: &DisplayHandle,
170            _client: &wayland_server::Client,
171            resource: wayland_server::New<ZwlrDataControlManagerV1>,
172            global_data: &DataControlManagerGlobalData,
173            data_init: &mut wayland_server::DataInit<'_, D>,
174        ) {
175            data_init.init(
176                resource,
177                DataControlManagerUserData {
178                    primary_selection_filter: Arc::clone(&global_data.primary_selection_filter),
179                },
180            );
181        }
182
183        fn can_view(client: Client, global_data: &DataControlManagerGlobalData) -> bool {
184            (global_data.filter)(&client)
185        }
186    }
187
188    impl<D> Dispatch<ZwlrDataControlManagerV1, DataControlManagerUserData, D> for DataControlState
189    where
190        D: Dispatch<ZwlrDataControlManagerV1, DataControlManagerUserData>,
191        D: Dispatch<ZwlrDataControlDeviceV1, DataControlDeviceUserData>,
192        D: Dispatch<ZwlrDataControlSourceV1, DataControlSourceUserData>,
193        D: DataControlHandler,
194        D: 'static,
195    {
196        fn request(
197            _handler: &mut D,
198            client: &wayland_server::Client,
199            _resource: &ZwlrDataControlManagerV1,
200            request: <ZwlrDataControlManagerV1 as wayland_server::Resource>::Request,
201            data: &DataControlManagerUserData,
202            dh: &DisplayHandle,
203            data_init: &mut wayland_server::DataInit<'_, D>,
204        ) {
205            match request {
206                zwlr_data_control_manager_v1::Request::CreateDataSource { id } => {
207                    data_init.init(id, DataControlSourceUserData::new(dh.clone()));
208                }
209                zwlr_data_control_manager_v1::Request::GetDataDevice { id, seat: wl_seat } => {
210                    match Seat::<D>::from_resource(&wl_seat) {
211                        Some(seat) => {
212                            seat.user_data()
213                                .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
214
215                            let device = SelectionDevice::WlrDataControl(data_init.init(
216                                id,
217                                DataControlDeviceUserData {
218                                    wl_seat,
219                                    primary_selection_filter: Arc::clone(&data.primary_selection_filter),
220                                },
221                            ));
222
223                            let mut seat_data = seat
224                                .user_data()
225                                .get::<RefCell<SeatData<D::SelectionUserData>>>()
226                                .unwrap()
227                                .borrow_mut();
228
229                            seat_data.add_device(device.clone());
230
231                            // NOTE: broadcast selection only to the newly created device.
232                            let device = Some(&device);
233                            seat_data.send_selection::<D>(dh, SelectionTarget::Clipboard, device, true);
234                            if (*data.primary_selection_filter)(client) {
235                                seat_data.send_selection::<D>(dh, SelectionTarget::Primary, device, true);
236                            }
237                        }
238                        None => {
239                            error!(
240                                data_control_device = ?id,
241                                client = ?client,
242                                "Unmanaged seat given to a primary selection device."
243                            );
244                        }
245                    }
246                }
247                zwlr_data_control_manager_v1::Request::Destroy => (),
248                _ => unreachable!(),
249            }
250        }
251    }
252}
