42impl<D> Dispatch<ExtDataControlSourceV1, ExtDataControlSourceUserData, D> for DataControlState
43where
44    D: Dispatch<ExtDataControlSourceV1, ExtDataControlSourceUserData>,
45    D: DataControlHandler,
46    D: 'static,
47{
48    fn request(
49        _state: &mut D,
50        _client: &wayland_server::Client,
51        _resource: &ExtDataControlSourceV1,
52        request: <ExtDataControlSourceV1 as wayland_server::Resource>::Request,
53        data: &ExtDataControlSourceUserData,
54        _dhandle: &DisplayHandle,
55        _data_init: &mut wayland_server::DataInit<'_, D>,
56    ) {
57        match request {
58            ext_data_control_source_v1::Request::Offer { mime_type } => {
59                let mut data = data.inner.lock().unwrap();
60                data.mime_types.push(mime_type);
61            }
62            ext_data_control_source_v1::Request::Destroy => (),
63            _ => unreachable!(),
64        }
65    }
66
67    fn destroyed(
68        state: &mut D,
69        _client: ClientId,
70        source: &ExtDataControlSourceV1,
71        data: &ExtDataControlSourceUserData,
72    ) {
73        // Remove the source from the used ones.
74        let seat = match state
75            .data_control_state()
76            .used_sources
77            .remove(source)
78            .as_ref()
79            .and_then(Seat::<D>::from_resource)
80        {
81            Some(seat) => seat,
82            None => return,
83        };
84
85        let mut seat_data = seat
86            .user_data()
87            .get::<RefCell<SeatData<D::SelectionUserData>>>()
88            .unwrap()
89            .borrow_mut();
90
91        for target in [SelectionTarget::Primary, SelectionTarget::Clipboard] {
92            let selection = match target {
93                SelectionTarget::Primary => seat_data.get_primary_selection(),
94                SelectionTarget::Clipboard => seat_data.get_clipboard_selection(),
95            };
96
97            match selection {
98                Some(OfferReplySource::Client(SelectionSourceProvider::ExtDataControl(set_source)))
99                    if set_source == source =>
100                {
101                    match target {
102                        SelectionTarget::Primary => {
103                            seat_data.set_primary_selection::<D>(&data.display_handle, None)
104                        }
105                        SelectionTarget::Clipboard => {
106                            seat_data.set_clipboard_selection::<D>(&data.display_handle, None)
107                        }
108                    }
109                }
110                _ => (),
111            };
112        }
113    }
114}
