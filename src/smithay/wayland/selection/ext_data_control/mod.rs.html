1//! Automatic handling of the `ext_data_control` protocol
2//!
3//! ## Initialization
4//!
5//! To initialize this implementation, create [`DataControlState`], store it in your `State`
6//! struct, and implement the required trait, as shown in the example:
7//!
8//! ```
9//! # extern crate wayland_server;
10//! # #[macro_use] extern crate smithay;
11//! use smithay::wayland::selection::SelectionHandler;
12//! use smithay::wayland::selection::ext_data_control::{DataControlState, DataControlHandler};
13//! # use smithay::input::{Seat, SeatHandler, SeatState, pointer::CursorImageStatus};
14//! # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
15//!
16//! # struct State { data_control_state: DataControlState }
17//! # let mut display = wayland_server::Display::<State>::new().unwrap();
18//! // Create the data_control state
19//! let data_control_state = DataControlState::new::<State, _>(
20//!     &display.handle(), None, |_| true
21//! );
22//!
23//! // insert the DataControlState into your state
24//! // ..
25//!
26//! // implement the necessary traits
27//! # impl SeatHandler for State {
28//! #     type KeyboardFocus = WlSurface;
29//! #     type PointerFocus = WlSurface;
30//! #     type TouchFocus = WlSurface;
31//! #     fn seat_state(&mut self) -> &mut SeatState<Self> { unimplemented!() }
32//! #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
33//! #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
34//! # }
35//! impl SelectionHandler for State {
36//!     type SelectionUserData = ();
37//! }
38//! impl DataControlHandler for State {
39//!     fn data_control_state(&mut self) -> &mut DataControlState { &mut self.data_control_state }
40//!     // ... override default implementations here to customize handling ...
41//! }
42//! delegate_ext_data_control!(State);
43//!
44//! // You're now ready to go!
45//! ```
46//!
47//! Be aware that data control clients rely on other selection providers to be implemneted, like
48//! wl_data_device or zwp_primary_selection.

73/// State of the data control.
74#[derive(Debug)]
75pub struct DataControlState {
76    manager_global: GlobalId,
77    /// Used sources.
78    ///
79    /// Protocol states that each source can only be used once. We
80    /// also use it during destruction to get seat data.
81    pub(crate) used_sources: HashMap<ExtDataControlSourceV1, WlSeat>,
82}
83
84impl DataControlState {
85    /// Register new [ExtDataControlManagerV1] global.
86    ///
87    /// Passing `primary_selection` will enable support for primary selection as well.
88    pub fn new<D, F>(
89        display: &DisplayHandle,
90        primary_selection: Option<&PrimarySelectionState>,
91        filter: F,
92    ) -> Self
93    where
94        D: GlobalDispatch<ExtDataControlManagerV1, ExtDataControlManagerGlobalData> + 'static,
95        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
96    {
97        let data = ExtDataControlManagerGlobalData {
98            primary: primary_selection.is_some(),
99            filter: Box::new(filter),
100        };
101        let manager_global = display.create_global::<D, ExtDataControlManagerV1, _>(1, data);
102        Self {
103            manager_global,
104            used_sources: Default::default(),
105        }
106    }
107
108    /// [ExtDataControlManagerV1]  GlobalId getter.
109    pub fn global(&self) -> GlobalId {
110        self.manager_global.clone()
111    }
112}

154    impl<D> GlobalDispatch<ExtDataControlManagerV1, ExtDataControlManagerGlobalData, D> for DataControlState
155    where
156        D: GlobalDispatch<ExtDataControlManagerV1, ExtDataControlManagerGlobalData>,
157        D: Dispatch<ExtDataControlManagerV1, ExtDataControlManagerUserData>,
158        D: Dispatch<ExtDataControlDeviceV1, ExtDataControlDeviceUserData>,
159        D: Dispatch<ExtDataControlSourceV1, ExtDataControlSourceUserData>,
160        D: DataControlHandler,
161        D: 'static,
162    {
163        fn bind(
164            _state: &mut D,
165            _handle: &DisplayHandle,
166            _client: &wayland_server::Client,
167            resource: wayland_server::New<ExtDataControlManagerV1>,
168            global_data: &ExtDataControlManagerGlobalData,
169            data_init: &mut wayland_server::DataInit<'_, D>,
170        ) {
171            data_init.init(
172                resource,
173                ExtDataControlManagerUserData {
174                    primary: global_data.primary,
175                },
176            );
177        }
178
179        fn can_view(client: Client, global_data: &ExtDataControlManagerGlobalData) -> bool {
180            (global_data.filter)(&client)
181        }
182    }
183
184    impl<D> Dispatch<ExtDataControlManagerV1, ExtDataControlManagerUserData, D> for DataControlState
185    where
186        D: Dispatch<ExtDataControlManagerV1, ExtDataControlManagerUserData>,
187        D: Dispatch<ExtDataControlDeviceV1, ExtDataControlDeviceUserData>,
188        D: Dispatch<ExtDataControlSourceV1, ExtDataControlSourceUserData>,
189        D: DataControlHandler,
190        D: 'static,
191    {
192        fn request(
193            _handler: &mut D,
194            client: &wayland_server::Client,
195            _resource: &ExtDataControlManagerV1,
196            request: <ExtDataControlManagerV1 as wayland_server::Resource>::Request,
197            data: &ExtDataControlManagerUserData,
198            dh: &DisplayHandle,
199            data_init: &mut wayland_server::DataInit<'_, D>,
200        ) {
201            match request {
202                ext_data_control_manager_v1::Request::CreateDataSource { id } => {
203                    data_init.init(id, ExtDataControlSourceUserData::new(dh.clone()));
204                }
205                ext_data_control_manager_v1::Request::GetDataDevice { id, seat: wl_seat } => {
206                    match Seat::<D>::from_resource(&wl_seat) {
207                        Some(seat) => {
208                            seat.user_data()
209                                .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
210
211                            let device = SelectionDevice::ExtDataControl(data_init.init(
212                                id,
213                                ExtDataControlDeviceUserData {
214                                    wl_seat,
215                                    primary: data.primary,
216                                },
217                            ));
218
219                            let mut seat_data = seat
220                                .user_data()
221                                .get::<RefCell<SeatData<D::SelectionUserData>>>()
222                                .unwrap()
223                                .borrow_mut();
224
225                            seat_data.add_device(device.clone());
226
227                            // NOTE: broadcast selection only to the newly created device.
228                            let device = Some(&device);
229                            seat_data.send_selection::<D>(dh, SelectionTarget::Clipboard, device, true);
230                            if data.primary {
231                                seat_data.send_selection::<D>(dh, SelectionTarget::Primary, device, true);
232                            }
233                        }
234                        None => {
235                            error!(
236                                data_control_device = ?id,
237                                client = ?client,
238                                "Unmanaged seat given to a primary selection device."
239                            );
240                        }
241                    }
242                }
243                ext_data_control_manager_v1::Request::Destroy => (),
244                _ => unreachable!(),
245            }
246        }
247    }
248}

254        $crate::reexports::wayland_server::delegate_global_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
255            $crate::reexports::wayland_protocols::ext::data_control::v1::server::ext_data_control_manager_v1::ExtDataControlManagerV1: $crate::wayland::selection::ext_data_control::ExtDataControlManagerGlobalData
256        ] => $crate::wayland::selection::ext_data_control::DataControlState);
257        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
258            $crate::reexports::wayland_protocols::ext::data_control::v1::server::ext_data_control_manager_v1::ExtDataControlManagerV1: $crate::wayland::selection::ext_data_control::ExtDataControlManagerUserData
259        ] => $crate::wayland::selection::ext_data_control::DataControlState);
260        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
