19/// The metadata describing a data source
20#[derive(Debug, Clone)]
21pub struct SourceMetadata {
22    /// The MIME types supported by this source
23    pub mime_types: Vec<String>,
24    /// The Drag'n'Drop actions supported by this source
25    pub dnd_action: DndAction,
26}
27
28impl Default for SourceMetadata {
29    fn default() -> Self {
30        Self {
31            mime_types: Vec::new(),
32            dnd_action: DndAction::None,
33        }
34    }
35}

37#[doc(hidden)]
38#[derive(Debug)]
39pub struct DataSourceUserData {

55impl<D> Dispatch<WlDataSource, DataSourceUserData, D> for DataDeviceState
56where
57    D: Dispatch<WlDataSource, DataSourceUserData>,
58    D: DataDeviceHandler,
59    D: 'static,
60{
61    fn request(
62        _state: &mut D,
63        _client: &wayland_server::Client,
64        _resource: &WlDataSource,
65        request: wl_data_source::Request,
66        data: &DataSourceUserData,
67        _dhandle: &DisplayHandle,
68        _data_init: &mut wayland_server::DataInit<'_, D>,
69    ) {
70        let mut data = data.inner.lock().unwrap();
71
72        match request {
73            wl_data_source::Request::Offer { mime_type } => {
74                data.mime_types.push(mime_type);
75            }
76            wl_data_source::Request::SetActions { dnd_actions } => match dnd_actions {
77                wayland_server::WEnum::Value(dnd_actions) => {
78                    data.dnd_action = dnd_actions;
79                }
80                wayland_server::WEnum::Unknown(action) => {
81                    error!("Unknown dnd_action: {:?}", action);
82                }
83            },
84            wl_data_source::Request::Destroy => {}
85            _ => unreachable!(),
86        }
87    }
88
89    fn destroyed(state: &mut D, _client: ClientId, source: &WlDataSource, data: &DataSourceUserData) {
90        data.alive_tracker.destroy_notify();
91
92        // Remove the source from the used ones.
93        let seat = match state
94            .data_device_state()
95            .used_sources
96            .remove(source)
97            .as_ref()
98            .and_then(Seat::<D>::from_resource)
99        {
100            Some(seat) => seat,
101            None => return,
102        };
103
104        let mut seat_data = seat
105            .user_data()
106            .get::<RefCell<SeatData<D::SelectionUserData>>>()
107            .unwrap()
108            .borrow_mut();
109
110        match seat_data.get_clipboard_selection() {
111            Some(OfferReplySource::Client(SelectionSourceProvider::DataDevice(set_source)))
112                if set_source == source =>
113            {
114                seat_data.set_clipboard_selection::<D>(&data.display_handle, None)
115            }
116            _ => (),
117        }
118    }
119}

121impl IsAlive for WlDataSource {
122    #[inline]
123    fn alive(&self) -> bool {
124        let data: &DataSourceUserData = self.data().unwrap();
125        data.alive_tracker.alive()
126    }

129/// Access the metadata of a data source
130pub fn with_source_metadata<T, F: FnOnce(&SourceMetadata) -> T>(
131    source: &WlDataSource,
132    f: F,
133) -> Result<T, crate::utils::UnmanagedResource> {
134    match source.data::<DataSourceUserData>() {
135        Some(data) => Ok(f(&data.inner.lock().unwrap())),
136        None => Err(crate::utils::UnmanagedResource),
137    }
