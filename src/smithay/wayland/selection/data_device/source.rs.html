47impl<D> Dispatch<WlDataSource, DataSourceUserData, D> for DataDeviceState
48where
49    D: Dispatch<WlDataSource, DataSourceUserData>,
50    D: DataDeviceHandler,
51    D: 'static,
52{
53    fn request(
54        _state: &mut D,
55        _client: &wayland_server::Client,
56        _resource: &WlDataSource,
57        request: wl_data_source::Request,
58        data: &DataSourceUserData,
59        _dhandle: &DisplayHandle,
60        _data_init: &mut wayland_server::DataInit<'_, D>,
61    ) {
62        let mut data = data.inner.lock().unwrap();
63
64        match request {
65            wl_data_source::Request::Offer { mime_type } => {
66                data.mime_types.push(mime_type);
67            }
68            wl_data_source::Request::SetActions { dnd_actions } => match dnd_actions {
69                wayland_server::WEnum::Value(dnd_actions) => {
70                    data.dnd_actions = DndAction::vec_from_wl(dnd_actions);
71                }
72                wayland_server::WEnum::Unknown(action) => {
73                    error!("Unknown dnd_action: {:?}", action);
74                }
75            },
76            wl_data_source::Request::Destroy => {}
77            _ => unreachable!(),
78        }
79    }
80
81    fn destroyed(state: &mut D, _client: ClientId, source: &WlDataSource, data: &DataSourceUserData) {
82        data.alive_tracker.destroy_notify();
83
84        // Remove the source from the used ones.
85        let seat = match state
86            .data_device_state()
87            .used_sources
88            .remove(source)
89            .as_ref()
90            .and_then(Seat::<D>::from_resource)
91        {
92            Some(seat) => seat,
93            None => return,
94        };
95
96        let mut seat_data = seat
97            .user_data()
98            .get::<RefCell<SeatData<D::SelectionUserData>>>()
99            .unwrap()
100            .borrow_mut();
101
102        match seat_data.get_clipboard_selection() {
103            Some(OfferReplySource::Client(SelectionSourceProvider::DataDevice(set_source)))
104                if set_source == source =>
105            {
106                seat_data.set_clipboard_selection::<D>(&data.display_handle, None)
107            }
108            _ => (),
109        }
110    }
111}

113impl IsAlive for WlDataSource {

121impl Source for WlDataSource {
122    fn metadata(&self) -> Option<SourceMetadata> {
123        self.data::<DataSourceUserData>()
124            .map(|data| data.inner.lock().unwrap().clone())
125    }
126
127    fn choose_action(&self, action: DndAction) {
128        self.action(action.into());
129    }
130
131    fn send(&self, mime_type: &str, fd: OwnedFd) {
132        debug!(?mime_type, "DnD transfer request");
133        self.send(mime_type.to_owned(), fd.as_fd());
134    }
135
136    fn drop_performed(&self) {
137        if self.version() >= wl_data_source::EVT_DND_DROP_PERFORMED_SINCE {
138            self.dnd_drop_performed();
139        }
140    }
141
142    fn cancel(&self) {
143        self.cancelled();
144    }
145
146    fn finished(&self) {
147        if self.version() >= wl_data_source::EVT_DND_FINISHED_SINCE {
148            self.dnd_finished();
149        }
150    }
151}

158    }

164    fn choose_action(&self, action: DndAction) {
165        let _ = action;

168    fn send(&self, mime_type: &str, fd: OwnedFd) {
169        let _ = (mime_type, fd);

171    }
