32#[derive(Debug, Clone)]
33pub enum OfferReplySource<U: Clone + Send + Sync + 'static> {
34    /// The selection is backend by client source.
35    Client(SelectionSourceProvider),
36    /// The selection is backed by the compositor.
37    Compositor(CompositorSelectionProvider<U>),
38}
39
40impl<U: Clone + Send + Sync + 'static> OfferReplySource<U> {
41    /// Get the copy of the underlying mime types.
42    pub fn mime_types(&self) -> Vec<String> {
43        match self {
44            OfferReplySource::Client(source) => source.mime_types(),
45            OfferReplySource::Compositor(source) => source.mime_types.clone(),
46        }
47    }
48
49    /// Check whether the source contains the given `mime_type`.
50    pub fn contains_mime_type(&self, mime_type: &String) -> bool {
51        match self {
52            OfferReplySource::Client(source) => source.contains_mime_type(mime_type),
53            OfferReplySource::Compositor(source) => source.mime_types.contains(mime_type),
54        }
55    }
56}

58/// Offer representing various selection offers.
59#[derive(Debug, Clone, PartialEq, Eq)]
60pub(crate) enum SelectionOffer {
61    DataDevice(WlDataOffer),
62    Primary(PrimaryOffer),
63    WlrDataControl(ZwlrDataControlOfferV1),
64    ExtDataControl(ExtDataControlOfferV1),
65}
66
67impl SelectionOffer {
68    pub fn new<D>(
69        dh: &DisplayHandle,
70        device: &SelectionDevice,
71        client_id: ClientId,
72        data: OfferReplySource<D::SelectionUserData>,
73    ) -> Self
74    where
75        D: SelectionHandler + 'static,
76    {
77        // NOTE: the types are tied to the `SelectionDevice`, so every
78        // RTTI like checking is safe and reliable.
79
80        let device_kind = device.device_kind();
81        let data = Arc::new(OfferReplyData {
82            device_kind,
83            seat: device.seat(),
84            source: data,
85        });
86        let backend = dh.backend_handle();
87
88        let interface = match device_kind {
89            DataDeviceKind::Core => WlDataOffer::interface(),
90            DataDeviceKind::Primary => PrimaryOffer::interface(),
91            DataDeviceKind::WlrDataControl => ZwlrDataControlOfferV1::interface(),
92            DataDeviceKind::ExtDataControl => ExtDataControlOfferV1::interface(),
93        };
94
95        let offer = backend
96            .create_object::<D>(client_id, interface, device.version(), data)
97            .unwrap();
98
99        match device_kind {
100            DataDeviceKind::Core => Self::DataDevice(WlDataOffer::from_id(dh, offer).unwrap()),
101            DataDeviceKind::Primary => Self::Primary(PrimaryOffer::from_id(dh, offer).unwrap()),
102            DataDeviceKind::WlrDataControl => {
103                Self::WlrDataControl(ZwlrDataControlOfferV1::from_id(dh, offer).unwrap())
104            }
105            DataDeviceKind::ExtDataControl => {
106                Self::ExtDataControl(ExtDataControlOfferV1::from_id(dh, offer).unwrap())
107            }
108        }
109    }
110
111    pub fn offer(&self, mime_type: String) {
112        selection_dispatch!(self; Self(offer) => offer.offer(mime_type))
113    }
114}

116struct OfferReplyData<U: Clone + Send + Sync + 'static> {
117    device_kind: DataDeviceKind,
118    source: OfferReplySource<U>,
119    seat: WlSeat,
120}
121
122impl<D> ObjectData<D> for OfferReplyData<D::SelectionUserData>
123where
124    D: SelectionHandler + 'static,
125{
126    fn request(
127        self: Arc<Self>,
128        dh: &Handle,
129        handle: &mut D,
130        _: ClientId,
131        msg: Message<ObjectId, OwnedFd>,
132    ) -> Option<Arc<dyn ObjectData<D>>> {
133        let dh = DisplayHandle::from(dh.clone());
134
135        // NOTE: we can't parse message more than once, since it expects the `OwnedFd` which
136        // we can't clone. To achieve that, we use RTTI passed along the selection data, to
137        // make the parsing work only once.
138        let (mime_type, fd, object_name) = match self.device_kind {
139            DataDeviceKind::Core => {
140                if let Ok((_resource, DataOfferRequest::Receive { mime_type, fd })) =
141                    WlDataOffer::parse_request(&dh, msg)
142                {
143                    (mime_type, fd, "wl_data_offer")
144                } else {
145                    return None;
146                }
147            }
148            DataDeviceKind::Primary => {
149                if let Ok((_resource, PrimaryRequest::Receive { mime_type, fd })) =
150                    PrimaryOffer::parse_request(&dh, msg)
151                {
152                    (mime_type, fd, "primary_selection_offer")
153                } else {
154                    return None;
155                }
156            }
157            DataDeviceKind::WlrDataControl => {
158                if let Ok((_resource, DataControlRequest::Receive { mime_type, fd })) =
159                    ZwlrDataControlOfferV1::parse_request(&dh, msg)
160                {
161                    (mime_type, fd, "wlr_data_control_offer")
162                } else {
163                    return None;
164                }
165            }
166            DataDeviceKind::ExtDataControl => {
167                if let Ok((_resource, ext_data_control_offer_v1::Request::Receive { mime_type, fd })) =
168                    ExtDataControlOfferV1::parse_request(&dh, msg)
169                {
170                    (mime_type, fd, "ext_data_control_offer")
171                } else {
172                    return None;
173                }
174            }
175        };
176
177        if !self.source.contains_mime_type(&mime_type) {
178            debug!("Denying a {object_name}.receive with invalid mime type");
179        } else {
180            match &self.source {
181                OfferReplySource::Client(source) => {
182                    source.send(mime_type, fd);
183                }
184                OfferReplySource::Compositor(source) => {
185                    if let Some(seat) = Seat::<D>::from_resource(&self.seat) {
186                        handle.send_selection(source.ty, mime_type, fd, seat, &source.user_data);
187                    }
188                }
189            }
190        }
191
192        None
193    }
194
195    fn destroyed(self: Arc<Self>, _: &Handle, _: &mut D, _: ClientId, _: ObjectId) {}
