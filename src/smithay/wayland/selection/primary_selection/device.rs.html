25#[doc(hidden)]
26#[derive(Debug)]
27pub struct PrimaryDeviceUserData {
28    pub(crate) wl_seat: WlSeat,
29}
30
31impl<D> Dispatch<PrimaryDevice, PrimaryDeviceUserData, D> for PrimarySelectionState
32where
33    D: Dispatch<PrimaryDevice, PrimaryDeviceUserData> + PrimarySelectionHandler,
34    D: SelectionHandler,
35    D: SeatHandler,
36    <D as SeatHandler>::KeyboardFocus: WaylandFocus,
37    D: 'static,
38{
39    fn request(
40        handler: &mut D,
41        client: &Client,
42        resource: &PrimaryDevice,
43        request: primary_device::Request,
44        data: &PrimaryDeviceUserData,
45        dh: &DisplayHandle,
46        _data_init: &mut DataInit<'_, D>,
47    ) {
48        let seat = match Seat::<D>::from_resource(&data.wl_seat) {
49            Some(seat) => seat,
50            None => return,
51        };
52
53        match request {
54            primary_device::Request::SetSelection { source, .. } => {
55                let seat_data = match seat.get_keyboard() {
56                    Some(keyboard) if keyboard.client_of_object_has_focus(&resource.id()) => seat
57                        .user_data()
58                        .get::<RefCell<SeatData<D::SelectionUserData>>>()
59                        .unwrap(),
60                    _ => {
61                        debug!(
62                            client = ?client,
63                            "denying setting selection by a non-focused client"
64                        );
65                        return;
66                    }
67                };
68
69                // NOTE: While protocol states that selection shouldn't be used more than once,
70                // no-one enforces it, thus we have clients around that do so and crashing them
71                // doesn't worth it at this point.
72                if let Some(source) = source.as_ref() {
73                    handler
74                        .primary_selection_state()
75                        .used_sources
76                        .insert(source.clone(), data.wl_seat.clone());
77                }
78
79                let source = source.map(SelectionSourceProvider::Primary);
80
81                handler.new_selection(
82                    SelectionTarget::Primary,
83                    source.clone().map(|provider| SelectionSource { provider }),
84                    seat.clone(),
85                );
86
87                // The client has kbd focus, it can set the selection
88                seat_data
89                    .borrow_mut()
90                    .set_primary_selection::<D>(dh, source.map(OfferReplySource::Client));
91            }
92            primary_device::Request::Destroy => seat
93                .user_data()
94                .get::<RefCell<SeatData<D::SelectionUserData>>>()
95                .unwrap()
96                .borrow_mut()
97                .retain_devices(|ndd| match ndd {
98                    SelectionDevice::Primary(ndd) => ndd != resource,
99                    _ => true,
100                }),
101            _ => unreachable!(),
102        }
103    }
104}
