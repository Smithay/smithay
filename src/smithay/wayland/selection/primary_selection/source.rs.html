15/// The metadata describing a data source
16#[derive(Debug, Default, Clone)]
17pub struct SourceMetadata {
18    /// The MIME types supported by this source
19    pub mime_types: Vec<String>,
20}

22#[doc(hidden)]
23#[derive(Debug)]
24pub struct PrimarySourceUserData {
25    pub(crate) inner: Mutex<SourceMetadata>,
26    display_handle: DisplayHandle,
27}
28
29impl PrimarySourceUserData {
30    pub(super) fn new(display_handle: DisplayHandle) -> Self {
31        Self {
32            inner: Default::default(),
33            display_handle,
34        }
35    }
36}

38impl<D> Dispatch<PrimarySource, PrimarySourceUserData, D> for PrimarySelectionState
39where
40    D: Dispatch<PrimarySource, PrimarySourceUserData>,
41    D: PrimarySelectionHandler,
42    D: 'static,
43{
44    fn request(
45        state: &mut D,
46        _client: &wayland_server::Client,
47        _resource: &PrimarySource,
48        request: primary_source::Request,
49        data: &PrimarySourceUserData,
50        _dhandle: &DisplayHandle,
51        _data_init: &mut wayland_server::DataInit<'_, D>,
52    ) {
53        let _primary_selection_state = state.primary_selection_state();
54        let mut data = data.inner.lock().unwrap();
55
56        match request {
57            primary_source::Request::Offer { mime_type } => {
58                data.mime_types.push(mime_type);
59            }
60            primary_source::Request::Destroy => {}
61            _ => unreachable!(),
62        }
63    }
64
65    fn destroyed(state: &mut D, _client: ClientId, source: &PrimarySource, data: &PrimarySourceUserData) {
66        // Remove the source from the used ones.
67        let seat = match state
68            .primary_selection_state()
69            .used_sources
70            .remove(source)
71            .as_ref()
72            .and_then(Seat::<D>::from_resource)
73        {
74            Some(seat) => seat,
75            None => return,
76        };
77
78        let mut seat_data = seat
79            .user_data()
80            .get::<RefCell<SeatData<D::SelectionUserData>>>()
81            .unwrap()
82            .borrow_mut();
83
84        match seat_data.get_primary_selection() {
85            Some(OfferReplySource::Client(SelectionSourceProvider::Primary(set_source)))
86                if set_source == source =>
87            {
88                seat_data.set_primary_selection::<D>(&data.display_handle, None)
89            }
90            _ => (),
91        }
92    }
93}
