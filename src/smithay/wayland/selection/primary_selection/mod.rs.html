1//! Utilities for manipulating the primary selection
2//!
3//! The primary selection is an additional protocol modeled after the data device to represent
4//! and additional selection (copy/paste), a concept taken from the X11 Server.
5//! This primary selection is a shortcut to the common clipboard selection,
6//! where text just needs to be selected in order to allow copying it elsewhere
7//! The de facto way to perform this action is the middle mouse button, although it is not limited to this one.
8//!
9//! This module provides the freestanding [`set_primary_focus`] function:
10//!   This function sets the data device focus for a given seat; you'd typically call it
11//!   whenever the keyboard focus changes, to follow it (for example in the focus hook of your keyboards).
12//!
13//! The module also provides an additional mechanism allowing your compositor to see and interact with
14//! the contents of the primary selection:
15//!
16//! - the freestanding function [`set_primary_selection`]
17//!   allows you to set the contents of the selection for your clients
18//! - the `PrimarySelectionHandle` gives you the option to inspect new selections
19//!   by overriding [`SelectionHandler::new_selection`].
20//!
21//! ## Initialization
22//!
23//! To initialize this implementation, create the [`PrimarySelectionState`], store it inside your `State` struct
24//! and implement the [`PrimarySelectionHandler`] and [`SelectionHandler`], as shown in this example:
25//!
26//! ```
27//! # extern crate wayland_server;
28//! # #[macro_use] extern crate smithay;
29//! use smithay::delegate_primary_selection;
30//! use smithay::wayland::selection::SelectionHandler;
31//! use smithay::wayland::selection::primary_selection::{PrimarySelectionState, PrimarySelectionHandler};
32//! # use smithay::input::{Seat, SeatHandler, SeatState, pointer::CursorImageStatus};
33//! # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
34//!
35//! # struct State { primary_selection_state: PrimarySelectionState }
36//! # let mut display = wayland_server::Display::<State>::new().unwrap();
37//! // Create the primary_selection state
38//! let primary_selection_state = PrimarySelectionState::new::<State>(
39//!     &display.handle(),
40//! );
41//!
42//! // insert the PrimarySelectionState into your state
43//! // ..
44//!
45//! // implement the necessary traits
46//! # impl SeatHandler for State {
47//! #     type KeyboardFocus = WlSurface;
48//! #     type PointerFocus = WlSurface;
49//! #     type TouchFocus = WlSurface;
50//! #     fn seat_state(&mut self) -> &mut SeatState<Self> { unimplemented!() }
51//! #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
52//! #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
53//! # }
54//! impl SelectionHandler for State {
55//!     type SelectionUserData = ();
56//! }
57//! impl PrimarySelectionHandler for State {
58//!     fn primary_selection_state(&mut self) -> &mut PrimarySelectionState { &mut self.primary_selection_state }
59//!     // ... override default implementations here to customize handling ...
60//! }
61//! delegate_primary_selection!(State);
62//!
63//! // You're now ready to go!
64//! ```

95/// Access the primary selection state.

97    /// [PrimarySelectionState] getter.
98    fn primary_selection_state(&mut self) -> &mut PrimarySelectionState;

101/// State of the primary selection.
102pub struct PrimarySelectionState {
103    manager_global: GlobalId,
104    pub(super) filter: Arc<Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>>,
105    /// Used sources.
106    ///
107    /// Protocol states that each source can only be used once. We
108    /// also use it during destruction to get seat data.
109    pub(crate) used_sources: HashMap<ZwpPrimarySelectionSourceV1, WlSeat>,
110}

112impl std::fmt::Debug for PrimarySelectionState {
113    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
114        f.debug_struct("PrimarySelectionState")
115            .field("manager_global", &self.manager_global)
116            .finish()
117    }
118}

120impl PrimarySelectionState {
121    /// Register new [`PrimaryDeviceManager`] global
122    pub fn new<D>(display: &DisplayHandle) -> Self
123    where
124        D: GlobalDispatch<PrimaryDeviceManager, PrimaryDeviceManagerGlobalData> + 'static,
125        D: PrimarySelectionHandler,
126    {
127        Self::new_with_filter::<D, _>(display, |_| true)
128    }
129
130    /// Register new [`PrimaryDeviceManager`] global with a filter.
131    ///
132    /// Filters can be used to limit visibility of a global to certain clients.
133    pub fn new_with_filter<D, F>(display: &DisplayHandle, filter: F) -> Self
134    where
135        D: GlobalDispatch<PrimaryDeviceManager, PrimaryDeviceManagerGlobalData> + 'static,
136        D: PrimarySelectionHandler,
137        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
138    {
139        let data = PrimaryDeviceManagerGlobalData {
140            filter: Arc::new(Box::new(filter)),
141        };
142        let filter = Arc::clone(&data.filter);
143        let manager_global = display.create_global::<D, PrimaryDeviceManager, _>(1, data);
144
145        Self {
146            manager_global,
147            filter,
148            used_sources: Default::default(),
149        }
150    }
151
152    /// [`PrimaryDeviceManager`] GlobalId getter
153    pub fn global(&self) -> GlobalId {
154        self.manager_global.clone()
155    }
156}

164/// Set the primary selection focus to a certain client for a given seat
165///
166/// If the focus is different from the existing focus, the current selection will be offered to the client.
167#[instrument(name = "wayland_primary_selection", level = "debug", skip(dh, seat, client), fields(seat = seat.name(), client = ?client.as_ref().map(|c| c.id())))]
168pub fn set_primary_focus<D>(dh: &DisplayHandle, seat: &Seat<D>, client: Option<Client>)
169where
170    D: SeatHandler + PrimarySelectionHandler + 'static,
171{
172    seat.user_data()
173        .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
174    let seat_data = seat
175        .user_data()
176        .get::<RefCell<SeatData<D::SelectionUserData>>>()
177        .unwrap();
178    seat_data.borrow_mut().set_primary_focus::<D>(dh, client);
179}

181/// Set a compositor-provided primary selection for this seat
182///
183/// You need to provide the available mime types for this selection.
184///
185/// Whenever a client requests to read the selection, your callback will
186/// receive a [`SelectionHandler::send_selection`] event.
187#[instrument(name = "wayland_primary_selection", level = "debug", skip(dh, seat, user_data), fields(seat = seat.name()))]
188pub fn set_primary_selection<D>(
189    dh: &DisplayHandle,
190    seat: &Seat<D>,
191    mime_types: Vec<String>,
192    user_data: D::SelectionUserData,
193) where
194    D: SeatHandler + PrimarySelectionHandler + 'static,
195{
196    seat.user_data()
197        .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
198    let seat_data = seat
199        .user_data()
200        .get::<RefCell<SeatData<D::SelectionUserData>>>()
201        .unwrap();
202
203    let selection = OfferReplySource::Compositor(CompositorSelectionProvider {
204        ty: SelectionTarget::Primary,
205        mime_types,
206        user_data,
207    });
208
209    seat_data
210        .borrow_mut()
211        .set_primary_selection::<D>(dh, Some(selection));
212}

214/// Errors happening when requesting selection contents
215#[derive(Debug, thiserror::Error)]
216pub enum SelectionRequestError {
217    /// Requested mime type is not available
218    #[error("Requested mime type is not available")]
219    InvalidMimetype,
220    /// Requesting server side selection contents is not supported
221    #[error("Current selection is server-side")]
222    ServerSideSelection,
223    /// There is no active selection
224    #[error("No active selection to query")]
225    NoSelection,
226}

228/// Request the current primary selection of the given seat
229/// to be written to the provided file descriptor with the given mime type.
230pub fn request_primary_client_selection<D>(
231    seat: &Seat<D>,
232    mime_type: String,
233    fd: OwnedFd,
234) -> Result<(), SelectionRequestError>
235where
236    D: SeatHandler + PrimarySelectionHandler + 'static,
237{
238    seat.user_data()
239        .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
240    let seat_data = seat
241        .user_data()
242        .get::<RefCell<SeatData<D::SelectionUserData>>>()
243        .unwrap();
244    match seat_data.borrow().get_primary_selection() {
245        None => Err(SelectionRequestError::NoSelection),
246        Some(OfferReplySource::Client(source)) => {
247            if !source.contains_mime_type(&mime_type) {
248                Err(SelectionRequestError::InvalidMimetype)
249            } else {
250                source.send(mime_type, fd);
251                Ok(())
252            }
253        }
254        Some(OfferReplySource::Compositor(selection)) => {
255            if !selection.mime_types.contains(&mime_type) {
256                Err(SelectionRequestError::InvalidMimetype)
257            } else {
258                Err(SelectionRequestError::ServerSideSelection)
259            }
260        }
261    }
262}

264/// Gets the user_data for the currently active selection, if set by the compositor
265#[instrument(name = "wayland_primary_selection", level = "debug", skip_all, fields(seat = seat.name()))]
266pub fn current_primary_selection_userdata<D>(seat: &Seat<D>) -> Option<Ref<'_, D::SelectionUserData>>
267where
268    D: SeatHandler + PrimarySelectionHandler + 'static,
269{
270    seat.user_data()
271        .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
272    let seat_data = seat
273        .user_data()
274        .get::<RefCell<SeatData<D::SelectionUserData>>>()
275        .unwrap();
276    Ref::filter_map(seat_data.borrow(), |data| match data.get_primary_selection() {
277        Some(OfferReplySource::Compositor(CompositorSelectionProvider { ref user_data, .. })) => {
278            Some(user_data)
279        }
280        _ => None,
281    })
282    .ok()
283}

348    impl<D> Dispatch<PrimaryDeviceManager, (), D> for PrimarySelectionState
349    where
350        D: Dispatch<PrimaryDeviceManager, ()>,
351        D: Dispatch<PrimarySource, PrimarySourceUserData>,
352        D: Dispatch<PrimaryDevice, PrimaryDeviceUserData>,
353        D: PrimarySelectionHandler,
354        D: SeatHandler,
355        D: 'static,
356    {
357        fn request(
358            _state: &mut D,
359            client: &wayland_server::Client,
360            _resource: &PrimaryDeviceManager,
361            request: primary_device_manager::Request,
362            _data: &(),
363            dhandle: &DisplayHandle,
364            data_init: &mut wayland_server::DataInit<'_, D>,
365        ) {
366            match request {
367                primary_device_manager::Request::CreateSource { id } => {
368                    data_init.init(id, PrimarySourceUserData::new(dhandle.clone()));
369                }
370                primary_device_manager::Request::GetDevice { id, seat: wl_seat } => {
371                    match Seat::<D>::from_resource(&wl_seat) {
372                        Some(seat) => {
373                            seat.user_data()
374                                .insert_if_missing(|| RefCell::new(SeatData::<D::SelectionUserData>::new()));
375
376                            let device = SelectionDevice::Primary(
377                                data_init.init(id, PrimaryDeviceUserData { wl_seat }),
378                            );
379
380                            let seat_data = seat
381                                .user_data()
382                                .get::<RefCell<SeatData<D::SelectionUserData>>>()
383                                .unwrap();
384
385                            seat_data.borrow_mut().add_device(device);
386                        }
387                        None => {
388                            error!(
389                                primary_selection_device = ?id,
390                                client = ?client,
391                                "Unmanaged seat given to a primary selection device."
392                            );
393                        }
394                    }
395                }
396                primary_device_manager::Request::Destroy => {}
397                _ => unreachable!(),
398            }
399        }
400    }
