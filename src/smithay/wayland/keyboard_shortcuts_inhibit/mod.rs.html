1//! Protocol for inhibiting the compositor keyboard shortcuts
2//!
3//! This protocol specifies a way for a client to request the compositor
4//! to ignore its own keyboard shortcuts for a given seat,
5//! so that all key events from that seat get forwarded to a surface.

28/// List of inhibitors associated with WlSeat
29#[derive(Debug, Default)]
30struct SeatInhibitors(Vec<KeyboardShortcutsInhibitor>);
31
32impl SeatInhibitors {
33    fn push(&mut self, value: KeyboardShortcutsInhibitor) {
34        self.0.push(value)
35    }
36
37    fn is_empty(&self) -> bool {
38        self.0.is_empty()
39    }
40
41    /// Check if any inhibitor is active
42    fn is_inhibited(&self) -> bool {
43        self.0.iter().any(|i| i.is_active())
44    }
45
46    fn remove(&mut self, id: ObjectId) -> Option<KeyboardShortcutsInhibitor> {
47        self.0
48            .iter()
49            .position(|i| i.inhibitor.id() == id)
50            .map(|id| self.0.remove(id))
51    }
52
53    fn surface_has_inhibitor(&self, surface: &WlSurface) -> bool {
54        self.inhibitor_for_surface(surface).is_some()
55    }
56
57    /// Find inhibitor_for WlSurface
58    fn inhibitor_for_surface(&self, surface: &WlSurface) -> Option<&KeyboardShortcutsInhibitor> {
59        self.0.iter().find(|i| i.wl_surface() == surface)
60    }
61}

63/// Delegate type for KeyboardShortcutsInhibit global.
64#[derive(Debug)]
65pub struct KeyboardShortcutsInhibitState {
66    manager_global: GlobalId,
67    inhibitors: HashMap<SeatId, Rc<RefCell<SeatInhibitors>>>,

70impl KeyboardShortcutsInhibitState {
71    /// Regiseter new [ZwpKeyboardShortcutsInhibitManagerV1] global
72    pub fn new<D>(display: &DisplayHandle) -> Self
73    where
74        D: GlobalDispatch<ZwpKeyboardShortcutsInhibitManagerV1, GlobalData>,
75        D: Dispatch<ZwpKeyboardShortcutsInhibitManagerV1, GlobalData>,
76        D: Dispatch<ZwpKeyboardShortcutsInhibitorV1, KeyboardShortcutsInhibitorUserData>,
77        D: 'static,
78    {
79        let manager_global =
80            display.create_global::<D, ZwpKeyboardShortcutsInhibitManagerV1, _>(1, GlobalData);

83            inhibitors: HashMap::new(),
84        }
85    }
86
87    /// [ZwpKeyboardShortcutsInhibitManagerV1] GlobalId getter
88    pub fn global(&self) -> GlobalId {
89        self.manager_global.clone()
90    }

93/// Context object for keyboard shortcuts inhibitor
94#[derive(Debug, Clone, PartialEq, Eq)]
95pub struct KeyboardShortcutsInhibitor {
96    inhibitor: ZwpKeyboardShortcutsInhibitorV1,
97}
98
99impl KeyboardShortcutsInhibitor {
100    fn data(&self) -> &KeyboardShortcutsInhibitorUserData {
101        self.inhibitor.data().unwrap()
102    }
103
104    fn set_is_active(&self, v: bool) {
105        self.data().is_active.store(v, atomic::Ordering::Release);
106    }
107
108    /// Seat that is being inhibited
109    fn seat_id(&self) -> &SeatId {
110        &self.data().seat
111    }
112
113    /// Seat that is being inhibited
114    pub fn seat(&self, dh: &DisplayHandle) -> Option<WlSeat> {
115        WlSeat::from_id(dh, self.seat_id().clone()).ok()
116    }
117
118    /// Seat that is being inhibited
119    #[inline]
120    pub fn wl_surface(&self) -> &WlSurface {
121        &self.data().surface
122    }
123
124    /// Is inhibitor active
125    pub fn is_active(&self) -> bool {
126        self.data().is_active.load(atomic::Ordering::Acquire)
127    }
128
129    /// This method indicates that the shortcut inhibitor is active.
130    ///
131    /// When active, the client may receive input events normally reserved by the compositor.
132    ///
133    /// Typically this is called when user instructs the compositor to enable inhibitor using any mechanism offered by the compositor,
134    /// either by accepting new inhibitor or re-enabling already existing inhibitor.
135    pub fn activate(&self) {
136        self.inhibitor.active();
137        self.set_is_active(true);
138    }
139
140    /// This event indicates that the shortcuts inhibitor is inactive, normal shortcuts processing is restored by the compositor.
141    pub fn inactivate(&self) {
142        self.inhibitor.inactive();
143        self.set_is_active(false);
144    }
145}

147#[derive(Debug, Default)]
148struct SeatData {
149    inhibitors: Rc<RefCell<SeatInhibitors>>,
150}
151
152impl SeatData {
153    fn get<D>(seat: &Seat<D>) -> &RefCell<Self>
154    where
155        D: SeatHandler,
156        D: 'static,
157    {
158        seat.user_data()
159            .insert_if_missing(|| RefCell::new(SeatData::default()));
160        seat.user_data().get().unwrap()
161    }
162
163    /// Check if any inhibitor is active
164    fn is_inhibited(&self) -> bool {
165        self.inhibitors.borrow().is_inhibited()
166    }
167
168    /// Find inhibitor for WlSurface
169    fn inhibitor_for_surface(&self, surface: &WlSurface) -> Option<KeyboardShortcutsInhibitor> {
170        self.inhibitors.borrow().inhibitor_for_surface(surface).cloned()
171    }
172}

174/// Seat extension used to check if shortcuts are inhibited
175pub trait KeyboardShortcutsInhibitorSeat {
176    /// Check if keyboard_shortcuts are inhibited
177    fn keyboard_shortcuts_inhibited(&self) -> bool;
178
179    /// Get inhibitors associated with given WlSurface
180    ///
181    /// Can be used to check if certain surface has inhibitor on it
182    /// ```no_run
183    /// use smithay::input::Seat;
184    /// use smithay::wayland::keyboard_shortcuts_inhibit::KeyboardShortcutsInhibitorSeat;
185    /// # use smithay::input::{SeatHandler, SeatState, pointer::CursorImageStatus};
186    /// # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
187    /// # struct State;
188    /// # impl SeatHandler for State {
189    /// #     type KeyboardFocus = WlSurface;
190    /// #     type PointerFocus = WlSurface;
191    /// #     type TouchFocus = WlSurface;
192    /// #     fn seat_state(&mut self) -> &mut SeatState<Self> { unimplemented!() }
193    /// #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
194    /// #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
195    /// # }
196    ///
197    /// # let wl_surface: WlSurface = todo!();
198    ///
199    /// let seat: Seat<State> = todo!();
200    ///
201    /// if let Some(inhibitor) = seat.keyboard_shortcuts_inhibitor_for_surface(&wl_surface) {
202    ///     dbg!(inhibitor.is_active());
203    /// }
204    /// ```
205    fn keyboard_shortcuts_inhibitor_for_surface(
206        &self,
207        surface: &WlSurface,
208    ) -> Option<KeyboardShortcutsInhibitor>;
209}
210
211impl<D> KeyboardShortcutsInhibitorSeat for Seat<D>
212where
213    D: SeatHandler,
214    D: 'static,
215{
216    fn keyboard_shortcuts_inhibited(&self) -> bool {
217        SeatData::get(self).borrow().is_inhibited()
218    }
219
220    fn keyboard_shortcuts_inhibitor_for_surface(
221        &self,
222        surface: &WlSurface,
223    ) -> Option<KeyboardShortcutsInhibitor> {
224        SeatData::get(self).borrow().inhibitor_for_surface(surface)
225    }
226}
