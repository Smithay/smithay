48impl<D> Dispatch<ZwpKeyboardShortcutsInhibitManagerV1, (), D> for KeyboardShortcutsInhibitState
49where
50    D: KeyboardShortcutsInhibitHandler,
51    D: SeatHandler,
52    D: Dispatch<ZwpKeyboardShortcutsInhibitManagerV1, ()>,
53    D: Dispatch<ZwpKeyboardShortcutsInhibitorV1, KeyboardShortcutsInhibitorUserData>,
54{
55    fn request(
56        handler: &mut D,
57        _client: &wayland_server::Client,
58        resource: &ZwpKeyboardShortcutsInhibitManagerV1,
59        request: zwp_keyboard_shortcuts_inhibit_manager_v1::Request,
60        _data: &(),
61        _dhandle: &wayland_server::DisplayHandle,
62        data_init: &mut wayland_server::DataInit<'_, D>,
63    ) {
64        match request {
65            zwp_keyboard_shortcuts_inhibit_manager_v1::Request::InhibitShortcuts { id, surface, seat } => {
66                let seat_id = seat.id();
67
68                if handler
69                    .keyboard_shortcuts_inhibit_state()
70                    .inhibitors
71                    .get(&seat_id)
72                    .map(|list| list.borrow().surface_has_inhibitor(&surface))
73                    .unwrap_or(false)
74                {
75                    resource.post_error(
76                        zwp_keyboard_shortcuts_inhibit_manager_v1::Error::AlreadyInhibited,
77                        "Keyboare shortcuts for this surface are already inhibited",
78                    );
79                    return;
80                }
81
82                let seat = Seat::<D>::from_resource(&seat).unwrap();
83                let seat_data = super::SeatData::get(&seat);
84
85                let inhibitor = data_init.init(
86                    id,
87                    KeyboardShortcutsInhibitorUserData {
88                        seat: seat_id.clone(),
89                        surface,
90                        is_active: AtomicBool::new(false),
91                    },
92                );
93
94                let inhibitor = super::KeyboardShortcutsInhibitor { inhibitor };
95
96                let list = handler
97                    .keyboard_shortcuts_inhibit_state()
98                    .inhibitors
99                    .entry(seat_id)
100                    .or_insert_with(|| seat_data.borrow().inhibitors.clone());
101
102                list.borrow_mut().push(inhibitor.clone());
103
104                handler.new_inhibitor(inhibitor);
105            }
106            zwp_keyboard_shortcuts_inhibit_manager_v1::Request::Destroy => {}
107            _ => unreachable!(),
108        }
109    }
110}

112impl<D> Dispatch<ZwpKeyboardShortcutsInhibitorV1, KeyboardShortcutsInhibitorUserData, D>
113    for KeyboardShortcutsInhibitState
114where
115    D: KeyboardShortcutsInhibitHandler,
116{
117    fn request(
118        _handler: &mut D,
119        _client: &wayland_server::Client,
120        _inhibitor: &ZwpKeyboardShortcutsInhibitorV1,
121        request: zwp_keyboard_shortcuts_inhibitor_v1::Request,
122        _data: &KeyboardShortcutsInhibitorUserData,
123        _dhandle: &wayland_server::DisplayHandle,
124        _data_init: &mut wayland_server::DataInit<'_, D>,
125    ) {
126        match request {
127            zwp_keyboard_shortcuts_inhibitor_v1::Request::Destroy => {}
128            _ => unreachable!(),
129        }
130    }
131
132    fn destroyed(
133        handler: &mut D,
134        _client: ClientId,
135        wl_inhibitor: &ZwpKeyboardShortcutsInhibitorV1,
136        data: &KeyboardShortcutsInhibitorUserData,
137    ) {
138        data.is_active.store(false, atomic::Ordering::Release);
139
140        let state = handler.keyboard_shortcuts_inhibit_state();
141
142        if let Entry::Occupied(mut entry) = state.inhibitors.entry(data.seat.clone()) {
143            let inhibitor = entry.get_mut().borrow_mut().remove(wl_inhibitor.id());
144
145            if entry.get().borrow().is_empty() {
146                entry.remove();
147            }
148
149            if let Some(inhibitor) = inhibitor {
150                handler.inhibitor_destroyed(inhibitor)
151            }
152        }
153    }
