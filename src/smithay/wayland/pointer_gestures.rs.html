1//! Utilities for pointer gestures support
2//!
3//! This protocol allows clients to receive touchpad gestures. Use the following methods to send
4//! gesture events to any respective objects created by the client:
5//!
6//! 1. Swipe gestures
7//!     - [`PointerHandle::gesture_swipe_begin`]
8//!     - [`PointerHandle::gesture_swipe_update`]
9//!     - [`PointerHandle::gesture_swipe_end`]
10//! 2. Pinch gesture:
11//!     - [`PointerHandle::gesture_pinch_begin`]
12//!     - [`PointerHandle::gesture_pinch_update`]
13//!     - [`PointerHandle::gesture_pinch_end`]
14//! 3. Hold gesture:
15//!     - [`PointerHandle::gesture_hold_begin`]
16//!     - [`PointerHandle::gesture_hold_end`]
17//!
18//! ```
19//! extern crate wayland_server;
20//! extern crate smithay;
21//!
22//! use smithay::wayland::pointer_gestures::PointerGesturesState;
23//! use smithay::delegate_pointer_gestures;
24//! # use smithay::backend::input::KeyState;
25//! # use smithay::input::{
26//! #   pointer::{PointerTarget, AxisFrame, MotionEvent, ButtonEvent, RelativeMotionEvent,
27//! #             GestureSwipeBeginEvent, GestureSwipeUpdateEvent, GestureSwipeEndEvent,
28//! #             GesturePinchBeginEvent, GesturePinchUpdateEvent, GesturePinchEndEvent,
29//! #             GestureHoldBeginEvent, GestureHoldEndEvent},
30//! #   keyboard::{KeyboardTarget, KeysymHandle, ModifiersState},
31//! #   touch::{DownEvent, UpEvent, MotionEvent as TouchMotionEvent, ShapeEvent, OrientationEvent, TouchTarget},
32//! #   Seat, SeatHandler, SeatState,
33//! # };
34//! # use smithay::utils::{IsAlive, Serial};
35//!
36//! # #[derive(Debug, Clone, PartialEq)]
37//! # struct Target;
38//! # impl IsAlive for Target {
39//! #   fn alive(&self) -> bool { true }
40//! # }
41//! # impl PointerTarget<State> for Target {
42//! #   fn enter(&self, seat: &Seat<State>, data: &mut State, event: &MotionEvent) {}
43//! #   fn motion(&self, seat: &Seat<State>, data: &mut State, event: &MotionEvent) {}
44//! #   fn relative_motion(&self, seat: &Seat<State>, data: &mut State, event: &RelativeMotionEvent) {}
45//! #   fn button(&self, seat: &Seat<State>, data: &mut State, event: &ButtonEvent) {}
46//! #   fn axis(&self, seat: &Seat<State>, data: &mut State, frame: AxisFrame) {}
47//! #   fn frame(&self, seat: &Seat<State>, data: &mut State) {}
48//! #   fn leave(&self, seat: &Seat<State>, data: &mut State, serial: Serial, time: u32) {}
49//! #   fn gesture_swipe_begin(&self, seat: &Seat<State>, data: &mut State, event: &GestureSwipeBeginEvent) {}
50//! #   fn gesture_swipe_update(&self, seat: &Seat<State>, data: &mut State, event: &GestureSwipeUpdateEvent) {}
51//! #   fn gesture_swipe_end(&self, seat: &Seat<State>, data: &mut State, event: &GestureSwipeEndEvent) {}
52//! #   fn gesture_pinch_begin(&self, seat: &Seat<State>, data: &mut State, event: &GesturePinchBeginEvent) {}
53//! #   fn gesture_pinch_update(&self, seat: &Seat<State>, data: &mut State, event: &GesturePinchUpdateEvent) {}
54//! #   fn gesture_pinch_end(&self, seat: &Seat<State>, data: &mut State, event: &GesturePinchEndEvent) {}
55//! #   fn gesture_hold_begin(&self, seat: &Seat<State>, data: &mut State, event: &GestureHoldBeginEvent) {}
56//! #   fn gesture_hold_end(&self, seat: &Seat<State>, data: &mut State, event: &GestureHoldEndEvent) {}
57//! # }
58//! # impl KeyboardTarget<State> for Target {
59//! #   fn enter(&self, seat: &Seat<State>, data: &mut State, keys: Vec<KeysymHandle<'_>>, serial: Serial) {}
60//! #   fn leave(&self, seat: &Seat<State>, data: &mut State, serial: Serial) {}
61//! #   fn key(
62//! #       &self,
63//! #       seat: &Seat<State>,
64//! #       data: &mut State,
65//! #       key: KeysymHandle<'_>,
66//! #       state: KeyState,
67//! #       serial: Serial,
68//! #       time: u32,
69//! #   ) {}
70//! #   fn modifiers(&self, seat: &Seat<State>, data: &mut State, modifiers: ModifiersState, serial: Serial) {}
71//! # }
72//! # impl TouchTarget<State> for Target {
73//! #   fn down(&self, seat: &Seat<State>, data: &mut State, event: &DownEvent, seq: Serial) {}
74//! #   fn up(&self, seat: &Seat<State>, data: &mut State, event: &UpEvent, seq: Serial) {}
75//! #   fn motion(&self, seat: &Seat<State>, data: &mut State, event: &TouchMotionEvent, seq: Serial) {}
76//! #   fn frame(&self, seat: &Seat<State>, data: &mut State, seq: Serial) {}
77//! #   fn cancel(&self, seat: &Seat<State>, data: &mut State, seq: Serial) {}
78//! #   fn shape(&self, seat: &Seat<State>, data: &mut State, event: &ShapeEvent, seq: Serial) {}
79//! #   fn orientation(&self, seat: &Seat<State>, data: &mut State, event: &OrientationEvent, seq: Serial) {}
80//! # }
81//! # struct State {
82//! #     seat_state: SeatState<Self>,
83//! # };
84//! # let mut display = wayland_server::Display::<State>::new().unwrap();
85//! # impl SeatHandler for State {
86//! #     type KeyboardFocus = Target;
87//! #     type PointerFocus = Target;
88//! #     type TouchFocus = Target;
89//! #
90//! #     fn seat_state(&mut self) -> &mut SeatState<Self> {
91//! #         &mut self.seat_state
92//! #     }
93//! # }
94//! let state = PointerGesturesState::new::<State>(&display.handle());
95//!
96//! delegate_pointer_gestures!(State);
97//! ```

149    pub(super) fn leave<D: SeatHandler + 'static>(&self, surface: &WlSurface, serial: Serial, time: u32) {
150        self.for_each_focused_swipe_gesture(surface, |gesture| {
151            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
152            let ongoing = data.in_progress_on.lock().unwrap().take();
153            if ongoing.is_some() {
154                // Cancel the ongoing gesture.
155                gesture.end(serial.into(), time, 1);
156            }
157        });
158        self.for_each_focused_pinch_gesture(surface, |gesture| {
159            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
160            let ongoing = data.in_progress_on.lock().unwrap().take();
161            if ongoing.is_some() {
162                // Cancel the ongoing gesture.
163                gesture.end(serial.into(), time, 1);
164            }
165        });
166        self.for_each_focused_hold_gesture(surface, |gesture| {
167            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
168            let ongoing = data.in_progress_on.lock().unwrap().take();
169            if ongoing.is_some() {
170                // Cancel the ongoing gesture.
171                gesture.end(serial.into(), time, 1);
172            }
173        });
174    }

176    pub(super) fn gesture_swipe_begin<D: SeatHandler + 'static>(
177        &self,
178        surface: &WlSurface,
179        event: &GestureSwipeBeginEvent,
180    ) {
181        self.for_each_focused_swipe_gesture(surface, |gesture| {
182            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
183            let ongoing = data.in_progress_on.lock().unwrap().replace(surface.clone());
184            if ongoing.is_some() {
185                // Cancel an ongoing gesture for a different surface.
186                gesture.end(event.serial.into(), event.time, 1);
187            }
188            gesture.begin(event.serial.into(), event.time, surface, event.fingers);
189        });
190    }
191
192    pub(super) fn gesture_swipe_update<D: SeatHandler + 'static>(
193        &self,
194        surface: &WlSurface,
195        event: &GestureSwipeUpdateEvent,
196    ) {
197        self.for_each_focused_swipe_gesture(surface, |gesture| {
198            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
199            let mut ongoing = data.in_progress_on.lock().unwrap();
200            // Check that the ongoing gesture is for this surface.
201            if ongoing.as_ref() == Some(surface) {
202                let client_scale = data.client_scale.load(Ordering::Acquire);
203                let delta = event.delta.to_client(client_scale);
204                gesture.update(event.time, delta.x, delta.y);
205            } else if ongoing.take().is_some() {
206                // If it was for a different surface, cancel it.
207                gesture.end(SERIAL_COUNTER.next_serial().into(), event.time, 1);
208            }
209        });
210    }
211
212    pub(super) fn gesture_swipe_end<D: SeatHandler + 'static>(
213        &self,
214        surface: &WlSurface,
215        event: &GestureSwipeEndEvent,
216    ) {
217        self.for_each_focused_swipe_gesture(surface, |gesture| {
218            let data = gesture.data::<PointerGestureUserData<D>>().unwrap();
219            let ongoing = data.in_progress_on.lock().unwrap().take();
220            // Check if the gesture was ongoing.
221            if ongoing.is_some() {
222                let cancelled = if ongoing.as_ref() == Some(surface) {
223                    event.cancelled
224                } else {
225                    // If the gesture was ongoing for any other surface then cancel it.
226                    true
227                };
228                gesture.end(event.serial.into(), event.time, cancelled.into());
229            }
230        });
231    }

363/// User data of ZwpPointerGesture*V1 objects
364#[derive(Debug)]
365pub struct PointerGestureUserData<D: SeatHandler> {
366    handle: Option<PointerHandle<D>>,
367    /// This gesture is in the middle between its begin() and end() on this surface.
368    pub(crate) in_progress_on: Mutex<Option<WlSurface>>,
369    client_scale: Arc<AtomicF64>,
370}

372/// State of the pointer gestures
373#[derive(Debug)]
374pub struct PointerGesturesState {
375    global: GlobalId,
376}

378impl PointerGesturesState {
379    /// Register new [ZwpPointerGesturesV1] global
380    pub fn new<D>(display: &DisplayHandle) -> Self
381    where
382        D: GlobalDispatch<ZwpPointerGesturesV1, ()>,
383        D: Dispatch<ZwpPointerGesturesV1, ()>,
384        D: Dispatch<ZwpPointerGestureSwipeV1, PointerGestureUserData<D>>,
385        D: Dispatch<ZwpPointerGesturePinchV1, PointerGestureUserData<D>>,
386        D: Dispatch<ZwpPointerGestureHoldV1, PointerGestureUserData<D>>,
387        D: SeatHandler,
388        D: 'static,
389    {
390        let global = display.create_global::<D, ZwpPointerGesturesV1, _>(MANAGER_VERSION, ());
391
392        Self { global }
393    }
394
395    /// [ZwpPointerGesturesV1] GlobalId getter
396    pub fn global(&self) -> GlobalId {
397        self.global.clone()
398    }

401impl<D> Dispatch<ZwpPointerGesturesV1, (), D> for PointerGesturesState
402where
403    D: Dispatch<ZwpPointerGesturesV1, ()>,
404    D: Dispatch<ZwpPointerGestureSwipeV1, PointerGestureUserData<D>>,
405    D: Dispatch<ZwpPointerGesturePinchV1, PointerGestureUserData<D>>,
406    D: Dispatch<ZwpPointerGestureHoldV1, PointerGestureUserData<D>>,
407    D: SeatHandler,
408    D: 'static,
409{
410    fn request(
411        _state: &mut D,
412        _client: &wayland_server::Client,
413        _pointer_gestures: &ZwpPointerGesturesV1,
414        request: zwp_pointer_gestures_v1::Request,
415        _data: &(),
416        _dh: &DisplayHandle,
417        data_init: &mut wayland_server::DataInit<'_, D>,
418    ) {
419        match request {
420            zwp_pointer_gestures_v1::Request::GetSwipeGesture { id, pointer } => {
421                let data = pointer.data::<PointerUserData<D>>().unwrap();
422                let user_data = PointerGestureUserData {
423                    handle: data.handle.clone(),
424                    in_progress_on: Mutex::new(None),
425                    client_scale: data.client_scale.clone(),
426                };
427                let gesture = data_init.init(id, user_data);
428                if let Some(handle) = &data.handle {
429                    handle.wp_pointer_gestures.new_swipe_gesture(gesture);
430                }
431            }
432            zwp_pointer_gestures_v1::Request::GetPinchGesture { id, pointer } => {
433                let data = pointer.data::<PointerUserData<D>>().unwrap();
434                let user_data = PointerGestureUserData {
435                    handle: data.handle.clone(),
436                    in_progress_on: Mutex::new(None),
437                    client_scale: data.client_scale.clone(),
438                };
439                let gesture = data_init.init(id, user_data);
440                if let Some(handle) = &data.handle {
441                    handle.wp_pointer_gestures.new_pinch_gesture(gesture);
442                }
443            }
444            zwp_pointer_gestures_v1::Request::GetHoldGesture { id, pointer } => {
445                let data = pointer.data::<PointerUserData<D>>().unwrap();
446                let user_data = PointerGestureUserData {
447                    handle: data.handle.clone(),
448                    in_progress_on: Mutex::new(None),
449                    client_scale: data.client_scale.clone(),
450                };
451                let gesture = data_init.init(id, user_data);
452                if let Some(handle) = &data.handle {
453                    handle.wp_pointer_gestures.new_hold_gesture(gesture);
454                }
455            }
456            zwp_pointer_gestures_v1::Request::Release => {}
457            _ => unreachable!(),
458        }
459    }
460}

512        }
513    }

574
575    fn destroyed(
576        _state: &mut D,
577        _: ClientId,
578        object: &ZwpPointerGestureHoldV1,
579        data: &PointerGestureUserData<D>,
580    ) {
581        if let Some(ref handle) = data.handle {
582            handle
583                .wp_pointer_gestures
584                .known_hold_gestures
585                .lock()
586                .unwrap()
587                .retain(|p| p.id() != object.id());
588        }
589    }
