1//! Utilities for handling the `wp_single_pixel_buffer` protocol
2//!
3//! ## How to use it
4//!
5//! ### Initialization
6//!
7//! To initialize this implementation, create [`SinglePixelBufferState`], store it in your `State` struct and
8//! implement the required traits, as shown in this example:
9//!
10//! ```
11//! use smithay::wayland::{
12//!     buffer::BufferHandler,
13//!     single_pixel_buffer::SinglePixelBufferState
14//! };
15//! use smithay::delegate_single_pixel_buffer;
16//!
17//! # struct State;
18//! # let mut display = wayland_server::Display::<State>::new().unwrap();
19//!
20//! // Create the single pixel buffer state:
21//! let single_pixel_buffer_state = SinglePixelBufferState::new::<State>(
22//!     &display.handle(), // the display
23//! );
24//!
25//! // Smithay's "SinglePixelBufferState" also requires the buffer management utilities, you need to implement
26//! // "BufferHandler".
27//! impl BufferHandler for State {
28//!     fn buffer_destroyed(&mut self, buffer: &wayland_server::protocol::wl_buffer::WlBuffer) {
29//!         // All renderers can handle buffer destruction at this point. Some parts of window management may
30//!         // also use this function.
31//!     }
32//! }
33//!
34//! // implement Dispatch for the SinglePixelBuffer types
35//! delegate_single_pixel_buffer!(State);
36//!
37//! // You're now ready to go!
38//! ```
39//!
40//! ### Access the single pixel buffer data
41//!
42//! The buffer data stores the color as RGBA values which can be retrieved by calling [`get_single_pixel_buffer`].
43//!
44//! ```no_run
45//! use smithay::wayland::single_pixel_buffer;
46//! # let buffer: smithay::reexports::wayland_server::protocol::wl_buffer::WlBuffer = { todo!() };
47//! let single_pixel_buffer_data = single_pixel_buffer::get_single_pixel_buffer(&buffer).unwrap();
48//! ```

57/// Delegate state of WpSinglePixelBuffer protocol
58#[derive(Debug)]
59pub struct SinglePixelBufferState {
60    global: GlobalId,
61}
62
63impl SinglePixelBufferState {
64    /// Create a new [`WpSinglePixelBufferManagerV1`] global
65    //
66    /// The id provided by [`SinglePixelBufferState::global`] may be used to
67    /// remove or disable this global in the future.
68    pub fn new<D>(display: &DisplayHandle) -> Self
69    where
70        D: GlobalDispatch<WpSinglePixelBufferManagerV1, ()>,
71        D: Dispatch<WpSinglePixelBufferManagerV1, ()>,
72        D: 'static,
73    {
74        let global = display.create_global::<D, WpSinglePixelBufferManagerV1, _>(1, ());
75
76        Self { global }
77    }
78
79    /// Returns the id of the [`WpSinglePixelBufferManagerV1`] global.
80    pub fn global(&self) -> GlobalId {
81        self.global.clone()
82    }
83}

85/// User data of `WlBuffer` backed by single pixel
86#[derive(Debug)]
87pub struct SinglePixelBufferUserData {
88    /// Value of the buffer's red channel
89    pub r: u32,
90    /// Value of the buffer's green channel
91    pub g: u32,
92    /// Value of the buffer's blue channel
93    pub b: u32,
94    /// Value of the buffer's alpha channel
95    pub a: u32,
96}
97
98impl SinglePixelBufferUserData {
99    /// Check if pixel has alpha
100    pub fn has_alpha(&self) -> bool {
101        self.a != u32::MAX
102    }
103
104    /// RGAB8888 color buffer
105    pub fn rgba8888(&self) -> [u8; 4] {
106        let divisor = u32::MAX / 255;
107
108        [
109            (self.r / divisor) as u8,
110            (self.g / divisor) as u8,
111            (self.b / divisor) as u8,
112            (self.a / divisor) as u8,
113        ]
114    }
115
116    /// Color represented as rgba32f
117    ///
118    /// Ranges from `0f32` to `1f32`
119    pub fn rgba32f(&self) -> [f32; 4] {
120        let divisor = u32::MAX as f32;
121
122        [
123            self.r as f32 / divisor,
124            self.g as f32 / divisor,
125            self.b as f32 / divisor,
126            self.a as f32 / divisor,
127        ]
128    }
129}

131/// Error that can occur when accessing an SinglePixelBuffer
132#[derive(Debug, thiserror::Error)]
133pub enum BufferAccessError {
134    /// This buffer is not managed by the SinglePixelBuffer handler
135    #[error("non-single-pixel buffer")]
136    NotManaged,
137}
138
139/// Gets the data of a `SinglePixelBuffer` backed [`WlBuffer`].
140pub fn get_single_pixel_buffer(buffer: &WlBuffer) -> Result<&SinglePixelBufferUserData, BufferAccessError> {
141    buffer
142        .data::<SinglePixelBufferUserData>()
143        .ok_or(BufferAccessError::NotManaged)
144}

146/// Macro used to delegate `WpSinglePixelBuffer` events
147#[macro_export]
148macro_rules! delegate_single_pixel_buffer {
149    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
150        const _: () = {
151            use $crate::{
152                reexports::{
153                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
154                    wayland_server::protocol::wl_buffer::WlBuffer,
155                    wayland_protocols::wp::single_pixel_buffer::v1::server::wp_single_pixel_buffer_manager_v1::WpSinglePixelBufferManagerV1,
156                },
157                wayland::single_pixel_buffer::{SinglePixelBufferState, SinglePixelBufferUserData},
158            };
159
160            delegate_global_dispatch!(
161                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
162                $ty: [WpSinglePixelBufferManagerV1: ()] => SinglePixelBufferState
163            );
164
165            delegate_dispatch!(
166                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
167                $ty: [WpSinglePixelBufferManagerV1: ()] => SinglePixelBufferState
168            );
169
170            delegate_dispatch!(
171                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
172                $ty: [WlBuffer: SinglePixelBufferUserData] => SinglePixelBufferState
173            );
174        };
175    };
176}
