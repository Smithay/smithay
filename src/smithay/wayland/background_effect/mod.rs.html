1//! Implementation of `ext_background_effect_manager_v1` protocol
2//!
3//! In order to advertise background effect global call [BackgroundEffectState::new].
4//! Currently attached effects are available in double-buffered [BackgroundEffectSurfaceCachedState]
5//!
6//! ```
7//! use smithay::wayland::background_effect::{BackgroundEffectState, ExtBackgroundEffectHandler, Capability};
8//! use wayland_server::protocol::wl_surface::WlSurface;
9//! use smithay::wayland::compositor;
10//!
11//! # struct State;
12//! # let mut display = wayland_server::Display::<State>::new().unwrap();
13//!
14//! BackgroundEffectState::new::<State>(
15//!     &display.handle(),
16//! );
17//!
18//! impl ExtBackgroundEffectHandler for State {
19//!     fn capabilities(&self) -> Capability {
20//!         Capability::Blur
21//!     }
22//!
23//!     fn set_blur_region(&mut self, wl_surface: WlSurface, region: compositor::RegionAttributes) {
24//!         // Called when blur becomes pending, and awaits surface commit.
25//!         // Blur region is stored in wl_surface [BackgroundEffectSurfaceCachedState]
26//!     }
27//! }
28//!
29//! smithay::delegate_dispatch2!(State);
30//! ```
31

52/// Handler trait for ext background effect events.
53pub trait ExtBackgroundEffectHandler: 'static {
54    /// Getter for background-effect capabilities of the compositor
55    fn capabilities(&self) -> Capability {
56        // For now only blur is supported, so it's safe to assume it's supported by default
57        Capability::Blur
58    }
59
60    /// Called when blur becomes pending, and awaits surface commit.
61    /// Blur region is stored in wl_surface [BackgroundEffectSurfaceCachedState]
62    fn set_blur_region(&mut self, wl_surface: WlSurface, region: compositor::RegionAttributes) {
63        let _ = wl_surface;
64        let _ = region;
65    }
66
67    /// Called when blur unset becomes pending, and awaits surface commit.
68    fn unset_blur_region(&mut self, wl_surface: WlSurface) {
69        let _ = wl_surface;
70    }
71}

73/// Cached state for background effect per surface (double-buffered)
74///
75/// ```no_run
76/// use smithay::wayland::compositor;
77/// use smithay::wayland::background_effect::BackgroundEffectSurfaceCachedState;
78///
79/// # let wl_surface = todo!();
80/// compositor::with_states(&wl_surface, |states| {
81///     let mut modifier_state = states.cached_state.get::<BackgroundEffectSurfaceCachedState>();
82///     dbg!(&modifier_state.current().blur_region);
83/// });
84/// ```
85#[derive(Debug, Clone, Default)]
86pub struct BackgroundEffectSurfaceCachedState {
87    /// Region of the surface that will have its background blurred.
88    pub blur_region: Option<compositor::RegionAttributes>,
89}
90
91impl Cacheable for BackgroundEffectSurfaceCachedState {
92    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
93        self.clone()
94    }
95
96    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
97        *into = self;
98    }
99}

101/// User data for ext_background_effect_surface_v1
102#[derive(Debug)]
103pub struct BackgroundEffectSurfaceUserData(Mutex<Weak<WlSurface>>);
104
105impl BackgroundEffectSurfaceUserData {
106    fn new(surface: WlSurface) -> Self {
107        Self(Mutex::new(surface.downgrade()))
108    }
109
110    fn wl_surface(&self) -> Option<WlSurface> {
111        self.0.lock().unwrap().upgrade().ok()
112    }
113}

115/// Tracks if a surface already has a background effect object
116#[derive(Debug)]
117struct BackgroundEffectSurfaceData {
118    is_resource_attached: AtomicBool,
119}
120
121impl BackgroundEffectSurfaceData {
122    fn new() -> Self {
123        Self {
124            is_resource_attached: AtomicBool::new(false),
125        }
126    }
127
128    fn set_is_resource_attached(&self, is_attached: bool) {
129        self.is_resource_attached.store(is_attached, Ordering::Release)
130    }
131
132    fn is_resource_attached(&self) -> bool {
133        self.is_resource_attached.load(Ordering::Acquire)
134    }
135}

137/// Global state for background effect protocol
138#[derive(Debug)]
139pub struct BackgroundEffectState {
140    global: GlobalId,
141}
142
143impl BackgroundEffectState {
144    /// Regiseter new [ExtBackgroundEffectManagerV1] global
145    pub fn new<D>(display: &DisplayHandle) -> BackgroundEffectState
146    where
147        D: ExtBackgroundEffectHandler + GlobalDispatch<ExtBackgroundEffectManagerV1, GlobalData>,
148    {
149        let global = display.create_global::<D, ExtBackgroundEffectManagerV1, _>(1, GlobalData);
150        BackgroundEffectState { global }
151    }
152
153    /// Returns the [ExtBackgroundEffectManagerV1] global id
154    pub fn global(&self) -> GlobalId {
155        self.global.clone()
156    }
