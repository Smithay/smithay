35impl<D> Dispatch<WpAlphaModifierV1, (), D> for AlphaModifierState
36where
37    D: Dispatch<WpAlphaModifierV1, ()>,
38    D: Dispatch<WpAlphaModifierSurfaceV1, AlphaModifierSurfaceUserData>,
39    D: 'static,
40{
41    fn request(
42        _state: &mut D,
43        _: &Client,
44        manager: &WpAlphaModifierV1,
45        request: wp_alpha_modifier_v1::Request,
46        _data: &(),
47        _dh: &DisplayHandle,
48        data_init: &mut DataInit<'_, D>,
49    ) {
50        match request {
51            wp_alpha_modifier_v1::Request::GetSurface { id, surface } => {
52                let already_taken = compositor::with_states(&surface, |states| {
53                    states
54                        .data_map
55                        .insert_if_missing_threadsafe(AlphaModifierSurfaceData::new);
56                    let data = states.data_map.get::<AlphaModifierSurfaceData>().unwrap();
57
58                    let already_taken = data.is_resource_attached();
59
60                    if !already_taken {
61                        data.set_is_resource_attached(true);
62                    }
63
64                    already_taken
65                });
66
67                if already_taken {
68                    manager.post_error(
69                        wp_alpha_modifier_v1::Error::AlreadyConstructed,
70                        "wl_surface already has a alpha modifier object attached",
71                    )
72                } else {
73                    data_init.init(id, AlphaModifierSurfaceUserData::new(surface));
74                }
75            }
76
77            wp_alpha_modifier_v1::Request::Destroy => {}
78            _ => unreachable!(),
79        }
80    }
81}

83impl<D> Dispatch<WpAlphaModifierSurfaceV1, AlphaModifierSurfaceUserData, D> for AlphaModifierState
84where
85    D: Dispatch<WpAlphaModifierSurfaceV1, AlphaModifierSurfaceUserData>,
86{
87    fn request(
88        _state: &mut D,
89        _: &Client,
90        obj: &WpAlphaModifierSurfaceV1,
91        request: wp_alpha_modifier_surface_v1::Request,
92        data: &AlphaModifierSurfaceUserData,
93        _dh: &DisplayHandle,
94        _: &mut DataInit<'_, D>,
95    ) {
96        match request {
97            wp_alpha_modifier_surface_v1::Request::SetMultiplier { factor } => {
98                let Some(surface) = data.wl_surface() else {
99                    obj.post_error(
100                        wp_alpha_modifier_surface_v1::Error::NoSurface,
101                        "wl_surface was destroyed",
102                    );
103                    return;
104                };
105
106                compositor::with_states(&surface, |states| {
107                    states
108                        .cached_state
109                        .get::<AlphaModifierSurfaceCachedState>()
110                        .pending()
111                        .multiplier = Some(factor);
112                })
113            }
114            // Switch back to not specifying the alpha multiplier of this surface.
115            wp_alpha_modifier_surface_v1::Request::Destroy => {
116                let Some(surface) = data.wl_surface() else {
117                    obj.post_error(
118                        wp_alpha_modifier_surface_v1::Error::NoSurface,
119                        "wl_surface was destroyed",
120                    );
121                    return;
122                };
123
124                compositor::with_states(&surface, |states| {
125                    states
126                        .data_map
127                        .get::<AlphaModifierSurfaceData>()
128                        .unwrap()
129                        .set_is_resource_attached(false);
130
131                    states
132                        .cached_state
133                        .get::<AlphaModifierSurfaceCachedState>()
134                        .pending()
135                        .multiplier = None;
136                });
137            }
138            _ => unreachable!(),
139        }
140    }
141
142    fn destroyed(
143        _state: &mut D,
144        _client: ClientId,
145        _object: &WpAlphaModifierSurfaceV1,
146        _data: &AlphaModifierSurfaceUserData,
147    ) {
148        // Nothing to do here, graceful Destroy is already handled with double buffering
149        // and in case of client close WlSurface destroyed handler will clean up the data anyway,
150        // so there is no point in queuing new update
151    }
