1//! Implementation of `wp_alpha_modifier` protocol
2//!
3//! [`WaylandSurfaceRenderElement`][`crate::backend::renderer::element::surface::WaylandSurfaceRenderElement`]
4//! takes the alpha multiplier into account automatically.
5//!
6//! ### Example
7//!
8//! ```no_run
9//! # extern crate wayland_server;
10//! #
11//! use wayland_server::{protocol::wl_surface::WlSurface, DisplayHandle};
12//! use smithay::{
13//!     delegate_alpha_modifier, delegate_compositor,
14//!     wayland::compositor::{self, CompositorState, CompositorClientState, CompositorHandler},
15//!     wayland::alpha_modifier::{AlphaModifierSurfaceCachedState, AlphaModifierState},
16//! };
17//!
18//! pub struct State {
19//!     compositor_state: CompositorState,
20//! };
21//! struct ClientState { compositor_state: CompositorClientState }
22//! impl wayland_server::backend::ClientData for ClientState {}
23//!
24//! delegate_alpha_modifier!(State);
25//! delegate_compositor!(State);
26//!
27//! impl CompositorHandler for State {
28//!    fn compositor_state(&mut self) -> &mut CompositorState {
29//!        &mut self.compositor_state
30//!    }
31//!
32//!    fn client_compositor_state<'a>(&self, client: &'a wayland_server::Client) -> &'a CompositorClientState {
33//!        &client.get_data::<ClientState>().unwrap().compositor_state    
34//!    }
35//!
36//!    fn commit(&mut self, surface: &WlSurface) {
37//!        compositor::with_states(&surface, |states| {
38//!            let mut modifier_state = states.cached_state.get::<AlphaModifierSurfaceCachedState>();
39//!            dbg!(modifier_state.current().multiplier());
40//!        });
41//!    }
42//! }
43//!
44//! let mut display = wayland_server::Display::<State>::new().unwrap();
45//!
46//! let compositor_state = CompositorState::new::<State>(&display.handle());
47//! AlphaModifierState::new::<State>(&display.handle());
48//!
49//! let state = State {
50//!     compositor_state,
51//! };
52//! ```

71/// Data associated with WlSurface
72/// Represents the client pending state
73///
74/// ```no_run
75/// use smithay::wayland::compositor;
76/// use smithay::wayland::alpha_modifier::AlphaModifierSurfaceCachedState;
77///
78/// # let wl_surface = todo!();
79/// compositor::with_states(&wl_surface, |states| {
80///     let mut modifier_state = states.cached_state.get::<AlphaModifierSurfaceCachedState>();
81///     dbg!(modifier_state.current().multiplier());
82/// });
83/// ```
84#[derive(Debug, Clone, Copy, Default)]
85pub struct AlphaModifierSurfaceCachedState {
86    multiplier: Option<u32>,
87}
88
89impl AlphaModifierSurfaceCachedState {
90    /// Alpha multiplier for the surface
91    pub fn multiplier(&self) -> Option<u32> {
92        self.multiplier
93    }
94
95    /// Alpha multiplier for the surface
96    pub fn multiplier_f32(&self) -> Option<f32> {
97        self.multiplier
98            .map(|multiplier| multiplier as f32 / u32::MAX as f32)
99    }
100
101    /// Alpha multiplier for the surface
102    pub fn multiplier_f64(&self) -> Option<f64> {
103        self.multiplier
104            .map(|multiplier| multiplier as f64 / u32::MAX as f64)
105    }
106}
107
108impl Cacheable for AlphaModifierSurfaceCachedState {
109    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
110        *self
111    }
112
113    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
114        *into = self;
115    }
116}

119struct AlphaModifierSurfaceData {

121}
122
123impl AlphaModifierSurfaceData {
124    fn new() -> Self {
125        Self {
126            is_resource_attached: AtomicBool::new(false),
127        }
128    }
129
130    fn set_is_resource_attached(&self, is_attached: bool) {
131        self.is_resource_attached
132            .store(is_attached, atomic::Ordering::Release)
133    }
134
135    fn is_resource_attached(&self) -> bool {
136        self.is_resource_attached.load(atomic::Ordering::Acquire)
137    }

142pub struct AlphaModifierSurfaceUserData(Mutex<Weak<WlSurface>>);

144impl AlphaModifierSurfaceUserData {

146        Self(Mutex::new(surface.downgrade()))

148

154/// Delegate type for [WpAlphaModifierV1] global.
155#[derive(Debug)]
156pub struct AlphaModifierState {
157    global: GlobalId,
158}
159
160impl AlphaModifierState {
161    /// Regiseter new [WpAlphaModifierV1] global
162    pub fn new<D>(display: &DisplayHandle) -> AlphaModifierState
163    where
164        D: GlobalDispatch<WpAlphaModifierV1, ()>
165            + Dispatch<WpAlphaModifierV1, ()>
166            + Dispatch<WpAlphaModifierSurfaceV1, AlphaModifierSurfaceUserData>
167            + 'static,
168    {
169        let global = display.create_global::<D, WpAlphaModifierV1, _>(1, ());
170
171        AlphaModifierState { global }
172    }
173
174    /// Returns the [WpAlphaModifierV1] global id
175    pub fn global(&self) -> GlobalId {
176        self.global.clone()
177    }
178}
