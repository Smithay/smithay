51impl<D> Dispatch<ZwlrLayerShellV1, (), D> for WlrLayerShellState
52where
53    D: Dispatch<ZwlrLayerShellV1, ()>,
54    D: Dispatch<ZwlrLayerSurfaceV1, WlrLayerSurfaceUserData>,
55    D: WlrLayerShellHandler,
56    D: 'static,
57{
58    fn request(
59        state: &mut D,
60        _client: &Client,
61        shell: &ZwlrLayerShellV1,
62        request: zwlr_layer_shell_v1::Request,
63        _data: &(),
64        _dh: &DisplayHandle,
65        data_init: &mut DataInit<'_, D>,
66    ) {
67        match request {
68            zwlr_layer_shell_v1::Request::GetLayerSurface {
69                id,
70                surface: wl_surface,
71                output,
72                layer,
73                namespace,
74            } => {
75                let layer: Layer = match layer.try_into() {
76                    Ok(layer) => layer,
77                    Err(layer) => {
78                        shell.post_error(
79                            zwlr_layer_shell_v1::Error::InvalidLayer,
80                            format!("invalid layer: {layer:?}"),
81                        );
82                        return;
83                    }
84                };
85
86                if compositor::give_role(&wl_surface, LAYER_SURFACE_ROLE).is_err() {
87                    shell.post_error(zwlr_layer_shell_v1::Error::Role, "Surface already has a role.");
88                    return;
89                }
90
91                let id = data_init.init(
92                    id,
93                    WlrLayerSurfaceUserData {
94                        shell_data: state.shell_state().clone(),
95                        wl_surface: wl_surface.downgrade(),
96                        alive_tracker: Default::default(),
97                    },
98                );
99
100                let initial = compositor::with_states(&wl_surface, |states| {
101                    let inserted = states
102                        .data_map
103                        .insert_if_missing_threadsafe(|| Mutex::new(LayerSurfaceAttributes::new(id.clone())));
104
105                    if !inserted {
106                        let mut attributes = states
107                            .data_map
108                            .get::<Mutex<LayerSurfaceAttributes>>()
109                            .unwrap()
110                            .lock()
111                            .unwrap();
112                        attributes.surface = id.clone();
113                    }
114
115                    states
116                        .cached_state
117                        .get::<LayerSurfaceCachedState>()
118                        .pending()
119                        .layer = layer;
120
121                    inserted
122                });
123
124                if initial {
125                    compositor::add_pre_commit_hook::<D, _>(
126                        &wl_surface,
127                        super::LayerSurface::pre_commit_hook,
128                    );
129                }
130
131                let handle = super::LayerSurface {
132                    wl_surface,
133                    shell_surface: id,
134                };
135
136                state
137                    .shell_state()
138                    .known_layers
139                    .lock()
140                    .unwrap()
141                    .push(handle.clone());
142
143                WlrLayerShellHandler::new_layer_surface(state, handle, output, layer, namespace);
144            }
145            zwlr_layer_shell_v1::Request::Destroy => {
146                // Handled by destructor
147            }
148            _ => {}
149        }
150    }
151}

157/// User data for wlr layer surface
158#[derive(Debug)]
159pub struct WlrLayerSurfaceUserData {
160    shell_data: WlrLayerShellState,
161    // `LayerSurfaceAttributes` stored in the surface `data_map`, contains a
162    // `ZwlrLayerSurfaceV1`. So this reference needs to be weak to avoid a
163    // cycle.
164    wl_surface: Weak<wl_surface::WlSurface>,
165    alive_tracker: AliveTracker,
166}
167
168impl IsAlive for ZwlrLayerSurfaceV1 {
169    #[inline]
170    fn alive(&self) -> bool {
171        let data: &WlrLayerSurfaceUserData = self.data().unwrap();
172        data.alive_tracker.alive()
173    }
174}

176impl<D> Dispatch<ZwlrLayerSurfaceV1, WlrLayerSurfaceUserData, D> for WlrLayerShellState
177where
178    D: Dispatch<ZwlrLayerSurfaceV1, WlrLayerSurfaceUserData>,
179    D: WlrLayerShellHandler,
180{
181    fn request(
182        state: &mut D,
183        _client: &Client,
184        layer_surface: &ZwlrLayerSurfaceV1,
185        request: zwlr_layer_surface_v1::Request,
186        data: &WlrLayerSurfaceUserData,
187        _dh: &DisplayHandle,
188        _data_init: &mut DataInit<'_, D>,
189    ) {
190        match request {
191            zwlr_layer_surface_v1::Request::SetSize { width, height } => {
192                let _ = with_surface_pending_state(layer_surface, |data| {
193                    data.size = (width as i32, height as i32).into();
194                });
195            }
196            zwlr_layer_surface_v1::Request::SetAnchor { anchor } => {
197                match Anchor::try_from(anchor) {
198                    Ok(anchor) => {
199                        let _ = with_surface_pending_state(layer_surface, |data| {
200                            data.anchor = Anchor::from_bits(anchor.bits()).unwrap_or_default();
201                        });
202                    }
203                    Err((err, msg)) => {
204                        layer_surface.post_error(err, msg);
205                    }
206                };
207            }
208            zwlr_layer_surface_v1::Request::SetExclusiveZone { zone } => {
209                let _ = with_surface_pending_state(layer_surface, |data| {
210                    data.exclusive_zone = zone.into();
211                });
212            }
213            zwlr_layer_surface_v1::Request::SetMargin {
214                top,
215                right,
216                bottom,
217                left,
218            } => {
219                let _ = with_surface_pending_state(layer_surface, |data| {
220                    data.margin = Margins {
221                        top,
222                        right,
223                        bottom,
224                        left,
225                    };
226                });
227            }
228            zwlr_layer_surface_v1::Request::SetKeyboardInteractivity {
229                keyboard_interactivity,
230            } => {
231                match KeyboardInteractivity::try_from(keyboard_interactivity) {
232                    Ok(keyboard_interactivity) => {
233                        let _ = with_surface_pending_state(layer_surface, |data| {
234                            data.keyboard_interactivity = keyboard_interactivity;
235                        });
236                    }
237                    Err((err, msg)) => {
238                        layer_surface.post_error(err, msg);
239                    }
240                };
241            }
242            zwlr_layer_surface_v1::Request::SetLayer { layer } => {
243                match Layer::try_from(layer) {
244                    Ok(layer) => {
245                        let _ = with_surface_pending_state(layer_surface, |data| {
246                            data.layer = layer;
247                        });
248                    }
249                    Err((err, msg)) => {
250                        layer_surface.post_error(err, msg);
251                    }
252                };
253            }
254            zwlr_layer_surface_v1::Request::SetExclusiveEdge { edge } => {
255                match Anchor::try_from(edge) {
256                    Ok(edge) => {
257                        let _ = with_surface_pending_state(layer_surface, |data| {
258                            if edge.is_empty() {
259                                data.exclusive_edge = None;
260                            } else if edge.bits().count_ones() == 1 {
261                                data.exclusive_edge = Some(edge);
262                            } else {
263                                layer_surface.post_error(
264                                    zwlr_layer_surface_v1::Error::InvalidExclusiveEdge,
265                                    "exclusive edge cannot have multiple edges",
266                                );
267                            }
268                        });
269                    }
270                    Err((err, msg)) => {
271                        layer_surface.post_error(err, msg);
272                    }
273                };
274            }
275            zwlr_layer_surface_v1::Request::GetPopup { popup } => {
276                let Ok(parent_surface) = data.wl_surface.upgrade() else {
277                    return;
278                };
279
280                let data = popup
281                    .data::<crate::wayland::shell::xdg::XdgShellSurfaceUserData>()
282                    .unwrap();
283
284                compositor::with_states(&data.wl_surface, move |states| {
285                    states
286                        .data_map
287                        .get::<XdgPopupSurfaceData>()
288                        .unwrap()
289                        .lock()
290                        .unwrap()
291                        .parent = Some(parent_surface);
292                });
293
294                WlrLayerShellHandler::new_popup(
295                    state,
296                    make_surface_handle(layer_surface).unwrap(),
297                    crate::wayland::shell::xdg::handlers::make_popup_handle(&popup),
298                );
299            }
300            zwlr_layer_surface_v1::Request::AckConfigure { serial } => {
301                let Ok(surface) = data.wl_surface.upgrade() else {
302                    return;
303                };
304
305                let serial = Serial::from(serial);
306
307                let found_configure = compositor::with_states(&surface, |states| {
308                    states
309                        .data_map
310                        .get::<LayerSurfaceData>()
311                        .unwrap()
312                        .lock()
313                        .unwrap()
314                        .ack_configure(serial)
315                });
316
317                let configure = match found_configure {
318                    Some(configure) => configure,
319                    None => {
320                        layer_surface.post_error(
321                            zwlr_layer_surface_v1::Error::InvalidSurfaceState,
322                            format!("wrong configure serial: {}", <u32>::from(serial)),
323                        );
324                        return;
325                    }
326                };
327
328                WlrLayerShellHandler::ack_configure(state, surface, configure);
329            }
330            zwlr_layer_surface_v1::Request::Destroy => {
331                // Handled by destroyed handler
332            }
333            _ => {}
334        }
335    }
336
337    fn destroyed(
338        state: &mut D,
339        _client_id: wayland_server::backend::ClientId,
340        layer_surface: &ZwlrLayerSurfaceV1,
341        data: &WlrLayerSurfaceUserData,
342    ) {
343        data.alive_tracker.destroy_notify();
344
345        // remove this surface from the known ones (as well as any leftover dead surface)
346        let mut layers = data.shell_data.known_layers.lock().unwrap();
347        if let Some(index) = layers
348            .iter()
349            .position(|layer| layer.shell_surface.id() == layer_surface.id())
350        {
351            let layer = layers.remove(index);
352            drop(layers);
353            let surface = layer.wl_surface().clone();
354            WlrLayerShellHandler::layer_destroyed(state, layer);
355            compositor::with_states(&surface, |states| {
356                let mut attributes = states
357                    .data_map
358                    .get::<Mutex<LayerSurfaceAttributes>>()
359                    .unwrap()
360                    .lock()
361                    .unwrap();
362                attributes.reset();
363
364                let mut guard = states.cached_state.get::<LayerSurfaceCachedState>();
365                *guard.pending() = Default::default();
366                *guard.current() = Default::default();
367            });
368        }
369    }
370}

372fn with_surface_pending_state<F, T>(
373    layer_surface: &zwlr_layer_surface_v1::ZwlrLayerSurfaceV1,
374    f: F,
375) -> Result<T, wayland_server::backend::InvalidId>
376where
377    F: FnOnce(&mut LayerSurfaceCachedState) -> T,
378{
379    let data = layer_surface.data::<WlrLayerSurfaceUserData>().unwrap();
380    let surface = data.wl_surface.upgrade()?;
381    Ok(compositor::with_states(&surface, |states| {
382        f(states.cached_state.get::<LayerSurfaceCachedState>().pending())
383    }))
384}

386pub fn make_surface_handle(
387    resource: &zwlr_layer_surface_v1::ZwlrLayerSurfaceV1,
388) -> Result<crate::wayland::shell::wlr_layer::LayerSurface, wayland_server::backend::InvalidId> {
389    let data = resource.data::<WlrLayerSurfaceUserData>().unwrap();
390    let wl_surface = data.wl_surface.upgrade()?;
391    Ok(crate::wayland::shell::wlr_layer::LayerSurface {
392        wl_surface,
393        shell_surface: resource.clone(),
394    })
