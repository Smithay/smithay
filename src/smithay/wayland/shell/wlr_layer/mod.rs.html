1//! Utilities for handling shell surfaces with the `wlr_layer_shell` protocol
2//!
3//! This interface should be suitable for the implementation of many desktop shell components,
4//! and a broad number of other applications that interact with the desktop.
5//!
6//! ### Initialization
7//!
8//! To initialize this handler, create the [`WlrLayerShellState`], store it inside your `State` and
9//! implement the [`WlrLayerShellHandler`], as shown in this example:
10//!
11//! ```no_run
12//! # extern crate wayland_server;
13//! #
14//! use smithay::delegate_layer_shell;
15//! use smithay::wayland::shell::wlr_layer::{WlrLayerShellState, WlrLayerShellHandler, LayerSurface, Layer};
16//! use smithay::reexports::wayland_server::protocol::wl_output::WlOutput;
17//!
18//! # struct State { layer_shell_state: WlrLayerShellState }
19//! # let mut display = wayland_server::Display::<State>::new().unwrap();
20//! let layer_shell_state = WlrLayerShellState::new::<State>(
21//!     &display.handle(),
22//! );
23//!
24//! // - Put your layer_shell_state into your `State`.
25//! // - And implement `LayerShellHandler
26//! impl WlrLayerShellHandler for State {
27//!     fn shell_state(&mut self) -> &mut WlrLayerShellState {
28//!         &mut self.layer_shell_state
29//!     }
30//!
31//!     fn new_layer_surface(
32//!         &mut self,
33//!         surface: LayerSurface,
34//!         output: Option<WlOutput>,
35//!         layer: Layer,
36//!         namespace: String,
37//!     ) {
38//!         # let _ = (surface, output, layer, namespace);
39//!         // your implementation
40//!     }
41//! }
42//! // let smithay implement wayland_server::DelegateDispatch
43//! delegate_layer_shell!(State);
44//!
45//! // You're now ready to go!
46//! ```

90/// Attributes for layer surface
91#[derive(Debug)]
92pub struct LayerSurfaceAttributes {
93    surface: zwlr_layer_surface_v1::ZwlrLayerSurfaceV1,
94    /// Holds the state if the surface has sent the initial
95    /// configure event to the client. It is expected that
96    /// during the first commit a initial
97    /// configure event is sent to the client
98    pub initial_configure_sent: bool,
99    /// Holds the configures the server has sent out
100    /// to the client waiting to be acknowledged by
101    /// the client. All pending configures that are older
102    /// than the acknowledged one will be discarded during
103    /// processing layer_surface.ack_configure.
104    pending_configures: Vec<LayerSurfaceConfigure>,
105    /// Holds the pending state as set by the server.
106    pub server_pending: Option<LayerSurfaceState>,
107    /// Holds the last configure that has been acknowledged by the client. This state should be
108    /// cloned to the current during a commit. Note that this state can be newer than the last
109    /// acked state at the time of the last commit.
110    pub last_acked: Option<LayerSurfaceConfigure>,
111}
112
113impl LayerSurfaceAttributes {
114    fn new(surface: zwlr_layer_surface_v1::ZwlrLayerSurfaceV1) -> Self {
115        Self {
116            surface,
117            initial_configure_sent: false,
118            pending_configures: Vec::new(),
119            server_pending: None,
120            last_acked: None,
121        }
122    }
123
124    fn ack_configure(&mut self, serial: Serial) -> Option<LayerSurfaceConfigure> {
125        let configure = self
126            .pending_configures
127            .iter()
128            .find(|configure| configure.serial == serial)
129            .cloned()?;
130
131        self.last_acked = Some(configure.clone());
132
133        self.pending_configures.retain(|c| c.serial > serial);
134        Some(configure)
135    }
136
137    fn reset(&mut self) {
138        self.initial_configure_sent = false;
139        self.pending_configures = Vec::new();
140        self.server_pending = None;
141        self.last_acked = None;
142    }
143
144    fn current_server_state(&self) -> LayerSurfaceState {
145        self.pending_configures
146            .last()
147            .map(|c| &c.state)
148            .or(self.last_acked.as_ref().map(|c| &c.state))
149            .cloned()
150            .unwrap_or_default()
151    }
152
153    fn has_pending_changes(&self) -> bool {
154        self.server_pending
155            .as_ref()
156            .map(|s| *s != self.current_server_state())
157            .unwrap_or(false)
158    }
159}

299impl LayerSurface {
300    /// Checks if the surface is still alive
301    #[inline]
302    pub fn alive(&self) -> bool {
303        self.wl_surface.alive() && self.shell_surface.alive()
304    }
305
306    /// Gets the current pending state for a configure
307    ///
308    /// Returns `Some` if either no initial configure has been sent or
309    /// the `server_pending` is `Some` and different from the last pending
310    /// configure or `last_acked` if there is no pending
311    ///
312    /// Returns `None` if either no `server_pending` or the pending
313    /// has already been sent to the client or the pending is equal
314    /// to the `last_acked`
315    fn get_pending_state(&self, attributes: &mut LayerSurfaceAttributes) -> Option<LayerSurfaceState> {
316        if !attributes.initial_configure_sent {
317            return Some(
318                attributes
319                    .server_pending
320                    .take()
321                    .unwrap_or_else(|| attributes.current_server_state().clone()),
322            );
323        }
324
325        // Check if the state really changed, it is possible
326        // that with_pending_state has been called without
327        // modifying the state.
328        if !attributes.has_pending_changes() {
329            return None;
330        }
331
332        attributes.server_pending.take()
333    }
334
335    /// Send a pending configure event to this layer surface to suggest it a new configuration
336    ///
337    /// If changes have occurred a configure event will be send to the clients and the serial will be returned
338    /// (for tracking the configure in [`WlrLayerShellHandler::ack_configure`] if desired).
339    /// If no changes occurred no event will be send and `None` will be returned.
340    ///
341    /// See [`send_configure`](LayerSurface::send_configure) and [`has_pending_changes`](LayerSurface::has_pending_changes)
342    /// for more information.
343    pub fn send_pending_configure(&self) -> Option<Serial> {
344        if self.has_pending_changes() {
345            Some(self.send_configure())
346        } else {
347            None
348        }
349    }
350
351    /// Send a configure event to this layer surface to suggest it a new configuration
352    ///
353    /// The serial of this configure will be tracked waiting for the client to ACK it.
354    ///
355    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
356    /// method.
357    ///
358    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
359    /// [`send_pending_configure`](LayerSurface::send_pending_configure)
360    pub fn send_configure(&self) -> Serial {
361        let configure = compositor::with_states(&self.wl_surface, |states| {
362            let mut attributes = states
363                .data_map
364                .get::<Mutex<LayerSurfaceAttributes>>()
365                .unwrap()
366                .lock()
367                .unwrap();
368
369            let state = self
370                .get_pending_state(&mut attributes)
371                .unwrap_or_else(|| attributes.current_server_state().clone());
372
373            let configure = LayerSurfaceConfigure {
374                serial: SERIAL_COUNTER.next_serial(),
375                state,
376            };
377
378            attributes.pending_configures.push(configure.clone());
379            attributes.initial_configure_sent = true;
380
381            configure
382        });
383
384        // send surface configure
385        let (width, height) = configure.state.size.unwrap_or_default().into();
386        let serial = configure.serial;
387        self.shell_surface
388            .configure(serial.into(), width as u32, height as u32);
389        serial
390    }
391
392    /// Send a "close" event to the client
393    pub fn send_close(&self) {
394        self.shell_surface.closed()
395    }

405    /// Allows the pending state of this layer to
406    /// be manipulated.
407    ///
408    /// This should be used to inform the client about size and state changes,
409    /// for example after a resize request from the client.
410    ///
411    /// The state will be sent to the client when calling [`send_configure`](#method.send_configure).
412    pub fn with_pending_state<F, T>(&self, f: F) -> T
413    where
414        F: FnOnce(&mut LayerSurfaceState) -> T,
415    {
416        compositor::with_states(&self.wl_surface, |states| {
417            let mut attributes = states
418                .data_map
419                .get::<Mutex<LayerSurfaceAttributes>>()
420                .unwrap()
421                .lock()
422                .unwrap();
423            if attributes.server_pending.is_none() {
424                attributes.server_pending = Some(attributes.current_server_state().clone());
425            }
426
427            let server_pending = attributes.server_pending.as_mut().unwrap();
428            f(server_pending)
429        })
430    }
431
432    /// Tests this [`LayerSurface`] for pending changes
433    ///
434    /// Returns `true` if [`with_pending_state`](LayerSurface::with_pending_state) was used to manipulate the state
435    /// and resulted in a different state or if the initial configure is still pending.
436    pub fn has_pending_changes(&self) -> bool {
437        compositor::with_states(&self.wl_surface, |states| {
438            let attributes = states
439                .data_map
440                .get::<Mutex<LayerSurfaceAttributes>>()
441                .unwrap()
442                .lock()
443                .unwrap();
444
445            !attributes.initial_configure_sent || attributes.has_pending_changes()
446        })
447    }

449    /// Provides access to the current committed cached state.
450    pub fn with_cached_state<F, T>(&self, f: F) -> T
451    where
452        F: FnOnce(&LayerSurfaceCachedState) -> T,
453    {
454        compositor::with_states(&self.wl_surface, |states| {
455            let mut guard = states.cached_state.get::<LayerSurfaceCachedState>();
456            f(guard.current())
457        })
458    }

460    /// Provides access to the current committed state.
461    ///
462    /// This is the state that the client last acked before making the current commit.
463    pub fn with_committed_state<F, T>(&self, f: F) -> T
464    where
465        F: FnOnce(Option<&LayerSurfaceState>) -> T,
466    {
467        self.with_cached_state(move |state| f(state.last_acked.as_ref().map(|c| &c.state)))
468    }

470    /// Handles the role specific commit error checking
471    ///
472    /// This should be called when the underlying WlSurface
473    /// handles a wl_surface.commit request.
474    pub(crate) fn pre_commit_hook<D: 'static>(
475        _state: &mut D,
476        _dh: &DisplayHandle,
477        surface: &wl_surface::WlSurface,
478    ) {
479        let _span = trace_span!("layer-surface pre-commit", surface = %surface.id()).entered();
480
481        compositor::with_states(surface, |states| {
482            let mut role = states.data_map.get::<LayerSurfaceData>().unwrap().lock().unwrap();
483
484            let mut guard_layer = states.cached_state.get::<LayerSurfaceCachedState>();
485            let pending = guard_layer.pending();
486
487            if pending.size.w == 0 && !pending.anchor.anchored_horizontally() {
488                role.surface.post_error(
489                    zwlr_layer_surface_v1::Error::InvalidSize,
490                    "width 0 requested without setting left and right anchors",
491                );
492                return;
493            }
494
495            if pending.size.h == 0 && !pending.anchor.anchored_vertically() {
496                role.surface.post_error(
497                    zwlr_layer_surface_v1::Error::InvalidSize,
498                    "height 0 requested without setting top and bottom anchors",
499                );
500                return;
501            }
502
503            if let Some(edge) = pending.exclusive_edge {
504                if !pending.anchor.contains(edge) {
505                    role.surface.post_error(
506                        zwlr_layer_surface_v1::Error::InvalidExclusiveEdge,
507                        "exclusive edge is not an anchor",
508                    );
509                    return;
510                }
511            }
512
513            // The presence of last_acked always follows the buffer assignment because the
514            // surface is not allowed to attach a buffer without acking the initial configure.
515            let had_buffer_before = pending.last_acked.is_some();
516
517            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
518            let has_buffer = match &guard_surface.pending().buffer {
519                Some(BufferAssignment::NewBuffer(_)) => true,
520                Some(BufferAssignment::Removed) => false,
521                None => had_buffer_before,
522            };
523            // Need to check had_buffer_before in case the client attaches a null buffer for the
524            // initial commit---we don't want to consider that as "got unmapped" and reset role.
525            // Reproducer: waybar.
526            let got_unmapped = had_buffer_before && !has_buffer;
527
528            if has_buffer {
529                let Some(last_acked) = role.last_acked.clone() else {
530                    role.surface.post_error(
531                        zwlr_layer_surface_v1::Error::InvalidSurfaceState,
532                        "must ack the initial configure before attaching buffer",
533                    );
534                    return;
535                };
536
537                // The surface remains, or became mapped, track the last acked state.
538                pending.last_acked = Some(last_acked);
539            } else {
540                // The surface remains, or became, unmapped, meaning that it's in the initial
541                // configure stage.
542                pending.last_acked = None;
543            }
544
545            if got_unmapped {
546                trace!(
547                    "got unmapped; resetting role and cached state; have {} pending configures",
548                    role.pending_configures.len()
549                );
550
551                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
552                // the list of pending configures because there's no way for a surface to tell
553                // an in-flight configure apart from our next initial configure after unmapping.
554                let pending_configures = std::mem::take(&mut role.pending_configures);
555                role.reset();
556                role.pending_configures = pending_configures;
557                *guard_layer.pending() = Default::default();
558            }
559        });
560    }

583/// Macro to delegate implementation of wlr layer shell to [`WlrLayerShellState`].
584///
585/// You must also implement [`WlrLayerShellHandler`] to use this.
586#[macro_export]
587macro_rules! delegate_layer_shell {
588    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
589        type __ZwlrLayerShellV1 =
590            $crate::reexports::wayland_protocols_wlr::layer_shell::v1::server::zwlr_layer_shell_v1::ZwlrLayerShellV1;
591        type __ZwlrLayerShellSurfaceV1 =
592            $crate::reexports::wayland_protocols_wlr::layer_shell::v1::server::zwlr_layer_surface_v1::ZwlrLayerSurfaceV1;
593
594        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
595            __ZwlrLayerShellV1: ()
596        ] => $crate::wayland::shell::wlr_layer::WlrLayerShellState);
597        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
598            __ZwlrLayerShellSurfaceV1: $crate::wayland::shell::wlr_layer::WlrLayerSurfaceUserData
599        ] => $crate::wayland::shell::wlr_layer::WlrLayerShellState);
600
601        $crate::reexports::wayland_server::delegate_global_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
602            __ZwlrLayerShellV1: $crate::wayland::shell::wlr_layer::WlrLayerShellGlobalData
603        ] => $crate::wayland::shell::wlr_layer::WlrLayerShellState);
604    };
