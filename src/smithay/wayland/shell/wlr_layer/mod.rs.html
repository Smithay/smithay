1//! Utilities for handling shell surfaces with the `wlr_layer_shell` protocol
2//!
3//! This interface should be suitable for the implementation of many desktop shell components,
4//! and a broad number of other applications that interact with the desktop.
5//!
6//! ### Initialization
7//!
8//! To initialize this handler, create the [`WlrLayerShellState`], store it inside your `State` and
9//! implement the [`WlrLayerShellHandler`], as shown in this example:
10//!
11//! ```no_run
12//! # extern crate wayland_server;
13//! #
14//! use smithay::delegate_layer_shell;
15//! use smithay::wayland::shell::wlr_layer::{WlrLayerShellState, WlrLayerShellHandler, LayerSurface, Layer};
16//! use smithay::reexports::wayland_server::protocol::wl_output::WlOutput;
17//!
18//! # struct State { layer_shell_state: WlrLayerShellState }
19//! # let mut display = wayland_server::Display::<State>::new().unwrap();
20//! let layer_shell_state = WlrLayerShellState::new::<State>(
21//!     &display.handle(),
22//! );
23//!
24//! // - Put your layer_shell_state into your `State`.
25//! // - And implement `LayerShellHandler
26//! impl WlrLayerShellHandler for State {
27//!     fn shell_state(&mut self) -> &mut WlrLayerShellState {
28//!         &mut self.layer_shell_state
29//!     }
30//!
31//!     fn new_layer_surface(
32//!         &mut self,
33//!         surface: LayerSurface,
34//!         output: Option<WlOutput>,
35//!         layer: Layer,
36//!         namespace: String,
37//!     ) {
38//!         # let _ = (surface, output, layer, namespace);
39//!         // your implementation
40//!     }
41//! }
42//! // let smithay implement wayland_server::DelegateDispatch
43//! delegate_layer_shell!(State);
44//!
45//! // You're now ready to go!
46//! ```

90/// Attributes for layer surface
91#[derive(Debug)]
92pub struct LayerSurfaceAttributes {
93    surface: zwlr_layer_surface_v1::ZwlrLayerSurfaceV1,
94    /// Holds the state if the surface has sent the initial
95    /// configure event to the client. It is expected that
96    /// during the first commit a initial
97    /// configure event is sent to the client
98    pub initial_configure_sent: bool,
99    /// Holds the configures the server has sent out
100    /// to the client waiting to be acknowledged by
101    /// the client. All pending configures that are older
102    /// than the acknowledged one will be discarded during
103    /// processing layer_surface.ack_configure.
104    pending_configures: Vec<LayerSurfaceConfigure>,
105    /// Holds the pending state as set by the server.
106    pub server_pending: Option<LayerSurfaceState>,
107    /// Holds the last configure that has been acknowledged by the client. This state should be
108    /// cloned to the current during a commit. Note that this state can be newer than the last
109    /// acked state at the time of the last commit.
110    pub last_acked: Option<LayerSurfaceConfigure>,
111}
112
113impl LayerSurfaceAttributes {
114    fn new(surface: zwlr_layer_surface_v1::ZwlrLayerSurfaceV1) -> Self {
115        Self {
116            surface,
117            initial_configure_sent: false,
118            pending_configures: Vec::new(),
119            server_pending: None,
120            last_acked: None,
121        }
122    }
123
124    fn ack_configure(&mut self, serial: Serial) -> Option<LayerSurfaceConfigure> {
125        let configure = self
126            .pending_configures
127            .iter()
128            .find(|configure| configure.serial == serial)
129            .cloned()?;
130
131        self.last_acked = Some(configure);
132
133        self.pending_configures.retain(|c| c.serial > serial);
134        Some(configure)
135    }
136
137    fn reset(&mut self) {
138        self.initial_configure_sent = false;
139        self.pending_configures = Vec::new();
140        self.server_pending = None;
141        self.last_acked = None;
142    }
143
144    fn current_server_state(&self) -> LayerSurfaceState {
145        self.pending_configures
146            .last()
147            .map(|c| &c.state)
148            .or(self.last_acked.as_ref().map(|c| &c.state))
149            .cloned()
150            .unwrap_or_default()
151    }
152
153    fn has_pending_changes(&self) -> bool {
154        self.server_pending
155            .as_ref()
156            .map(|s| *s != self.current_server_state())
157            .unwrap_or(false)
158    }
159}

303impl LayerSurface {
304    /// Checks if the surface is still alive
305    #[inline]
306    pub fn alive(&self) -> bool {
307        self.wl_surface.alive() && self.shell_surface.alive()
308    }
309
310    /// Gets the current pending state for a configure
311    ///
312    /// Returns `Some` if either no initial configure has been sent or
313    /// the `server_pending` is `Some` and different from the last pending
314    /// configure or `last_acked` if there is no pending
315    ///
316    /// Returns `None` if either no `server_pending` or the pending
317    /// has already been sent to the client or the pending is equal
318    /// to the `last_acked`
319    fn get_pending_state(&self, attributes: &mut LayerSurfaceAttributes) -> Option<LayerSurfaceState> {
320        if !attributes.initial_configure_sent {
321            return Some(
322                attributes
323                    .server_pending
324                    .take()
325                    .unwrap_or_else(|| attributes.current_server_state()),
326            );
327        }
328
329        // Check if the state really changed, it is possible
330        // that with_pending_state has been called without
331        // modifying the state.
332        if !attributes.has_pending_changes() {
333            return None;
334        }
335
336        attributes.server_pending.take()
337    }
338
339    /// Send a pending configure event to this layer surface to suggest it a new configuration
340    ///
341    /// If changes have occurred a configure event will be send to the clients and the serial will be returned
342    /// (for tracking the configure in [`WlrLayerShellHandler::ack_configure`] if desired).
343    /// If no changes occurred no event will be send and `None` will be returned.
344    ///
345    /// See [`send_configure`](LayerSurface::send_configure) and [`has_pending_changes`](LayerSurface::has_pending_changes)
346    /// for more information.
347    pub fn send_pending_configure(&self) -> Option<Serial> {
348        if self.has_pending_changes() {
349            Some(self.send_configure())
350        } else {
351            None
352        }
353    }
354
355    /// Send a configure event to this layer surface to suggest it a new configuration
356    ///
357    /// The serial of this configure will be tracked waiting for the client to ACK it.
358    ///
359    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
360    /// method.
361    ///
362    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
363    /// [`send_pending_configure`](LayerSurface::send_pending_configure)
364    pub fn send_configure(&self) -> Serial {
365        let configure = compositor::with_states(&self.wl_surface, |states| {
366            let mut attributes = states
367                .data_map
368                .get::<Mutex<LayerSurfaceAttributes>>()
369                .unwrap()
370                .lock()
371                .unwrap();
372
373            let state = self
374                .get_pending_state(&mut attributes)
375                .unwrap_or_else(|| attributes.current_server_state());
376
377            let configure = LayerSurfaceConfigure {
378                serial: SERIAL_COUNTER.next_serial(),
379                state,
380            };
381
382            attributes.pending_configures.push(configure);
383            attributes.initial_configure_sent = true;
384
385            configure
386        });
387
388        // send surface configure
389        let (width, height) = configure.state.size.unwrap_or_default().into();
390        let serial = configure.serial;
391        self.shell_surface
392            .configure(serial.into(), width as u32, height as u32);
393        serial
394    }
395
396    /// Send a "close" event to the client
397    pub fn send_close(&self) {
398        self.shell_surface.closed()
399    }

409    /// Allows the pending state of this layer to
410    /// be manipulated.
411    ///
412    /// This should be used to inform the client about size and state changes,
413    /// for example after a resize request from the client.
414    ///
415    /// The state will be sent to the client when calling [`send_configure`](#method.send_configure).
416    pub fn with_pending_state<F, T>(&self, f: F) -> T
417    where
418        F: FnOnce(&mut LayerSurfaceState) -> T,
419    {
420        compositor::with_states(&self.wl_surface, |states| {
421            let mut attributes = states
422                .data_map
423                .get::<Mutex<LayerSurfaceAttributes>>()
424                .unwrap()
425                .lock()
426                .unwrap();
427            if attributes.server_pending.is_none() {
428                attributes.server_pending = Some(attributes.current_server_state());
429            }
430
431            let server_pending = attributes.server_pending.as_mut().unwrap();
432            f(server_pending)
433        })
434    }
435
436    /// Tests this [`LayerSurface`] for pending changes
437    ///
438    /// Returns `true` if [`with_pending_state`](LayerSurface::with_pending_state) was used to manipulate the state
439    /// and resulted in a different state or if the initial configure is still pending.
440    pub fn has_pending_changes(&self) -> bool {
441        compositor::with_states(&self.wl_surface, |states| {
442            let attributes = states
443                .data_map
444                .get::<Mutex<LayerSurfaceAttributes>>()
445                .unwrap()
446                .lock()
447                .unwrap();
448
449            !attributes.initial_configure_sent || attributes.has_pending_changes()
450        })
451    }

453    /// Provides access to the current committed cached state.
454    pub fn with_cached_state<F, T>(&self, f: F) -> T
455    where
456        F: FnOnce(&LayerSurfaceCachedState) -> T,
457    {
458        compositor::with_states(&self.wl_surface, |states| {
459            let mut guard = states.cached_state.get::<LayerSurfaceCachedState>();
460            f(guard.current())
461        })
462    }
463
464    /// Provides access to the current committed state.
465    ///
466    /// This is the state that the client last acked before making the current commit.
467    pub fn with_committed_state<F, T>(&self, f: F) -> T
468    where
469        F: FnOnce(Option<&LayerSurfaceState>) -> T,
470    {
471        self.with_cached_state(move |state| f(state.last_acked.as_ref().map(|c| &c.state)))
472    }

474    /// Handles the role specific commit error checking
475    ///
476    /// This should be called when the underlying WlSurface
477    /// handles a wl_surface.commit request.
478    pub(crate) fn pre_commit_hook<D: 'static>(
479        _state: &mut D,
480        _dh: &DisplayHandle,
481        surface: &wl_surface::WlSurface,
482    ) {
483        let _span = trace_span!("layer-surface pre-commit", surface = %surface.id()).entered();
484
485        compositor::with_states(surface, |states| {
486            let mut role = states.data_map.get::<LayerSurfaceData>().unwrap().lock().unwrap();
487
488            let mut guard_layer = states.cached_state.get::<LayerSurfaceCachedState>();
489            let pending = guard_layer.pending();
490
491            if pending.size.w == 0 && !pending.anchor.anchored_horizontally() {
492                role.surface.post_error(
493                    zwlr_layer_surface_v1::Error::InvalidSize,
494                    "width 0 requested without setting left and right anchors",
495                );
496                return;
497            }
498
499            if pending.size.h == 0 && !pending.anchor.anchored_vertically() {
500                role.surface.post_error(
501                    zwlr_layer_surface_v1::Error::InvalidSize,
502                    "height 0 requested without setting top and bottom anchors",
503                );
504                return;
505            }
506
507            if let Some(edge) = pending.exclusive_edge {
508                if !pending.anchor.contains(edge) {
509                    role.surface.post_error(
510                        zwlr_layer_surface_v1::Error::InvalidExclusiveEdge,
511                        "exclusive edge is not an anchor",
512                    );
513                    return;
514                }
515            }
516
517            // The presence of last_acked always follows the buffer assignment because the
518            // surface is not allowed to attach a buffer without acking the initial configure.
519            let had_buffer_before = pending.last_acked.is_some();
520
521            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
522            let has_buffer = match &guard_surface.pending().buffer {
523                Some(BufferAssignment::NewBuffer(_)) => true,
524                Some(BufferAssignment::Removed) => false,
525                None => had_buffer_before,
526            };
527            // Need to check had_buffer_before in case the client attaches a null buffer for the
528            // initial commit---we don't want to consider that as "got unmapped" and reset role.
529            // Reproducer: waybar.
530            let got_unmapped = had_buffer_before && !has_buffer;
531
532            if has_buffer {
533                let Some(last_acked) = role.last_acked else {
534                    role.surface.post_error(
535                        zwlr_layer_surface_v1::Error::InvalidSurfaceState,
536                        "must ack the initial configure before attaching buffer",
537                    );
538                    return;
539                };
540
541                // The surface remains, or became mapped, track the last acked state.
542                pending.last_acked = Some(last_acked);
543            } else {
544                // The surface remains, or became, unmapped, meaning that it's in the initial
545                // configure stage.
546                pending.last_acked = None;
547            }
548
549            if got_unmapped {
550                trace!(
551                    "got unmapped; resetting role and cached state; have {} pending configures",
552                    role.pending_configures.len()
553                );
554
555                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
556                // the list of pending configures because there's no way for a surface to tell
557                // an in-flight configure apart from our next initial configure after unmapping.
558                let pending_configures = std::mem::take(&mut role.pending_configures);
559                role.reset();
560                role.pending_configures = pending_configures;
561                *guard_layer.pending() = Default::default();
562            }
563        });
564    }

587/// Macro to delegate implementation of wlr layer shell to [`WlrLayerShellState`].
588///
589/// You must also implement [`WlrLayerShellHandler`] to use this.
590#[macro_export]
591macro_rules! delegate_layer_shell {
592    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
593        const _: () = {
594            use $crate::{
595                reexports::{
596                    wayland_protocols_wlr::layer_shell::v1::server::{
597                        zwlr_layer_shell_v1::ZwlrLayerShellV1, zwlr_layer_surface_v1::ZwlrLayerSurfaceV1,
598                    },
599                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
600                },
601                wayland::shell::wlr_layer::{
602                    WlrLayerShellGlobalData, WlrLayerShellState, WlrLayerSurfaceUserData,
603                },
604            };
605
606            delegate_dispatch!(
607                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
608                $ty: [ZwlrLayerShellV1: ()] => WlrLayerShellState
609            );
610
611            delegate_dispatch!(
612                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
613                $ty: [ZwlrLayerSurfaceV1: WlrLayerSurfaceUserData] => WlrLayerShellState
614            );
615
616            delegate_global_dispatch!(
617                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
618                $ty: [ZwlrLayerShellV1: WlrLayerShellGlobalData] => WlrLayerShellState
619            );
620        };
621    };
