1//! KDE Window decoration manager
2//!
3//! This interface allows a compositor to announce support for KDE's legacy server-side decorations.
4//!
5//! A client can use this protocol to request being decorated by a supporting compositor.
6//!
7//! ```
8//! extern crate wayland_server;
9//! extern crate smithay;
10//!
11//! use smithay::wayland::shell::kde::decoration::{KdeDecorationHandler, KdeDecorationState};
12//! use wayland_protocols_misc::server_decoration::server::org_kde_kwin_server_decoration_manager::Mode;
13//!
14//! # struct State { kde_decoration_state: KdeDecorationState };
15//! # let mut display = wayland_server::Display::<State>::new().unwrap();
16//!
17//! // Create the new KdeDecorationState.
18//! let state = KdeDecorationState::new::<State>(&display.handle(), Mode::Server);
19//!
20//! // Insert KdeDecorationState into your compositor state.
21//! // …
22//!
23//! // Implement KDE server decoration handlers.
24//! impl KdeDecorationHandler for State {
25//!     fn kde_decoration_state(&self) -> &KdeDecorationState {
26//!         &self.kde_decoration_state
27//!     }
28//! }
29//!
30//! smithay::delegate_dispatch2!(State);
31//! ```
32

45/// KDE server decoration handler.
46pub trait KdeDecorationHandler {
47    /// Return the KDE server decoration state.
48    fn kde_decoration_state(&self) -> &KdeDecorationState;
49
50    /// Handle new decoration object creation.
51    ///
52    /// Called whenever a new decoration object is created, usually this happens when a new window
53    /// is opened.
54    fn new_decoration(&mut self, _surface: &WlSurface, _decoration: &OrgKdeKwinServerDecoration) {}
55
56    /// Handle surface decoration mode requests.
57    ///
58    /// Called when a surface requests a specific decoration mode or acknowledged the compositor's
59    /// decoration request.
60    ///
61    /// **It is up to the compositor to prevent feedback loops**, a client is free to ignore modes
62    /// suggested by [`OrgKdeKwinServerDecoration::mode`] and instead request their preferred mode
63    /// instead.
64    fn request_mode(
65        &mut self,
66        _surface: &WlSurface,
67        decoration: &OrgKdeKwinServerDecoration,
68        mode: WEnum<Mode>,
69    ) {
70        if let WEnum::Value(mode) = mode {
71            decoration.mode(mode);
72        }
73    }
74
75    /// Handle decoration object removal for a surface.
76    fn release(&mut self, _decoration: &OrgKdeKwinServerDecoration, _surface: &WlSurface) {}
77}

79/// KDE server decoration state.
80#[derive(Debug)]
81pub struct KdeDecorationState {
82    pub(crate) default_mode: DefaultMode,
83
84    kde_decoration_manager: GlobalId,
85}

87/// Data associated with a KdeDecorationManager global.
88#[allow(missing_debug_implementations)]
89pub struct KdeDecorationManagerGlobalData {
90    pub(crate) filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
91}
92
93#[doc(hidden)]
94#[derive(Debug)]
95pub struct KwinServerDecorationData(pub(crate) WlSurface);
96
97impl KdeDecorationState {
98    /// Create a new KDE server decoration global.
99    pub fn new<D>(display: &DisplayHandle, default_mode: DefaultMode) -> Self
100    where
101        D: GlobalDispatch<OrgKdeKwinServerDecorationManager, KdeDecorationManagerGlobalData>
102            + Dispatch<OrgKdeKwinServerDecorationManager, GlobalData>
103            + Dispatch<OrgKdeKwinServerDecoration, KwinServerDecorationData>
104            + KdeDecorationHandler
105            + 'static,
106    {
107        Self::new_with_filter::<D, _>(display, default_mode, |_| true)
108    }
109
110    /// Create a new KDE server decoration global with a filter.
111    ///
112    /// Filters can be used to limit visibility of a global to certain clients.
113    pub fn new_with_filter<D, F>(display: &DisplayHandle, default_mode: DefaultMode, filter: F) -> Self
114    where
115        D: GlobalDispatch<OrgKdeKwinServerDecorationManager, KdeDecorationManagerGlobalData>
116            + Dispatch<OrgKdeKwinServerDecorationManager, GlobalData>
117            + Dispatch<OrgKdeKwinServerDecoration, KwinServerDecorationData>
118            + KdeDecorationHandler
119            + 'static,
120        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
121    {
122        let data = KdeDecorationManagerGlobalData {
123            filter: Box::new(filter),
124        };
125        let kde_decoration_manager =
126            display.create_global::<D, OrgKdeKwinServerDecorationManager, _>(1, data);
127
128        Self {
129            kde_decoration_manager,
130            default_mode,
131        }
132    }
133
134    /// Returns the id of the [`OrgKdeKwinServerDecorationManager`] global.
135    pub fn global(&self) -> GlobalId {
136        self.kde_decoration_manager.clone()
137    }
138
139    /// Updates the default decoration mode.
140    ///
141    /// The new default mode will only be sent to new binds to the manager object; it is up to you
142    /// to update any existing decoration instances with the new mode, if desired.
143    pub fn set_default_mode(&mut self, default_mode: DefaultMode) {
144        self.default_mode = default_mode;
145    }
146}
