1//! KDE Window decoration manager
2//!
3//! This interface allows a compositor to announce support for KDE's legacy server-side decorations.
4//!
5//! A client can use this protocol to request being decorated by a supporting compositor.
6//!
7//! ```
8//! extern crate wayland_server;
9//! extern crate smithay;
10//!
11//! use smithay::delegate_kde_decoration;
12//! use smithay::wayland::shell::kde::decoration::{KdeDecorationHandler, KdeDecorationState};
13//! use wayland_protocols_misc::server_decoration::server::org_kde_kwin_server_decoration_manager::Mode;
14//!
15//! # struct State { kde_decoration_state: KdeDecorationState };
16//! # let mut display = wayland_server::Display::<State>::new().unwrap();
17//!
18//! // Create the new KdeDecorationState.
19//! let state = KdeDecorationState::new::<State>(&display.handle(), Mode::Server);
20//!
21//! // Insert KdeDecorationState into your compositor state.
22//! // â€¦
23//!
24//! // Implement KDE server decoration handlers.
25//! impl KdeDecorationHandler for State {
26//!     fn kde_decoration_state(&self) -> &KdeDecorationState {
27//!         &self.kde_decoration_state
28//!     }
29//! }
30//!
31//! delegate_kde_decoration!(State);
32//! ```

44/// KDE server decoration handler.
45pub trait KdeDecorationHandler {
46    /// Return the KDE server decoration state.
47    fn kde_decoration_state(&self) -> &KdeDecorationState;
48
49    /// Handle new decoration object creation.
50    ///
51    /// Called whenever a new decoration object is created, usually this happens when a new window
52    /// is opened.
53    fn new_decoration(&mut self, _surface: &WlSurface, _decoration: &OrgKdeKwinServerDecoration) {}
54
55    /// Handle surface decoration mode requests.
56    ///
57    /// Called when a surface requests a specific decoration mode or acknowledged the compositor's
58    /// decoration request.
59    ///
60    /// **It is up to the compositor to prevent feedback loops**, a client is free to ignore modes
61    /// suggested by [`OrgKdeKwinServerDecoration::mode`] and instead request their preferred mode
62    /// instead.
63    fn request_mode(
64        &mut self,
65        _surface: &WlSurface,
66        decoration: &OrgKdeKwinServerDecoration,
67        mode: WEnum<Mode>,
68    ) {
69        if let WEnum::Value(mode) = mode {
70            decoration.mode(mode);
71        }
72    }
73
74    /// Handle decoration object removal for a surface.
75    fn release(&mut self, _decoration: &OrgKdeKwinServerDecoration, _surface: &WlSurface) {}
76}

78/// KDE server decoration state.
79#[derive(Debug)]
80pub struct KdeDecorationState {
81    pub(crate) default_mode: DefaultMode,
82
83    kde_decoration_manager: GlobalId,
84}
85
86/// Data associated with a KdeDecorationManager global.
87#[allow(missing_debug_implementations)]
88pub struct KdeDecorationManagerGlobalData {
89    pub(crate) filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
90}
91
92impl KdeDecorationState {
93    /// Create a new KDE server decoration global.
94    pub fn new<D>(display: &DisplayHandle, default_mode: DefaultMode) -> Self
95    where
96        D: GlobalDispatch<OrgKdeKwinServerDecorationManager, KdeDecorationManagerGlobalData>
97            + Dispatch<OrgKdeKwinServerDecorationManager, ()>
98            + Dispatch<OrgKdeKwinServerDecoration, WlSurface>
99            + KdeDecorationHandler
100            + 'static,
101    {
102        Self::new_with_filter::<D, _>(display, default_mode, |_| true)
103    }
104
105    /// Create a new KDE server decoration global with a filter.
106    ///
107    /// Filters can be used to limit visibility of a global to certain clients.
108    pub fn new_with_filter<D, F>(display: &DisplayHandle, default_mode: DefaultMode, filter: F) -> Self
109    where
110        D: GlobalDispatch<OrgKdeKwinServerDecorationManager, KdeDecorationManagerGlobalData>
111            + Dispatch<OrgKdeKwinServerDecorationManager, ()>
112            + Dispatch<OrgKdeKwinServerDecoration, WlSurface>
113            + KdeDecorationHandler
114            + 'static,
115        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
116    {
117        let data = KdeDecorationManagerGlobalData {
118            filter: Box::new(filter),
119        };
120        let kde_decoration_manager =
121            display.create_global::<D, OrgKdeKwinServerDecorationManager, _>(1, data);
122
123        Self {
124            kde_decoration_manager,
125            default_mode,
126        }
127    }
128
129    /// Returns the id of the [`OrgKdeKwinServerDecorationManager`] global.
130    pub fn global(&self) -> GlobalId {
131        self.kde_decoration_manager.clone()
132    }
133
134    /// Updates the default decoration mode.
135    ///
136    /// The new default mode will only be sent to new binds to the manager object; it is up to you
137    /// to update any existing decoration instances with the new mode, if desired.
138    pub fn set_default_mode(&mut self, default_mode: DefaultMode) {
139        self.default_mode = default_mode;
140    }
141}
