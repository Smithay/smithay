17impl<D> Dispatch<XdgPopup, XdgShellSurfaceUserData, D> for XdgShellState
18where
19    D: Dispatch<XdgPopup, XdgShellSurfaceUserData>,
20    D: XdgShellHandler,
21    D: 'static,
22{
23    fn request(
24        state: &mut D,
25        _client: &wayland_server::Client,
26        popup: &XdgPopup,
27        request: xdg_popup::Request,
28        data: &XdgShellSurfaceUserData,
29        _dh: &DisplayHandle,
30        _data_init: &mut DataInit<'_, D>,
31    ) {
32        match request {
33            xdg_popup::Request::Destroy => {
34                if let Some(surface_data) = data.xdg_surface.data::<XdgSurfaceUserData>() {
35                    surface_data.has_active_role.store(false, Ordering::Release);
36                }
37            }
38            xdg_popup::Request::Grab { seat, serial } => {
39                let handle = crate::wayland::shell::xdg::PopupSurface {
40                    wl_surface: data.wl_surface.clone(),
41                    shell_surface: popup.clone(),
42                };
43
44                let serial = Serial::from(serial);
45
46                with_states(handle.wl_surface(), |states| {
47                    states
48                        .data_map
49                        .get::<XdgPopupSurfaceData>()
50                        .unwrap()
51                        .lock()
52                        .unwrap()
53                        .requested_grab = Some((seat, serial));
54                });
55            }
56            xdg_popup::Request::Reposition { positioner, token } => {
57                let handle = crate::wayland::shell::xdg::PopupSurface {
58                    wl_surface: data.wl_surface.clone(),
59                    shell_surface: popup.clone(),
60                };
61
62                let positioner_data = *positioner
63                    .data::<XdgPositionerUserData>()
64                    .unwrap()
65                    .inner
66                    .lock()
67                    .unwrap();
68
69                XdgShellHandler::reposition_request(state, handle, positioner_data, token);
70            }
71            _ => unreachable!(),
72        }
73    }
74
75    fn destroyed(state: &mut D, _client_id: ClientId, xdg_popup: &XdgPopup, data: &XdgShellSurfaceUserData) {
76        data.alive_tracker.destroy_notify();
77
78        // remove this surface from the known ones (as well as any leftover dead surface)
79        if let Some(index) = state
80            .xdg_shell_state()
81            .known_popups
82            .iter()
83            .position(|pop| pop.shell_surface.id() == xdg_popup.id())
84        {
85            let popup = state.xdg_shell_state().known_popups.remove(index);
86            let surface = popup.wl_surface().clone();
87            XdgShellHandler::popup_destroyed(state, popup);
88            compositor::with_states(&surface, |states| {
89                *states
90                    .data_map
91                    .get::<XdgPopupSurfaceData>()
92                    .unwrap()
93                    .lock()
94                    .unwrap() = Default::default();
95
96                let mut guard = states.cached_state.get::<SurfaceCachedState>();
97                *guard.pending() = Default::default();
98                *guard.current() = Default::default();
99
100                let mut guard = states.cached_state.get::<PopupCachedState>();
101                *guard.pending() = Default::default();
102                *guard.current() = Default::default();
103            })
104        }
105    }
106}

108pub fn send_popup_configure(resource: &XdgPopup, configure: PopupConfigure) {
109    let data = resource.data::<XdgShellSurfaceUserData>().unwrap();
110
111    let serial = configure.serial;
112    let geometry = configure.state.geometry;
113
114    // Send repositioned if token is set
115    if let Some(token) = configure.reposition_token {
116        resource.repositioned(token);
117    }
118
119    // Send the popup configure
120    resource.configure(geometry.loc.x, geometry.loc.y, geometry.size.w, geometry.size.h);
121
122    // Send the base xdg_surface configure event to mark
123    // the configure as finished
124    data.xdg_surface.configure(serial.into());
125}
