18impl<D> Dispatch<XdgPopup, XdgShellSurfaceUserData, D> for XdgShellState
19where
20    D: Dispatch<XdgPopup, XdgShellSurfaceUserData>,
21    D: XdgShellHandler,
22    D: SeatHandler,
23    D: 'static,
24{
25    fn request(
26        state: &mut D,
27        _client: &wayland_server::Client,
28        popup: &XdgPopup,
29        request: xdg_popup::Request,
30        data: &XdgShellSurfaceUserData,
31        _dh: &DisplayHandle,
32        _data_init: &mut DataInit<'_, D>,
33    ) {
34        match request {
35            xdg_popup::Request::Destroy => {
36                if let Some(surface_data) = data.xdg_surface.data::<XdgSurfaceUserData>() {
37                    surface_data.has_active_role.store(false, Ordering::Release);
38                }
39            }
40            xdg_popup::Request::Grab { seat, serial } => {
41                let handle = crate::wayland::shell::xdg::PopupSurface {
42                    wl_surface: data.wl_surface.clone(),
43                    shell_surface: popup.clone(),
44                };
45
46                let serial = Serial::from(serial);
47
48                with_states(handle.wl_surface(), |states| {
49                    states
50                        .data_map
51                        .get::<XdgPopupSurfaceData>()
52                        .unwrap()
53                        .lock()
54                        .unwrap()
55                        .requested_grab = Some((seat, serial));
56                });
57            }
58            xdg_popup::Request::Reposition { positioner, token } => {
59                let handle = crate::wayland::shell::xdg::PopupSurface {
60                    wl_surface: data.wl_surface.clone(),
61                    shell_surface: popup.clone(),
62                };
63
64                let positioner_data = *positioner
65                    .data::<XdgPositionerUserData>()
66                    .unwrap()
67                    .inner
68                    .lock()
69                    .unwrap();
70
71                XdgShellHandler::reposition_request(state, handle, positioner_data, token);
72            }
73            _ => unreachable!(),
74        }
75    }
76
77    fn destroyed(state: &mut D, _client_id: ClientId, xdg_popup: &XdgPopup, data: &XdgShellSurfaceUserData) {
78        data.alive_tracker.destroy_notify();
79
80        // remove this surface from the known ones (as well as any leftover dead surface)
81        if let Some(index) = state
82            .xdg_shell_state()
83            .known_popups
84            .iter()
85            .position(|pop| pop.shell_surface.id() == xdg_popup.id())
86        {
87            let popup = state.xdg_shell_state().known_popups.remove(index);
88            let surface = popup.wl_surface().clone();
89            XdgShellHandler::popup_destroyed(state, popup);
90            compositor::with_states(&surface, |states| {
91                *states
92                    .data_map
93                    .get::<XdgPopupSurfaceData>()
94                    .unwrap()
95                    .lock()
96                    .unwrap() = Default::default();
97
98                let mut guard = states.cached_state.get::<SurfaceCachedState>();
99                *guard.pending() = Default::default();
100                *guard.current() = Default::default();
101
102                let mut guard = states.cached_state.get::<PopupCachedState>();
103                *guard.pending() = Default::default();
104                *guard.current() = Default::default();
105            })
106        }
107    }
108}

110pub fn send_popup_configure(resource: &XdgPopup, configure: PopupConfigure) {
111    let data = resource.data::<XdgShellSurfaceUserData>().unwrap();
112
113    let serial = configure.serial;
114    let geometry = configure.state.geometry;
115
116    // Send repositioned if token is set
117    if let Some(token) = configure.reposition_token {
118        resource.repositioned(token);
119    }
120
121    // Send the popup configure
122    resource.configure(geometry.loc.x, geometry.loc.y, geometry.size.w, geometry.size.h);
123
124    // Send the base xdg_surface configure event to mark
125    // the configure as finished
126    data.xdg_surface.configure(serial.into());
127}
