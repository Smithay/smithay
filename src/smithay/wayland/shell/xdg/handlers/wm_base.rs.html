18impl<D> GlobalDispatch2<XdgWmBase, D> for GlobalData
19where
20    D: Dispatch<XdgWmBase, XdgWmBaseUserData>,
21    D: Dispatch<XdgSurface, XdgSurfaceUserData>,
22    D: Dispatch<XdgPositioner, XdgPositionerUserData>,
23    D: XdgShellHandler,
24    D: 'static,
25{
26    fn bind(
27        &self,
28        state: &mut D,
29        _dh: &DisplayHandle,
30        _client: &wayland_server::Client,
31        resource: New<XdgWmBase>,
32        data_init: &mut DataInit<'_, D>,
33    ) {
34        let shell = data_init.init(resource, XdgWmBaseUserData::default());
35
36        XdgShellHandler::new_client(state, ShellClient::new(&shell));
37    }

40impl<D> Dispatch2<XdgWmBase, D> for XdgWmBaseUserData
41where
42    D: Dispatch<XdgSurface, XdgSurfaceUserData>,
43    D: Dispatch<XdgPositioner, XdgPositionerUserData>,
44    D: XdgShellHandler,
45    D: 'static,
46{
47    fn request(
48        &self,
49        state: &mut D,
50        _client: &wayland_server::Client,
51        wm_base: &XdgWmBase,
52        request: xdg_wm_base::Request,
53        _dh: &DisplayHandle,
54        data_init: &mut DataInit<'_, D>,
55    ) {
56        match request {
57            xdg_wm_base::Request::CreatePositioner { id } => {
58                data_init.init(id, XdgPositionerUserData::default());
59            }
60            xdg_wm_base::Request::GetXdgSurface { id, surface } => {
61                // Do not assign a role to the surface here
62                // xdg_surface is not role, only xdg_toplevel and
63                // xdg_popup are defined as roles
64                let xdg_surface = data_init.init(
65                    id,
66                    XdgSurfaceUserData {
67                        known_surfaces: self.known_surfaces.clone(),
68                        wl_surface: surface,
69                        wm_base: wm_base.clone(),
70                        has_active_role: AtomicBool::new(false),
71                    },
72                );
73                self.known_surfaces
74                    .lock()
75                    .unwrap()
76                    .insert(xdg_surface.downgrade());
77            }
78            xdg_wm_base::Request::Pong { serial } => {
79                let serial = Serial::from(serial);
80                let valid = {
81                    let mut guard = self.client_data.lock().unwrap();
82                    if guard.pending_ping == Some(serial) {
83                        guard.pending_ping = None;
84                        true
85                    } else {
86                        false
87                    }
88                };
89                if valid {
90                    XdgShellHandler::client_pong(state, ShellClient::new(wm_base));
91                }
92            }
93            xdg_wm_base::Request::Destroy => {
94                if !self.known_surfaces.lock().unwrap().is_empty() {
95                    wm_base.post_error(
96                        xdg_wm_base::Error::DefunctSurfaces,
97                        "xdg_wm_base was destroyed before children",
98                    );
99                }
100            }
101            _ => unreachable!(),
102        }
103    }
104
105    fn destroyed(&self, state: &mut D, _client_id: ClientId, wm_base: &XdgWmBase) {
106        XdgShellHandler::client_destroyed(state, ShellClient::new(wm_base));
107        self.alive_tracker.destroy_notify();
108    }
109}

115        data.alive_tracker.alive()
116    }
