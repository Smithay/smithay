39/// User data of XdgSurface
40#[derive(Debug)]
41pub struct XdgSurfaceUserData {
42    pub(crate) known_surfaces: Arc<Mutex<IndexSet<Weak<xdg_surface::XdgSurface>>>>,
43    pub(crate) wl_surface: wl_surface::WlSurface,
44    pub(crate) wm_base: xdg_wm_base::XdgWmBase,
45    pub(crate) has_active_role: AtomicBool,
46}
47
48impl<D> Dispatch<XdgSurface, XdgSurfaceUserData, D> for XdgShellState
49where
50    D: Dispatch<XdgSurface, XdgSurfaceUserData>,
51    D: Dispatch<XdgToplevel, XdgShellSurfaceUserData>,
52    D: Dispatch<XdgPopup, XdgShellSurfaceUserData>,
53    D: XdgShellHandler,
54    D: 'static,
55{
56    fn request(
57        state: &mut D,
58        _client: &wayland_server::Client,
59        xdg_surface: &XdgSurface,
60        request: xdg_surface::Request,
61        data: &XdgSurfaceUserData,
62        _dh: &DisplayHandle,
63        data_init: &mut DataInit<'_, D>,
64    ) {
65        match request {
66            xdg_surface::Request::Destroy => {
67                data.known_surfaces
68                    .lock()
69                    .unwrap()
70                    .swap_remove(&xdg_surface.downgrade());
71
72                if !data.wl_surface.alive() {
73                    // the wl_surface is destroyed, this means the client is not
74                    // trying to change the role but it's a cleanup (possibly a
75                    // disconnecting client), ignore the protocol check.
76                    return;
77                }
78
79                if compositor::get_role(&data.wl_surface).is_none() {
80                    // No role assigned to the surface, we can exit early.
81                    return;
82                }
83
84                if data.has_active_role.load(Ordering::Acquire) {
85                    data.wm_base.post_error(
86                        xdg_wm_base::Error::Role,
87                        "xdg_surface was destroyed before its role object",
88                    );
89                }
90            }
91            xdg_surface::Request::GetToplevel { id } => {
92                // We now can assign a role to the surface
93                let surface = &data.wl_surface;
94                let shell = &data.wm_base;
95
96                if compositor::give_role(surface, XDG_TOPLEVEL_ROLE).is_err() {
97                    shell.post_error(xdg_wm_base::Error::Role, "Surface already has a role.");
98                    return;
99                }
100
101                data.has_active_role.store(true, Ordering::Release);
102
103                let initial = compositor::with_states(surface, |states| {
104                    let initial = states.data_map.insert_if_missing_threadsafe(|| {
105                        Mutex::new(XdgToplevelSurfaceRoleAttributes::default())
106                    });
107
108                    // Initialize the toplevel capabilities from the default capabilities
109                    let default_capabilities = &state.xdg_shell_state().default_capabilities;
110                    let mut attributes = states
111                        .data_map
112                        .get::<Mutex<XdgToplevelSurfaceRoleAttributes>>()
113                        .unwrap()
114                        .lock()
115                        .unwrap();
116                    if attributes.server_pending.is_none() {
117                        let current = attributes.current_server_state();
118                        attributes.server_pending = Some(current);
119                    }
120                    let current_capabilties = &mut attributes.server_pending.as_mut().unwrap().capabilities;
121                    current_capabilties.replace(default_capabilities.capabilities.iter().copied());
122
123                    initial
124                });
125
126                if initial {
127                    compositor::add_pre_commit_hook::<D, _>(
128                        surface,
129                        super::super::ToplevelSurface::pre_commit_hook,
130                    );
131                }
132
133                let toplevel = data_init.init(
134                    id,
135                    XdgShellSurfaceUserData {
136                        wl_surface: data.wl_surface.clone(),
137                        xdg_surface: xdg_surface.clone(),
138                        wm_base: data.wm_base.clone(),
139                        decoration: Default::default(),
140                        dialog: Default::default(),
141                        alive_tracker: Default::default(),
142                    },
143                );
144
145                state
146                    .xdg_shell_state()
147                    .known_toplevels
148                    .push(make_toplevel_handle(&toplevel));
149
150                let handle = make_toplevel_handle(&toplevel);
151
152                XdgShellHandler::new_toplevel(state, handle);
153            }
154            xdg_surface::Request::GetPopup {
155                id,
156                parent,
157                positioner,
158            } => {
159                let positioner_data = *positioner
160                    .data::<XdgPositionerUserData>()
161                    .unwrap()
162                    .inner
163                    .lock()
164                    .unwrap();
165
166                let parent_surface = parent.map(|parent| {
167                    let parent_data = parent.data::<XdgSurfaceUserData>().unwrap();
168                    parent_data.wl_surface.clone()
169                });
170
171                // We now can assign a role to the surface
172                let surface = &data.wl_surface;
173                let shell = &data.wm_base;
174
175                let attributes = XdgPopupSurfaceRoleAttributes {
176                    parent: parent_surface,
177                    server_pending: Some(PopupState {
178                        // Set the positioner data as the popup geometry
179                        geometry: positioner_data.get_geometry(),
180                        positioner: positioner_data,
181                    }),
182                    ..Default::default()
183                };
184                if compositor::give_role(surface, XDG_POPUP_ROLE).is_err() {
185                    shell.post_error(xdg_wm_base::Error::Role, "Surface already has a role.");
186                    return;
187                }
188
189                data.has_active_role.store(true, Ordering::Release);
190
191                let initial = compositor::with_states(surface, |states| {
192                    let inserted = states.data_map.insert_if_missing_threadsafe(|| {
193                        Mutex::new(XdgPopupSurfaceRoleAttributes::default())
194                    });
195                    *states
196                        .data_map
197                        .get::<XdgPopupSurfaceData>()
198                        .unwrap()
199                        .lock()
200                        .unwrap() = attributes;
201                    inserted
202                });
203
204                if initial {
205                    compositor::add_pre_commit_hook::<D, _>(
206                        surface,
207                        super::super::PopupSurface::pre_commit_hook,
208                    );
209                }
210
211                let popup = data_init.init(
212                    id,
213                    XdgShellSurfaceUserData {
214                        wl_surface: data.wl_surface.clone(),
215                        xdg_surface: xdg_surface.clone(),
216                        wm_base: data.wm_base.clone(),
217                        decoration: Default::default(),
218                        dialog: Default::default(),
219                        alive_tracker: Default::default(),
220                    },
221                );
222
223                state
224                    .xdg_shell_state()
225                    .known_popups
226                    .push(make_popup_handle(&popup));
227
228                let handle = make_popup_handle(&popup);
229
230                XdgShellHandler::new_popup(state, handle, positioner_data);
231            }
232            xdg_surface::Request::SetWindowGeometry { x, y, width, height } => {
233                // Check the role of the surface, this can be either xdg_toplevel
234                // or xdg_popup. If none of the role matches the xdg_surface has no role set
235                // which is a protocol error.
236                let surface = &data.wl_surface;
237
238                let role = compositor::get_role(surface);
239
240                if role.is_none() {
241                    xdg_surface.post_error(
242                        xdg_surface::Error::NotConstructed,
243                        "xdg_surface must have a role.",
244                    );
245                    return;
246                }
247
248                if role != Some(XDG_TOPLEVEL_ROLE) && role != Some(XDG_POPUP_ROLE) {
249                    data.wm_base.post_error(
250                        xdg_wm_base::Error::Role,
251                        "xdg_surface must have a role of xdg_toplevel or xdg_popup.",
252                    );
253                }
254
255                compositor::with_states(surface, |states| {
256                    states.cached_state.get::<SurfaceCachedState>().pending().geometry =
257                        Some(Rectangle::new((x, y).into(), (width, height).into()));
258                });
259            }
260            xdg_surface::Request::AckConfigure { serial } => {
261                let serial = Serial::from(serial);
262                let surface = &data.wl_surface;
263
264                // Check the role of the surface, this can be either xdg_toplevel
265                // or xdg_popup. If none of the role matches the xdg_surface has no role set
266                // which is a protocol error.
267                if compositor::get_role(surface).is_none() {
268                    xdg_surface.post_error(
269                        xdg_surface::Error::NotConstructed,
270                        "xdg_surface must have a role.",
271                    );
272                    return;
273                }
274
275                // Find the correct configure state for the provided serial
276                // discard all configure states that are older than the provided
277                // serial.
278                // If no matching serial can be found raise a protocol error
279                //
280                // Invoke the user impl with the found configuration
281                // This has to include the serial and the role specific data.
282                // - For xdg_popup there is no data.
283                // - For xdg_toplevel send the state data including
284                //   width, height, min/max size, maximized, fullscreen, resizing, activated
285                //
286                // This can be used to integrate custom protocol extensions
287                let found_configure = compositor::with_states(surface, |states| {
288                    if states.role == Some(XDG_TOPLEVEL_ROLE) {
289                        Ok(states
290                            .data_map
291                            .get::<XdgToplevelSurfaceData>()
292                            .unwrap()
293                            .lock()
294                            .unwrap()
295                            .ack_configure(serial))
296                    } else if states.role == Some(XDG_POPUP_ROLE) {
297                        Ok(states
298                            .data_map
299                            .get::<XdgPopupSurfaceData>()
300                            .unwrap()
301                            .lock()
302                            .unwrap()
303                            .ack_configure(serial))
304                    } else {
305                        Err(())
306                    }
307                });
308
309                let configure = match found_configure {
310                    Ok(Some(configure)) => configure,
311                    Ok(None) => {
312                        data.wm_base.post_error(
313                            xdg_wm_base::Error::InvalidSurfaceState,
314                            format!("wrong configure serial: {}", <u32>::from(serial)),
315                        );
316                        return;
317                    }
318                    Err(()) => {
319                        data.wm_base.post_error(
320                            xdg_wm_base::Error::Role as u32,
321                            "xdg_surface must have a role of xdg_toplevel or xdg_popup.",
322                        );
323                        return;
324                    }
325                };
326
327                XdgShellHandler::ack_configure(state, surface.clone(), configure);
328            }
329            _ => unreachable!(),
330        }
331    }
332}

336pub struct XdgShellSurfaceUserData {

346impl IsAlive for XdgToplevel {
347    #[inline]
348    fn alive(&self) -> bool {
349        let data: &XdgShellSurfaceUserData = self.data().unwrap();
350        data.alive_tracker.alive()
351    }
