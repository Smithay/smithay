1//! XDG Window decoration manager
2//!
3//! This interface allows a compositor to announce support for server-side decorations.
4//!
5//! A client can use this protocol to request being decorated by a supporting compositor.
6//!
7//!
8//! ```no_run
9//! # extern crate wayland_server;
10//! #
11//! use smithay::{delegate_xdg_decoration, delegate_xdg_shell};
12//! use smithay::wayland::shell::xdg::{ToplevelSurface, XdgShellHandler};
13//! # use smithay::utils::Serial;
14//! # use smithay::wayland::shell::xdg::{XdgShellState, PopupSurface, PositionerState};
15//! # use smithay::reexports::wayland_server::protocol::{wl_seat, wl_surface};
16//! use smithay::wayland::shell::xdg::decoration::{XdgDecorationState, XdgDecorationHandler};
17//! use smithay::reexports::wayland_protocols::xdg::decoration::zv1::server::zxdg_toplevel_decoration_v1::Mode;
18//!
19//! # struct State { decoration_state: XdgDecorationState, seat_state: SeatState<Self> }
20//! # let mut display = wayland_server::Display::<State>::new().unwrap();
21//!
22//! // Create a decoration state
23//! let decoration_state = XdgDecorationState::new::<State>(
24//!     &display.handle(),
25//! );
26//!
27//! // store that state inside your compositor state
28//! // ...
29//!
30//! // implement the necessary traits
31//! impl XdgShellHandler for State {
32//!     # fn xdg_shell_state(&mut self) -> &mut XdgShellState { unimplemented!() }
33//!     # fn new_toplevel(&mut self, surface: ToplevelSurface) { unimplemented!() }
34//!     # fn new_popup(
35//!     #     &mut self,
36//!     #     surface: PopupSurface,
37//!     #     positioner: PositionerState,
38//!     # ) { unimplemented!() }
39//!     # fn grab(
40//!     #     &mut self,
41//!     #     surface: PopupSurface,
42//!     #     seat: wl_seat::WlSeat,
43//!     #     serial: Serial,
44//!     # ) { unimplemented!() }
45//!     # fn reposition_request(
46//!     #     &mut self,
47//!     #     surface: PopupSurface,
48//!     #     positioner: PositionerState,
49//!     #     token: u32,
50//!     # ) { unimplemented!() }
51//!     // ...
52//! }
53//! impl XdgDecorationHandler for State {
54//!     fn new_decoration(&mut self, toplevel: ToplevelSurface) {
55//!         toplevel.with_pending_state(|state| {
56//!             // Advertise server side decoration
57//!             state.decoration_mode = Some(Mode::ServerSide);
58//!         });
59//!         toplevel.send_configure();
60//!     }
61//!     fn request_mode(&mut self, toplevel: ToplevelSurface, mode: Mode) { /* ... */ }
62//!     fn unset_mode(&mut self, toplevel: ToplevelSurface) { /* ... */ }
63//! }
64//!
65//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
66//!
67//! type Target = wl_surface::WlSurface;
68//! impl SeatHandler for State {
69//!     type KeyboardFocus = Target;
70//!     type PointerFocus = Target;
71//!     type TouchFocus = Target;
72//!
73//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
74//!         &mut self.seat_state
75//!     }
76//!
77//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&Target>) {
78//!         // handle focus changes, if you need to ...
79//!     }
80//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) {
81//!         // handle new images for the cursor ...
82//!     }
83//! }
84//! delegate_xdg_shell!(State);
85//! delegate_xdg_decoration!(State);
86//!
87//! // You are ready to go!

101/// Delegate type for handling xdg decoration events.
102#[derive(Debug)]
103pub struct XdgDecorationState {
104    global: GlobalId,
105}

107/// Data associated with a XdgDecorationManager global.
108#[allow(missing_debug_implementations)]
109pub struct XdgDecorationManagerGlobalData {
110    filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
111}

113impl XdgDecorationState {
114    /// Creates a new delegate type for handling xdg decoration events.
115    ///
116    /// A global id is also returned to allow destroying the global in the future.
117    pub fn new<D>(display: &DisplayHandle) -> XdgDecorationState
118    where
119        D: GlobalDispatch<
120                zxdg_decoration_manager_v1::ZxdgDecorationManagerV1,
121                XdgDecorationManagerGlobalData,
122            > + Dispatch<zxdg_decoration_manager_v1::ZxdgDecorationManagerV1, ()>
123            + 'static,
124    {
125        Self::new_with_filter::<D, _>(display, |_| true)
126    }
127
128    /// Creates a new delegate type for handling xdg decoration events with a filter.
129    ///
130    /// Filters can be used to limit visibility of a global to certain clients.
131    ///
132    /// A global id is also returned to allow destroying the global in the future.
133    pub fn new_with_filter<D, F>(display: &DisplayHandle, filter: F) -> XdgDecorationState
134    where
135        D: GlobalDispatch<
136                zxdg_decoration_manager_v1::ZxdgDecorationManagerV1,
137                XdgDecorationManagerGlobalData,
138            > + Dispatch<zxdg_decoration_manager_v1::ZxdgDecorationManagerV1, ()>
139            + 'static,
140        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
141    {
142        let data = XdgDecorationManagerGlobalData {
143            filter: Box::new(filter),
144        };
145        let global =
146            display.create_global::<D, zxdg_decoration_manager_v1::ZxdgDecorationManagerV1, _>(1, data);
147
148        XdgDecorationState { global }
149    }
150
151    /// Returns the xdg-decoration global.
152    pub fn global(&self) -> GlobalId {
153        self.global.clone()
154    }
155}

157/// Handler trait for xdg decoration events.
158pub trait XdgDecorationHandler {
159    /// Notification the client supports server side decoration on the toplevel.
160    fn new_decoration(&mut self, toplevel: ToplevelSurface);
161
162    /// Notification the client prefers the provided decoration decoration mode on the toplevel.
163    fn request_mode(&mut self, toplevel: ToplevelSurface, mode: Mode);
164
165    /// Notification the client does not prefer a particular decoration mode on the toplevel.
166    fn unset_mode(&mut self, toplevel: ToplevelSurface);
167}

169/// Macro to delegate implementation of the xdg decoration to [`XdgDecorationState`].

171/// You must also implement [`XdgDecorationHandler`] to use this.
172#[macro_export]
173macro_rules! delegate_xdg_decoration {
174    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
175        $crate::reexports::wayland_server::delegate_global_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
176            $crate::reexports::wayland_protocols::xdg::decoration::zv1::server::zxdg_decoration_manager_v1::ZxdgDecorationManagerV1: $crate::wayland::shell::xdg::decoration::XdgDecorationManagerGlobalData
177        ] => $crate::wayland::shell::xdg::decoration::XdgDecorationState);
178
179        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
180            $crate::reexports::wayland_protocols::xdg::decoration::zv1::server::zxdg_decoration_manager_v1::ZxdgDecorationManagerV1: ()
181        ] => $crate::wayland::shell::xdg::decoration::XdgDecorationState);
182        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
183            $crate::reexports::wayland_protocols::xdg::decoration::zv1::server::zxdg_toplevel_decoration_v1::ZxdgToplevelDecorationV1: $crate::wayland::shell::xdg::ToplevelSurface
184        ] => $crate::wayland::shell::xdg::decoration::XdgDecorationState);
185    };

221impl<D> Dispatch<zxdg_decoration_manager_v1::ZxdgDecorationManagerV1, (), D> for XdgDecorationState
222where
223    D: Dispatch<zxdg_decoration_manager_v1::ZxdgDecorationManagerV1, ()>
224        + Dispatch<zxdg_toplevel_decoration_v1::ZxdgToplevelDecorationV1, ToplevelSurface>
225        + XdgShellHandler
226        + XdgDecorationHandler
227        + 'static,
228{
229    fn request(
230        state: &mut D,
231        _: &Client,
232        resource: &zxdg_decoration_manager_v1::ZxdgDecorationManagerV1,
233        request: zxdg_decoration_manager_v1::Request,
234        _: &(),
235        _dh: &DisplayHandle,
236        data_init: &mut DataInit<'_, D>,
237    ) {
238        use self::zxdg_decoration_manager_v1::Request;
239
240        match request {
241            Request::GetToplevelDecoration { id, toplevel } => {
242                let data = toplevel.data::<XdgShellSurfaceUserData>().unwrap();
243
244                if !data.alive_tracker.alive() {
245                    resource.post_error(
246                        zxdg_toplevel_decoration_v1::Error::Orphaned,
247                        "toplevel is already destroyed",
248                    );
249                    return;
250                }
251
252                let mut decoration_guard = data.decoration.lock().unwrap();
253
254                if decoration_guard.is_some() {
255                    resource.post_error(
256                        zxdg_toplevel_decoration_v1::Error::AlreadyConstructed,
257                        "toplevel decoration is already constructed",
258                    );
259                    return;
260                }
261
262                let toplevel = state.xdg_shell_state().get_toplevel(&toplevel).unwrap();
263                let toplevel_decoration = data_init.init(id, toplevel.clone());
264
265                *decoration_guard = Some(toplevel_decoration);
266                drop(decoration_guard);
267
268                state.new_decoration(toplevel);
269            }
270
271            Request::Destroy => {}
272
273            _ => unreachable!(),
274        }
275    }
276}

280impl<D> Dispatch<zxdg_toplevel_decoration_v1::ZxdgToplevelDecorationV1, ToplevelSurface, D>
281    for XdgDecorationState
282where
283    D: Dispatch<zxdg_toplevel_decoration_v1::ZxdgToplevelDecorationV1, ToplevelSurface>
284        + XdgDecorationHandler,
285{
286    fn request(
287        state: &mut D,
288        _: &Client,
289        _: &zxdg_toplevel_decoration_v1::ZxdgToplevelDecorationV1,
290        request: zxdg_toplevel_decoration_v1::Request,
291        data: &ToplevelSurface,
292        _dh: &DisplayHandle,
293        _: &mut DataInit<'_, D>,
294    ) {
295        use self::zxdg_toplevel_decoration_v1::Request;
296
297        match request {
298            Request::SetMode { mode } => {
299                if let WEnum::Value(mode) = mode {
300                    state.request_mode(data.clone(), mode);
301                }
302            }
303
304            Request::UnsetMode => {
305                state.unset_mode(data.clone());
306            }
307
308            Request::Destroy => {
309                if let Some(data) = data.xdg_toplevel().data::<XdgShellSurfaceUserData>() {
310                    data.decoration.lock().unwrap().take();
311                }
312            }
313
314            _ => unreachable!(),
315        }
316    }
