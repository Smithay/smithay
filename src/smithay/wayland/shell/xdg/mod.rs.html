1//! Utilities for handling shell surfaces with the `xdg_shell` protocol
2//!
3//! This module provides automatic handling of shell surfaces objects, by being registered
4//! as a global handler for `xdg_shell`.
5//!
6//! ## Why use this implementation
7//!
8//! This implementation can track for you the various shell surfaces defined by the
9//! clients by handling the `xdg_shell` protocol.
10//!
11//! It allows you to easily access a list of all shell surfaces defined by your clients
12//! access their associated metadata and underlying `wl_surface`s.
13//!
14//! This handler only handles the protocol exchanges with the client to present you the
15//! information in a coherent and relatively easy to use manner. All the actual drawing
16//! and positioning logic of windows is out of its scope.
17//!
18//! ## How to use it
19//!
20//! ### Initialization
21//!
22//! To initialize this handler, create [`XdgShellState`], store it in your `State` struct and
23//! implement the [`XdgShellHandler`], as shown in this example:
24//!
25//! ```no_run
26//! # extern crate wayland_server;
27//! #
28//! use smithay::delegate_xdg_shell;
29//! use smithay::reexports::wayland_server::protocol::{wl_seat, wl_surface};
30//! use smithay::wayland::shell::xdg::{XdgShellState, XdgShellHandler, ToplevelSurface, PopupSurface, PositionerState};
31//! use smithay::utils::Serial;
32//!
33//! # struct State { xdg_shell_state: XdgShellState, seat_state: SeatState<Self> }
34//! # let mut display = wayland_server::Display::<State>::new().unwrap();
35//! let xdg_shell_state = XdgShellState::new::<State>(
36//!     &display.handle(),
37//! );
38//!
39//! // insert the xdg_shell_state into your state
40//! // ...
41//!
42//! // implement the necessary traits
43//! impl XdgShellHandler for State {
44//!     fn xdg_shell_state(&mut self) -> &mut XdgShellState {
45//!         &mut self.xdg_shell_state
46//!     }
47//!
48//!     // handle the shell requests here.
49//!     // more optional methods can be used to further customized
50//!     fn new_toplevel(&mut self, surface: ToplevelSurface) {
51//!         // ...
52//!     }
53//!     fn new_popup(
54//!         &mut self,
55//!         surface: PopupSurface,
56//!         positioner: PositionerState,
57//!     ) {
58//!         // ...
59//!     }
60//!     fn grab(
61//!         &mut self,
62//!         surface: PopupSurface,
63//!         seat: wl_seat::WlSeat,
64//!         serial: Serial,
65//!     ) {
66//!         // ...
67//!     }
68//!     fn reposition_request(
69//!         &mut self,
70//!         surface: PopupSurface,
71//!         positioner: PositionerState,
72//!         token: u32,
73//!     ) {
74//!         // ...
75//!     }
76//! }
77//!
78//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
79//!
80//! type Target = wl_surface::WlSurface;
81//! impl SeatHandler for State {
82//!     type KeyboardFocus = Target;
83//!     type PointerFocus = Target;
84//!     type TouchFocus = Target;
85//!
86//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
87//!         &mut self.seat_state
88//!     }
89//!
90//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&Target>) {
91//!         // handle focus changes, if you need to ...
92//!     }
93//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) {
94//!         // handle new images for the cursor ...
95//!     }
96//! }
97//! delegate_xdg_shell!(State);
98//!
99//! // You're now ready to go!
100//! ```
101//!
102//! ### Access to shell surface and clients data
103//!
104//! There are mainly 3 kind of objects that you'll manipulate from this implementation:
105//!
106//! - [`ShellClient`]:
107//!   This is a handle representing an instantiation of a shell global
108//!   you can associate client-wise metadata to it through an [`UserDataMap`].
109//! - [`ToplevelSurface`]:
110//!   This is a handle representing a toplevel surface, you can
111//!   retrieve a list of all currently alive toplevel surface from the
112//!   [`XdgShellState`].
113//! - [`PopupSurface`]:
114//!   This is a handle representing a popup/tooltip surface. Similarly,
115//!   you can get a list of all currently alive popup surface from the
116//!   [`XdgShellState`].
117//!
118//! You'll obtain these objects though two means: either via the callback methods of
119//! the [`XdgShellHandler`], or via methods on the [`XdgShellState`].

160macro_rules! xdg_role {
161    ($state:ty,
162     $(#[$configure_meta:meta])* $configure_name:ident $({$($(#[$configure_field_meta:meta])* $configure_field_vis:vis$configure_field_name:ident:$configure_field_type:ty),*}),*,
163     $(#[$attributes_meta:meta])* $attributes_name:ident {$($(#[$attributes_field_meta:meta])* $attributes_field_vis:vis$attributes_field_name:ident:$attributes_field_type:ty),*},
164     $(#[$cached_state_meta:meta])* $cached_state_name:ident) => {
165
166        $(#[$configure_meta])*
167        pub struct $configure_name {
168                /// The state associated with this configure
169                pub state: $state,
170                /// A serial number to track ACK from the client
171                ///
172                /// This should be an ever increasing number, as the ACK-ing
173                /// from a client for a serial will validate all pending lower
174                /// serials.
175                pub serial: Serial,
176
177                $($(
178                    $(#[$configure_field_meta])*
179                    $configure_field_vis $configure_field_name: $configure_field_type,
180                )*)*
181        }
182
183        $(#[$attributes_meta])*
184        pub struct $attributes_name {
185                /// Holds the state if the surface has sent the initial
186                /// configure event to the client. It is expected that
187                /// during the first commit a initial
188                /// configure event is sent to the client
189                pub initial_configure_sent: bool,
190                /// Holds the configures the server has sent out
191                /// to the client waiting to be acknowledged by
192                /// the client. All pending configures that are older
193                /// than the acknowledged one will be discarded during
194                /// processing xdg_surface.ack_configure.
195                pending_configures: Vec<$configure_name>,
196                /// Holds the pending state as set by the server.
197                pub server_pending: Option<$state>,
198                /// Holds the last configure that has been acknowledged by the client. This state
199                /// should be cloned to the current during a commit. Note that this state can be
200                /// newer than the last acked state at the time of the last commit.
201                pub last_acked: Option<$configure_name>,
202
203                $(
204                    $(#[$attributes_field_meta])*
205                    $attributes_field_vis $attributes_field_name: $attributes_field_type,
206                )*
207        }
208
209        impl $attributes_name {
210            fn ack_configure(&mut self, serial: Serial) -> Option<Configure> {
211                let configure = match self
212                    .pending_configures
213                    .iter()
214                    .find(|configure| configure.serial == serial)
215                {
216                    Some(configure) => (*configure).clone(),
217                    None => {
218                        return None;
219                    }
220                };
221
222                // Save the configure as the last acked configure
223                self.last_acked = Some(configure.clone());
224
225                // Clean old configures
226                self.pending_configures.retain(|c| c.serial > serial);
227
228                Some(configure.into())
229            }
230
231            /// Gets the latest state that has been configured
232            /// on the server and sent to the client.
233            ///
234            /// The state includes all changes that have been
235            /// made on the server, including all not yet
236            /// acked or committed changes, but excludes the
237            /// current [`server_pending`](#structfield.server_pending) state.
238            ///
239            /// This can be used for example to check if the
240            /// [`server_pending`](#structfield.server_pending) state
241            /// is different from the last configured.
242            pub fn current_server_state(&self) -> $state {
243                // We check if there is already a non-acked pending
244                // configure and use its state or otherwise we could
245                // loose some state that was previously configured
246                // and sent, but not acked before calling with_pending_state
247                // again. If there is no pending state we try to use the
248                // last acked state which could contain state changes
249                // already acked but not committed to the current state.
250                // In both cases the state already contains all previous
251                // sent states. This way all pending state is accumulated
252                // into the current state.
253                self.pending_configures
254                    .last()
255                    .map(|c| &c.state)
256                    .or_else(|| self.last_acked.as_ref().map(|c| &c.state))
257                    .cloned()
258                    .unwrap_or_default()
259            }
260
261            /// Check if the state has pending changes that have
262            /// not been sent to the client.
263            ///
264            /// This differs from just checking if the [`server_pending`](#structfield.server_pending)
265            /// state is [`Some`] in that it also checks if a current pending
266            /// state is different from the [`current_server_state`](#method.current_server_state).
267            pub fn has_pending_changes(&self) -> bool {
268                self.server_pending.as_ref().map(|s| *s != self.current_server_state()).unwrap_or(false)
269            }
270
271            /// Returns a list of configures sent to, but not yet acknowledged by the client.
272            ///
273            /// The list is ordered by age, so the last configure in the list is the last one sent
274            /// to the client.
275            pub fn pending_configures(&self) -> &[$configure_name] {
276                &self.pending_configures
277            }
278        }
279
280        impl Default for $attributes_name {
281            fn default() -> Self {
282                Self {
283                    pending_configures: Vec::new(),
284                    initial_configure_sent: false,
285                    server_pending: None,
286                    last_acked: None,
287
288                    $(
289                        $attributes_field_name: Default::default(),
290                    )*
291                }
292            }
293        }
294
295        $(#[$cached_state_meta])*
296        pub struct $cached_state_name {
297            /// Configure last acknowledged by the client at the time of the commit.
298            ///
299            /// Reset to `None` when the surface unmaps.
300            pub last_acked: Option<$configure_name>,
301        }
302
303        impl Cacheable for $cached_state_name {
304            fn commit(&mut self, _dh: &DisplayHandle) -> Self {
305                self.clone()
306            }
307            fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
308                *into = self;
309            }
310        }
311    };
312}

539impl PositionerState {
540    pub(crate) fn anchor_has_edge(&self, edge: xdg_positioner::Anchor) -> bool {
541        match edge {
542            xdg_positioner::Anchor::Top => {
543                self.anchor_edges == xdg_positioner::Anchor::Top
544                    || self.anchor_edges == xdg_positioner::Anchor::TopLeft
545                    || self.anchor_edges == xdg_positioner::Anchor::TopRight
546            }
547            xdg_positioner::Anchor::Bottom => {
548                self.anchor_edges == xdg_positioner::Anchor::Bottom
549                    || self.anchor_edges == xdg_positioner::Anchor::BottomLeft
550                    || self.anchor_edges == xdg_positioner::Anchor::BottomRight
551            }
552            xdg_positioner::Anchor::Left => {
553                self.anchor_edges == xdg_positioner::Anchor::Left
554                    || self.anchor_edges == xdg_positioner::Anchor::TopLeft
555                    || self.anchor_edges == xdg_positioner::Anchor::BottomLeft
556            }
557            xdg_positioner::Anchor::Right => {
558                self.anchor_edges == xdg_positioner::Anchor::Right
559                    || self.anchor_edges == xdg_positioner::Anchor::TopRight
560                    || self.anchor_edges == xdg_positioner::Anchor::BottomRight
561            }
562            _ => unreachable!(),
563        }
564    }
565
566    /// Get the anchor point for a popup as defined by this positioner.
567    ///
568    /// Defined by `xdg_positioner.set_anchor_rect` and
569    /// `xdg_positioner.set_anchor`.
570    pub fn get_anchor_point(&self) -> Point<i32, Logical> {
571        let mut point = self.anchor_rect.loc;
572
573        point.y += if self.anchor_has_edge(xdg_positioner::Anchor::Top) {
574            0
575        } else if self.anchor_has_edge(xdg_positioner::Anchor::Bottom) {
576            self.anchor_rect.size.h
577        } else {
578            self.anchor_rect.size.h / 2
579        };
580
581        point.x += if self.anchor_has_edge(xdg_positioner::Anchor::Left) {
582            0
583        } else if self.anchor_has_edge(xdg_positioner::Anchor::Right) {
584            self.anchor_rect.size.w
585        } else {
586            self.anchor_rect.size.w / 2
587        };
588
589        point
590    }

594            xdg_positioner::Gravity::Top => {
595                self.gravity == xdg_positioner::Gravity::Top
596                    || self.gravity == xdg_positioner::Gravity::TopLeft
597                    || self.gravity == xdg_positioner::Gravity::TopRight
598            }
599            xdg_positioner::Gravity::Bottom => {
600                self.gravity == xdg_positioner::Gravity::Bottom
601                    || self.gravity == xdg_positioner::Gravity::BottomLeft
602                    || self.gravity == xdg_positioner::Gravity::BottomRight
603            }
604            xdg_positioner::Gravity::Left => {
605                self.gravity == xdg_positioner::Gravity::Left
606                    || self.gravity == xdg_positioner::Gravity::TopLeft
607                    || self.gravity == xdg_positioner::Gravity::BottomLeft
608            }
609            xdg_positioner::Gravity::Right => {

611                    || self.gravity == xdg_positioner::Gravity::TopRight
612                    || self.gravity == xdg_positioner::Gravity::BottomRight
613            }
614            _ => unreachable!(),
615        }
616    }

618    /// Get the geometry for a popup as defined by this positioner.
619    ///
620    /// `Rectangle::width` and `Rectangle::height` corresponds to the
621    /// size set by `xdg_positioner.set_size`.
622    ///
623    /// `Rectangle::x` and `Rectangle::y` define the position of the
624    /// popup relative to its parent surface's `window_geometry`.
625    /// The position is calculated according to the rules defined
626    /// in the `xdg_shell` protocol.
627    /// The `constraint_adjustment` will not be considered by this
628    /// implementation and the position and size should be re-calculated
629    /// in the compositor if the compositor implements `constraint_adjustment`
630    ///
631    /// [`PositionerState::get_unconstrained_geometry`] does take `constraint_adjustment` into account.
632    pub fn get_geometry(&self) -> Rectangle<i32, Logical> {
633        // From the `xdg_shell` prococol specification:
634        //
635        // set_offset:
636        //
637        //  Specify the surface position offset relative to the position of the
638        //  anchor on the anchor rectangle and the anchor on the surface. For
639        //  example if the anchor of the anchor rectangle is at (x, y), the surface
640        //  has the gravity bottom|right, and the offset is (ox, oy), the calculated
641        //  surface position will be (x + ox, y + oy)
642        let mut geometry = Rectangle::new(self.offset, self.rect_size);
643
644        // Defines the anchor point for the anchor rectangle. The specified anchor
645        // is used derive an anchor point that the child surface will be
646        // positioned relative to. If a corner anchor is set (e.g. 'top_left' or
647        // 'bottom_right'), the anchor point will be at the specified corner;
648        // otherwise, the derived anchor point will be centered on the specified
649        // edge, or in the center of the anchor rectangle if no edge is specified.
650        geometry.loc += self.get_anchor_point();
651
652        // Defines in what direction a surface should be positioned, relative to
653        // the anchor point of the parent surface. If a corner gravity is
654        // specified (e.g. 'bottom_right' or 'top_left'), then the child surface
655        // will be placed towards the specified gravity; otherwise, the child
656        // surface will be centered over the anchor point on any axis that had no
657        // gravity specified.
658        if self.gravity_has_edge(xdg_positioner::Gravity::Top) {
659            geometry.loc.y -= geometry.size.h;
660        } else if !self.gravity_has_edge(xdg_positioner::Gravity::Bottom) {
661            geometry.loc.y -= geometry.size.h / 2;
662        }
663
664        if self.gravity_has_edge(xdg_positioner::Gravity::Left) {
665            geometry.loc.x -= geometry.size.w;
666        } else if !self.gravity_has_edge(xdg_positioner::Gravity::Right) {
667            geometry.loc.x -= geometry.size.w / 2;
668        }
669
670        geometry
671    }
672
673    /// Get the geometry for a popup as defined by this positioner, after trying to fit the popup into the
674    /// target rectangle.
675    ///
676    /// `Rectangle::width` and `Rectangle::height` corresponds to the size set by `xdg_positioner.set_size`.
677    ///
678    /// `Rectangle::x` and `Rectangle::y` define the position of the popup relative to its parent surface's
679    /// `window_geometry`. The position is calculated according to the rules defined in the `xdg_shell`
680    /// protocol.
681    ///
682    /// This method does consider `constrain_adjustment` by trying to fit the popup into the provided target
683    /// rectangle. The target rectangle is in the same coordinate system as the rectangle returned by this
684    /// method. So, it is relative to the parent surface's geometry.
685    pub fn get_unconstrained_geometry(mut self, target: Rectangle<i32, Logical>) -> Rectangle<i32, Logical> {
686        // The protocol defines the following order for adjustments: flip, slide, resize. If the flip fails
687        // to remove the constraints, it is reverted.
688        //
689        // The adjustments are applied individually between axes. We can do that reasonably safely, given
690        // that both our target and our popup are simple rectangles. The code is grouped per adjustment for
691        // easier copy-paste checking, and because flips replace the geometry entirely, while further
692        // adjustments change individual fields.
693        let mut geo = self.get_geometry();
694        let (mut off_left, mut off_right, mut off_top, mut off_bottom) = compute_offsets(target, geo);
695
696        // Try to flip horizontally.
697        if (off_left > 0 || off_right > 0) && self.constraint_adjustment.contains(ConstraintAdjustment::FlipX)
698        {
699            let mut new = self;
700            new.anchor_edges = invert_anchor_x(new.anchor_edges);
701            new.gravity = invert_gravity_x(new.gravity);
702            let new_geo = new.get_geometry();
703            let (new_off_left, new_off_right, _, _) = compute_offsets(target, new_geo);
704
705            // Apply flip only if it removed the constraint.
706            if new_off_left <= 0 && new_off_right <= 0 {
707                self = new;
708                geo = new_geo;
709                off_left = 0;
710                off_right = 0;
711                // off_top and off_bottom are unchanged since we're using rectangles.
712            }
713        }
714
715        // Try to flip vertically.
716        if (off_top > 0 || off_bottom > 0) && self.constraint_adjustment.contains(ConstraintAdjustment::FlipY)
717        {
718            let mut new = self;
719            new.anchor_edges = invert_anchor_y(new.anchor_edges);
720            new.gravity = invert_gravity_y(new.gravity);
721            let new_geo = new.get_geometry();
722            let (_, _, new_off_top, new_off_bottom) = compute_offsets(target, new_geo);
723
724            // Apply flip only if it removed the constraint.
725            if new_off_top <= 0 && new_off_bottom <= 0 {
726                self = new;
727                geo = new_geo;
728                off_top = 0;
729                off_bottom = 0;
730                // off_left and off_right are unchanged since we're using rectangles.
731            }
732        }
733
734        // Try to slide horizontally.
735        if (off_left > 0 || off_right > 0)
736            && self.constraint_adjustment.contains(ConstraintAdjustment::SlideX)
737        {
738            // Prefer to show the top-left corner of the popup so that we can easily do a resize
739            // adjustment next.
740            if off_left > 0 {
741                geo.loc.x += off_left;
742            } else if off_right > 0 {
743                geo.loc.x -= min(off_right, -off_left);
744            }
745
746            (off_left, off_right, _, _) = compute_offsets(target, geo);
747            // off_top and off_bottom are the same since we're using rectangles.
748        }
749
750        // Try to slide vertically.
751        if (off_top > 0 || off_bottom > 0)
752            && self.constraint_adjustment.contains(ConstraintAdjustment::SlideY)
753        {
754            // Prefer to show the top-left corner of the popup so that we can easily do a resize
755            // adjustment next.
756            if off_top > 0 {
757                geo.loc.y += off_top;
758            } else if off_bottom > 0 {
759                geo.loc.y -= min(off_bottom, -off_top);
760            }
761
762            (_, _, off_top, off_bottom) = compute_offsets(target, geo);
763            // off_left and off_right are the same since we're using rectangles.
764        }
765
766        // Try to resize horizontally.
767        if self.constraint_adjustment.contains(ConstraintAdjustment::ResizeX) {
768            // Unconstrain both edges by clamping the left and right sides of the popup rectangle.
769            // Skip if the offset is larger than the width, in which case the entirety of the popup
770            // is outside the target rectangle and a clamp would result in a zero-sized geometry.
771
772            if off_left > 0 && off_left < geo.size.w {
773                geo.loc.x += off_left;
774                geo.size.w -= off_left;
775            }
776            if off_right > 0 && off_right < geo.size.w {
777                geo.size.w -= off_right;
778            }
779        }
780
781        // Try to resize vertically.
782        if self.constraint_adjustment.contains(ConstraintAdjustment::ResizeY) {
783            // Unconstrain both edges by clamping the top and bottom sides of the popup rectangle.
784            // Skip if the offset is larger than the height, in which case the entirety of the popup
785            // is outside the target rectangle and a clamp would result in a zero-sized geometry.
786
787            if off_top > 0 && off_top < geo.size.h {
788                geo.loc.y += off_top;
789                geo.size.h -= off_top;
790            }
791            if off_bottom > 0 && off_bottom < geo.size.h {
792                geo.size.h -= off_bottom;
793            }
794        }
795
796        geo
797    }
798}

800fn compute_offsets(target: Rectangle<i32, Logical>, popup: Rectangle<i32, Logical>) -> (i32, i32, i32, i32) {
801    let off_left = target.loc.x - popup.loc.x;
802    let off_right = (popup.loc.x + popup.size.w) - (target.loc.x + target.size.w);
803    let off_top = target.loc.y - popup.loc.y;
804    let off_bottom = (popup.loc.y + popup.size.h) - (target.loc.y + target.size.h);
805    (off_left, off_right, off_top, off_bottom)
806}

809    match anchor {
810        Anchor::Left => Anchor::Right,
811        Anchor::Right => Anchor::Left,
812        Anchor::TopLeft => Anchor::TopRight,
813        Anchor::TopRight => Anchor::TopLeft,
814        Anchor::BottomLeft => Anchor::BottomRight,
815        Anchor::BottomRight => Anchor::BottomLeft,
816        x => x,
817    }
818}

823        Anchor::Bottom => Anchor::Top,
824        Anchor::TopLeft => Anchor::BottomLeft,
825        Anchor::TopRight => Anchor::BottomRight,
826        Anchor::BottomLeft => Anchor::TopLeft,
827        Anchor::BottomRight => Anchor::TopRight,
828        x => x,
829    }

894/// Container holding the states for a `XdgToplevel`
895///
896/// This container will prevent the `XdgToplevel` from
897/// having the same `xdg_toplevel::State` multiple times
898/// and simplifies setting and un-setting a particularly
899/// `xdg_toplevel::State`
900#[derive(Debug, Default, Clone, PartialEq, Eq)]
901pub struct ToplevelStateSet {
902    states: Vec<xdg_toplevel::State>,
903}
904
905impl ToplevelStateSet {
906    /// Returns `true` if the states contains a state.
907    pub fn contains(&self, state: xdg_toplevel::State) -> bool {
908        self.states.contains(&state)
909    }
910
911    /// Adds a state to the states.
912    ///
913    /// If the states did not have this state present, `true` is returned.
914    ///
915    /// If the states did have this state present, `false` is returned.
916    pub fn set(&mut self, state: xdg_toplevel::State) -> bool {
917        if self.contains(state) {
918            false
919        } else {
920            self.states.push(state);
921            true
922        }
923    }
924
925    /// Removes a state from the states. Returns whether the state was
926    /// present in the states.
927    pub fn unset(&mut self, state: xdg_toplevel::State) -> bool {
928        if !self.contains(state) {
929            false
930        } else {
931            self.states.retain(|s| *s != state);
932            true
933        }
934    }
935
936    /// Filter the states according to the provided version
937    /// of the [`XdgToplevel`]
938    pub(crate) fn into_filtered_states(self, version: u32) -> Vec<xdg_toplevel::State> {
939        let tiled_supported = version >= XDG_TOPLEVEL_STATE_TILED_SINCE;
940        let suspended_supported = version >= XDG_TOPLEVEL_STATE_SUSPENDED_SINCE;
941
942        // If the client version supports the suspended states
943        // we can directly return the states which will save
944        // us from allocating another vector
945        if suspended_supported {
946            return self.states;
947        }
948
949        let is_suspended = |state: &xdg_toplevel::State| *state == xdg_toplevel::State::Suspended;
950        let contains_suspended = self.states.contains(&xdg_toplevel::State::Suspended);
951
952        // If tiled is supported and there is no suspend state there is nothing to filter out
953        if tiled_supported && !contains_suspended {
954            return self.states;
955        }
956
957        let is_tiled = |state: &xdg_toplevel::State| {
958            matches!(
959                state,
960                xdg_toplevel::State::TiledTop
961                    | xdg_toplevel::State::TiledBottom
962                    | xdg_toplevel::State::TiledLeft
963                    | xdg_toplevel::State::TiledRight
964            )
965        };
966        let contains_tiled = self.states.iter().any(is_tiled);
967
968        // If it does not contain unsupported values
969        if !contains_suspended && !contains_tiled {
970            return self.states;
971        }
972
973        self.states
974            .into_iter()
975            .filter(|state| {
976                if tiled_supported {
977                    !is_suspended(state)
978                } else {
979                    !is_suspended(state) && !is_tiled(state)
980                }
981            })
982            .collect()
983    }
984}

1339    ///
1340    /// You'll receive the reply as a [`XdgShellHandler::client_pong`] request.
1341    ///
1342    /// A typical use is to start a timer at the same time you send this ping
1343    /// request, and cancel it when you receive the pong. If the timer runs
1344    /// down to 0 before a pong is received, mark the client as unresponsive.
1345    ///
1346    /// Fails if this shell client already has a pending ping or is already dead.

1349            return Err(PingError::DeadSurface);
1350        }

1354            return Err(PingError::PingAlreadyPending(pending_ping));
1355        }
1356        guard.pending_ping = Some(serial);
1357        self.kind.ping(serial.into());
1358
1359        Ok(())
1360    }

1386        let mut guard = data.client_data.lock().unwrap();

1388    }

1466    /// See [`send_configure`](ToplevelSurface::send_configure) and [`has_pending_changes`](ToplevelSurface::has_pending_changes)

1476    /// Send a configure event to this toplevel surface to suggest it a new configuration
1477    ///
1478    /// The serial of this configure will be tracked waiting for the client to ACK it.
1479    ///
1480    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
1481    /// method.
1482    ///
1483    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
1484    /// [`send_pending_configure`](ToplevelSurface::send_pending_configure)
1485    pub fn send_configure(&self) -> Serial {
1486        let shell_surface_data = self.shell_surface.data::<XdgShellSurfaceUserData>();
1487        let decoration =
1488            shell_surface_data.and_then(|data| data.decoration.lock().unwrap().as_ref().cloned());
1489        let (configure, decoration_mode_changed, bounds_changed, capabilities_changed) =
1490            compositor::with_states(&self.wl_surface, |states| {
1491                let mut attributes = states
1492                    .data_map
1493                    .get::<XdgToplevelSurfaceData>()
1494                    .unwrap()
1495                    .lock()
1496                    .unwrap();
1497
1498                let pending = self
1499                    .get_pending_state(&mut attributes)
1500                    .unwrap_or_else(|| attributes.current_server_state());
1501                // retrieve the current state before adding it to the
1502                // pending state so that we can compare what has changed
1503                let current = attributes.current_server_state();
1504
1505                // test if we should send the decoration mode, either because it changed
1506                // or we never sent it
1507                let decoration_mode_changed = !attributes.initial_decoration_configure_sent
1508                    || (pending.decoration_mode != current.decoration_mode);
1509
1510                // test if we should send a bounds configure event, either because the
1511                // bounds changed or we never sent one
1512                let bounds_changed = !attributes.initial_configure_sent || (pending.bounds != current.bounds);
1513
1514                // test if we should send a capabilities event, either because the
1515                // capabilities changed or we never sent one
1516                let capabilities_changed =
1517                    !attributes.initial_configure_sent || (pending.capabilities != current.capabilities);
1518
1519                let configure = ToplevelConfigure {
1520                    serial: SERIAL_COUNTER.next_serial(),
1521                    state: pending,
1522                };
1523
1524                attributes.pending_configures.push(configure.clone());
1525                attributes.initial_configure_sent = true;
1526                if decoration.is_some() {
1527                    attributes.initial_decoration_configure_sent = true;
1528                }
1529
1530                (
1531                    configure,
1532                    decoration_mode_changed,
1533                    bounds_changed,
1534                    capabilities_changed,
1535                )
1536            });
1537
1538        if decoration_mode_changed {
1539            if let Some(decoration) = &decoration {
1540                self::decoration::send_decoration_configure(
1541                    decoration,
1542                    configure
1543                        .state
1544                        .decoration_mode
1545                        .unwrap_or(zxdg_toplevel_decoration_v1::Mode::ClientSide),
1546                );
1547            }
1548        }
1549
1550        let serial = configure.serial;
1551        self::handlers::send_toplevel_configure(
1552            &self.shell_surface,
1553            configure,
1554            bounds_changed,
1555            capabilities_changed,
1556        );
1557
1558        serial
1559    }

1577    /// Handles the role specific commit error checking
1578    ///
1579    /// This should be called when the underlying WlSurface
1580    /// handles a wl_surface.commit request.
1581    pub(crate) fn pre_commit_hook<D: XdgShellHandler + 'static>(
1582        state: &mut D,
1583        _dh: &DisplayHandle,
1584        surface: &wl_surface::WlSurface,
1585    ) {
1586        let _span = trace_span!("xdg-toplevel pre-commit", surface = %surface.id()).entered();
1587
1588        let error = compositor::with_states(surface, |states| {
1589            let mut role = states
1590                .data_map
1591                .get::<XdgToplevelSurfaceData>()
1592                .unwrap()
1593                .lock()
1594                .unwrap();
1595
1596            let mut guard_toplevel = states.cached_state.get::<ToplevelCachedState>();
1597            let pending = guard_toplevel.pending();
1598
1599            // The presence of last_acked always follows the buffer assignment because the
1600            // surface is not allowed to attach a buffer without acking the initial configure.
1601            let had_buffer_before = pending.last_acked.is_some();
1602
1603            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
1604            let has_buffer = match &guard_surface.pending().buffer {
1605                Some(BufferAssignment::NewBuffer(_)) => true,
1606                Some(BufferAssignment::Removed) => false,
1607                None => had_buffer_before,
1608            };
1609            // Need to check had_buffer_before in case the client attaches a null buffer for the
1610            // initial commit---we don't want to consider that as "got unmapped" and reset role.
1611            let got_unmapped = had_buffer_before && !has_buffer;
1612
1613            if has_buffer {
1614                let Some(last_acked) = role.last_acked.clone() else {
1615                    return Some((
1616                        xdg_surface::Error::UnconfiguredBuffer,
1617                        "must ack the initial configure before attaching buffer",
1618                    ));
1619                };
1620
1621                // The surface remains, or became mapped, track the last acked state.
1622                pending.last_acked = Some(last_acked);
1623            } else {
1624                // The surface remains, or became, unmapped, meaning that it's in the initial
1625                // configure stage.
1626                pending.last_acked = None;
1627            }
1628
1629            if got_unmapped {
1630                trace!(
1631                    "got unmapped; resetting role and cached state; have {} pending configures",
1632                    role.pending_configures.len()
1633                );
1634
1635                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
1636                // the list of pending configures because there's no way for a surface to tell
1637                // an in-flight configure apart from our next initial configure after unmapping.
1638                //
1639                // This can be reproduced with wleird-unmap if, say, and Activated configure is
1640                // timed right.
1641                let pending_configures = std::mem::take(&mut role.pending_configures);
1642                *role = Default::default();
1643                role.pending_configures = pending_configures;
1644                *guard_toplevel.pending() = Default::default();
1645
1646                // Don't forget to prepare the default capabilities though.
1647                let default_capabilities = &state.xdg_shell_state().default_capabilities;
1648                role.server_pending = Some(Default::default());
1649                let current_capabilties = &mut role.server_pending.as_mut().unwrap().capabilities;
1650                current_capabilties.replace(default_capabilities.capabilities.iter().copied());
1651            }
1652
1653            None
1654        });
1655
1656        if let Some((error, msg)) = error {
1657            // The surface attached a buffer without acking the initial configure.
1658            let toplevels = &state.xdg_shell_state().known_toplevels;
1659            if let Some(toplevel) = toplevels.iter().find(|handle| handle.wl_surface == *surface) {
1660                let data = toplevel
1661                    .shell_surface
1662                    .data::<self::handlers::XdgShellSurfaceUserData>()
1663                    .unwrap();
1664                data.xdg_surface.post_error(error, msg);
1665            } else {
1666                error!("surface missing from known toplevels");
1667            }
1668        }
1669    }

1867    /// If no changes occurred no event will be send and `Ok(None)` will be returned.
1868    ///
1869    /// See [`send_configure`](PopupSurface::send_configure) and [`has_pending_changes`](PopupSurface::has_pending_changes)
1870    /// for more information.
1871    pub fn send_pending_configure(&self) -> Result<Option<Serial>, PopupConfigureError> {
1872        if self.has_pending_changes() {
1873            self.send_configure().map(Some)
1874        } else {
1875            Ok(None)
1876        }
1877    }

1879    /// Send a configure event to this popup surface to suggest it a new configuration
1880    ///
1881    /// The serial of this configure will be tracked waiting for the client to ACK it.
1882    ///
1883    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
1884    /// method.
1885    ///
1886    /// Returns `Err(`[`PopupConfigureError`]`)` if the initial configure has already been sent and
1887    /// the client protocol version disallows a re-configure or the current [`PositionerState`]
1888    /// is not reactive.
1889    ///
1890    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
1891    /// [`send_pending_configure`](PopupSurface::send_pending_configure)
1892    pub fn send_configure(&self) -> Result<Serial, PopupConfigureError> {
1893        // Check if we are allowed to send a configure
1894        compositor::with_states(&self.wl_surface, |states| {
1895            let attributes = states
1896                .data_map
1897                .get::<XdgPopupSurfaceData>()
1898                .unwrap()
1899                .lock()
1900                .unwrap();
1901
1902            if attributes.initial_configure_sent && self.version() < xdg_popup::EVT_REPOSITIONED_SINCE {
1903                // Return error, initial configure already sent and client
1904                // does not support re-configure
1905                return Err(PopupConfigureError::AlreadyConfigured);
1906            }
1907
1908            let mut cached = states.cached_state.get::<PopupCachedState>();
1909            let is_reactive = cached
1910                .current()
1911                .last_acked
1912                .as_ref()
1913                .is_some_and(|c| c.state.positioner.reactive);
1914
1915            if attributes.initial_configure_sent && !is_reactive {
1916                // Return error, the positioner does not allow re-configure
1917                return Err(PopupConfigureError::NotReactive);
1918            }
1919
1920            Ok(())
1921        })?;
1922
1923        let serial = self.send_configure_internal(None);
1924
1925        Ok(serial)
1926    }

1952    /// Handles the role specific commit error checking
1953    ///
1954    /// This should be called when the underlying WlSurface
1955    /// handles a wl_surface.commit request.
1956    pub(crate) fn pre_commit_hook<D: XdgShellHandler + 'static>(
1957        state: &mut D,
1958        _dh: &DisplayHandle,
1959        surface: &wl_surface::WlSurface,
1960    ) {
1961        let _span = trace_span!("xdg-popup pre-commit", surface = %surface.id()).entered();
1962
1963        let res = compositor::with_states(surface, |states| {
1964            let mut role = states
1965                .data_map
1966                .get::<XdgPopupSurfaceData>()
1967                .unwrap()
1968                .lock()
1969                .unwrap();
1970            if role.parent.is_none() {
1971                return Err((
1972                    xdg_surface::Error::NotConstructed,
1973                    "xdg_popup must have parent before mapping",
1974                ));
1975            }
1976
1977            let mut guard_popup = states.cached_state.get::<PopupCachedState>();
1978            let pending = guard_popup.pending();
1979
1980            // The presence of last_acked always follows the buffer assignment because the
1981            // surface is not allowed to attach a buffer without acking the initial configure.
1982            let had_buffer_before = pending.last_acked.is_some();
1983
1984            let grab = role.requested_grab.take();
1985            if grab.is_some() && had_buffer_before {
1986                let popups = &state.xdg_shell_state().known_popups;
1987                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
1988                    popup
1989                        .shell_surface
1990                        .post_error(xdg_popup::Error::InvalidGrab, "tried to grab after being mapped");
1991                } else {
1992                    error!("surface missing from known popups");
1993                }
1994                return Ok(None);
1995            }
1996
1997            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
1998            let has_buffer = match &guard_surface.pending().buffer {
1999                Some(BufferAssignment::NewBuffer(_)) => true,
2000                Some(BufferAssignment::Removed) => false,
2001                None => had_buffer_before,
2002            };
2003            // Need to check had_buffer_before in case the client attaches a null buffer for the
2004            // initial commit---we don't want to consider that as "got unmapped" and reset role.
2005            let got_unmapped = had_buffer_before && !has_buffer;
2006
2007            if has_buffer {
2008                let Some(last_acked) = role.last_acked else {
2009                    return Err((
2010                        xdg_surface::Error::UnconfiguredBuffer,
2011                        "must ack the initial configure before attaching buffer",
2012                    ));
2013                };
2014
2015                // The surface remains, or became mapped, track the last acked state.
2016                pending.last_acked = Some(last_acked);
2017            } else {
2018                // The surface remains, or became, unmapped, meaning that it's in the initial
2019                // configure stage.
2020                pending.last_acked = None;
2021            }
2022
2023            if got_unmapped {
2024                trace!(
2025                    "got unmapped; resetting role and cached state; have {} pending configures",
2026                    role.pending_configures.len()
2027                );
2028
2029                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
2030                // the list of pending configures because there's no way for a surface to tell
2031                // an in-flight configure apart from our next initial configure after unmapping.
2032                let pending_configures = std::mem::take(&mut role.pending_configures);
2033                *role = Default::default();
2034                role.pending_configures = pending_configures;
2035                *guard_popup.pending() = Default::default();
2036            }
2037
2038            Ok(grab)
2039        });
2040
2041        match res {
2042            Ok(Some((seat, serial))) => {
2043                let popups = &state.xdg_shell_state().known_popups;
2044                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
2045                    let popup = popup.clone();
2046                    XdgShellHandler::grab(state, popup, seat, serial);
2047                } else {
2048                    error!("surface missing from known popups");
2049                }
2050            }
2051            Ok(None) => (),
2052            Err((error, msg)) => {
2053                let popups = &state.xdg_shell_state().known_popups;
2054                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
2055                    let data = popup
2056                        .shell_surface
2057                        .data::<self::handlers::XdgShellSurfaceUserData>()
2058                        .unwrap();
2059                    data.xdg_surface.post_error(error, msg);
2060                } else {
2061                    error!("surface missing from known popups");
2062                }
2063            }
2064        }
2065    }
