1//! Utilities for handling shell surfaces with the `xdg_shell` protocol
2//!
3//! This module provides automatic handling of shell surfaces objects, by being registered
4//! as a global handler for `xdg_shell`.
5//!
6//! ## Why use this implementation
7//!
8//! This implementation can track for you the various shell surfaces defined by the
9//! clients by handling the `xdg_shell` protocol.
10//!
11//! It allows you to easily access a list of all shell surfaces defined by your clients
12//! access their associated metadata and underlying `wl_surface`s.
13//!
14//! This handler only handles the protocol exchanges with the client to present you the
15//! information in a coherent and relatively easy to use manner. All the actual drawing
16//! and positioning logic of windows is out of its scope.
17//!
18//! ## How to use it
19//!
20//! ### Initialization
21//!
22//! To initialize this handler, create [`XdgShellState`], store it in your `State` struct and
23//! implement the [`XdgShellHandler`], as shown in this example:
24//!
25//! ```no_run
26//! # extern crate wayland_server;
27//! #
28//! use smithay::delegate_xdg_shell;
29//! use smithay::reexports::wayland_server::protocol::{wl_seat, wl_surface};
30//! use smithay::wayland::shell::xdg::{XdgShellState, XdgShellHandler, ToplevelSurface, PopupSurface, PositionerState};
31//! use smithay::utils::Serial;
32//!
33//! # struct State { xdg_shell_state: XdgShellState, seat_state: SeatState<Self> }
34//! # let mut display = wayland_server::Display::<State>::new().unwrap();
35//! let xdg_shell_state = XdgShellState::new::<State>(
36//!     &display.handle(),
37//! );
38//!
39//! // insert the xdg_shell_state into your state
40//! // ...
41//!
42//! // implement the necessary traits
43//! impl XdgShellHandler for State {
44//!     fn xdg_shell_state(&mut self) -> &mut XdgShellState {
45//!         &mut self.xdg_shell_state
46//!     }
47//!
48//!     // handle the shell requests here.
49//!     // more optional methods can be used to further customized
50//!     fn new_toplevel(&mut self, surface: ToplevelSurface) {
51//!         // ...
52//!     }
53//!     fn new_popup(
54//!         &mut self,
55//!         surface: PopupSurface,
56//!         positioner: PositionerState,
57//!     ) {
58//!         // ...
59//!     }
60//!     fn grab(
61//!         &mut self,
62//!         surface: PopupSurface,
63//!         seat: wl_seat::WlSeat,
64//!         serial: Serial,
65//!     ) {
66//!         // ...
67//!     }
68//!     fn reposition_request(
69//!         &mut self,
70//!         surface: PopupSurface,
71//!         positioner: PositionerState,
72//!         token: u32,
73//!     ) {
74//!         // ...
75//!     }
76//! }
77//!
78//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
79//!
80//! type Target = wl_surface::WlSurface;
81//! impl SeatHandler for State {
82//!     type KeyboardFocus = Target;
83//!     type PointerFocus = Target;
84//!     type TouchFocus = Target;
85//!
86//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
87//!         &mut self.seat_state
88//!     }
89//!
90//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&Target>) {
91//!         // handle focus changes, if you need to ...
92//!     }
93//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) {
94//!         // handle new images for the cursor ...
95//!     }
96//! }
97//! delegate_xdg_shell!(State);
98//!
99//! // You're now ready to go!
100//! ```
101//!
102//! ### Access to shell surface and clients data
103//!
104//! There are mainly 3 kind of objects that you'll manipulate from this implementation:
105//!
106//! - [`ShellClient`]:
107//!   This is a handle representing an instantiation of a shell global
108//!   you can associate client-wise metadata to it through an [`UserDataMap`].
109//! - [`ToplevelSurface`]:
110//!   This is a handle representing a toplevel surface, you can
111//!   retrieve a list of all currently alive toplevel surface from the
112//!   [`XdgShellState`].
113//! - [`PopupSurface`]:
114//!   This is a handle representing a popup/tooltip surface. Similarly,
115//!   you can get a list of all currently alive popup surface from the
116//!   [`XdgShellState`].
117//!
118//! You'll obtain these objects though two means: either via the callback methods of
119//! the [`XdgShellHandler`], or via methods on the [`XdgShellState`].

162macro_rules! xdg_role {
163    ($state:ty,
164     $(#[$configure_meta:meta])* $configure_name:ident $({$($(#[$configure_field_meta:meta])* $configure_field_vis:vis$configure_field_name:ident:$configure_field_type:ty),*}),*,
165     $(#[$attributes_meta:meta])* $attributes_name:ident {$($(#[$attributes_field_meta:meta])* $attributes_field_vis:vis$attributes_field_name:ident:$attributes_field_type:ty),*},
166     $(#[$cached_state_meta:meta])* $cached_state_name:ident) => {
167
168        $(#[$configure_meta])*
169        pub struct $configure_name {
170                /// The state associated with this configure
171                pub state: $state,
172                /// A serial number to track ACK from the client
173                ///
174                /// This should be an ever increasing number, as the ACK-ing
175                /// from a client for a serial will validate all pending lower
176                /// serials.
177                pub serial: Serial,
178
179                $($(
180                    $(#[$configure_field_meta])*
181                    $configure_field_vis $configure_field_name: $configure_field_type,
182                )*)*
183        }
184
185        $(#[$attributes_meta])*
186        pub struct $attributes_name {
187                /// Holds the state if the surface has sent the initial
188                /// configure event to the client. It is expected that
189                /// during the first commit a initial
190                /// configure event is sent to the client
191                pub initial_configure_sent: bool,
192                /// Holds the configures the server has sent out
193                /// to the client waiting to be acknowledged by
194                /// the client. All pending configures that are older
195                /// than the acknowledged one will be discarded during
196                /// processing xdg_surface.ack_configure.
197                pending_configures: Vec<$configure_name>,
198                /// Holds the pending state as set by the server.
199                pub server_pending: Option<$state>,
200                /// Holds the last configure that has been acknowledged by the client. This state
201                /// should be cloned to the current during a commit. Note that this state can be
202                /// newer than the last acked state at the time of the last commit.
203                pub last_acked: Option<$configure_name>,
204
205                $(
206                    $(#[$attributes_field_meta])*
207                    $attributes_field_vis $attributes_field_name: $attributes_field_type,
208                )*
209        }
210
211        impl $attributes_name {
212            fn ack_configure(&mut self, serial: Serial) -> Option<Configure> {
213                let configure = match self
214                    .pending_configures
215                    .iter()
216                    .find(|configure| configure.serial == serial)
217                {
218                    Some(configure) => (*configure).clone(),
219                    None => {
220                        return None;
221                    }
222                };
223
224                // Save the configure as the last acked configure
225                self.last_acked = Some(configure.clone());
226
227                // Clean old configures
228                self.pending_configures.retain(|c| c.serial > serial);
229
230                Some(configure.into())
231            }
232
233            /// Gets the latest state that has been configured
234            /// on the server and sent to the client.
235            ///
236            /// The state includes all changes that have been
237            /// made on the server, including all not yet
238            /// acked or committed changes, but excludes the
239            /// current [`server_pending`](#structfield.server_pending) state.
240            ///
241            /// This can be used for example to check if the
242            /// [`server_pending`](#structfield.server_pending) state
243            /// is different from the last configured.
244            pub fn current_server_state(&self) -> $state {
245                // We check if there is already a non-acked pending
246                // configure and use its state or otherwise we could
247                // loose some state that was previously configured
248                // and sent, but not acked before calling with_pending_state
249                // again. If there is no pending state we try to use the
250                // last acked state which could contain state changes
251                // already acked but not committed to the current state.
252                // In both cases the state already contains all previous
253                // sent states. This way all pending state is accumulated
254                // into the current state.
255                self.pending_configures
256                    .last()
257                    .map(|c| &c.state)
258                    .or_else(|| self.last_acked.as_ref().map(|c| &c.state))
259                    .cloned()
260                    .unwrap_or_default()
261            }
262
263            /// Check if the state has pending changes that have
264            /// not been sent to the client.
265            ///
266            /// This differs from just checking if the [`server_pending`](#structfield.server_pending)
267            /// state is [`Some`] in that it also checks if a current pending
268            /// state is different from the [`current_server_state`](#method.current_server_state).
269            pub fn has_pending_changes(&self) -> bool {
270                self.server_pending.as_ref().map(|s| *s != self.current_server_state()).unwrap_or(false)
271            }
272
273            /// Returns a list of configures sent to, but not yet acknowledged by the client.
274            ///
275            /// The list is ordered by age, so the last configure in the list is the last one sent
276            /// to the client.
277            pub fn pending_configures(&self) -> &[$configure_name] {
278                &self.pending_configures
279            }
280        }
281
282        impl Default for $attributes_name {
283            fn default() -> Self {
284                Self {
285                    pending_configures: Vec::new(),
286                    initial_configure_sent: false,
287                    server_pending: None,
288                    last_acked: None,
289
290                    $(
291                        $attributes_field_name: Default::default(),
292                    )*
293                }
294            }
295        }
296
297        $(#[$cached_state_meta])*
298        pub struct $cached_state_name {
299            /// Configure last acknowledged by the client at the time of the commit.
300            ///
301            /// Reset to `None` when the surface unmaps.
302            pub last_acked: Option<$configure_name>,
303        }
304
305        impl Cacheable for $cached_state_name {
306            fn commit(&mut self, _dh: &DisplayHandle) -> Self {
307                self.clone()
308            }
309            fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
310                *into = self;
311            }
312        }
313    };
314}

540impl PositionerState {
541    pub(crate) fn anchor_has_edge(&self, edge: xdg_positioner::Anchor) -> bool {
542        match edge {
543            xdg_positioner::Anchor::Top => {
544                self.anchor_edges == xdg_positioner::Anchor::Top
545                    || self.anchor_edges == xdg_positioner::Anchor::TopLeft
546                    || self.anchor_edges == xdg_positioner::Anchor::TopRight
547            }
548            xdg_positioner::Anchor::Bottom => {
549                self.anchor_edges == xdg_positioner::Anchor::Bottom
550                    || self.anchor_edges == xdg_positioner::Anchor::BottomLeft
551                    || self.anchor_edges == xdg_positioner::Anchor::BottomRight
552            }
553            xdg_positioner::Anchor::Left => {
554                self.anchor_edges == xdg_positioner::Anchor::Left
555                    || self.anchor_edges == xdg_positioner::Anchor::TopLeft
556                    || self.anchor_edges == xdg_positioner::Anchor::BottomLeft
557            }
558            xdg_positioner::Anchor::Right => {
559                self.anchor_edges == xdg_positioner::Anchor::Right
560                    || self.anchor_edges == xdg_positioner::Anchor::TopRight
561                    || self.anchor_edges == xdg_positioner::Anchor::BottomRight
562            }
563            _ => unreachable!(),
564        }
565    }
566
567    /// Get the anchor point for a popup as defined by this positioner.
568    ///
569    /// Defined by `xdg_positioner.set_anchor_rect` and
570    /// `xdg_positioner.set_anchor`.
571    pub fn get_anchor_point(&self) -> Point<i32, Logical> {
572        let mut point = self.anchor_rect.loc;
573
574        point.y += if self.anchor_has_edge(xdg_positioner::Anchor::Top) {
575            0
576        } else if self.anchor_has_edge(xdg_positioner::Anchor::Bottom) {
577            self.anchor_rect.size.h
578        } else {
579            self.anchor_rect.size.h / 2
580        };
581
582        point.x += if self.anchor_has_edge(xdg_positioner::Anchor::Left) {
583            0
584        } else if self.anchor_has_edge(xdg_positioner::Anchor::Right) {
585            self.anchor_rect.size.w
586        } else {
587            self.anchor_rect.size.w / 2
588        };
589
590        point
591    }

619    /// Get the geometry for a popup as defined by this positioner.
620    ///
621    /// `Rectangle::width` and `Rectangle::height` corresponds to the
622    /// size set by `xdg_positioner.set_size`.
623    ///
624    /// `Rectangle::x` and `Rectangle::y` define the position of the
625    /// popup relative to its parent surface's `window_geometry`.
626    /// The position is calculated according to the rules defined
627    /// in the `xdg_shell` protocol.
628    /// The `constraint_adjustment` will not be considered by this
629    /// implementation and the position and size should be re-calculated
630    /// in the compositor if the compositor implements `constraint_adjustment`
631    ///
632    /// [`PositionerState::get_unconstrained_geometry`] does take `constraint_adjustment` into account.
633    pub fn get_geometry(&self) -> Rectangle<i32, Logical> {
634        // From the `xdg_shell` prococol specification:
635        //
636        // set_offset:
637        //
638        //  Specify the surface position offset relative to the position of the
639        //  anchor on the anchor rectangle and the anchor on the surface. For
640        //  example if the anchor of the anchor rectangle is at (x, y), the surface
641        //  has the gravity bottom|right, and the offset is (ox, oy), the calculated
642        //  surface position will be (x + ox, y + oy)
643        let mut geometry = Rectangle::new(self.offset, self.rect_size);
644
645        // Defines the anchor point for the anchor rectangle. The specified anchor
646        // is used derive an anchor point that the child surface will be
647        // positioned relative to. If a corner anchor is set (e.g. 'top_left' or
648        // 'bottom_right'), the anchor point will be at the specified corner;
649        // otherwise, the derived anchor point will be centered on the specified
650        // edge, or in the center of the anchor rectangle if no edge is specified.
651        geometry.loc += self.get_anchor_point();
652
653        // Defines in what direction a surface should be positioned, relative to
654        // the anchor point of the parent surface. If a corner gravity is
655        // specified (e.g. 'bottom_right' or 'top_left'), then the child surface
656        // will be placed towards the specified gravity; otherwise, the child
657        // surface will be centered over the anchor point on any axis that had no
658        // gravity specified.
659        if self.gravity_has_edge(xdg_positioner::Gravity::Top) {
660            geometry.loc.y -= geometry.size.h;
661        } else if !self.gravity_has_edge(xdg_positioner::Gravity::Bottom) {
662            geometry.loc.y -= geometry.size.h / 2;
663        }
664
665        if self.gravity_has_edge(xdg_positioner::Gravity::Left) {
666            geometry.loc.x -= geometry.size.w;
667        } else if !self.gravity_has_edge(xdg_positioner::Gravity::Right) {
668            geometry.loc.x -= geometry.size.w / 2;
669        }
670
671        geometry
672    }
673
674    /// Get the geometry for a popup as defined by this positioner, after trying to fit the popup into the
675    /// target rectangle.
676    ///
677    /// `Rectangle::width` and `Rectangle::height` corresponds to the size set by `xdg_positioner.set_size`.
678    ///
679    /// `Rectangle::x` and `Rectangle::y` define the position of the popup relative to its parent surface's
680    /// `window_geometry`. The position is calculated according to the rules defined in the `xdg_shell`
681    /// protocol.
682    ///
683    /// This method does consider `constrain_adjustment` by trying to fit the popup into the provided target
684    /// rectangle. The target rectangle is in the same coordinate system as the rectangle returned by this
685    /// method. So, it is relative to the parent surface's geometry.
686    pub fn get_unconstrained_geometry(mut self, target: Rectangle<i32, Logical>) -> Rectangle<i32, Logical> {
687        // The protocol defines the following order for adjustments: flip, slide, resize. If the flip fails
688        // to remove the constraints, it is reverted.
689        //
690        // The adjustments are applied individually between axes. We can do that reasonably safely, given
691        // that both our target and our popup are simple rectangles. The code is grouped per adjustment for
692        // easier copy-paste checking, and because flips replace the geometry entirely, while further
693        // adjustments change individual fields.
694        let mut geo = self.get_geometry();
695        let (mut off_left, mut off_right, mut off_top, mut off_bottom) = compute_offsets(target, geo);
696
697        // Try to flip horizontally.
698        if (off_left > 0 || off_right > 0) && self.constraint_adjustment.contains(ConstraintAdjustment::FlipX)
699        {
700            let mut new = self;
701            new.anchor_edges = invert_anchor_x(new.anchor_edges);
702            new.gravity = invert_gravity_x(new.gravity);
703            let new_geo = new.get_geometry();
704            let (new_off_left, new_off_right, _, _) = compute_offsets(target, new_geo);
705
706            // Apply flip only if it removed the constraint.
707            if new_off_left <= 0 && new_off_right <= 0 {
708                self = new;
709                geo = new_geo;
710                off_left = 0;
711                off_right = 0;
712                // off_top and off_bottom are unchanged since we're using rectangles.
713            }
714        }
715
716        // Try to flip vertically.
717        if (off_top > 0 || off_bottom > 0) && self.constraint_adjustment.contains(ConstraintAdjustment::FlipY)
718        {
719            let mut new = self;
720            new.anchor_edges = invert_anchor_y(new.anchor_edges);
721            new.gravity = invert_gravity_y(new.gravity);
722            let new_geo = new.get_geometry();
723            let (_, _, new_off_top, new_off_bottom) = compute_offsets(target, new_geo);
724
725            // Apply flip only if it removed the constraint.
726            if new_off_top <= 0 && new_off_bottom <= 0 {
727                self = new;
728                geo = new_geo;
729                off_top = 0;
730                off_bottom = 0;
731                // off_left and off_right are unchanged since we're using rectangles.
732            }
733        }
734
735        // Try to slide horizontally.
736        if (off_left > 0 || off_right > 0)
737            && self.constraint_adjustment.contains(ConstraintAdjustment::SlideX)
738        {
739            // Prefer to show the top-left corner of the popup so that we can easily do a resize
740            // adjustment next.
741            if off_left > 0 {
742                geo.loc.x += off_left;
743            } else if off_right > 0 {
744                geo.loc.x -= min(off_right, -off_left);
745            }
746
747            (off_left, off_right, _, _) = compute_offsets(target, geo);
748            // off_top and off_bottom are the same since we're using rectangles.
749        }
750
751        // Try to slide vertically.
752        if (off_top > 0 || off_bottom > 0)
753            && self.constraint_adjustment.contains(ConstraintAdjustment::SlideY)
754        {
755            // Prefer to show the top-left corner of the popup so that we can easily do a resize
756            // adjustment next.
757            if off_top > 0 {
758                geo.loc.y += off_top;
759            } else if off_bottom > 0 {
760                geo.loc.y -= min(off_bottom, -off_top);
761            }
762
763            (_, _, off_top, off_bottom) = compute_offsets(target, geo);
764            // off_left and off_right are the same since we're using rectangles.
765        }
766
767        // Try to resize horizontally.
768        if self.constraint_adjustment.contains(ConstraintAdjustment::ResizeX) {
769            // Unconstrain both edges by clamping the left and right sides of the popup rectangle.
770            // Skip if the offset is larger than the width, in which case the entirety of the popup
771            // is outside the target rectangle and a clamp would result in a zero-sized geometry.
772
773            if off_left > 0 && off_left < geo.size.w {
774                geo.loc.x += off_left;
775                geo.size.w -= off_left;
776            }
777            if off_right > 0 && off_right < geo.size.w {
778                geo.size.w -= off_right;
779            }
780        }
781
782        // Try to resize vertically.
783        if self.constraint_adjustment.contains(ConstraintAdjustment::ResizeY) {
784            // Unconstrain both edges by clamping the top and bottom sides of the popup rectangle.
785            // Skip if the offset is larger than the height, in which case the entirety of the popup
786            // is outside the target rectangle and a clamp would result in a zero-sized geometry.
787
788            if off_top > 0 && off_top < geo.size.h {
789                geo.loc.y += off_top;
790                geo.size.h -= off_top;
791            }
792            if off_bottom > 0 && off_bottom < geo.size.h {
793                geo.size.h -= off_bottom;
794            }
795        }
796
797        geo
798    }
799}

801fn compute_offsets(target: Rectangle<i32, Logical>, popup: Rectangle<i32, Logical>) -> (i32, i32, i32, i32) {
802    let off_left = target.loc.x - popup.loc.x;
803    let off_right = (popup.loc.x + popup.size.w) - (target.loc.x + target.size.w);
804    let off_top = target.loc.y - popup.loc.y;
805    let off_bottom = (popup.loc.y + popup.size.h) - (target.loc.y + target.size.h);
806    (off_left, off_right, off_top, off_bottom)
807}

809fn invert_anchor_x(anchor: Anchor) -> Anchor {
810    match anchor {
811        Anchor::Left => Anchor::Right,
812        Anchor::Right => Anchor::Left,
813        Anchor::TopLeft => Anchor::TopRight,
814        Anchor::TopRight => Anchor::TopLeft,
815        Anchor::BottomLeft => Anchor::BottomRight,
816        Anchor::BottomRight => Anchor::BottomLeft,
817        x => x,
818    }
819}

822    match anchor {
823        Anchor::Top => Anchor::Bottom,
824        Anchor::Bottom => Anchor::Top,
825        Anchor::TopLeft => Anchor::BottomLeft,
826        Anchor::TopRight => Anchor::BottomRight,
827        Anchor::BottomLeft => Anchor::TopLeft,
828        Anchor::BottomRight => Anchor::TopRight,
829        x => x,
830    }
831}

895/// Container holding the states for a `XdgToplevel`
896///
897/// This container will prevent the `XdgToplevel` from
898/// having the same `xdg_toplevel::State` multiple times
899/// and simplifies setting and un-setting a particularly
900/// `xdg_toplevel::State`
901#[derive(Debug, Default, Clone, PartialEq, Eq)]
902pub struct ToplevelStateSet {
903    states: Vec<xdg_toplevel::State>,
904}
905
906impl ToplevelStateSet {
907    /// Returns `true` if the states contains a state.
908    pub fn contains(&self, state: xdg_toplevel::State) -> bool {
909        self.states.contains(&state)
910    }
911
912    /// Adds a state to the states.
913    ///
914    /// If the states did not have this state present, `true` is returned.
915    ///
916    /// If the states did have this state present, `false` is returned.
917    pub fn set(&mut self, state: xdg_toplevel::State) -> bool {
918        if self.contains(state) {
919            false
920        } else {
921            self.states.push(state);
922            true
923        }
924    }
925
926    /// Removes a state from the states. Returns whether the state was
927    /// present in the states.
928    pub fn unset(&mut self, state: xdg_toplevel::State) -> bool {
929        if !self.contains(state) {
930            false
931        } else {
932            self.states.retain(|s| *s != state);
933            true
934        }
935    }
936
937    /// Filter the states according to the provided version
938    /// of the [`XdgToplevel`]
939    pub(crate) fn into_filtered_states(mut self, version: u32) -> Vec<xdg_toplevel::State> {
940        let constrained_supported = version >= XDG_TOPLEVEL_STATE_CONSTRAINED_SINCE;
941
942        // If constrained is supported, everything is, so we can just return
943        if constrained_supported {
944            return self.states;
945        }
946
947        fn version_needed_for_state(state: &xdg_toplevel::State) -> u32 {
948            match state {
949                xdg_toplevel::State::TiledLeft
950                | xdg_toplevel::State::TiledRight
951                | xdg_toplevel::State::TiledTop
952                | xdg_toplevel::State::TiledBottom => XDG_TOPLEVEL_STATE_TILED_SINCE,
953                xdg_toplevel::State::ConstrainedLeft
954                | xdg_toplevel::State::ConstrainedRight
955                | xdg_toplevel::State::ConstrainedTop
956                | xdg_toplevel::State::ConstrainedBottom => XDG_TOPLEVEL_STATE_CONSTRAINED_SINCE,
957                xdg_toplevel::State::Suspended => XDG_TOPLEVEL_STATE_SUSPENDED_SINCE,
958                _ => 1,
959            }
960        }
961
962        self.states
963            .retain(|state| version >= version_needed_for_state(state));
964
965        self.states
966    }
967}

1333    /// Send a ping request to this shell client
1334    ///
1335    /// You'll receive the reply as a [`XdgShellHandler::client_pong`] request.
1336    ///
1337    /// A typical use is to start a timer at the same time you send this ping
1338    /// request, and cancel it when you receive the pong. If the timer runs
1339    /// down to 0 before a pong is received, mark the client as unresponsive.
1340    ///
1341    /// Fails if this shell client already has a pending ping or is already dead.
1342    pub fn send_ping(&self, serial: Serial) -> Result<(), PingError> {
1343        if !self.alive() {
1344            return Err(PingError::DeadSurface);
1345        }
1346        let user_data = self.kind.data::<self::handlers::XdgWmBaseUserData>().unwrap();

1348        if let Some(pending_ping) = guard.pending_ping {
1349            return Err(PingError::PingAlreadyPending(pending_ping));
1350        }
1351        guard.pending_ping = Some(serial);
1352        self.kind.ping(serial.into());
1353
1354        Ok(())
1355    }

1363        }

1366            "client did not respond to ping on time",

1460    ///
1461    /// See [`send_configure`](ToplevelSurface::send_configure) and [`has_pending_changes`](ToplevelSurface::has_pending_changes)

1471    /// Send a configure event to this toplevel surface to suggest it a new configuration
1472    ///
1473    /// The serial of this configure will be tracked waiting for the client to ACK it.
1474    ///
1475    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
1476    /// method.
1477    ///
1478    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
1479    /// [`send_pending_configure`](ToplevelSurface::send_pending_configure)
1480    pub fn send_configure(&self) -> Serial {
1481        let shell_surface_data = self.shell_surface.data::<XdgShellSurfaceUserData>();
1482        let decoration =
1483            shell_surface_data.and_then(|data| data.decoration.lock().unwrap().as_ref().cloned());
1484        let (configure, decoration_mode_changed, bounds_changed, capabilities_changed) =
1485            compositor::with_states(&self.wl_surface, |states| {
1486                let mut attributes = states
1487                    .data_map
1488                    .get::<XdgToplevelSurfaceData>()
1489                    .unwrap()
1490                    .lock()
1491                    .unwrap();
1492
1493                let pending = self
1494                    .get_pending_state(&mut attributes)
1495                    .unwrap_or_else(|| attributes.current_server_state());
1496                // retrieve the current state before adding it to the
1497                // pending state so that we can compare what has changed
1498                let current = attributes.current_server_state();
1499
1500                // test if we should send the decoration mode, either because it changed
1501                // or we never sent it
1502                let decoration_mode_changed = !attributes.initial_decoration_configure_sent
1503                    || (pending.decoration_mode != current.decoration_mode);
1504
1505                // test if we should send a bounds configure event, either because the
1506                // bounds changed or we never sent one
1507                let bounds_changed = !attributes.initial_configure_sent || (pending.bounds != current.bounds);
1508
1509                // test if we should send a capabilities event, either because the
1510                // capabilities changed or we never sent one
1511                let capabilities_changed =
1512                    !attributes.initial_configure_sent || (pending.capabilities != current.capabilities);
1513
1514                let configure = ToplevelConfigure {
1515                    serial: SERIAL_COUNTER.next_serial(),
1516                    state: pending,
1517                };
1518
1519                attributes.pending_configures.push(configure.clone());
1520                attributes.initial_configure_sent = true;
1521                if decoration.is_some() {
1522                    attributes.initial_decoration_configure_sent = true;
1523                }
1524
1525                (
1526                    configure,
1527                    decoration_mode_changed,
1528                    bounds_changed,
1529                    capabilities_changed,
1530                )
1531            });
1532
1533        if decoration_mode_changed {
1534            if let Some(decoration) = &decoration {
1535                self::decoration::send_decoration_configure(
1536                    decoration,
1537                    configure
1538                        .state
1539                        .decoration_mode
1540                        .unwrap_or(zxdg_toplevel_decoration_v1::Mode::ClientSide),
1541                );
1542            }
1543        }
1544
1545        let serial = configure.serial;
1546        self::handlers::send_toplevel_configure(
1547            &self.shell_surface,
1548            configure,
1549            bounds_changed,
1550            capabilities_changed,
1551        );
1552
1553        serial
1554    }

1572    /// Handles the role specific commit error checking
1573    ///
1574    /// This should be called when the underlying WlSurface
1575    /// handles a wl_surface.commit request.
1576    pub(crate) fn pre_commit_hook<D: XdgShellHandler + 'static>(
1577        state: &mut D,
1578        _dh: &DisplayHandle,
1579        surface: &wl_surface::WlSurface,
1580    ) {
1581        let _span = trace_span!("xdg-toplevel pre-commit", surface = %surface.id()).entered();
1582
1583        let error = compositor::with_states(surface, |states| {
1584            let mut role = states
1585                .data_map
1586                .get::<XdgToplevelSurfaceData>()
1587                .unwrap()
1588                .lock()
1589                .unwrap();
1590
1591            let mut guard_toplevel = states.cached_state.get::<ToplevelCachedState>();
1592            let pending = guard_toplevel.pending();
1593
1594            // The presence of last_acked always follows the buffer assignment because the
1595            // surface is not allowed to attach a buffer without acking the initial configure.
1596            let had_buffer_before = pending.last_acked.is_some();
1597
1598            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
1599            let has_buffer = match &guard_surface.pending().buffer {
1600                Some(BufferAssignment::NewBuffer(_)) => true,
1601                Some(BufferAssignment::Removed) => false,
1602                None => had_buffer_before,
1603            };
1604            // Need to check had_buffer_before in case the client attaches a null buffer for the
1605            // initial commit---we don't want to consider that as "got unmapped" and reset role.
1606            let got_unmapped = had_buffer_before && !has_buffer;
1607
1608            if has_buffer {
1609                let Some(last_acked) = role.last_acked.clone() else {
1610                    return Some((
1611                        xdg_surface::Error::UnconfiguredBuffer,
1612                        "must ack the initial configure before attaching buffer",
1613                    ));
1614                };
1615
1616                // The surface remains, or became mapped, track the last acked state.
1617                pending.last_acked = Some(last_acked);
1618            } else {
1619                // The surface remains, or became, unmapped, meaning that it's in the initial
1620                // configure stage.
1621                pending.last_acked = None;
1622            }
1623
1624            if got_unmapped {
1625                trace!(
1626                    "got unmapped; resetting role and cached state; have {} pending configures",
1627                    role.pending_configures.len()
1628                );
1629
1630                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
1631                // the list of pending configures because there's no way for a surface to tell
1632                // an in-flight configure apart from our next initial configure after unmapping.
1633                //
1634                // This can be reproduced with wleird-unmap if, say, and Activated configure is
1635                // timed right.
1636                let pending_configures = std::mem::take(&mut role.pending_configures);
1637                *role = Default::default();
1638                role.pending_configures = pending_configures;
1639                *guard_toplevel.pending() = Default::default();
1640
1641                // Don't forget to prepare the default capabilities though.
1642                let default_capabilities = &state.xdg_shell_state().default_capabilities;
1643                role.server_pending = Some(Default::default());
1644                let current_capabilities = &mut role.server_pending.as_mut().unwrap().capabilities;
1645                current_capabilities.replace(default_capabilities.capabilities.iter().copied());
1646            }
1647
1648            None
1649        });
1650
1651        if let Some((error, msg)) = error {
1652            // The surface attached a buffer without acking the initial configure.
1653            let toplevels = &state.xdg_shell_state().known_toplevels;
1654            if let Some(toplevel) = toplevels.iter().find(|handle| handle.wl_surface == *surface) {
1655                let data = toplevel
1656                    .shell_surface
1657                    .data::<self::handlers::XdgShellSurfaceUserData>()
1658                    .unwrap();
1659                data.xdg_surface.post_error(error, msg);
1660            } else {
1661                error!("surface missing from known toplevels");
1662            }
1663        }
1664    }

1861    /// (for tracking the configure in [`XdgShellHandler::ack_configure`] if desired).
1862    /// If no changes occurred no event will be send and `Ok(None)` will be returned.
1863    ///
1864    /// See [`send_configure`](PopupSurface::send_configure) and [`has_pending_changes`](PopupSurface::has_pending_changes)
1865    /// for more information.
1866    pub fn send_pending_configure(&self) -> Result<Option<Serial>, PopupConfigureError> {
1867        if self.has_pending_changes() {
1868            self.send_configure().map(Some)
1869        } else {
1870            Ok(None)
1871        }
1872    }

1874    /// Send a configure event to this popup surface to suggest it a new configuration
1875    ///
1876    /// The serial of this configure will be tracked waiting for the client to ACK it.
1877    ///
1878    /// You can manipulate the state that will be sent to the client with the [`with_pending_state`](#method.with_pending_state)
1879    /// method.
1880    ///
1881    /// Returns `Err(`[`PopupConfigureError`]`)` if the initial configure has already been sent and
1882    /// the client protocol version disallows a re-configure or the current [`PositionerState`]
1883    /// is not reactive.
1884    ///
1885    /// Note: This will always send a configure event, if you intend to only send a configure event on changes take a look at
1886    /// [`send_pending_configure`](PopupSurface::send_pending_configure)
1887    pub fn send_configure(&self) -> Result<Serial, PopupConfigureError> {
1888        // Check if we are allowed to send a configure
1889        compositor::with_states(&self.wl_surface, |states| {
1890            let attributes = states
1891                .data_map
1892                .get::<XdgPopupSurfaceData>()
1893                .unwrap()
1894                .lock()
1895                .unwrap();
1896
1897            if attributes.initial_configure_sent && self.version() < xdg_popup::EVT_REPOSITIONED_SINCE {
1898                // Return error, initial configure already sent and client
1899                // does not support re-configure
1900                return Err(PopupConfigureError::AlreadyConfigured);
1901            }
1902
1903            let mut cached = states.cached_state.get::<PopupCachedState>();
1904            let is_reactive = cached
1905                .current()
1906                .last_acked
1907                .as_ref()
1908                .is_some_and(|c| c.state.positioner.reactive);
1909
1910            if attributes.initial_configure_sent && !is_reactive {
1911                // Return error, the positioner does not allow re-configure
1912                return Err(PopupConfigureError::NotReactive);
1913            }
1914
1915            Ok(())
1916        })?;
1917
1918        let serial = self.send_configure_internal(None);
1919
1920        Ok(serial)
1921    }

1947    /// Handles the role specific commit error checking
1948    ///
1949    /// This should be called when the underlying WlSurface
1950    /// handles a wl_surface.commit request.
1951    pub(crate) fn pre_commit_hook<D: XdgShellHandler + 'static>(
1952        state: &mut D,
1953        _dh: &DisplayHandle,
1954        surface: &wl_surface::WlSurface,
1955    ) {
1956        let _span = trace_span!("xdg-popup pre-commit", surface = %surface.id()).entered();
1957
1958        let res = compositor::with_states(surface, |states| {
1959            let mut role = states
1960                .data_map
1961                .get::<XdgPopupSurfaceData>()
1962                .unwrap()
1963                .lock()
1964                .unwrap();
1965            if role.parent.is_none() {
1966                return Err((
1967                    xdg_surface::Error::NotConstructed,
1968                    "xdg_popup must have parent before mapping",
1969                ));
1970            }
1971
1972            let mut guard_popup = states.cached_state.get::<PopupCachedState>();
1973            let pending = guard_popup.pending();
1974
1975            // The presence of last_acked always follows the buffer assignment because the
1976            // surface is not allowed to attach a buffer without acking the initial configure.
1977            let had_buffer_before = pending.last_acked.is_some();
1978
1979            let grab = role.requested_grab.take();
1980            if grab.is_some() && had_buffer_before {
1981                let popups = &state.xdg_shell_state().known_popups;
1982                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
1983                    popup
1984                        .shell_surface
1985                        .post_error(xdg_popup::Error::InvalidGrab, "tried to grab after being mapped");
1986                } else {
1987                    error!("surface missing from known popups");
1988                }
1989                return Ok(None);
1990            }
1991
1992            let mut guard_surface = states.cached_state.get::<SurfaceAttributes>();
1993            let has_buffer = match &guard_surface.pending().buffer {
1994                Some(BufferAssignment::NewBuffer(_)) => true,
1995                Some(BufferAssignment::Removed) => false,
1996                None => had_buffer_before,
1997            };
1998            // Need to check had_buffer_before in case the client attaches a null buffer for the
1999            // initial commit---we don't want to consider that as "got unmapped" and reset role.
2000            let got_unmapped = had_buffer_before && !has_buffer;
2001
2002            if has_buffer {
2003                let Some(last_acked) = role.last_acked else {
2004                    return Err((
2005                        xdg_surface::Error::UnconfiguredBuffer,
2006                        "must ack the initial configure before attaching buffer",
2007                    ));
2008                };
2009
2010                // The surface remains, or became mapped, track the last acked state.
2011                pending.last_acked = Some(last_acked);
2012            } else {
2013                // The surface remains, or became, unmapped, meaning that it's in the initial
2014                // configure stage.
2015                pending.last_acked = None;
2016            }
2017
2018            if got_unmapped {
2019                trace!(
2020                    "got unmapped; resetting role and cached state; have {} pending configures",
2021                    role.pending_configures.len()
2022                );
2023
2024                // All attributes are discarded when an xdg_surface is unmapped. Though, we keep
2025                // the list of pending configures because there's no way for a surface to tell
2026                // an in-flight configure apart from our next initial configure after unmapping.
2027                let pending_configures = std::mem::take(&mut role.pending_configures);
2028                *role = Default::default();
2029                role.pending_configures = pending_configures;
2030                *guard_popup.pending() = Default::default();
2031            }
2032
2033            Ok(grab)
2034        });
2035
2036        match res {
2037            Ok(Some((seat, serial))) => {
2038                let popups = &state.xdg_shell_state().known_popups;
2039                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
2040                    let popup = popup.clone();
2041                    XdgShellHandler::grab(state, popup, seat, serial);
2042                } else {
2043                    error!("surface missing from known popups");
2044                }
2045            }
2046            Ok(None) => (),
2047            Err((error, msg)) => {
2048                let popups = &state.xdg_shell_state().known_popups;
2049                if let Some(popup) = popups.iter().find(|handle| handle.wl_surface == *surface) {
2050                    let data = popup
2051                        .shell_surface
2052                        .data::<self::handlers::XdgShellSurfaceUserData>()
2053                        .unwrap();
2054                    data.xdg_surface.post_error(error, msg);
2055                } else {
2056                    error!("surface missing from known popups");
2057                }
2058            }
2059        }
2060    }
