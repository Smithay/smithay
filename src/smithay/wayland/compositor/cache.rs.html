1// The caching logic is used to process surface synchronization. It creates
2// an effective decoupling between the moment the client sends wl_surface.commit
3// and the moment where the state that was commited is actually applied by the
4// compositor.
5//
6// The way this is modelled in Smithay is through the `Cache` type, which is a container
7// representing a cached state for a particular type. The full cached state of a surface
8// is thus composed of a a set of `Cache<T>` for all relevant `T`, as modelled by the
9// `MultiCache`.
10//
11// The logic of the `Cache` is as follows:
12//
13// - The protocol handlers mutably access the `pending` state to modify it accord to
14//   the client requests
15// - On commit, a snapshot of this pending state is created by invoking `Cacheable::commit`
16//   and stored in the cache alongside an externally provided id
17// - When the compositor decices that a given state (represented by its commit id) should
18//   become active, `Cache::apply_state` is invoked with that commit id. The associated state
19//   is then applied to the `current` state, that the compositor can then use as a reference
20//   for the current window state. Note that, to preserve the commit ordering, all states
21//   with a commit id older than the one requested are applied as well, in order.
22//
23// The logic for generating these commit ids and deciding when to apply them is implemented
24// and described in `transaction.rs`.
25

36/// Trait representing a value that can be used in double-buffered storage
37///
38/// The type needs to implement the [`Default`] trait, which will be used
39/// to initialize. You further need to provide two methods:
40/// [`Cacheable::commit`] and [`Cacheable::merge_into`].
41///
42/// Double-buffered state works by having a "pending" instance of your type,
43/// into which new values provided by the client are inserted. When the client
44/// sends `wl_surface.commit`, the [`Cacheable::commit`] method will be
45/// invoked on your value. This method is expected to produce a new instance of
46/// your type, that will be stored in the cache, and eventually merged into the
47/// current state.
48///
49/// In most cases, this method will simply produce a copy of the pending state,
50/// but you might need additional logic in some cases, such as for handling
51/// non-cloneable resources (which thus need to be moved into the produce value).
52///
53/// Then at some point the [`Cacheable::merge_into`] method of your type will be
54/// invoked. In this method, `self` acts as the update that should be merged into
55/// the current state provided as argument. In simple cases, the action would just
56/// be to copy `self` into the current state, but more complex cases require
57/// additional logic.
58pub trait Cacheable: Default {
59    /// Produce a new state to be cached from the pending state
60    fn commit(&mut self, dh: &DisplayHandle) -> Self;
61    /// Merge a state update into the current state
62    fn merge_into(self, into: &mut Self, dh: &DisplayHandle);
63}
64
65/// Double buffered cached state of type `T`
66#[derive(Debug)]
67pub struct CachedState<T> {
68    pending: T,
69    cache: VecDeque<(Serial, T)>,
70    current: T,
71}
72
73impl<T: Default> Default for CachedState<T> {
74    fn default() -> Self {
75        CachedState {
76            pending: T::default(),
77            cache: VecDeque::new(),
78            current: T::default(),
79        }
80    }
81}
82
83impl<T> CachedState<T> {
84    /// Access the current state for `T`
85    pub fn current(&mut self) -> &mut T {
86        &mut self.current
87    }
88
89    /// Access the pending state for `T`
90    pub fn pending(&mut self) -> &mut T {
91        &mut self.pending
92    }
93}

95trait Cache: Downcast {
96    fn commit(&self, commit_id: Option<Serial>, dh: &DisplayHandle);
97    fn apply_state(&self, commit_id: Serial, dh: &DisplayHandle);
98}
99
100impl_downcast!(Cache);
101
102impl<T: Cacheable + 'static> Cache for Mutex<CachedState<T>> {
103    fn commit(&self, commit_id: Option<Serial>, dh: &DisplayHandle) {
104        let mut guard = self.lock().unwrap();
105        let me = &mut *guard;
106        let new_state = me.pending.commit(dh);
107        if let Some(id) = commit_id {
108            match me.cache.back_mut() {
109                Some(&mut (cid, ref mut state)) if cid == id => new_state.merge_into(state, dh),
110                _ => me.cache.push_back((id, new_state)),
111            }
112        } else {
113            for (_, state) in me.cache.drain(..) {
114                state.merge_into(&mut me.current, dh);
115            }
116            new_state.merge_into(&mut me.current, dh);
117        }
118    }
119
120    fn apply_state(&self, commit_id: Serial, dh: &DisplayHandle) {
121        let mut me = self.lock().unwrap();
122        loop {
123            if me.cache.front().map(|&(s, _)| s > commit_id).unwrap_or(true) {
124                // if the cache is empty or the next state has a commit_id greater than the requested one
125                break;
126            }
127            me.cache.pop_front().unwrap().1.merge_into(&mut me.current, dh);
128        }
129    }
130}

132/// A typemap-like container for double-buffered values
133///
134/// All values inserted into this container must implement the [`Cacheable`] trait,
135/// which defines their buffering semantics. They furthermore must be [`Send`] as the surface state
136/// can be accessed from multiple threads (but [`Sync`] is not required, the surface internally synchronizes
137/// access to its state).
138///
139/// [`MultiCache::get`] provides access to the [`CachedState`] associated with a particular type.
140///
141/// Consumers of surface state (like compositor applications using Smithay) will mostly be concerned
142/// with the [`CachedState::current`] method, which gives access to the current state of the surface for
143/// a particular type.
144///
145/// Writers of protocol extensions logic will mostly be concerned with the [`CachedState::pending`] method,
146/// which provides access to the pending state of the surface, in which new state from clients will be
147/// stored.
148///
149/// This contained has [`Mutex`]-like semantics: values of multiple stored types can be accessed at the
150/// same time, but accessing the same value multiple times will cause a deadlock.
151/// The stored values are initialized lazily the first time [`get`][Self::get] is invoked with this type as argument.
152pub struct MultiCache {
153    caches: appendlist::AppendList<Box<dyn Cache + Send>>,
154}
155
156impl std::fmt::Debug for MultiCache {
157    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
158        f.debug_struct("MultiCache").finish_non_exhaustive()
159    }
160}
161
162impl MultiCache {
163    pub(crate) fn new() -> Self {
164        Self {
165            caches: appendlist::AppendList::new(),
166        }
167    }
168
169    fn find_or_insert<T: Cacheable + Send + 'static>(&self) -> &Mutex<CachedState<T>> {
170        for cache in &self.caches {
171            if let Some(v) = (**cache).as_any().downcast_ref() {
172                return v;
173            }
174        }
175        // if we reach here, then the value is not yet in the list, insert it
176        self.caches
177            .push(Box::new(Mutex::new(CachedState::<T>::default())) as Box<_>);
178        (*self.caches[self.caches.len() - 1])
179            .as_any()
180            .downcast_ref()
181            .unwrap()
182    }
183
184    /// Access the [`CachedState`] associated with type `T`
185    pub fn get<T: Cacheable + Send + 'static>(&self) -> MutexGuard<'_, CachedState<T>> {
186        self.find_or_insert::<T>().lock().unwrap()
187    }
188
189    /// Check if the container currently contains values for type `T`
190    pub fn has<T: Cacheable + Send + 'static>(&self) -> bool {
191        self.caches
192            .iter()
193            .any(|c| (**c).as_any().is::<Mutex<CachedState<T>>>())
194    }
195
196    /// Commits the pending state, invoking Cacheable::commit()
197    ///
198    /// If commit_id is None, then the pending state is directly merged
199    /// into the current state. Otherwise, this id is used to store the
200    /// cached state. An id ca no longer be re-used as soon as not new id
201    /// has been used in between. Provided IDs are expected to be provided
202    /// in increasing order according to `Serial` semantics.
203    ///
204    /// If a None commit is given but there are some cached states, they'll
205    /// all be merged into the current state before merging the pending one.
206    pub(crate) fn commit(&mut self, commit_id: Option<Serial>, dh: &DisplayHandle) {
207        // none of the underlying borrow_mut() can panic, as we hold
208        // a &mut reference to the container, non are borrowed.
209        for cache in &self.caches {
210            cache.commit(commit_id, dh);
211        }
212    }
213
214    /// Apply given identified cached state to the current one
215    ///
216    /// All other preceding states are applied as well, to preserve commit ordering
217    pub(crate) fn apply_state(&self, commit_id: Serial, dh: &DisplayHandle) {
218        // none of the underlying borrow_mut() can panic, as we hold
219        // a &mut reference to the container, non are borrowed.
220        for cache in &self.caches {
221            cache.apply_state(commit_id, dh);
222        }
223    }
224}
