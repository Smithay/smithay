1// The transaction model for handling surface states in Smithay
2//
3// The caching logic in `cache.rs` provides surfaces with a queue of
4// pending states identified with numeric commit ids, allowing the compositor
5// to precisely control *when* a state become active. This file is the second
6// half: these identified states are grouped into transactions, which allow the
7// synchronization of updates across surfaces.
8//
9// There are 2 main cases when the state of multiple surfaces must be updated
10// atomically:
11// - synchronized subsurface must have their state updated at the same time as their parents
12// - The upcoming `wp_transaction` protocol
13//
14// In these situations, the individual states in a surface queue are grouped into a transaction
15// and are all applied atomically when the transaction itself is applied. The logic for creating
16// new transactions is currently the following:
17//
18// - Each surface has an implicit "pending" transaction, into which its newly committed state is
19//   recorded
20// - Furthermore, on commit, the pending transaction of all synchronized child subsurfaces is merged
21//   into the current surface's pending transaction, and a new implicit transaction is started for those
22//   children (logic is implemented in `handlers.rs`, in `PrivateSurfaceData::commit`).
23// - Then, still on commit, if the surface is not a synchronized subsurface, its pending transaction is
24//   directly applied
25//
26// This last step will change once we have support for explicit synchronization (and further in the future,
27// of the wp_transaction protocol). Explicit synchronization introduces a notion of blockers: the transaction
28// cannot be applied before all blockers are released, and thus must wait for it to be the case.
29//
30// For those situations, the (currently unused) `TransactionQueue` will come into play. It is a per-client
31// queue of transactions, that stores and applies them by both respecting their topological order
32// (ensuring that for each surface, states are applied in the correct order) and that all transactions
33// wait before all their blockers are resolved to be merged. If a blocker is cancelled, the whole transaction
34// it blocks is cancelled as well, and simply dropped. Thanks to the logic of `Cache::apply_state`, the
35// associated state will be applied automatically when the next valid transaction is applied, ensuring
36// global coherence.
37
38// A significant part of the logic of this module is not yet used,
39// but will be once proper transaction & blockers support is
40// added to smithay

53/// Types potentially blocking state changes
54pub trait Blocker {
55    /// Retrieve the current state of the blocker
56    fn state(&self) -> BlockerState;
57}
58
59/// States of a [`Blocker`]
60#[derive(Debug, Clone, Copy, PartialEq, Eq, Hash)]
61pub enum BlockerState {
62    /// The block is pending and not resolved yet
63    Pending,
64    /// The block got released and changes can be applied
65    Released,
66    /// The block got cancelled and changes should be discarded
67    Cancelled,
68}

70/// A simple [`Blocker`] barrier
71#[derive(Debug, Clone)]
72pub struct Barrier(Arc<AtomicBool>);
73
74impl PartialEq for Barrier {
75    #[inline]
76    fn eq(&self, other: &Self) -> bool {
77        Arc::ptr_eq(&self.0, &other.0)
78    }
79}
80impl Eq for Barrier {}
81
82impl Barrier {
83    /// Initialize a new [`Barrier`] with the provided state
84    pub fn new(signaled: bool) -> Self {
85        Self(Arc::new(AtomicBool::new(signaled)))
86    }
87
88    /// Query if this barrier has been signaled
89    #[inline]
90    pub fn is_signaled(&self) -> bool {
91        self.0.load(std::sync::atomic::Ordering::Acquire)
92    }
93
94    /// Signal this barrier
95    #[inline]
96    pub fn signal(&self) {
97        self.0.store(true, std::sync::atomic::Ordering::Release)
98    }
99}
100
101impl Blocker for Barrier {
102    fn state(&self) -> BlockerState {
103        if self.is_signaled() {
104            BlockerState::Released
105        } else {
106            BlockerState::Pending
107        }
108    }
109}

111#[derive(Default)]
112struct TransactionState {
113    surfaces: Vec<(Weak<WlSurface>, Serial)>,
114    blockers: Vec<Box<dyn Blocker + Send>>,
115}
116
117impl TransactionState {
118    fn insert(&mut self, surface: WlSurface, id: Serial) {
119        if let Some(place) = self.surfaces.iter_mut().find(|place| place.0 == surface) {
120            // the surface is already in the list, update the serial
121            if place.1 < id {
122                place.1 = id;
123            }
124        } else {
125            // the surface is not in the list, insert it
126            self.surfaces.push((surface.downgrade(), id));
127        }
128    }
129}

131enum TransactionInner {
132    Data(TransactionState),
133    Fused(Arc<Mutex<TransactionInner>>),
134}
135
136pub(crate) struct PendingTransaction {
137    inner: Arc<Mutex<TransactionInner>>,
138}
139
140impl Default for PendingTransaction {
141    fn default() -> Self {
142        PendingTransaction {
143            inner: Arc::new(Mutex::new(TransactionInner::Data(Default::default()))),
144        }
145    }
146}
147
148impl PendingTransaction {
149    fn with_inner_state<T, F: FnOnce(&mut TransactionState) -> T>(&self, f: F) -> T {
150        let mut next = self.inner.clone();
151        loop {
152            let tmp = match *next.lock().unwrap() {
153                TransactionInner::Data(ref mut state) => return f(state),
154                TransactionInner::Fused(ref into) => into.clone(),
155            };
156            next = tmp;
157        }
158    }
159
160    pub(crate) fn insert_state(&self, surface: WlSurface, id: Serial) {
161        self.with_inner_state(|state| state.insert(surface, id))
162    }
163
164    pub(crate) fn add_blocker<B: Blocker + Send + 'static>(&self, blocker: B) {
165        self.with_inner_state(|state| state.blockers.push(Box::new(blocker) as Box<_>))
166    }
167
168    pub(crate) fn is_same_as(&self, other: &PendingTransaction) -> bool {
169        let ptr1 = self.with_inner_state(|state| state as *const _);
170        let ptr2 = other.with_inner_state(|state| state as *const _);
171        ptr1 == ptr2
172    }
173
174    pub(crate) fn merge_into(&self, into: &PendingTransaction) {
175        if self.is_same_as(into) {
176            // nothing to do
177            return;
178        }
179        // extract our pending surfaces and change our link
180        let mut next = self.inner.clone();
181        let my_state;
182        loop {
183            let tmp = {
184                let mut guard = next.lock().unwrap();
185                match *guard {
186                    TransactionInner::Data(ref mut state) => {
187                        my_state = std::mem::take(state);
188                        *guard = TransactionInner::Fused(into.inner.clone());
189                        break;
190                    }
191                    TransactionInner::Fused(ref into) => into.clone(),
192                }
193            };
194            next = tmp;
195        }
196        // fuse our surfaces into our new transaction state
197        self.with_inner_state(|state| {
198            for (surface, id) in my_state.surfaces {
199                if let Ok(surface) = surface.upgrade() {
200                    state.insert(surface, id);
201                }
202            }
203            state.blockers.extend(my_state.blockers);
204        });
205    }
206
207    pub(crate) fn finalize(mut self) -> Transaction {
208        // When finalizing a transaction, this *must* be the last handle to this transaction
209        loop {
210            let inner = match Arc::try_unwrap(self.inner) {
211                Ok(mutex) => mutex.into_inner().unwrap(),
212                Err(_) => panic!("Attempting to finalize a transaction but handle is not the last."),
213            };
214            match inner {
215                TransactionInner::Data(TransactionState {
216                    surfaces, blockers, ..
217                }) => return Transaction { surfaces, blockers },
218                TransactionInner::Fused(into) => self.inner = into,
219            }
220        }
221    }
222}

224#[derive(Debug)]
225pub(crate) struct Transaction {
226    surfaces: Vec<(Weak<WlSurface>, Serial)>,
227    blockers: Vec<Box<dyn Blocker + Send>>,
228}

230impl fmt::Debug for Box<dyn Blocker + Send> {
231    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
232        f.debug_struct("Blocker").field("state", &self.state()).finish()
233    }
234}
235
236impl Transaction {
237    /// Computes the global state of the transaction with regard to its blockers
238    ///
239    /// The logic is:
240    ///
241    /// - if at least one blocker is cancelled, the transaction is cancelled
242    /// - otherwise, if at least one blocker is pending, the transaction is pending
243    /// - otherwise, all blockers are released, and the transaction is also released
244    pub(crate) fn state(&self) -> BlockerState {
245        // In case all of our surfaces have been destroyed we can cancel this transaction
246        // as we won't apply its state anyway
247        if !self.surfaces.iter().any(|surface| surface.0.is_alive()) {
248            return BlockerState::Cancelled;
249        }
250
251        use BlockerState::*;
252        self.blockers
253            .iter()
254            .fold(Released, |acc, blocker| match (acc, blocker.state()) {
255                (Cancelled, _) | (_, Cancelled) => Cancelled,
256                (Pending, _) | (_, Pending) => Pending,
257                (Released, Released) => Released,
258            })
259    }
260
261    pub(crate) fn apply<C: CompositorHandler + 'static>(self, dh: &DisplayHandle, state: &mut C) {
262        for (surface, id) in self.surfaces {
263            let Ok(surface) = surface.upgrade() else {
264                continue;
265            };
266
267            PrivateSurfaceData::with_states(&surface, |states| {
268                states.cached_state.apply_state(id, dh);
269            });
270
271            PrivateSurfaceData::invoke_post_commit_hooks::<C>(state, dh, &surface);
272
273            tracing::trace!("Calling user implementation for wl_surface.commit");
274
275            state.commit(&surface);
276        }
277    }
278}

280// This queue should be per-client
281#[derive(Debug, Default)]
282pub(crate) struct TransactionQueue {
283    transactions: Vec<Transaction>,
284    // we keep the hashset around to reuse allocations
285    seen_surfaces: HashSet<u32>,
286}
287
288impl TransactionQueue {
289    pub(crate) fn append(&mut self, t: Transaction) {
290        self.transactions.push(t);
291    }
292
293    pub(crate) fn take_ready(&mut self) -> Vec<Transaction> {
294        // FIXME: Get rid of this allocation here
295        let mut ready_transactions = Vec::new();
296        // this is a very non-optimized implementation
297        // we just iterate over the queue of transactions, keeping track of which
298        // surface we have seen as they encode transaction dependencies
299        self.seen_surfaces.clear();
300        // manually iterate as we're going to modify the Vec while iterating on it
301        let mut i = 0;
302        // the loop will terminate, as at every iteration either i is incremented by 1
303        // or the length of self.transactions is reduced by 1.
304        while i < self.transactions.len() {
305            let mut skip = false;
306            // does the transaction have any active blocker?
307            match self.transactions[i].state() {
308                BlockerState::Cancelled => {
309                    // this transaction is cancelled, remove it without further processing
310                    self.transactions.remove(i);
311                    continue;
312                }
313                BlockerState::Pending => {
314                    skip = true;
315                }
316                BlockerState::Released => {}
317            }
318            // if not, does this transaction depend on any previous transaction?
319            if !skip {
320                for (s, _) in &self.transactions[i].surfaces {
321                    // TODO: is this alive check still needed?
322                    if !s.is_alive() {
323                        continue;
324                    }
325                    if self.seen_surfaces.contains(&s.id().protocol_id()) {
326                        skip = true;
327                        break;
328                    }
329                }
330            }
331
332            if skip {
333                // this transaction is not yet ready and should be skipped, add its surfaces to our
334                // seen list
335                for (s, _) in &self.transactions[i].surfaces {
336                    // TODO: is this alive check still needed?
337                    if !s.is_alive() {
338                        continue;
339                    }
340                    self.seen_surfaces.insert(s.id().protocol_id());
341                }
342                i += 1;
343            } else {
344                // this transaction is to be applied, yay!
345                ready_transactions.push(self.transactions.remove(i));
346            }
347        }
348
349        ready_transactions
350    }
