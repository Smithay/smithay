1//! DRM syncobj protocol
2//!
3//! This module implement the `linux-drm-syncobj-v1` protocol, used to support
4//! explicit sync.
5//!
6//! Currently, the implementation here assumes acquire fences are already signalled
7//! when the surface transaction is ready. Use [`DrmSyncPointBlocker`].
8//!
9//! The server should only expose the protocol if [`supports_syncobj_eventfd`] returns
10//! `true`. Or it won't be possible to create the blocker. This is similar to other
11//! implementations.
12//!
13//! The release fence is signalled when all references to a
14//! [`Buffer`][crate::backend::renderer::utils::Buffer] are dropped.
15//!
16//! ```no_run
17//! # use smithay::delegate_drm_syncobj;
18//! # use smithay::wayland::drm_syncobj::*;
19//!
20//! pub struct State {
21//!     syncobj_state: Option<DrmSyncobjState>,
22//! }
23//!
24//! impl DrmSyncobjHandler for State {
25//!     fn drm_syncobj_state(&mut self) -> Option<&mut DrmSyncobjState> {
26//!         self.syncobj_state.as_mut()
27//!     }
28//! }
29//!
30//! # let mut display = wayland_server::Display::<State>::new().unwrap();
31//! # let display_handle = display.handle();
32//! # let import_device = todo!();
33//! let syncobj_state = if supports_syncobj_eventfd(&import_device) {
34//!     Some(DrmSyncobjState::new::<State>(&display_handle, import_device))
35//! } else {
36//!     None
37//! };
38//!
39//! delegate_drm_syncobj!(State);
40//! ```

67/// Test if DRM device supports `syncobj_eventfd`.
68// Similar to test used in Mutter
69pub fn supports_syncobj_eventfd(device: &DrmDeviceFd) -> bool {
70    // Pass device as palceholder for eventfd as well, since `drm_ffi` requires
71    // a valid fd.
72    match drm_ffi::syncobj::eventfd(device.as_fd(), 0, 0, device.as_fd(), false) {
73        Ok(_) => unreachable!(),
74        Err(err) => err.kind() == std::io::ErrorKind::NotFound,
75    }
76}

90/// Pending DRM syncobj sync point state
91#[derive(Debug, Default)]
92pub struct DrmSyncobjCachedState {
93    /// Timeline point signaled when buffer is ready to read
94    pub acquire_point: Option<DrmSyncPoint>,
95    /// Timeline point to be signaled when server is done with buffer
96    pub release_point: Option<DrmSyncPoint>,
97}
98
99impl Cacheable for DrmSyncobjCachedState {
100    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
101        Self {
102            acquire_point: self.acquire_point.take(),
103            release_point: self.release_point.take(),
104        }
105    }
106
107    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
108        if self.acquire_point.is_some() && self.release_point.is_some() {
109            if let Some(release_point) = &into.release_point {
110                if let Err(err) = release_point.signal() {
111                    tracing::error!("Failed to signal syncobj release point: {}", err);
112                }
113            }
114            into.acquire_point = self.acquire_point;
115            into.release_point = self.release_point;
116        }
117    }
118}

120/// Delegate type for a `wp_linux_drm_syncobj_manager_v1` global
121#[derive(Debug)]
122pub struct DrmSyncobjState {
123    global: GlobalId,
124    import_device: DrmDeviceFd,
125    known_timelines: Vec<Weak<DrmTimelineInner>>,
126}
127
128impl DrmSyncobjState {
129    /// Create a new `wp_linux_drm_syncobj_manager_v1` global
130    ///
131    /// The `import_device` will be used to import the syncobj fds, and wait on them.
132    pub fn new<D>(display: &DisplayHandle, import_device: DrmDeviceFd) -> Self
133    where
134        D: GlobalDispatch<WpLinuxDrmSyncobjManagerV1, DrmSyncobjGlobalData>,
135        D: 'static,
136    {
137        Self::new_with_filter::<D, _>(display, import_device, |_| true)
138    }
139
140    /// Create a new `wp_linuxdrm_syncobj_manager_v1` global with a client filter
141    ///
142    /// The `import_device` will be used to import the syncobj fds, and wait on them.
143    pub fn new_with_filter<D, F>(display: &DisplayHandle, import_device: DrmDeviceFd, filter: F) -> Self
144    where
145        D: GlobalDispatch<WpLinuxDrmSyncobjManagerV1, DrmSyncobjGlobalData>,
146        D: 'static,
147        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
148    {
149        let global = display.create_global::<D, WpLinuxDrmSyncobjManagerV1, DrmSyncobjGlobalData>(
150            1,
151            DrmSyncobjGlobalData {
152                filter: Box::new(filter),
153            },
154        );
155
156        Self {
157            global,
158            import_device,
159            known_timelines: Vec::new(),
160        }
161    }
162
163    /// Sets a new `import_device` to import the syncobj fds and wait on them.
164    ///
165    /// Note: This will not update already existing timeline objects,
166    /// which will continue to use the previous device.
167    pub fn update_device(&mut self, import_device: DrmDeviceFd) {
168        for timeline in self.known_timelines.iter().filter_map(Weak::upgrade) {
169            if let Err(err) = timeline.update_device(&import_device) {
170                warn!(?err, "Failed to update existing timeline");
171            }
172        }
173        self.import_device = import_device;
174    }
175
176    /// Destroys the state and returns the `GlobalId` for compositors to disable/destroy.
177    ///
178    /// Note: This will cause any future timeline import to raise a protocol error for
179    /// clients that have bound this protocol until [`DrmSyncobjHandler::drm_syncobj_state`] returns `Some` again.
180    pub fn into_global(self) -> GlobalId {
181        for timeline in self.known_timelines.iter().filter_map(Weak::upgrade) {
182            timeline.invalidate();
183        }
184        self.global
185    }
186}

208fn commit_hook<D: DrmSyncobjHandler>(_data: &mut D, _dh: &DisplayHandle, surface: &WlSurface) {
209    compositor::with_states(surface, |states| {
210        let mut cached = states.cached_state.get::<SurfaceAttributes>();
211        let pending = cached.pending();
212        let new_buffer = pending.buffer.as_ref().and_then(|buffer| match buffer {
213            BufferAssignment::NewBuffer(buffer) => Some(buffer),
214            _ => None,
215        });
216        if let Some(data) = states
217            .data_map
218            .get::<RefCell<Option<WpLinuxDrmSyncobjSurfaceV1>>>()
219        {
220            if let Some(syncobj_surface) = data.borrow().as_ref() {
221                let mut cached = states.cached_state.get::<DrmSyncobjCachedState>();
222                let pending = cached.pending();
223                if pending.acquire_point.is_some() && new_buffer.is_none() {
224                    syncobj_surface.post_error(
225                        wp_linux_drm_syncobj_surface_v1::Error::NoBuffer as u32,
226                        "acquire point without buffer".to_string(),
227                    );
228                } else if pending.acquire_point.is_some() && pending.release_point.is_none() {
229                    syncobj_surface.post_error(
230                        wp_linux_drm_syncobj_surface_v1::Error::NoReleasePoint as u32,
231                        "acquire point without release point".to_string(),
232                    );
233                } else if pending.acquire_point.is_none() && pending.release_point.is_some() {
234                    syncobj_surface.post_error(
235                        wp_linux_drm_syncobj_surface_v1::Error::NoAcquirePoint as u32,
236                        "release point without acquire point".to_string(),
237                    );
238                } else if let (Some(acquire), Some(release)) =
239                    (pending.acquire_point.as_ref(), pending.release_point.as_ref())
240                {
241                    if acquire.timeline == release.timeline && release.point <= acquire.point {
242                        syncobj_surface.post_error(
243                            wp_linux_drm_syncobj_surface_v1::Error::ConflictingPoints as u32,
244                            format!(
245                                "release point {} is not greater than acquire point {}",
246                                release.point, acquire.point
247                            ),
248                        );
249                    }
250                    if let Some(buffer) = new_buffer {
251                        if get_dmabuf(buffer).is_err() {
252                            syncobj_surface.post_error(
253                                wp_linux_drm_syncobj_surface_v1::Error::UnsupportedBuffer as u32,
254                                "sync points with non-dmabuf buffer".to_string(),
255                            );
256                        }
257                    }
258                }
259            }
260        }
261    });
262}
263
264fn destruction_hook<D: DrmSyncobjHandler>(_data: &mut D, surface: &WlSurface) {
265    compositor::with_states(surface, |states| {
266        let mut cached = states.cached_state.get::<DrmSyncobjCachedState>();
267        if let Some(release_point) = &cached.pending().release_point {
268            if let Err(err) = release_point.signal() {
269                tracing::error!("Failed to signal syncobj release point: {}", err);
270            }
271        }
272        if let Some(release_point) = &cached.current().release_point {
273            if let Err(err) = release_point.signal() {
274                tracing::error!("Failed to signal syncobj release point: {}", err);
275            }
276        }
277    });
278}

280impl<D> Dispatch<WpLinuxDrmSyncobjManagerV1, (), D> for DrmSyncobjState
281where
282    D: Dispatch<WpLinuxDrmSyncobjSurfaceV1, DrmSyncobjSurfaceData>,
283    D: Dispatch<WpLinuxDrmSyncobjTimelineV1, DrmSyncobjTimelineData>,
284    D: DrmSyncobjHandler,
285{
286    fn request(
287        state: &mut D,
288        _client: &Client,
289        resource: &WpLinuxDrmSyncobjManagerV1,
290        request: wp_linux_drm_syncobj_manager_v1::Request,
291        _data: &(),
292        _dh: &DisplayHandle,
293        data_init: &mut DataInit<'_, D>,
294    ) {
295        match request {
296            wp_linux_drm_syncobj_manager_v1::Request::GetSurface { id, surface } => {
297                let already_exists = with_states(&surface, |states| {
298                    states
299                        .data_map
300                        .get::<RefCell<Option<WpLinuxDrmSyncobjSurfaceV1>>>()
301                        .map(|v| v.borrow().is_some())
302                        .unwrap_or(false)
303                });
304                if already_exists {
305                    resource.post_error(
306                        wp_linux_drm_syncobj_manager_v1::Error::SurfaceExists as u32,
307                        "the surface already has a syncobj_surface object associated".to_string(),
308                    );
309                    return;
310                }
311                let commit_hook_id = compositor::add_pre_commit_hook::<D, _>(&surface, commit_hook);
312                let destruction_hook_id =
313                    compositor::add_destruction_hook::<D, _>(&surface, destruction_hook);
314                let syncobj_surface = data_init.init::<_, _>(
315                    id,
316                    DrmSyncobjSurfaceData {
317                        surface: surface.downgrade(),
318                        commit_hook_id,
319                        destruction_hook_id,
320                    },
321                );
322                with_states(&surface, |states| {
323                    states
324                        .data_map
325                        .insert_if_missing(|| RefCell::new(Some(syncobj_surface)))
326                });
327            }
328            wp_linux_drm_syncobj_manager_v1::Request::ImportTimeline { id, fd } => {
329                if let Some(state) = state.drm_syncobj_state() {
330                    match DrmTimeline::new(&state.import_device, fd) {
331                        Ok(timeline) => {
332                            state.known_timelines.push(Arc::downgrade(&timeline.0));
333                            data_init.init::<_, _>(id, DrmSyncobjTimelineData { timeline });
334                        }
335                        Err(err) => {
336                            resource.post_error(
337                                wp_linux_drm_syncobj_manager_v1::Error::InvalidTimeline as u32,
338                                format!("failed to import syncobj timeline: {err}"),
339                            );
340                        }
341                    }
342                } else {
343                    resource.post_error(
344                        wp_linux_drm_syncobj_manager_v1::Error::InvalidTimeline as u32,
345                        "global orphaned",
346                    )
347                }
348            }
349            wp_linux_drm_syncobj_manager_v1::Request::Destroy => {}
350            _ => unreachable!(),
351        }
352    }
353}

355/// Data attached to wp_linux_drm_syncobj_surface_v1 objects
356#[derive(Debug)]
357pub struct DrmSyncobjSurfaceData {
358    surface: WlWeak<WlSurface>,
359    commit_hook_id: HookId,
360    destruction_hook_id: HookId,
361}
362
363impl<D> Dispatch<WpLinuxDrmSyncobjSurfaceV1, DrmSyncobjSurfaceData, D> for DrmSyncobjState
364where
365    D: DrmSyncobjHandler,
366{
367    fn request(
368        _state: &mut D,
369        _client: &Client,
370        resource: &WpLinuxDrmSyncobjSurfaceV1,
371        request: wp_linux_drm_syncobj_surface_v1::Request,
372        data: &DrmSyncobjSurfaceData,
373        _dh: &DisplayHandle,
374        _data_init: &mut DataInit<'_, D>,
375    ) {
376        match request {
377            wp_linux_drm_syncobj_surface_v1::Request::Destroy => {
378                if let Ok(surface) = data.surface.upgrade() {
379                    compositor::remove_pre_commit_hook(&surface, data.commit_hook_id.clone());
380                    compositor::remove_destruction_hook(&surface, data.destruction_hook_id.clone());
381                    with_states(&surface, |states| {
382                        *states
383                            .data_map
384                            .get::<RefCell<Option<WpLinuxDrmSyncobjSurfaceV1>>>()
385                            .unwrap()
386                            .borrow_mut() = None;
387                        // Committed sync points should still be used, but pending points can
388                        // be cleared.
389                        let mut cached = states.cached_state.get::<DrmSyncobjCachedState>();
390                        cached.pending().acquire_point = None;
391                        if let Some(release_point) = cached.pending().release_point.take() {
392                            if let Err(err) = release_point.signal() {
393                                tracing::error!("Failed to signal syncobj release point: {}", err);
394                            }
395                        }
396                    });
397                }
398            }
399            wp_linux_drm_syncobj_surface_v1::Request::SetAcquirePoint {
400                timeline,
401                point_hi,
402                point_lo,
403            } => {
404                let Ok(surface) = data.surface.upgrade() else {
405                    resource.post_error(
406                        wp_linux_drm_syncobj_surface_v1::Error::NoSurface,
407                        "Set acquire point for destroyed surface.",
408                    );
409                    return;
410                };
411
412                let sync_point = DrmSyncPoint {
413                    timeline: timeline
414                        .data::<DrmSyncobjTimelineData>()
415                        .unwrap()
416                        .timeline
417                        .clone(),
418                    point: ((point_hi as u64) << 32) + (point_lo as u64),
419                };
420                with_states(&surface, |states| {
421                    let mut cached = states.cached_state.get::<DrmSyncobjCachedState>();
422                    let cached_state = cached.pending();
423                    cached_state.acquire_point = Some(sync_point);
424                });
425            }
426            wp_linux_drm_syncobj_surface_v1::Request::SetReleasePoint {
427                timeline,
428                point_hi,
429                point_lo,
430            } => {
431                let Ok(surface) = data.surface.upgrade() else {
432                    resource.post_error(
433                        wp_linux_drm_syncobj_surface_v1::Error::NoSurface,
434                        "Set release point for destroyed surface.",
435                    );
436                    return;
437                };
438
439                let sync_point = DrmSyncPoint {
440                    timeline: timeline
441                        .data::<DrmSyncobjTimelineData>()
442                        .unwrap()
443                        .timeline
444                        .clone(),
445                    point: ((point_hi as u64) << 32) + (point_lo as u64),
446                };
447                with_states(&surface, |states| {
448                    let mut cached = states.cached_state.get::<DrmSyncobjCachedState>();
449                    let cached_state = cached.pending();
450                    cached_state.release_point = Some(sync_point);
451                });
452            }
453            _ => unreachable!(),
454        }
455    }
456}

477            wp_linux_drm_syncobj_timeline_v1::Request::Destroy => {}

482    fn destroyed(
483        state: &mut D,
484        _client: wayland_server::backend::ClientId,
485        _resource: &WpLinuxDrmSyncobjTimelineV1,
486        data: &DrmSyncobjTimelineData,
487    ) {
488        if let Some(state) = state.drm_syncobj_state() {
489            state
490                .known_timelines
491                .retain(|t| t.upgrade().is_some_and(|t| !Arc::ptr_eq(&t, &data.timeline.0)))
492        }
493    }

496/// Macro to delegate implementation of the drm syncobj protocol to [`DrmSyncobjState`].
497///
498/// You must also implement [`DrmSyncobjHandler`] to use this.
499#[macro_export]
500macro_rules! delegate_drm_syncobj {
501    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
502        $crate::reexports::wayland_server::delegate_global_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
503            $crate::reexports::wayland_protocols::wp::linux_drm_syncobj::v1::server::wp_linux_drm_syncobj_manager_v1::WpLinuxDrmSyncobjManagerV1: $crate::wayland::drm_syncobj::DrmSyncobjGlobalData
504        ] => $crate::wayland::drm_syncobj::DrmSyncobjState);
505        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
506            $crate::reexports::wayland_protocols::wp::linux_drm_syncobj::v1::server::wp_linux_drm_syncobj_manager_v1::WpLinuxDrmSyncobjManagerV1: ()
507        ] => $crate::wayland::drm_syncobj::DrmSyncobjState);
508        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
509            $crate::reexports::wayland_protocols::wp::linux_drm_syncobj::v1::server::wp_linux_drm_syncobj_surface_v1::WpLinuxDrmSyncobjSurfaceV1: $crate::wayland::drm_syncobj::DrmSyncobjSurfaceData
510        ] => $crate::wayland::drm_syncobj::DrmSyncobjState);
511        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
512            $crate::reexports::wayland_protocols::wp::linux_drm_syncobj::v1::server::wp_linux_drm_syncobj_timeline_v1::WpLinuxDrmSyncobjTimelineV1: $crate::wayland::drm_syncobj::DrmSyncobjTimelineData
513        ] => $crate::wayland::drm_syncobj::DrmSyncobjState);
514    }
