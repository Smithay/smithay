22impl<D> GlobalDispatch<WlOutput, WlOutputData, D> for OutputManagerState
23where
24    D: GlobalDispatch<WlOutput, WlOutputData>,
25    D: Dispatch<WlOutput, OutputUserData>,
26    D: OutputHandler,
27    D: CompositorHandler,
28    D: 'static,
29{
30    fn bind(
31        state: &mut D,
32        _dh: &DisplayHandle,
33        client: &Client,
34        resource: New<WlOutput>,
35        global_data: &WlOutputData,
36        data_init: &mut DataInit<'_, D>,
37    ) {
38        let client_scale = state.client_compositor_state(client).clone_client_scale();
39        let output = data_init.init(
40            resource,
41            OutputUserData {
42                output: global_data.output.downgrade(),
43                last_client_scale: AtomicF64::new(client_scale.load(Ordering::Acquire)),
44                client_scale,
45            },
46        );
47
48        let mut inner = global_data.output.inner.0.lock().unwrap();
49
50        let span = warn_span!("output_bind", name = inner.name);
51        let _enter = span.enter();
52
53        trace!("New WlOutput global instantiated");
54
55        if inner.modes.is_empty() {
56            warn!("Output is used with no modes set");
57        }
58        if inner.current_mode.is_none() {
59            warn!("Output is used with no current mod set");
60        }
61        if inner.preferred_mode.is_none() {
62            warn!("Output is used with not preferred mode set");
63        }
64
65        inner.send_geometry_to(&output);
66
67        for &mode in &inner.modes {
68            let mut flags = WMode::empty();
69            if Some(mode) == inner.current_mode {
70                flags |= WMode::Current;
71            }
72            if Some(mode) == inner.preferred_mode {
73                flags |= WMode::Preferred;
74            }
75            output.mode(flags, mode.size.w, mode.size.h, mode.refresh);
76        }
77
78        if output.version() >= 4 {
79            output.name(inner.name.clone());
80            output.description(inner.description.clone())
81        }
82
83        if output.version() >= 2 {
84            output.scale(inner.scale.integer_scale());
85            output.done();
86        }
87
88        // Send enter for surfaces already on this output.
89        for surface in &inner.surfaces {
90            if let Ok(surface) = surface.upgrade() {
91                if surface.client().as_ref() == Some(client) {
92                    surface.enter(&output);
93                }
94            }
95        }
96
97        inner.instances.push(output.downgrade());
98
99        drop(inner);
100        state.output_bound(global_data.output.clone(), output);
101    }
102}

113        _data: &OutputUserData,
114        _dhandle: &DisplayHandle,
115        _data_init: &mut DataInit<'_, D>,
116    ) {

118
119    fn destroyed(
120        _state: &mut D,
121        _client_id: wayland_server::backend::ClientId,
122        output: &WlOutput,
123        data: &OutputUserData,
124    ) {
125        if let Some(o) = data.output.upgrade() {
126            o.inner
127                .0
128                .lock()
129                .unwrap()
130                .instances
131                .retain(|o| o.id() != output.id());
132        }
133    }
134}

159impl<D> Dispatch<ZxdgOutputManagerV1, (), D> for OutputManagerState
160where
161    D: Dispatch<ZxdgOutputManagerV1, ()>,
162    D: Dispatch<ZxdgOutputV1, XdgOutputUserData>,
163    D: CompositorHandler,
164    D: 'static,
165{
166    fn request(
167        state: &mut D,
168        client: &Client,
169        _resource: &ZxdgOutputManagerV1,
170        request: zxdg_output_manager_v1::Request,
171        _data: &(),
172        _dh: &DisplayHandle,
173        data_init: &mut DataInit<'_, D>,
174    ) {
175        match request {
176            zxdg_output_manager_v1::Request::GetXdgOutput {
177                id,
178                output: wl_output,
179            } => {
180                let output = Output::from_resource(&wl_output).unwrap();
181                let mut inner = output.inner.0.lock().unwrap();
182
183                let xdg_output = XdgOutput::new(&inner);
184
185                if inner.xdg_output.is_none() {
186                    inner.xdg_output = Some(xdg_output.clone());
187                }
188
189                let client_scale = state.client_compositor_state(client).clone_client_scale();
190                let id = data_init.init(
191                    id,
192                    XdgOutputUserData {
193                        xdg_output,
194                        last_client_scale: AtomicF64::new(client_scale.load(Ordering::Acquire)),
195                        client_scale,
196                    },
197                );
198
199                inner.xdg_output.as_ref().unwrap().add_instance(&id, &wl_output);
200            }
201            zxdg_output_manager_v1::Request::Destroy => {}
202            _ => {}
203        }
204    }
205}
