1//! XDG Output advertising capabilities
2//!
3//! This protocol is meant for describing outputs in a way
4//! which is more in line with the concept of an output on desktop oriented systems.
5
6use std::sync::{atomic::Ordering, Arc, Mutex};
7
8use tracing::trace;
9use wayland_protocols::xdg::xdg_output::zv1::server::zxdg_output_v1::ZxdgOutputV1;
10use wayland_server::{protocol::wl_output::WlOutput, Resource, Weak};
11
12use crate::utils::{Logical, Physical, Point, Size, Transform};
13
14use super::{Mode, OutputUserData, Scale, XdgOutputUserData};
15
16#[derive(Debug)]
17pub(crate) struct Inner {
18    name: String,
19    description: String,
20    pub(super) logical_position: Point<i32, Logical>,
21
22    pub(super) physical_size: Option<Size<i32, Physical>>,
23    pub(super) scale: Scale,
24    transform: Transform,
25
26    pub instances: Vec<Weak<ZxdgOutputV1>>,
27}
28
29#[derive(Debug, Clone)]
30pub(crate) struct XdgOutput {
31    pub(crate) inner: Arc<Mutex<Inner>>,
32}
33
34impl XdgOutput {
35    pub(super) fn new(output: &super::Inner) -> Self {
36        trace!(name = output.name, "Creating new xdg_output");
37
38        let physical_size = output.current_mode.map(|mode| mode.size);
39
40        Self {
41            inner: Arc::new(Mutex::new(Inner {
42                name: output.name.clone(),
43                description: output.description.clone(),
44                logical_position: output.location,
45
46                physical_size,
47                scale: output.scale,
48                transform: output.transform,
49
50                instances: Vec::new(),
51            })),
52        }
53    }
54
55    pub(super) fn add_instance(&self, xdg_output: &ZxdgOutputV1, wl_output: &WlOutput) {
56        let mut inner = self.inner.lock().unwrap();
57        let client_scale = wl_output
58            .data::<OutputUserData>()
59            .unwrap()
60            .client_scale
61            .load(Ordering::Acquire);
62
63        let logical_position = inner.logical_position.to_client_precise_round(client_scale);
64        xdg_output.logical_position(logical_position.x, logical_position.y);
65
66        if let Some(size) = inner.physical_size {
67            let logical_size = size
68                .to_f64()
69                .to_logical(inner.scale.fractional_scale())
70                .to_client(client_scale)
71                .to_i32_round();
72            let transformed_size = inner.transform.transform_size(logical_size);
73            xdg_output.logical_size(transformed_size.w, transformed_size.h);
74        }
75
76        if xdg_output.version() >= 2 {
77            xdg_output.name(inner.name.clone());
78            xdg_output.description(inner.description.clone());
79        }
80
81        // xdg_output.done() is deprecated since version 3
82        if xdg_output.version() < 3 {
83            xdg_output.done();
84        }
85
86        wl_output.done();
87
88        inner.instances.push(xdg_output.downgrade());
89    }
90
91    pub(super) fn change_current_state(
92        &self,
93        new_mode: Option<Mode>,
94        new_scale: Option<Scale>,
95        new_location: Option<Point<i32, Logical>>,
96        new_transform: Option<impl Into<Transform>>,
97    ) {
98        let mut output = self.inner.lock().unwrap();
99
100        let new_transform = new_transform.map(|x| x.into());
101
102        if let Some(new_mode) = new_mode {
103            output.physical_size = Some(new_mode.size);
104        }
105        if let Some(new_scale) = new_scale {
106            output.scale = new_scale;
107        }
108        if let Some(new_location) = new_location {
109            output.logical_position = new_location;
110        }
111        if let Some(new_transform) = new_transform {
112            output.transform = new_transform;
113        }
114
115        for instance in output.instances.iter() {
116            let Ok(instance) = instance.upgrade() else {
117                continue;
118            };
119
120            let data = instance.data::<XdgOutputUserData>().unwrap();
121            let client_scale = data.client_scale.load(Ordering::Acquire);
122            let scale_changed = client_scale != data.last_client_scale.swap(client_scale, Ordering::AcqRel);
123
124            if new_mode.is_some() || new_scale.is_some() || new_transform.is_some() || scale_changed {
125                if let Some(size) = output.physical_size {
126                    let logical_size = size
127                        .to_f64()
128                        .to_logical(output.scale.fractional_scale())
129                        .to_client(client_scale)
130                        .to_i32_round();
131                    let transformed_size = output.transform.transform_size(logical_size);
132                    instance.logical_size(transformed_size.w, transformed_size.h);
133                }
134            }
135
136            if new_location.is_some() || scale_changed {
137                let logical_position = output.logical_position.to_client_precise_round(client_scale);
138                instance.logical_position(logical_position.x, logical_position.y);
139            }
140
141            // xdg_output.done() is deprecated since version 3
142            if instance.version() < 3 {
143                instance.done();
144            }
145
146            // No need for wl_output.done() here, it will be called by caller (super::Output::change_current_state)
147        }
148    }
149}
