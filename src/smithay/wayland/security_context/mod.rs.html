22/// Handler for security context protocol
23pub trait SecurityContextHandler {
24    /// A client has created a security context. `source` is a callop `EventSource` that listens on
25    /// the socket and produces streams when clients connect. `context` has the metadata associated
26    /// with the security context.
27    fn context_created(&mut self, source: SecurityContextListenerSource, context: SecurityContext);
28}

30/// State of the [WpSecurityContextManagerV1] global
31#[derive(Debug)]
32pub struct SecurityContextState {
33    global: GlobalId,
34}
35
36/// User data for a `WpSecurityContextV1`
37#[derive(Debug)]
38pub struct SecurityContextUserData(Mutex<Option<SecurityContextBuilder>>);
39
40#[derive(Debug)]
41struct SecurityContextBuilder {
42    listen_fd: UnixListener,
43    close_fd: OwnedFd,
44    sandbox_engine: Option<String>,
45    app_id: Option<String>,
46    instance_id: Option<String>,
47}

49/// A security context associated with a listener created through security
50/// context protocol, and with clients connected through that listener.
51#[derive(Clone, Debug)]
52pub struct SecurityContext {
53    /// Name of the sandbox engine associated with the security context
54    pub sandbox_engine: Option<String>,
55    /// Opaque sandbox-specific ID for an application
56    pub app_id: Option<String>,
57    /// Opaque sandbox-specific ID for an instance of an application
58    pub instance_id: Option<String>,
59    /// Client that created the security context
60    pub creator_client_id: ClientId,
61}
62
63impl SecurityContextState {
64    /// Register new [WpSecurityContextManagerV1] global
65    ///
66    /// Filter determines if what clients see the global. It *must* exclude clients
67    /// created through a security context for the protcol to be correct and secure.
68    pub fn new<D, F>(display: &DisplayHandle, filter: F) -> Self
69    where
70        D: GlobalDispatch<WpSecurityContextManagerV1, SecurityContextGlobalData>,
71        D: Dispatch<WpSecurityContextManagerV1, ()>,
72        D: Dispatch<WpSecurityContextV1, SecurityContextUserData>,
73        D: 'static,
74        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
75    {
76        let global_data = SecurityContextGlobalData {
77            filter: Box::new(filter),
78        };
79        let global = display.create_global::<D, WpSecurityContextManagerV1, _>(MANAGER_VERSION, global_data);
80
81        Self { global }
82    }
83
84    /// [WpSecurityContextManagerV1] GlobalId getter
85    pub fn global(&self) -> GlobalId {
86        self.global.clone()
87    }
88}

90/// User data for a `WpSecurityContextManagerV1` global
91#[allow(missing_debug_implementations)]
92pub struct SecurityContextGlobalData {
93    filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
94}

96impl<D> GlobalDispatch<WpSecurityContextManagerV1, SecurityContextGlobalData, D> for SecurityContextState
97where
98    D: GlobalDispatch<WpSecurityContextManagerV1, SecurityContextGlobalData>,
99    D: Dispatch<WpSecurityContextManagerV1, ()>,
100    D: 'static,
101{
102    fn bind(
103        _state: &mut D,
104        _dh: &DisplayHandle,
105        _client: &Client,
106        resource: New<WpSecurityContextManagerV1>,
107        _global_data: &SecurityContextGlobalData,
108        data_init: &mut DataInit<'_, D>,
109    ) {
110        data_init.init(resource, ());
111    }
112
113    fn can_view(client: Client, global_data: &SecurityContextGlobalData) -> bool {
114        (global_data.filter)(&client)
115    }
116}

118impl<D> Dispatch<WpSecurityContextManagerV1, (), D> for SecurityContextState
119where
120    D: Dispatch<WpSecurityContextManagerV1, ()>,
121    D: Dispatch<WpSecurityContextV1, SecurityContextUserData>,
122    D: 'static,
123{
124    fn request(
125        _state: &mut D,
126        _client: &wayland_server::Client,
127        _manager: &WpSecurityContextManagerV1,
128        request: wp_security_context_manager_v1::Request,
129        _data: &(),
130        _dh: &DisplayHandle,
131        data_init: &mut wayland_server::DataInit<'_, D>,
132    ) {
133        match request {
134            wp_security_context_manager_v1::Request::CreateListener {
135                id,
136                listen_fd,
137                close_fd,
138            } => {
139                let data = SecurityContextUserData(Mutex::new(Some(SecurityContextBuilder {
140                    listen_fd: UnixListener::from(listen_fd),
141                    close_fd,
142                    sandbox_engine: None,
143                    app_id: None,
144                    instance_id: None,
145                })));
146                data_init.init(id, data);
147            }
148            wp_security_context_manager_v1::Request::Destroy => {}
149            _ => unreachable!(),
150        }
151    }
152}

154impl<D> Dispatch<WpSecurityContextV1, SecurityContextUserData, D> for SecurityContextState
155where
156    D: Dispatch<WpSecurityContextV1, SecurityContextUserData> + 'static,
157    D: SecurityContextHandler,
158{
159    fn request(
160        state: &mut D,
161        client: &wayland_server::Client,
162        context: &WpSecurityContextV1,
163        request: wp_security_context_v1::Request,
164        data: &SecurityContextUserData,
165        _dh: &DisplayHandle,
166        _data_init: &mut wayland_server::DataInit<'_, D>,
167    ) {
168        let mut data = data.0.lock().unwrap();
169
170        if matches!(request, wp_security_context_v1::Request::Destroy) {
171            return;
172        }
173
174        let Some(builder) = &mut *data else {
175            context.post_error(
176                wp_security_context_v1::Error::AlreadyUsed,
177                "Security context already used",
178            );
179            return;
180        };
181
182        match request {
183            wp_security_context_v1::Request::SetSandboxEngine { name } => {
184                if builder.sandbox_engine.is_some() {
185                    context.post_error(
186                        wp_security_context_v1::Error::AlreadySet,
187                        "Security context already has a sandbox engine",
188                    );
189                }
190                builder.sandbox_engine = Some(name);
191            }
192            wp_security_context_v1::Request::SetAppId { app_id } => {
193                if builder.app_id.is_some() {
194                    context.post_error(
195                        wp_security_context_v1::Error::AlreadySet,
196                        "Security context already has an app id",
197                    );
198                }
199                builder.app_id = Some(app_id);
200            }
201            wp_security_context_v1::Request::SetInstanceId { instance_id } => {
202                if builder.instance_id.is_some() {
203                    context.post_error(
204                        wp_security_context_v1::Error::AlreadySet,
205                        "Security context already has an instance id",
206                    );
207                }
208                builder.instance_id = Some(instance_id);
209            }
210            wp_security_context_v1::Request::Commit => {
211                let builder = data.take().unwrap();
212                let listener_source = SecurityContextListenerSource::new(builder.listen_fd, builder.close_fd);
213                let security_context = SecurityContext {
214                    sandbox_engine: builder.sandbox_engine,
215                    app_id: builder.app_id,
216                    instance_id: builder.instance_id,
217                    creator_client_id: client.id(),
218                };
219                match listener_source {
220                    Ok(listener_source) => state.context_created(listener_source, security_context),
221                    Err(e) => {
222                        error!(error = ?e, "Failed to create security context listener source");
223                    }
224                }
225            }
226            _ => unreachable!(),
227        }
228    }
229}
