1//! XDG Toplevel Tag
2//!
3//! This protocol allows clients to set a tag and description of a toplevel.
4//!
5//! In order to advertise toplevel tag global call [XdgToplevelTagManager::new] and delegate
6//! events to it with [`delegate_xdg_toplevel_tag`][crate::delegate_xdg_toplevel_tag].
7//! Currently attached tag is available either via [XdgToplevelTagHandler] or in [XdgToplevelTagSurfaceData]
8//!
9//! ```
10//! use smithay::wayland::xdg_toplevel_tag::{XdgToplevelTagManager, XdgToplevelTagHandler};
11//! use wayland_protocols::xdg::shell::server::xdg_toplevel::XdgToplevel;
12//! use wayland_server::protocol::wl_surface::WlSurface;
13//! use smithay::delegate_xdg_toplevel_tag;
14//!
15//! # struct State;
16//! # let mut display = wayland_server::Display::<State>::new().unwrap();
17//!
18//! XdgToplevelTagManager::new::<State>(
19//!     &display.handle(),
20//! );
21//!
22//! impl XdgToplevelTagHandler for State {
23//!     fn set_tag(&mut self, toplevel: XdgToplevel, tag: String) {
24//!         dbg!(tag);
25//!     }
26//!
27//!     fn set_description(&mut self, toplevel: XdgToplevel, description: String) {
28//!         dbg!(description);
29//!     }
30//! }
31//!
32//! delegate_xdg_toplevel_tag!(State);
33//! ```

48/// Data associated with WlSurface
49/// Represents the client pending state
50///
51/// ```no_run
52/// use smithay::wayland::compositor;
53/// use smithay::wayland::xdg_toplevel_tag::XdgToplevelTagSurfaceData;
54///
55/// # let wl_surface = todo!();
56/// compositor::with_states(&wl_surface, |states| {
57///     let mut data = states.data_map.get::<XdgToplevelTagSurfaceData>().unwrap();
58///     dbg!(data.tag());
59///     dbg!(data.description());
60/// });
61/// ```
62#[derive(Default, Debug)]
63pub struct XdgToplevelTagSurfaceData {
64    tag: Mutex<Option<Arc<str>>>,
65    description: Mutex<Option<Arc<str>>>,
66}
67
68impl XdgToplevelTagSurfaceData {
69    /// Get tag of a toplevel.
70    ///
71    /// The tag may be shown to the user in UI,
72    /// so it's preferable for it to be human readable,
73    /// but it must be suitable for configuration files and should not be translated.
74    ///
75    /// Example of a tag would be "main window", "settings", "e-mail composer".
76    pub fn tag(&self) -> Option<Arc<str>> {
77        self.tag.lock().unwrap().clone()
78    }
79
80    /// Get description of a toplevel.
81    ///
82    /// This description may be shown to the user in UI or read by a screen reader for accessibility purposes.
83    pub fn description(&self) -> Option<Arc<str>> {
84        self.description.lock().unwrap().clone()
85    }
86}

88/// Handler trait for xdg toplevel tag events.
89pub trait XdgToplevelTagHandler:
90    GlobalDispatch<XdgToplevelTagManagerV1, ()> + Dispatch<XdgToplevelTagManagerV1, ()> + 'static
91{
92    /// Toplevel tag was set/updated.
93    #[allow(unused)]
94    fn set_tag(&mut self, toplevel: XdgToplevel, tag: String) {}
95
96    /// Toplevel description was set/updated.
97    #[allow(unused)]
98    fn set_description(&mut self, toplevel: XdgToplevel, description: String) {}
99}

101/// Delegate type for handling xdg toplevel tag events.
102#[derive(Debug)]
103pub struct XdgToplevelTagManager {
104    global: GlobalId,
105}
106
107impl XdgToplevelTagManager {
108    /// Creates a new delegate type for handling xdg toplevel tag events.
109    pub fn new<D: XdgToplevelTagHandler>(display: &DisplayHandle) -> Self {
110        let global = display.create_global::<D, XdgToplevelTagManagerV1, _>(1, ());
111        XdgToplevelTagManager { global }
112    }
113
114    /// Returns the [XdgToplevelTagManagerV1] global id.
115    pub fn global(&self) -> GlobalId {
116        self.global.clone()
117    }
118}

120impl<D: XdgToplevelTagHandler> GlobalDispatch<XdgToplevelTagManagerV1, (), D> for XdgToplevelTagManager {
121    fn bind(
122        _state: &mut D,
123        _dh: &DisplayHandle,
124        _client: &Client,
125        resource: New<XdgToplevelTagManagerV1>,
126        _data: &(),
127        data_init: &mut DataInit<'_, D>,
128    ) {
129        data_init.init(resource, ());
130    }
131}
132
133impl<D: XdgToplevelTagHandler> Dispatch<XdgToplevelTagManagerV1, (), D> for XdgToplevelTagManager {
134    fn request(
135        state: &mut D,
136        _client: &Client,
137        _resource: &XdgToplevelTagManagerV1,
138        request: xdg_toplevel_tag_manager_v1::Request,
139        _data: &(),
140        _dh: &DisplayHandle,
141        _data_init: &mut DataInit<'_, D>,
142    ) {
143        use xdg_toplevel_tag_manager_v1::Request;
144
145        match request {
146            Request::SetToplevelTag { toplevel, tag } => {
147                let data = toplevel.data::<XdgShellSurfaceUserData>().unwrap();
148
149                compositor::with_states(&data.wl_surface, |states| {
150                    let data = states
151                        .data_map
152                        .get_or_insert_threadsafe(XdgToplevelTagSurfaceData::default);
153                    *data.tag.lock().unwrap() = Some(tag.clone().into());
154                });
155
156                state.set_tag(toplevel, tag);
157            }
158            Request::SetToplevelDescription {
159                toplevel,
160                description,
161            } => {
162                let data = toplevel.data::<XdgShellSurfaceUserData>().unwrap();
163
164                compositor::with_states(&data.wl_surface, |states| {
165                    let data = states
166                        .data_map
167                        .get_or_insert_threadsafe(XdgToplevelTagSurfaceData::default);
168                    *data.tag.lock().unwrap() = Some(description.clone().into());
169                });
170
171                state.set_description(toplevel, description);
172            }
173            Request::Destroy => {}
174            _ => unreachable!(),
175        }
176    }
177}

179/// Macro to delegate implementation of the xdg toplevel tag to [`XdgToplevelTagManager`].
180///
181/// You must also implement [`XdgToplevelTagHandler`] to use this.
182#[macro_export]
183macro_rules! delegate_xdg_toplevel_tag {
184    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
185        const _: () = {
186            use $crate::{
187                reexports::{
188                    wayland_protocols::xdg::toplevel_tag::v1::server::xdg_toplevel_tag_manager_v1::XdgToplevelTagManagerV1,
189                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
190                },
191                wayland::xdg_toplevel_tag::XdgToplevelTagManager,
192            };
193
194            delegate_global_dispatch!(
195                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
196                $ty: [XdgToplevelTagManagerV1: ()] => XdgToplevelTagManager
197            );
198
199            delegate_dispatch!(
200                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
201                $ty: [XdgToplevelTagManagerV1: ()] => XdgToplevelTagManager
202            );
203        };
204    };
