33/// Handler for pointer constraints
34pub trait PointerConstraintsHandler: SeatHandler {
35    /// Pointer lock or confinement constraint created for `pointer` on `surface`
36    ///
37    /// Use [`with_pointer_constraint`] to access the constraint.
38    fn new_constraint(&mut self, surface: &WlSurface, pointer: &PointerHandle<Self>);
39
40    /// The client holding a LockedPointer has committed a cursor position hint.
41    ///
42    /// This is emitted upon a surface commit if the cursor position hint has been updated.
43    ///
44    /// Use [`with_pointer_constraint`] to access the constraint and check if it is active.
45    fn cursor_position_hint(
46        &mut self,
47        surface: &WlSurface,
48        pointer: &PointerHandle<Self>,
49        location: Point<f64, Logical>,
50    );
51}

94/// A constraint imposed on the pointer instance
95#[derive(Debug)]
96pub enum PointerConstraint {
97    /// Pointer is confined to a region of the surface
98    Confined(ConfinedPointer),
99    /// Pointer is locked in place
100    Locked(LockedPointer),
101}
102
103/// A reference to a pointer constraint that can be activated or deactivated.
104///
105/// The derefs to `[PointerConstraint]`.
106#[derive(Debug)]
107pub struct PointerConstraintRef<'a, D: SeatHandler + 'static> {
108    entry: hash_map::OccupiedEntry<'a, PointerHandle<D>, PointerConstraint>,
109}
110
111impl<D: SeatHandler + 'static> ops::Deref for PointerConstraintRef<'_, D> {
112    type Target = PointerConstraint;
113
114    fn deref(&self) -> &Self::Target {
115        self.entry.get()
116    }
117}

119impl<D: SeatHandler + 'static> PointerConstraintRef<'_, D> {
120    /// Send `locked`/`unlocked`
121    ///
122    /// This is not sent automatically since compositors may have different
123    /// policies about when to allow and activate constraints.
124    pub fn activate(&self) {
125        match self.entry.get() {
126            PointerConstraint::Confined(confined) => {
127                if !confined.active.swap(true, Ordering::SeqCst) {
128                    confined.handle.confined();
129                }
130            }
131            PointerConstraint::Locked(locked) => {
132                if !locked.active.swap(true, Ordering::SeqCst) {
133                    locked.handle.locked();
134                }
135            }
136        }
137    }
138
139    /// Send `unlocked`/`unconfined`
140    ///
141    /// For oneshot constraints, will destroy the constraint.
142    ///
143    /// This is sent automatically when the surface loses pointer focus, but
144    /// may also be invoked while the surface is focused.
145    pub fn deactivate(self) {
146        let deactivated = match self.entry.get() {
147            PointerConstraint::Confined(confined) => {
148                if confined.active.swap(false, Ordering::SeqCst) {
149                    confined.handle.unconfined();
150                    true
151                } else {
152                    false
153                }
154            }
155            PointerConstraint::Locked(locked) => {
156                if locked.active.swap(false, Ordering::SeqCst) {
157                    locked.handle.unlocked();
158                    true
159                } else {
160                    false
161                }
162            }
163        };
164
165        if deactivated && self.lifetime() == WEnum::Value(Lifetime::Oneshot) {
166            self.entry.remove_entry();
167        }
168    }
169}
170
171impl PointerConstraint {
172    /// Constraint is active
173    pub fn is_active(&self) -> bool {
174        match self {
175            PointerConstraint::Confined(confined) => &confined.active,
176            PointerConstraint::Locked(locked) => &locked.active,
177        }
178        .load(Ordering::SeqCst)
179    }
180
181    /// Region in which to lock or confine the pointer
182    pub fn region(&self) -> Option<&RegionAttributes> {
183        match self {
184            PointerConstraint::Confined(confined) => confined.region(),
185            PointerConstraint::Locked(locked) => locked.region(),
186        }
187    }
188
189    fn lifetime(&self) -> WEnum<Lifetime> {
190        match self {
191            PointerConstraint::Confined(confined) => confined.lifetime,
192            PointerConstraint::Locked(locked) => locked.lifetime,
193        }
194    }
195
196    /// Commits the pending state of the constraint, and returns the cursor position hint if it has changed.
197    fn commit(&mut self) -> Option<Point<f64, Logical>> {
198        match self {
199            Self::Confined(confined) => {
200                confined.region.clone_from(&confined.pending_region);
201                None
202            }
203            Self::Locked(locked) => {
204                locked.region.clone_from(&locked.pending_region);
205                locked.pending_cursor_position_hint.take().inspect(|hint| {
206                    locked.cursor_position_hint = Some(*hint);
207                })
208            }
209        }
210    }
211}

252// TODO Public method to get current constraints for surface/seat
253/// Get constraint for surface and pointer, if any
254pub fn with_pointer_constraint<
255    D: SeatHandler + 'static,
256    T,
257    F: FnOnce(Option<PointerConstraintRef<'_, D>>) -> T,
258>(
259    surface: &WlSurface,
260    pointer: &PointerHandle<D>,
261    f: F,
262) -> T {
263    with_constraint_data::<D, _, _>(surface, |data| {
264        let constraint = data.and_then(|data| match data.constraints.entry(pointer.clone()) {
265            hash_map::Entry::Occupied(entry) => Some(PointerConstraintRef { entry }),
266            hash_map::Entry::Vacant(_) => None,
267        });
268        f(constraint)
269    })
270}

272fn commit_hook<D: SeatHandler + PointerConstraintsHandler + 'static>(
273    state: &mut D,
274    _dh: &DisplayHandle,
275    surface: &WlSurface,
276) {
277    // `with_constraint_data` locks the pointer constraints,
278    // so we collect the hints first into a Vec, then release the mutex
279    // and only once the mutex is released, we call the handler method.
280    //
281    // This is to avoid deadlocks when the handler method might try to access the constraints again.
282    // It's not a hypothetical, it bit me while implementing the position hint functionality.
283    let position_hints = with_constraint_data::<D, _, _>(surface, |data| {
284        let data = data.unwrap();
285        data.constraints
286            .iter_mut()
287            .filter_map(|(pointer, constraint)| constraint.commit().map(|hint| (pointer.clone(), hint)))
288            .collect::<Vec<_>>()
289    });
290
291    for (pointer, hint) in position_hints {
292        state.cursor_position_hint(surface, &pointer, hint);
293    }
294}

296/// Get `PointerConstraintData` associated with a surface, if any.
297fn with_constraint_data<
298    D: SeatHandler + 'static,
299    T,
300    F: FnOnce(Option<&mut PointerConstraintData<D>>) -> T,
301>(
302    surface: &WlSurface,
303    f: F,
304) -> T {
305    compositor::with_states(surface, |states| {
306        let data = states.data_map.get::<Mutex<PointerConstraintData<D>>>();
307        if let Some(data) = data {
308            f(Some(&mut data.lock().unwrap()))
309        } else {
310            f(None)
311        }
312    })
313}

315/// Add constraint for surface, or raise protocol error if one exists
316fn add_constraint<D: SeatHandler + PointerConstraintsHandler + 'static>(
317    pointer_constraints: &ZwpPointerConstraintsV1,
318    surface: &WlSurface,
319    pointer: &PointerHandle<D>,
320    constraint: PointerConstraint,
321) {
322    let mut added = false;
323    compositor::with_states(surface, |states| {
324        added = states.data_map.insert_if_missing_threadsafe(|| {
325            Mutex::new(PointerConstraintData::<D> {
326                constraints: HashMap::new(),
327            })
328        });
329        let data = states.data_map.get::<Mutex<PointerConstraintData<D>>>().unwrap();
330        let mut data = data.lock().unwrap();
331
332        if data.constraints.contains_key(pointer) {
333            pointer_constraints.post_error(
334                zwp_pointer_constraints_v1::Error::AlreadyConstrained,
335                "pointer constraint already exists for surface and seat",
336            );
337        } else {
338            data.constraints.insert(pointer.clone(), constraint);
339        }
340    });
341
342    if added {
343        compositor::add_post_commit_hook(surface, commit_hook::<D>);
344    }
345}
346
347fn remove_constraint<D: SeatHandler + 'static>(surface: &WlSurface, pointer: &PointerHandle<D>) {
348    with_constraint_data::<D, _, _>(surface, |data| {
349        if let Some(data) = data {
350            data.constraints.remove(pointer);
351        }
352    });
353}

355impl<D> Dispatch<ZwpPointerConstraintsV1, (), D> for PointerConstraintsState
356where
357    D: Dispatch<ZwpPointerConstraintsV1, ()>,
358    D: Dispatch<ZwpConfinedPointerV1, PointerConstraintUserData<D>>,
359    D: Dispatch<ZwpLockedPointerV1, PointerConstraintUserData<D>>,
360    D: SeatHandler,
361    D: PointerConstraintsHandler,
362    D: 'static,
363{
364    fn request(
365        state: &mut D,
366        _client: &wayland_server::Client,
367        pointer_constraints: &ZwpPointerConstraintsV1,
368        request: zwp_pointer_constraints_v1::Request,
369        _data: &(),
370        _dh: &DisplayHandle,
371        data_init: &mut wayland_server::DataInit<'_, D>,
372    ) {
373        match request {
374            zwp_pointer_constraints_v1::Request::LockPointer {
375                id,
376                surface,
377                pointer,
378                region,
379                lifetime,
380            } => {
381                let region = region.as_ref().map(compositor::get_region_attributes);
382                let pointer = pointer.data::<PointerUserData<D>>().unwrap().handle.clone();
383                let handle = data_init.init(
384                    id,
385                    PointerConstraintUserData {
386                        surface: surface.clone(),
387                        pointer: pointer.clone(),
388                    },
389                );
390                if let Some(pointer) = pointer {
391                    add_constraint(
392                        pointer_constraints,
393                        &surface,
394                        &pointer,
395                        PointerConstraint::Locked(LockedPointer {
396                            handle,
397                            region: region.clone(),
398                            pending_region: region,
399                            lifetime,
400                            cursor_position_hint: None,
401                            pending_cursor_position_hint: None,
402                            active: AtomicBool::new(false),
403                        }),
404                    );
405                    state.new_constraint(&surface, &pointer);
406                }
407            }
408            zwp_pointer_constraints_v1::Request::ConfinePointer {
409                id,
410                surface,
411                pointer,
412                region,
413                lifetime,
414            } => {
415                let region = region.as_ref().map(compositor::get_region_attributes);
416                let pointer = pointer.data::<PointerUserData<D>>().unwrap().handle.clone();
417                let handle = data_init.init(
418                    id,
419                    PointerConstraintUserData {
420                        surface: surface.clone(),
421                        pointer: pointer.clone(),
422                    },
423                );
424                if let Some(pointer) = pointer {
425                    add_constraint(
426                        pointer_constraints,
427                        &surface,
428                        &pointer,
429                        PointerConstraint::Confined(ConfinedPointer {
430                            handle,
431                            region: region.clone(),
432                            pending_region: region,
433                            lifetime,
434                            active: AtomicBool::new(false),
435                        }),
436                    );
437                    state.new_constraint(&surface, &pointer);
438                }
439            }
440            zwp_pointer_constraints_v1::Request::Destroy => {}
441            _ => unreachable!(),
442        }
443    }
444}

465impl<D> Dispatch<ZwpConfinedPointerV1, PointerConstraintUserData<D>, D> for PointerConstraintsState
466where
467    D: Dispatch<ZwpConfinedPointerV1, PointerConstraintUserData<D>>,
468    D: SeatHandler,
469    D: 'static,
470{
471    fn request(
472        _state: &mut D,
473        _client: &wayland_server::Client,
474        _confined_pointer: &ZwpConfinedPointerV1,
475        request: zwp_confined_pointer_v1::Request,
476        data: &PointerConstraintUserData<D>,
477        _dh: &DisplayHandle,
478        _data_init: &mut wayland_server::DataInit<'_, D>,
479    ) {
480        let Some(pointer) = &data.pointer else {
481            return;
482        };
483
484        match request {
485            zwp_confined_pointer_v1::Request::SetRegion { region } => {
486                with_pointer_constraint(&data.surface, pointer, |constraint| {
487                    if let Some(PointerConstraint::Confined(confined)) =
488                        constraint.map(|x| x.entry.into_mut())
489                    {
490                        confined.pending_region = region.as_ref().map(compositor::get_region_attributes);
491                    }
492                });
493            }
494            zwp_confined_pointer_v1::Request::Destroy => {}
495            _ => unreachable!(),
496        }
497    }
498
499    fn destroyed(
500        _state: &mut D,
501        _client: wayland_server::backend::ClientId,
502        _resource: &ZwpConfinedPointerV1,
503        data: &PointerConstraintUserData<D>,
504    ) {
505        let Some(pointer) = &data.pointer else {
506            return;
507        };
508
509        remove_constraint(&data.surface, pointer);
510    }
511}

513impl<D> Dispatch<ZwpLockedPointerV1, PointerConstraintUserData<D>, D> for PointerConstraintsState
514where
515    D: Dispatch<ZwpLockedPointerV1, PointerConstraintUserData<D>>,
516    D: SeatHandler,
517    D: 'static,
518{
519    fn request(
520        _state: &mut D,
521        _client: &wayland_server::Client,
522        _locked_pointer: &ZwpLockedPointerV1,
523        request: zwp_locked_pointer_v1::Request,
524        data: &PointerConstraintUserData<D>,
525        _dh: &DisplayHandle,
526        _data_init: &mut wayland_server::DataInit<'_, D>,
527    ) {
528        let Some(pointer) = &data.pointer else {
529            return;
530        };
531
532        match request {
533            zwp_locked_pointer_v1::Request::SetCursorPositionHint { surface_x, surface_y } => {
534                with_pointer_constraint(&data.surface, pointer, |constraint| {
535                    if let Some(PointerConstraint::Locked(locked)) = constraint.map(|x| x.entry.into_mut()) {
536                        locked.pending_cursor_position_hint = Some((surface_x, surface_y).into());
537                    }
538                });
539            }
540            zwp_locked_pointer_v1::Request::SetRegion { region } => {
541                with_pointer_constraint(&data.surface, pointer, |constraint| {
542                    if let Some(PointerConstraint::Locked(locked)) = constraint.map(|x| x.entry.into_mut()) {
543                        locked.pending_region = region.as_ref().map(compositor::get_region_attributes);
544                    }
545                });
546            }
547            zwp_locked_pointer_v1::Request::Destroy => {}
548            _ => unreachable!(),
549        }
550    }
551
552    fn destroyed(
553        _state: &mut D,
554        _client: wayland_server::backend::ClientId,
555        _resource: &ZwpLockedPointerV1,
556        data: &PointerConstraintUserData<D>,
557    ) {
558        let Some(pointer) = &data.pointer else {
559            return;
560        };
561
562        remove_constraint(&data.surface, pointer);
563    }
564}
