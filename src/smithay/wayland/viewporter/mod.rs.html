1//! Utilities for handling the `wp_viewporter` protocol
2//!
3//! ## How to use it
4//!
5//! ### Initialization
6//!
7//! To initialize this implementation, create [`ViewporterState`], store it in your `State` struct and
8//! implement the required traits, as shown in this example:
9//!
10//! ```
11//! use smithay::wayland::viewporter::ViewporterState;
12//! use smithay::delegate_viewporter;
13//! # use smithay::delegate_compositor;
14//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
15//! # use smithay::reexports::wayland_server::{Client, protocol::wl_surface::WlSurface};
16//!
17//! # struct State;
18//! # let mut display = wayland_server::Display::<State>::new().unwrap();
19//!
20//! // Create the viewporter state:
21//! let viewporter_state = ViewporterState::new::<State>(
22//!     &display.handle(), // the display
23//! );
24//!
25//! // implement Dispatch for the Viewporter types
26//! delegate_viewporter!(State);
27//!
28//! # impl CompositorHandler for State {
29//! #     fn compositor_state(&mut self) -> &mut CompositorState { unimplemented!() }
30//! #     fn client_compositor_state<'a>(&self, client: &'a Client) -> &'a CompositorClientState { unimplemented!() }
31//! #     fn commit(&mut self, surface: &WlSurface) {}
32//! # }
33//! # delegate_compositor!(State);
34//!
35//! // You're now ready to go!
36//! ```
37//!
38//! ### Use the viewport state
39//!
40//! The [`viewport state`](ViewportCachedState) is double-buffered and
41//! can be accessed by using the [`with_states`] function
42//!
43//! ```no_compile
44//! let viewport = with_states(surface, |states| {
45//!     states.cached_state.get::<ViewportCachedState>().current()
46//! });
47//! ```
48//!
49//! Before accessing the state you should call [`ensure_viewport_valid`]
50//! to ensure the viewport is valid.
51//!
52//! Note: If you already hand over buffer management to smithay by using
53//! [`on_commit_buffer_handler`](crate::backend::renderer::utils::on_commit_buffer_handler)
54//! the implementation will already call [`ensure_viewport_valid`] for you.

76impl ViewporterState {
77    /// Create new [`wp_viewporter`] global.
78    ///
79    /// It returns the viewporter state, which you can drop to remove these global from
80    /// the event loop in the future.
81    pub fn new<D>(display: &DisplayHandle) -> ViewporterState
82    where
83        D: GlobalDispatch<wp_viewporter::WpViewporter, ()>
84            + Dispatch<wp_viewporter::WpViewporter, ()>
85            + Dispatch<wp_viewport::WpViewport, ViewportState>
86            + 'static,
87    {
88        ViewporterState {
89            global: display.create_global::<D, wp_viewporter::WpViewporter, ()>(1, ()),
90        }
91    }
92
93    /// Returns the viewporter global.
94    pub fn global(&self) -> GlobalId {
95        self.global.clone()
96    }
97}

117impl<D> Dispatch<wp_viewporter::WpViewporter, (), D> for ViewporterState
118where
119    D: GlobalDispatch<wp_viewporter::WpViewporter, ()>,
120    D: Dispatch<wp_viewporter::WpViewporter, ()>,
121    D: Dispatch<wp_viewport::WpViewport, ViewportState>,
122{
123    fn request(
124        _state: &mut D,
125        _client: &wayland_server::Client,
126        _resource: &wp_viewporter::WpViewporter,
127        request: <wp_viewporter::WpViewporter as wayland_server::Resource>::Request,
128        _data: &(),
129        _dhandle: &DisplayHandle,
130        data_init: &mut wayland_server::DataInit<'_, D>,
131    ) {
132        match request {
133            wp_viewporter::Request::GetViewport { id, surface } => {
134                let already_has_viewport = with_states(&surface, |states| {
135                    states
136                        .data_map
137                        .get::<ViewporterSurfaceState>()
138                        .map(|v| v.lock().unwrap().is_some())
139                        .unwrap_or(false)
140                });
141
142                if already_has_viewport {
143                    surface.post_error(
144                        wp_viewporter::Error::ViewportExists as u32,
145                        "the surface already has a viewport object associated".to_string(),
146                    );
147                    return;
148                }
149
150                let viewport = data_init.init(
151                    id,
152                    ViewportState {
153                        surface: surface.downgrade(),
154                    },
155                );
156                let initial = with_states(&surface, |states| {
157                    let inserted = states
158                        .data_map
159                        .insert_if_missing_threadsafe::<ViewporterSurfaceState, _>(|| {
160                            Mutex::new(Some(ViewportMarker(viewport.downgrade())))
161                        });
162
163                    // if we did not insert the marker it will be None as
164                    // checked in already_has_viewport and we have to update
165                    // it now
166                    if !inserted {
167                        *states
168                            .data_map
169                            .get::<ViewporterSurfaceState>()
170                            .unwrap()
171                            .lock()
172                            .unwrap() = Some(ViewportMarker(viewport.downgrade()));
173                    }
174
175                    inserted
176                });
177
178                // only add the pre-commit hook once for the surface
179                if initial {
180                    compositor::add_pre_commit_hook::<D, _>(&surface, viewport_pre_commit_hook);
181                }
182            }
183            wp_viewporter::Request::Destroy => {
184                // All is already handled by our destructor
185            }
186            _ => unreachable!(),
187        }
188    }
189}

191impl<D> Dispatch<wp_viewport::WpViewport, ViewportState, D> for ViewportState
192where
193    D: GlobalDispatch<wp_viewporter::WpViewporter, ()>,
194    D: Dispatch<wp_viewporter::WpViewporter, ()>,
195    D: Dispatch<wp_viewport::WpViewport, ViewportState>,
196    D: CompositorHandler,
197{
198    fn request(
199        state: &mut D,
200        client: &wayland_server::Client,
201        resource: &wp_viewport::WpViewport,
202        request: <wp_viewport::WpViewport as wayland_server::Resource>::Request,
203        data: &ViewportState,
204        _dhandle: &DisplayHandle,
205        _data_init: &mut wayland_server::DataInit<'_, D>,
206    ) {
207        match request {
208            wp_viewport::Request::Destroy => {
209                if let Ok(surface) = data.surface.upgrade() {
210                    with_states(&surface, |states| {
211                        states
212                            .data_map
213                            .get::<ViewporterSurfaceState>()
214                            .unwrap()
215                            .lock()
216                            .unwrap()
217                            .take();
218                        *states.cached_state.get::<ViewportCachedState>().pending() =
219                            ViewportCachedState::default();
220                    });
221                }
222            }
223            wp_viewport::Request::SetSource { x, y, width, height } => {
224                // If all of x, y, width and height are -1.0, the source rectangle is unset instead.
225                // Any other set of values where width or height are zero or negative,
226                // or x or y are negative, raise the bad_value protocol error.
227                let is_unset = x == -1.0 && y == -1.0 && width == -1.0 && height == -1.0;
228                let is_valid_src = x >= 0.0 && y >= 0.0 && width > 0.0 && height > 0.0;
229
230                if !is_unset && !is_valid_src {
231                    resource.post_error(
232                        wp_viewport::Error::BadValue as u32,
233                        "negative or zero values in width or height or negative values in x or y".to_string(),
234                    );
235                    return;
236                }
237
238                // If the wl_surface associated with the wp_viewport is destroyed,
239                // all wp_viewport requests except 'destroy' raise the protocol error no_surface.
240                let Ok(surface) = data.surface.upgrade() else {
241                    resource.post_error(
242                        wp_viewport::Error::NoSurface as u32,
243                        "the wl_surface was destroyed".to_string(),
244                    );
245                    return;
246                };
247
248                with_states(&surface, |states| {
249                    let mut guard = states.cached_state.get::<ViewportCachedState>();
250                    let viewport_state = guard.pending();
251                    let src = if is_unset {
252                        None
253                    } else {
254                        let src = Rectangle::new((x, y).into(), (width, height).into());
255                        trace!(surface = ?surface, src = ?src, "setting surface viewport src");
256                        Some(src)
257                    };
258                    viewport_state.src = src;
259                });
260            }
261            wp_viewport::Request::SetDestination { width, height } => {
262                // If width is -1 and height is -1, the destination size is unset instead.
263                // Any other pair of values for width and height that contains zero or
264                // negative values raises the bad_value protocol error.
265                let is_unset = width == -1 && height == -1;
266                let is_valid_size = width > 0 && height > 0;
267
268                if !is_unset && !is_valid_size {
269                    resource.post_error(
270                        wp_viewport::Error::BadValue as u32,
271                        "negative or zero values in width or height".to_string(),
272                    );
273                    return;
274                }
275
276                // If the wl_surface associated with the wp_viewport is destroyed,
277                // all wp_viewport requests except 'destroy' raise the protocol error no_surface.
278                let Ok(surface) = data.surface.upgrade() else {
279                    resource.post_error(
280                        wp_viewport::Error::NoSurface as u32,
281                        "the wl_surface was destroyed".to_string(),
282                    );
283                    return;
284                };
285
286                with_states(&surface, |states| {
287                    let mut guard = states.cached_state.get::<ViewportCachedState>();
288                    let viewport_state = guard.pending();
289                    let size = if is_unset {
290                        None
291                    } else {
292                        let client_scale = state.client_compositor_state(client).client_scale();
293                        let dst = Size::<_, Client>::from((width, height))
294                            .to_f64()
295                            .to_logical(client_scale)
296                            .to_i32_round();
297                        trace!(surface = ?surface, size = ?dst, "setting surface viewport destination size");
298                        Some(dst)
299                    };
300                    viewport_state.dst = size;
301                });
302            }
303            _ => unreachable!(),
304        }
305    }
306}

314pub(crate) struct ViewportMarker(Weak<wp_viewport::WpViewport>);
315
316fn viewport_pre_commit_hook<D: 'static>(
317    _state: &mut D,
318    _dh: &DisplayHandle,
319    surface: &wl_surface::WlSurface,
320) {
321    with_states(surface, |states| {
322        states
323            .data_map
324            .insert_if_missing_threadsafe::<ViewporterSurfaceState, _>(|| Mutex::new(None));
325        let viewport = states
326            .data_map
327            .get::<ViewporterSurfaceState>()
328            .unwrap()
329            .lock()
330            .unwrap();
331        if let Some(viewport) = &*viewport {
332            let mut guard = states.cached_state.get::<ViewportCachedState>();
333            let viewport_state = guard.pending();
334
335            // If src_width or src_height are not integers and destination size is not set,
336            // the bad_size protocol error is raised when the surface state is applied.
337            let src_size = viewport_state.src.map(|src| src.size);
338            if viewport_state.dst.is_none()
339                && src_size != src_size.map(|s| Size::from((s.w as i32, s.h as i32)).to_f64())
340            {
341                if let Ok(viewport) = viewport.0.upgrade() {
342                    viewport.post_error(
343                        wp_viewport::Error::BadSize as u32,
344                        "destination size is not integer".to_string(),
345                    );
346                }
347            }
348        }
349    });
350}

352/// Ensures that the viewport, if any, is valid accordingly to the protocol specification.
353///
354/// If the viewport violates any protocol checks a protocol error will be raised and `false`
355/// is returned.
356pub fn ensure_viewport_valid(states: &SurfaceData, buffer_size: Size<i32, Logical>) -> bool {
357    states
358        .data_map
359        .insert_if_missing_threadsafe::<ViewporterSurfaceState, _>(|| Mutex::new(None));
360    let viewport = states
361        .data_map
362        .get::<ViewporterSurfaceState>()
363        .unwrap()
364        .lock()
365        .unwrap();
366
367    if let Some(viewport) = &*viewport {
368        let mut guard = states.cached_state.get::<ViewportCachedState>();
369        let state = guard.current();
370
371        let buffer_rect = Rectangle::from_size(buffer_size.to_f64());
372        let src = state.src.unwrap_or(buffer_rect);
373        let valid = buffer_rect.contains_rect(src);
374        if !valid {
375            if let Ok(viewport) = viewport.0.upgrade() {
376                viewport.post_error(
377                    wp_viewport::Error::OutOfBuffer as u32,
378                    format!(
379                        "source rectangle x={},y={},w={},h={} extends outside of the content area x={},y={},w={},h={}", 
380                        src.loc.x, src.loc.y, src.size.w, src.size.h,
381                        buffer_rect.loc.x, buffer_rect.loc.y, buffer_rect.size.w, buffer_rect.size.h),
382                );
383            }
384        }
385        valid
386    } else {
387        true
388    }
389}

391/// Represents the double-buffered viewport
392/// state of a [`WlSurface`](wl_surface::WlSurface)
393#[derive(Debug, Default, Clone, Copy)]
394pub struct ViewportCachedState {
395    /// Defines the source [`Rectangle`] of the [`WlSurface`](wl_surface::WlSurface) in [`Logical`]
396    /// coordinates used for cropping.
397    pub src: Option<Rectangle<f64, Logical>>,
398    /// Defines the destination [`Size`] of the [`WlSurface`](wl_surface::WlSurface) in [`Logical`]
399    /// coordinates used for scaling.
400    pub dst: Option<Size<i32, Logical>>,
401}
402
403impl ViewportCachedState {
404    /// Gets the actual size the [`WlSurface`](wl_surface::WlSurface) should have on screen in
405    /// [`Logical`] coordinates.
406    ///
407    /// This will return the destination size if set or the size of the source rectangle.
408    /// If both are unset `None` is returned.
409    pub fn size(&self) -> Option<Size<i32, Logical>> {
410        self.dst.or_else(|| {
411            self.src
412                .map(|src| Size::from((src.size.w as i32, src.size.h as i32)))
413        })
414    }
415}
416
417impl Cacheable for ViewportCachedState {
418    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
419        ViewportCachedState {
420            src: self.src,
421            dst: self.dst,
422        }
423    }
424
425    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
426        into.src = self.src;
427        into.dst = self.dst;
428    }
429}
