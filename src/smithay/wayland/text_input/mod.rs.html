1//! Utilities for text input support
2//!
3//! This module provides you with utilities to handle text input surfaces,
4//! it is usually used in conjunction with the input method module.
5//!
6//! Text input focus is automatically set to the same surface that has keyboard focus.
7//!
8//! ```
9//! use smithay::{
10//!     delegate_seat, delegate_text_input_manager,
11//! #   delegate_compositor,
12//! };
13//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
14//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
15//! use smithay::wayland::text_input::TextInputManagerState;
16//! use smithay::reexports::wayland_server::{Display, protocol::wl_surface::WlSurface};
17//! # use smithay::reexports::wayland_server::Client;
18//!
19//! # struct State { seat_state: SeatState<Self> };
20//!
21//! delegate_seat!(State);
22//! // Delegate text input handling for State to TextInputManagerState.
23//! delegate_text_input_manager!(State);
24//!
25//! # let mut display = Display::<State>::new().unwrap();
26//! # let display_handle = display.handle();
27//!
28//! let seat_state = SeatState::<State>::new();
29//!
30//! // implement the required traits
31//! impl SeatHandler for State {
32//!     type KeyboardFocus = WlSurface;
33//!     type PointerFocus = WlSurface;
34//!     type TouchFocus = WlSurface;
35//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
36//!         &mut self.seat_state
37//!     }
38//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
39//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
40//! }
41//!
42//! // Add the seat state to your state and create manager global
43//! TextInputManagerState::new::<State>(&display_handle);
44//!
45//! # impl CompositorHandler for State {
46//! #     fn compositor_state(&mut self) -> &mut CompositorState { unimplemented!() }
47//! #     fn client_compositor_state<'a>(&self, client: &'a Client) -> &'a CompositorClientState { unimplemented!() }
48//! #     fn commit(&mut self, surface: &WlSurface) {}
49//! # }
50//! # delegate_compositor!(State);
51//! ```
52//!

71/// Extends [Seat] with text input functionality
72pub trait TextInputSeat {
73    /// Get text input associated with this seat
74    fn text_input(&self) -> &TextInputHandle;
75}
76
77impl<D: SeatHandler + 'static> TextInputSeat for Seat<D> {
78    fn text_input(&self) -> &TextInputHandle {
79        let user_data = self.user_data();
80        user_data.insert_if_missing(TextInputHandle::default);
81        user_data.get::<TextInputHandle>().unwrap()
82    }
83}

85/// State of wp text input protocol
86#[derive(Debug)]
87pub struct TextInputManagerState {
88    global: GlobalId,
89}
90
91impl TextInputManagerState {
92    /// Initialize a text input manager global.
93    pub fn new<D>(display: &DisplayHandle) -> Self
94    where
95        D: GlobalDispatch<ZwpTextInputManagerV3, ()>,
96        D: Dispatch<ZwpTextInputManagerV3, ()>,
97        D: Dispatch<ZwpTextInputV3, TextInputUserData>,
98        D: 'static,
99    {
100        let global = display.create_global::<D, ZwpTextInputManagerV3, _>(MANAGER_VERSION, ());
101
102        Self { global }
103    }
104
105    /// Get the id of ZwpTextInputManagerV3 global
106    pub fn global(&self) -> GlobalId {
107        self.global.clone()
108    }
109}

130impl<D> Dispatch<ZwpTextInputManagerV3, (), D> for TextInputManagerState
131where
132    D: Dispatch<ZwpTextInputManagerV3, ()>,
133    D: Dispatch<ZwpTextInputV3, TextInputUserData>,
134    D: SeatHandler,
135    D: 'static,
136{
137    fn request(
138        _state: &mut D,
139        _client: &Client,
140        _resource: &ZwpTextInputManagerV3,
141        request: zwp_text_input_manager_v3::Request,
142        _data: &(),
143        _dhandle: &DisplayHandle,
144        data_init: &mut DataInit<'_, D>,
145    ) {
146        match request {
147            zwp_text_input_manager_v3::Request::GetTextInput { id, seat } => {
148                let seat = Seat::<D>::from_resource(&seat).unwrap();
149
150                let user_data = seat.user_data();
151                user_data.insert_if_missing(TextInputHandle::default);
152                user_data.insert_if_missing(InputMethodHandle::default);
153                let handle = user_data.get::<TextInputHandle>().unwrap();
154                let input_method_handle = user_data.get::<InputMethodHandle>().unwrap();
155                let instance = data_init.init(
156                    id,
157                    TextInputUserData {
158                        handle: handle.clone(),
159                        input_method_handle: input_method_handle.clone(),
160                    },
161                );
162                handle.add_instance(&instance);
163                if input_method_handle.has_instance() {
164                    handle.enter();
165                }
166            }
167            zwp_text_input_manager_v3::Request::Destroy => {
168                // Nothing to do
169            }
170            _ => unreachable!(),
171        }
172    }
173}
