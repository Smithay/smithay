1//! Utilities for handling activation requests with the `xdg_activation` protocol
2//!
3//! ### Example
4//!
5//! ```no_run
6//! # extern crate wayland_server;
7//! #
8//! use wayland_server::{protocol::wl_surface::WlSurface, DisplayHandle};
9//! use smithay::{
10//!     delegate_xdg_activation,
11//!     wayland::xdg_activation::{XdgActivationHandler, XdgActivationState, XdgActivationToken, XdgActivationTokenData}
12//! };
13//!
14//! pub struct State {
15//!     activation_state: XdgActivationState,
16//! }
17//!
18//! impl XdgActivationHandler for State {
19//!     fn activation_state(&mut self) -> &mut XdgActivationState {
20//!         &mut self.activation_state
21//!     }
22//!
23//!     fn request_activation(
24//!         &mut self,
25//!         token: XdgActivationToken,
26//!         token_data: XdgActivationTokenData,
27//!         surface: WlSurface
28//!     ) {
29//!         if token_data.timestamp.elapsed().as_secs() < 10 {
30//!             // Request surface activation
31//!         }
32//!     }
33//! }
34//!
35//! // Delegate xdg activation handling for State to XdgActivationState.
36//! delegate_xdg_activation!(State);
37//!
38//! # let mut display = wayland_server::Display::<State>::new().unwrap();
39//! # let display_handle = display.handle();
40//! let state = State {
41//!     activation_state: XdgActivationState::new::<State>(&display_handle),
42//! };
43//!
44//! // Rest of the compositor goes here...
45//! ```

67/// Contains the unique string token of activation request
68#[derive(Debug, Clone, Hash, PartialEq, Eq)]
69pub struct XdgActivationToken(String);
70
71impl XdgActivationToken {
72    fn new() -> Self {
73        Self(Alphanumeric.sample_string(&mut rand::rng(), 32))
74    }
75
76    /// Extracts a string slice containing the entire token.
77    #[inline]
78    pub fn as_str(&self) -> &str {
79        &self.0
80    }
81}
82
83impl ops::Deref for XdgActivationToken {
84    type Target = str;
85    #[inline]
86    fn deref(&self) -> &str {
87        &self.0
88    }
89}

92    #[inline]
93    fn from(s: String) -> Self {
94        Self(s)
95    }
96}

98impl From<XdgActivationToken> for String {
99    #[inline]
100    fn from(s: XdgActivationToken) -> Self {
101        s.0
102    }

105/// Activation data asosiated with the [`XdgActivationToken`]
106
107#[derive(Debug, Clone)]
108pub struct XdgActivationTokenData {
109    /// Client that requested the token
110    pub client_id: Option<ClientId>,
111    /// Provides information about the seat and serial event that requested the token.
112    ///
113    /// The serial can come from an input or focus event.
114    /// For instance, if a click triggers the launch of a third-party client,
115    /// this field should contain serial and seat from the wl_pointer.button event.
116    ///
117    /// Some compositors might refuse to activate toplevels
118    /// when the token doesn't have a valid and recent enough event serial.
119    pub serial: Option<(Serial, WlSeat)>,
120    /// The requesting client can specify an app_id to associate the token being created with it.
121    pub app_id: Option<String>,
122    /// The surface requesting the activation.
123    ///
124    /// Note, this is different from the surface that will be activated.
125    pub surface: Option<WlSurface>,
126    /// Timestamp of the token
127    ///
128    /// You can use this to ignore tokens based on time.
129    /// For example you could ignore all tokens older than 5s.
130    pub timestamp: Instant,
131    /// Additional user data attached
132    pub user_data: Arc<UserDataMap>,
133}
134
135impl XdgActivationTokenData {
136    fn new(
137        client_id: Option<ClientId>,
138        serial: Option<(Serial, WlSeat)>,
139        app_id: Option<String>,
140        surface: Option<WlSurface>,
141    ) -> (XdgActivationToken, XdgActivationTokenData) {
142        (
143            XdgActivationToken::new(),
144            XdgActivationTokenData {
145                client_id,
146                serial,
147                app_id,
148                surface,
149                timestamp: Instant::now(),
150                user_data: Arc::new(UserDataMap::new()),
151            },
152        )
153    }
154}
155
156impl Default for XdgActivationTokenData {
157    fn default() -> Self {
158        Self {
159            client_id: None,
160            serial: None,
161            app_id: None,
162            surface: None,
163            timestamp: Instant::now(),
164            user_data: Arc::new(UserDataMap::new()),
165        }
166    }
167}

169/// Tracks the list of pending and current activation requests
170#[derive(Debug)]
171pub struct XdgActivationState {
172    global: GlobalId,
173    known_tokens: HashMap<XdgActivationToken, XdgActivationTokenData>,
174}
175
176impl XdgActivationState {
177    /// Creates a new xdg activation global.
178    ///
179    /// In order to use this abstraction, your `D` type needs to implement [`XdgActivationHandler`].
180    pub fn new<D>(display: &DisplayHandle) -> XdgActivationState
181    where
182        D: GlobalDispatch<xdg_activation_v1::XdgActivationV1, ()>
183            + Dispatch<xdg_activation_v1::XdgActivationV1, ()>
184            + XdgActivationHandler
185            + 'static,
186    {
187        let global = display.create_global::<D, xdg_activation_v1::XdgActivationV1, _>(1, ());
188
189        XdgActivationState {
190            global,
191            known_tokens: HashMap::new(),
192        }
193    }
194
195    /// Create a token without any client association, e.g. for spawning processes from the compositor.
196    ///
197    /// This will not invoke [`XdgActivationHandler::token_created`] like client-created tokens,
198    /// instead use the return arguments to handle any initialization of the data you might need and to copy the token.
199    pub fn create_external_token(
200        &mut self,
201        data: impl Into<Option<XdgActivationTokenData>>,
202    ) -> (&XdgActivationToken, &XdgActivationTokenData) {
203        let token = XdgActivationToken::new();
204        let data = data.into().unwrap_or_default();
205        self.known_tokens.insert(token.clone(), data);
206        self.known_tokens.get_key_value(&token).unwrap()
207    }
208
209    /// Iterate over all known tokens and their associated data
210    pub fn tokens(&self) -> impl Iterator<Item = (&XdgActivationToken, &XdgActivationTokenData)> {
211        self.known_tokens.iter()
212    }
213
214    /// Access the data of a known token
215    pub fn data_for_token(&self, token: &XdgActivationToken) -> Option<&XdgActivationTokenData> {
216        self.known_tokens.get(token)
217    }
218
219    /// Retain pending tokens
220    ///
221    /// You may want to remove super old tokens
222    /// that were never turned into activation request for some reason
223    pub fn retain_tokens<F>(&mut self, mut f: F)
224    where
225        F: FnMut(&XdgActivationToken, &XdgActivationTokenData) -> bool,
226    {
227        self.known_tokens.retain(|k, v| f(k, v))
228    }
229
230    /// Removes an activation token from the internal storage.
231    ///
232    /// Returns `true` if the token was found and subsequently removed.
233    pub fn remove_token(&mut self, token: &XdgActivationToken) -> bool {
234        self.known_tokens.remove(token).is_some()
235    }
236
237    /// Returns the xdg activation global.
238    pub fn global(&self) -> GlobalId {
239        self.global.clone()
240    }
241}

243/// A trait implemented to be notified of activation requests using the xdg activation protocol.
244pub trait XdgActivationHandler {
245    /// Returns the activation state.
246    fn activation_state(&mut self) -> &mut XdgActivationState;
247
248    /// A client has created a new token.
249    ///
250    /// If the token isn't considered valid, it can be immediately untracked by returning `false`.
251    /// The default implementation considers every token valid and will always return `true`.
252    ///
253    /// This method may also be used to attach user_data to the token.
254    fn token_created(&mut self, token: XdgActivationToken, data: XdgActivationTokenData) -> bool {
255        let _ = (token, data);
256        true
257    }
258
259    /// A client has requested surface activation.
260    ///
261    /// The compositor may know which client requested this by checking the token data and may decide whether
262    /// or not to follow through with the activation if it's considered unwanted.
263    ///
264    /// The token remains in the pool and might be used to issue other requests
265    /// until the compositor decides to remove it using [`XdgActivationState::remove_token`] or [`XdgActivationState::retain_tokens`].
266    fn request_activation(
267        &mut self,
268        token: XdgActivationToken,
269        token_data: XdgActivationTokenData,
270        surface: WlSurface,
271    );
272}

282/// Macro to delegate implementation of the xdg activation to [`XdgActivationState`].
283///
284/// You must also implement [`XdgActivationHandler`] to use this.
285#[macro_export]
286macro_rules! delegate_xdg_activation {
287    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
288        const _: () = {
289            use $crate::{
290                reexports::{
291                    wayland_protocols::xdg::activation::v1::server::{
292                        xdg_activation_token_v1::XdgActivationTokenV1, xdg_activation_v1::XdgActivationV1,
293                    },
294                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
295                },
296                wayland::xdg_activation::{ActivationTokenData, XdgActivationState},
297            };
298
299            delegate_dispatch!(
300                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
301                $ty: [XdgActivationV1: ()] => XdgActivationState
302            );
303
304            delegate_dispatch!(
305                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
306                $ty: [XdgActivationTokenV1: ActivationTokenData] => XdgActivationState
307            );
308
309            delegate_global_dispatch!(
310                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
311                $ty: [XdgActivationV1: ()] => XdgActivationState
312            );
313        };
314    };
315}
