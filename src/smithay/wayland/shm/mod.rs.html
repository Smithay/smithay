1//! SHM handling helpers
2//!
3//! This module provides helpers to handle SHM-based buffers from wayland clients.
4//!
5//! SHM (Shared Memory) is the most basic way wayland clients can send content to
6//! the compositor: by sending a file descriptor to some (likely RAM-backed) storage
7//! containing the actual data. This helper handles for you most of the logic for
8//! handling these file descriptor and accessing their contents as simple `&[u8]` slices.
9//!
10//! This module is heavily inspired from
11//! [the similar helpers](https://cgit.freedesktop.org/wayland/wayland/tree/src/wayland-shm.c)
12//! of the wayland C libraries.
13//!
14//! To use it, first add a `ShmGlobal` to your display, specifying the formats
15//! you want to support (ARGB8888 and XRGB8888 are always considered as supported,
16//! as specified by the wayland protocol).
17//!
18//! ```
19//! extern crate wayland_server;
20//! extern crate smithay;
21//!
22//! use smithay::wayland::buffer::BufferHandler;
23//! use smithay::wayland::shm::{ShmState, ShmHandler};
24//! use smithay::delegate_shm;
25//! use wayland_server::protocol::wl_shm::Format;
26//!
27//! # struct State { shm_state: ShmState };
28//! # let mut display = wayland_server::Display::<State>::new().unwrap();
29//! // Create the ShmState.
30//! // Here, we specify that Yuyv and C8 format are supported
31//! // additionally to the standard Argb8888 and Xrgb8888.
32//! let state = ShmState::new::<State>(
33//!     &display.handle(),
34//!     vec![Format::Yuyv, Format::C8],
35//! );
36//!
37//! // insert the shmstate into your compositor state.
38//! // ..
39//!
40//! // provide the necessary trait implementations
41//! impl BufferHandler for State {
42//!     fn buffer_destroyed(&mut self, buffer: &wayland_server::protocol::wl_buffer::WlBuffer) {
43//!         // All renderers can handle buffer destruction at this point.
44//!         // Some parts of window management may also use this function.
45//!     }
46//! }
47//! impl ShmHandler for State {
48//!     fn shm_state(&self) -> &ShmState {
49//!         &self.shm_state
50//!     }
51//! }
52//! delegate_shm!(State);
53//! ```
54//!
55//! Then, when you have a [`WlBuffer`](wayland_server::protocol::wl_buffer::WlBuffer)
56//! and need to retrieve its contents, use the
57//! [`with_buffer_contents`] function to do it:
58//!
59//! ```
60//! # extern crate wayland_server;
61//! # extern crate smithay;
62//! # fn wrap(buffer: &wayland_server::protocol::wl_buffer::WlBuffer) {
63//! use smithay::wayland::shm::{with_buffer_contents, BufferData, BufferAccessError};
64//!
65//! let content = with_buffer_contents(&buffer,
66//!     |ptr: *const u8, len: usize, buffer_metadata: BufferData| {
67//!         // do something to extract the contents of the buffer
68//!     }
69//! );
70//!
71//! match content {
72//!     Ok(something) =>  {
73//!         /* `something` is the value you returned from the closure */
74//!     },
75//!     Err(BufferAccessError::NotManaged) => {
76//!         /* This buffer is not managed by the SHM global, but by something else */
77//!     },
78//!     Err(BufferAccessError::BadMap) => {
79//!         /* The client supplied invalid content specification for this buffer,
80//!            and was killed.
81//!          */
82//!     },
83//!     Err(BufferAccessError::NotReadable) => {
84//!         /* The client has not allowed reads to this buffer */
85//!     },
86//!     Err(BufferAccessError::NotWritable) => unreachable!("cannot be triggered by with_buffer_contents"),
87//! }
88//! # }
89//! ```
90//!
91//! **Note**
92//!
93//! This handler makes itself safe regarding the client providing a wrong size for the memory pool
94//! by using a SIGBUS handler.
95//!
96//! If you are already using an handler for this signal, you probably don't want to use this handler.

126/// State of SHM module
127#[derive(Debug)]
128pub struct ShmState {
129    formats: HashSet<wl_shm::Format>,
130    shm: GlobalId,
131}
132
133impl ShmState {
134    /// Create a new SHM global advertizing the given formats.
135    ///
136    /// This global will always advertize `ARGB8888` and `XRGB8888` since these formats are required by the
137    /// protocol. Formats given as an argument are also advertized.
138    ///
139    /// The global is directly created on the provided [`Display`](wayland_server::Display),
140    /// and this function returns the a delegate type. The id provided by [`ShmState::global`] may be used to
141    /// remove this global in the future.
142    pub fn new<D>(display: &DisplayHandle, formats: impl IntoIterator<Item = wl_shm::Format>) -> ShmState
143    where
144        D: GlobalDispatch<WlShm, ()>
145            + Dispatch<WlShm, ()>
146            + Dispatch<WlShmPool, ShmPoolUserData>
147            + BufferHandler
148            + ShmHandler
149            + 'static,
150    {
151        let mut formats = formats.into_iter().collect::<HashSet<_>>();
152
153        // Mandatory formats
154        formats.insert(wl_shm::Format::Argb8888);
155        formats.insert(wl_shm::Format::Xrgb8888);
156
157        let shm = display.create_global::<D, WlShm, _>(2, ());
158
159        ShmState { formats, shm }
160    }
161
162    /// Returns the id of the [`WlShm`] global.
163    pub fn global(&self) -> GlobalId {
164        self.shm.clone()
165    }
166
167    /// Updates the list of formats advertised by the global.
168    ///
169    /// This will only affect new binds to the wl_shm global.
170    ///
171    /// Removing formats will cause old clients trying to create
172    /// a buffer of a now unsupported format to be killed.
173    ///
174    /// This function will never remove the mandatory formats `ARGB8888` and `XRGB8888`.
175    pub fn update_formats(&mut self, formats: impl IntoIterator<Item = wl_shm::Format>) {
176        self.formats = formats.into_iter().collect::<HashSet<_>>();
177        // Mandatory formats
178        self.formats.insert(wl_shm::Format::Argb8888);
179        self.formats.insert(wl_shm::Format::Xrgb8888);
180    }
181}

189/// Error that can occur when accessing an SHM buffer
190#[derive(Debug, thiserror::Error)]
191pub enum BufferAccessError {
192    /// This buffer is not managed by the SHM handler
193    #[error("non-SHM buffer")]
194    NotManaged,
195
196    /// An error occurred while accessing the memory map
197    ///
198    /// This can happen if the client advertized a wrong size
199    /// for the memory map.
200    ///
201    /// If this error occurs, the client has been killed as a result.
202    #[error("invalid client buffer")]
203    BadMap,
204
205    /// This buffer cannot be read by the compositor
206    #[error("Client has not indicated read permission for the buffer")]
207    NotReadable,
208
209    /// This buffer cannot be written to by the compositor
210    #[error("Client has not indicated write permission for the buffer")]
211    NotWritable,
212}
213
214impl From<UnmanagedResource> for BufferAccessError {
215    #[inline]
216    fn from(_: UnmanagedResource) -> Self {
217        Self::NotManaged
218    }
219}

221/// Call given closure with the contents of the given buffer
222///
223/// If the buffer is managed by the provided `ShmGlobal`, its contents are
224/// extracted and the closure is extracted with them:
225///
226/// - The first argument is a pointer to the contents of the pool
227/// - The second argument is the length of the contents of the pool
228/// - The third argument is the specification of this buffer is this pool
229///
230/// The pool cannot be provided as a slice since it is shared memory that could be mutated
231/// by the client.
232///
233/// If the buffer is not managed by the provided `ShmGlobal`, the closure is not called
234/// and this method will return `Err(BufferAccessError::NotManaged)` (this will be the case for an
235/// EGL buffer for example).
236///
237/// # Safety
238/// The pointer passed to the callback is only valid while the callback is running. The shared
239/// memory may be mutated by the client at any time. Creating a reference or slice into this memory
240/// is undefined behavior if it is mutated by the client while the reference exists.
241pub fn with_buffer_contents<F, T>(buffer: &wl_buffer::WlBuffer, f: F) -> Result<T, BufferAccessError>
242where
243    F: FnOnce(*const u8, usize, BufferData) -> T,
244{
245    let data = buffer
246        .data::<ShmBufferUserData>()
247        .ok_or(BufferAccessError::NotManaged)?;
248
249    match data.pool.with_data(|ptr, len| f(ptr, len, data.data)) {
250        Ok(t) => Ok(t),
251        Err(()) => {
252            // SIGBUS error occurred
253            buffer.post_error(wl_shm::Error::InvalidFd, "Bad pool size.");
254            Err(BufferAccessError::BadMap)
255        }
256    }
257}

259/// Call given closure with the contents of the given buffer for mutable access
260///
261/// If the buffer is managed by the provided `ShmGlobal`, its contents are
262/// extracted and the closure is extracted with them:
263///
264/// - The first argument is a pointer to the contents of the pool
265/// - The second argument is the length of the contents of the pool
266/// - The third argument is the specification of this buffer is this pool
267///
268/// The pool cannot be provided as a slice since it is shared memory that could be mutated
269/// by the client.
270///
271/// If the buffer is not managed by the provided `ShmGlobal`, the closure is not called
272/// and this method will return `Err(BufferAccessError::NotManaged)` (this will be the case for an
273/// EGL buffer for example).
274///
275/// # Safety
276/// The pointer passed to the callback is only valid while the callback is running. The shared
277/// memory may be mutated by the client at any time. Creating a reference or slice into this memory
278/// is undefined behavior if it is mutated by the client while the reference exists.
279pub fn with_buffer_contents_mut<F, T>(buffer: &wl_buffer::WlBuffer, f: F) -> Result<T, BufferAccessError>
280where
281    F: FnOnce(*mut u8, usize, BufferData) -> T,
282{
283    let data = buffer
284        .data::<ShmBufferUserData>()
285        .ok_or(BufferAccessError::NotManaged)?;
286
287    match data.pool.with_data_mut(|ptr, len| f(ptr, len, data.data)) {
288        Ok(t) => Ok(t),
289        Err(()) => {
290            // SIGBUS error occurred
291            buffer.post_error(wl_shm::Error::InvalidFd, "Bad pool size.");
292            Err(BufferAccessError::BadMap)
293        }
294    }
295}

297/// Returns the bpp of the format
298///
299/// Note: This will return 0 for formats that don't have a specified width.
300pub fn wl_bytes_per_pixel(format: WEnum<wl_shm::Format>) -> i32 {
301    match format {
302        WEnum::Value(f) => {
303            shm_format_to_fourcc(f).map_or(0, |fourcc| get_bpp(fourcc).map_or(0, |bpp| bpp / 8))
304        }
305        WEnum::Unknown(_) => 0,
306    }
307    .try_into()
308    .unwrap()
309}

311macro_rules! shm_format_table {
312    (
313        $(
314            $fourcc: ident => $shm: ident
315        ),* $(,)?
316    ) => {
317        /// Convert from a [`Fourcc`](crate::backend::allocator::Fourcc) format to a wl_shm format
318        pub const fn fourcc_to_shm_format(value: $crate::backend::allocator::Fourcc) -> Option<$crate::reexports::wayland_server::protocol::wl_shm::Format> {
319            match value {
320                $(
321                    $crate::backend::allocator::Fourcc::$fourcc => Some($crate::reexports::wayland_server::protocol::wl_shm::Format::$shm),
322                )*
323                _ => None,
324            }
325        }
326
327        /// Convert from a wl_shm format to a [`Fourcc`](crate::backend::allocator::Fourcc) format
328        pub const fn shm_format_to_fourcc(value: $crate::reexports::wayland_server::protocol::wl_shm::Format) -> Option<$crate::backend::allocator::Fourcc> {
329            match value {
330                $(
331                    $crate::reexports::wayland_server::protocol::wl_shm::Format::$shm => Some($crate::backend::allocator::Fourcc::$fourcc),
332                )*
333                _ => None,
334            }
335        }
336    }

473type DestructionHook = dyn Fn(&mut dyn Any, &wl_buffer::WlBuffer) + Send + Sync;
474
475/// User data of shm WlBuffer
476#[derive(Debug)]
477pub struct ShmBufferUserData {
478    pub(crate) pool: Arc<Pool>,
479    pub(crate) data: BufferData,
480    destruction_hooks: Mutex<Vec<Hook<DestructionHook>>>,
481}
482
483impl ShmBufferUserData {
484    pub(crate) fn add_destruction_hook(
485        &self,
486        hook: impl Fn(&mut dyn Any, &wl_buffer::WlBuffer) + Send + Sync + 'static,
487    ) -> HookId {
488        let hook: Hook<DestructionHook> = Hook::new(Arc::new(hook));
489        let id = hook.id.clone();
490        self.destruction_hooks.lock().unwrap().push(hook);
491        id
492    }
493
494    pub(crate) fn remove_destruction_hook(&self, hook_id: HookId) {
495        let mut guard = self.destruction_hooks.lock().unwrap();
496        if let Some(id) = guard.iter().position(|hook| hook.id != hook_id) {
497            guard.remove(id);
498        }
499    }
500}

504macro_rules! delegate_shm {
505    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
506        const _: () = {
507            use $crate::{
508                reexports::{
509                    wayland_server::protocol::{wl_buffer::WlBuffer, wl_shm::WlShm, wl_shm_pool::WlShmPool},
510                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
511                },
512                wayland::shm::{ShmBufferUserData, ShmPoolUserData, ShmState},
513            };
514
515            delegate_global_dispatch!(
516                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
517                $ty: [WlShm: ()] => ShmState
518            );
519
520            delegate_dispatch!(
521                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
522                $ty: [WlShm: ()] => ShmState
523            );
524
525            delegate_dispatch!(
526                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
527                $ty: [WlShmPool: ShmPoolUserData] => ShmState
528            );
529
530            delegate_dispatch!(
531                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
532                $ty: [WlBuffer: ShmBufferUserData] => ShmState
533            );
534        };
535    };
