19// Dropping Pool is actually pretty slow. Unmapping the memory can take 1-2 ms, but the real
20// offender is closing the file descriptor, which I've seen take up to 6 ms. It's waiting on some
21// spinlock in the kernel.
22//
23// Blocking the main thread for 6 ms is quite bad. In fact, 6 ms is almost the entire time budget
24// for a 165 Hz frame. To make matters worse, some clients will cause repeated creation and
25// dropping of shm pools, like Firefox during a focus-out animation. This results in dropped
26// frames.
27//
28// To work around this problem, we spawn a separate thread whose sole purpose is dropping stuff we
29// send it through a channel. Conveniently, Pool is already Send, so there's no problem doing this.
30static DROP_THIS: LazyLock<Sender<InnerPool>> = LazyLock::new(|| {
31    let (tx, rx) = channel();
32    thread::Builder::new()
33        .name("Shm dropping thread".to_owned())
34        .spawn(move || {
35            while let Ok(x) = rx.recv() {
36                profiling::scope!("dropping Pool");
37                drop(x);
38            }
39        })
40        .unwrap();
41    tx
42});

64pub enum ResizeError {
65    InvalidSize,
66    MremapFailed,
67}
68
69impl InnerPool {
70    #[instrument(level = "trace", skip_all, name = "wayland_shm")]
71    pub fn new(fd: OwnedFd, size: NonZeroUsize) -> Result<InnerPool, OwnedFd> {
72        let memmap = match MemMap::new(fd.as_fd(), size) {
73            Ok(memmap) => memmap,
74            Err(_) => {
75                return Err(fd);
76            }
77        };
78        trace!(fd = ?fd, size = ?size, "Creating new shm pool");
79        Ok(InnerPool {
80            map: RwLock::new(memmap),
81            fd,
82        })
83    }
84
85    pub fn resize(&self, newsize: NonZeroUsize) -> Result<(), ResizeError> {
86        let mut guard = self.map.write().unwrap();
87        let oldsize = guard.size();
88
89        if oldsize > usize::from(newsize) {
90            return Err(ResizeError::InvalidSize);
91        }
92
93        trace!(fd = ?self.fd, oldsize = oldsize, newsize = ?newsize, "Resizing shm pool");
94        guard.remap(self.fd.as_fd(), newsize).map_err(|()| {
95            debug!(fd = ?self.fd, oldsize = oldsize, newsize = ?newsize, "SHM pool resize failed");
96            ResizeError::MremapFailed
97        })
98    }
99
100    pub fn size(&self) -> usize {
101        self.map.read().unwrap().size
102    }
103
104    #[instrument(level = "trace", skip_all, name = "wayland_shm")]
105    pub fn with_data<T, F: FnOnce(*const u8, usize) -> T>(&self, f: F) -> Result<T, ()> {
106        // Place the sigbus handler
107        unsafe { place_sigbus_handler() };
108
109        let pool_guard = self.map.read().unwrap();
110
111        trace!(fd = ?self.fd, "Buffer access on shm pool");
112
113        // Prepare the access
114        SIGBUS_GUARD.with(|guard| {
115            let (p, _) = guard.get();
116            if !p.is_null() {
117                // Recursive call of this method is not supported
118                panic!("Recursive access to a SHM pool content is not supported.");
119            }
120            guard.set((&*pool_guard as *const MemMap, false))
121        });
122
123        let t = f(pool_guard.ptr as *const _, pool_guard.size);
124
125        // Cleanup Post-access
126        SIGBUS_GUARD.with(|guard| {
127            let (_, triggered) = guard.get();
128            guard.set((ptr::null_mut(), false));
129            if triggered {
130                debug!(fd = ?self.fd, "SIGBUS caught on access on shm pool");
131                Err(())
132            } else {
133                Ok(t)
134            }
135        })
136    }
137
138    #[instrument(level = "trace", skip_all, name = "wayland_shm")]
139    pub fn with_data_mut<T, F: FnOnce(*mut u8, usize) -> T>(&self, f: F) -> Result<T, ()> {
140        // Place the sigbus handler
141        unsafe { place_sigbus_handler() };
142
143        // This is actually a write access.
144        #[allow(clippy::readonly_write_lock)]
145        let pool_guard = self.map.write().unwrap();
146
147        trace!(fd = ?self.fd, "Mutable buffer access on shm pool");
148
149        // Prepare the access
150        SIGBUS_GUARD.with(|guard| {
151            let (p, _) = guard.get();
152            if !p.is_null() {
153                // Recursive call of this method is not supported
154                panic!("Recursive access to a SHM pool content is not supported.");
155            }
156            guard.set((&*pool_guard as *const MemMap, false))
157        });
158
159        let t = f(pool_guard.ptr, pool_guard.size);
160
161        // Cleanup Post-access
162        SIGBUS_GUARD.with(|guard| {
163            let (_, triggered) = guard.get();
164            guard.set((ptr::null_mut(), false));
165            if triggered {
166                debug!(fd = ?self.fd, "SIGBUS caught on access on shm pool");
167                Err(())
168            } else {
169                Ok(t)
170            }
171        })
172    }
173}

203#[derive(Debug)]
204struct MemMap {
205    ptr: *mut u8,
206    size: usize,
207}
208
209impl MemMap {
210    fn new(fd: BorrowedFd<'_>, size: NonZeroUsize) -> Result<MemMap, ()> {
211        Ok(MemMap {
212            ptr: unsafe { map(fd, size) }?,
213            size: size.into(),
214        })
215    }
216
217    fn remap(&mut self, fd: BorrowedFd<'_>, newsize: NonZeroUsize) -> Result<(), ()> {
218        if self.ptr.is_null() {
219            return Err(());
220        }
221        // memunmap cannot fail, as we are unmapping a pre-existing map
222        let _ = unsafe { unmap(self.ptr, self.size) };
223        // remap the fd with the new size
224        match unsafe { map(fd, newsize) } {
225            Ok(ptr) => {
226                // update the parameters
227                self.ptr = ptr;
228                self.size = usize::from(newsize);
229                Ok(())
230            }
231            Err(()) => {
232                // set ourselves in an empty state
233                self.ptr = ptr::null_mut();
234                self.size = 0;
235                Err(())
236            }
237        }
238    }
239
240    fn size(&self) -> usize {
241        self.size
242    }
243
244    fn contains(&self, ptr: *mut u8) -> bool {
245        ptr >= self.ptr && ptr < unsafe { self.ptr.add(self.size) }
246    }
247
248    fn nullify(&self) -> Result<(), ()> {
249        unsafe { nullify_map(self.ptr, self.size) }
250    }
251}
252
253impl Drop for MemMap {
254    fn drop(&mut self) {
255        if !self.ptr.is_null() {
256            let _ = unsafe { unmap(self.ptr, self.size) };
257        }
258    }
259}

261/// A simple wrapper with some default arguments for `nix::mman::mmap`.
262unsafe fn map(fd: BorrowedFd<'_>, size: NonZeroUsize) -> Result<*mut u8, ()> {
263    let ret = unsafe {
264        mm::mmap(
265            ptr::null_mut(),
266            size.into(),
267            mm::ProtFlags::READ | mm::ProtFlags::WRITE,
268            mm::MapFlags::SHARED,
269            fd,
270            0,
271        )
272    };
273    ret.map(|p| p as *mut u8).map_err(|_| ())
274}

276/// A simple wrapper for `nix::mman::munmap`.
277#[profiling::function]
278unsafe fn unmap(ptr: *mut u8, size: usize) -> Result<(), ()> {
279    let ret = unsafe { mm::munmap(ptr as *mut _, size) };
280    ret.map_err(|_| ())
281}

283unsafe fn nullify_map(ptr: *mut u8, size: usize) -> Result<(), ()> {
284    let ret = unsafe {
285        mm::mmap_anonymous(
286            ptr as *mut std::ffi::c_void,
287            size,
288            mm::ProtFlags::READ | mm::ProtFlags::WRITE,
289            mm::MapFlags::PRIVATE | mm::MapFlags::FIXED,
290        )
291    };
292    ret.map(|_| ()).map_err(|_| ())
293}

295/// The sigbus handler will be placed only once
296unsafe fn place_sigbus_handler() {
297    let _ = OLD_SIGBUS_HANDLER.get_or_init(|| {
298        // create our sigbus handler
299        unsafe {
300            // We use `mem::zeroed()` because regular struct init as well as struct update syntax require all fields to be public
301            // and libc does not guarantee that for all targets
302            let mut action: libc::sigaction = mem::zeroed();
303            action.sa_sigaction = sigbus_handler as _;
304            action.sa_flags = libc::SA_SIGINFO | libc::SA_NODEFER;
305
306            let mut old_action = mem::zeroed();
307            if libc::sigaction(libc::SIGBUS, &action, &mut old_action) == -1 {
308                let e = rustix::io::Errno::from_raw_os_error(errno::errno().0);
309                panic!("sigaction failed for SIGBUS handler: {e:?}");
310            }
311
312            old_action
313        }
314    });
315}
316
317unsafe fn reraise_sigbus() {
318    // reset the old sigaction
319    unsafe {
320        libc::sigaction(libc::SIGBUS, OLD_SIGBUS_HANDLER.get().unwrap(), ptr::null_mut());
321        libc::raise(libc::SIGBUS);
322    }
323}

325extern "C" fn sigbus_handler(_signum: libc::c_int, info: *mut libc::siginfo_t, _context: *mut libc::c_void) {
326    let faulty_ptr = unsafe { siginfo_si_addr(info) } as *mut u8;
327    SIGBUS_GUARD.with(|guard| {
328        let (memmap, _) = guard.get();
329        match unsafe { memmap.as_ref() }.map(|m| (m, m.contains(faulty_ptr))) {
330            Some((m, true)) => {
331                // we are in a faulty memory pool !
332                // remember that it was faulty
333                guard.set((memmap, true));
334                // nullify the pool
335                if m.nullify().is_err() {
336                    // something terrible occurred !
337                    unsafe { reraise_sigbus() }
338                }
339            }
340            _ => {
341                // something else occurred, let's die honorably
342                unsafe { reraise_sigbus() }
343            }
344        }
345    });
346}

355#[cfg(any(target_os = "linux", target_os = "android"))]
356unsafe fn siginfo_si_addr(info: *mut libc::siginfo_t) -> *mut libc::c_void {
357    #[repr(C)]
358    #[allow(non_camel_case_types)]
359    struct siginfo_t {
360        a: [libc::c_int; 3], // si_signo, si_errno, si_code
361        si_addr: *mut libc::c_void,
362    }
363
364    unsafe { (*(info as *const siginfo_t)).si_addr }
365}
