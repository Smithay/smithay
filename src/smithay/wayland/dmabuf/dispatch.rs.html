48impl<D> Dispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufData, D> for DmabufState
49where
50    D: Dispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufData>
51        + Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData>
52        + Dispatch<zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1, DmabufFeedbackData>
53        + DmabufHandler
54        + 'static,
55{
56    fn request(
57        state: &mut D,
58        _client: &Client,
59        _resource: &zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1,
60        request: zwp_linux_dmabuf_v1::Request,
61        data: &DmabufData,
62        _dh: &DisplayHandle,
63        data_init: &mut DataInit<'_, D>,
64    ) {
65        match request {
66            zwp_linux_dmabuf_v1::Request::Destroy => {}
67
68            zwp_linux_dmabuf_v1::Request::CreateParams { params_id } => {
69                data_init.init(
70                    params_id,
71                    DmabufParamsData {
72                        id: data.id,
73                        used: AtomicBool::new(false),
74                        formats: data.formats.clone(),
75                        modifier: Mutex::new(None),
76                        planes: Mutex::new(Vec::with_capacity(MAX_PLANES)),
77                    },
78                );
79            }
80
81            zwp_linux_dmabuf_v1::Request::GetDefaultFeedback { id } => {
82                let feedback = data_init.init(
83                    id,
84                    DmabufFeedbackData {
85                        known_default_feedbacks: data.known_default_feedbacks.clone(),
86                        surface: None,
87                    },
88                );
89
90                data.known_default_feedbacks
91                    .lock()
92                    .unwrap()
93                    .push(feedback.downgrade());
94
95                data.default_feedback
96                    .as_ref()
97                    .unwrap()
98                    .lock()
99                    .unwrap()
100                    .send(&feedback);
101            }
102
103            zwp_linux_dmabuf_v1::Request::GetSurfaceFeedback { id, surface } => {
104                let feedback = data_init.init(
105                    id,
106                    DmabufFeedbackData {
107                        known_default_feedbacks: data.known_default_feedbacks.clone(),
108                        surface: Some(surface.downgrade()),
109                    },
110                );
111
112                let surface_feedback = compositor::with_states(&surface, |states| {
113                    states
114                        .data_map
115                        .insert_if_missing_threadsafe(SurfaceDmabufFeedbackState::default);
116                    let feedback_state = states.data_map.get::<SurfaceDmabufFeedbackState>().unwrap();
117                    feedback_state.add_instance(&feedback, || {
118                        state
119                            .new_surface_feedback(&surface, &DmabufGlobal { id: data.id })
120                            .unwrap_or_else(|| {
121                                data.default_feedback.as_ref().unwrap().lock().unwrap().clone()
122                            })
123                    })
124                });
125
126                surface_feedback.send(&feedback);
127            }
128
129            _ => unreachable!(),
130        }
131    }
132}

134impl<D> Dispatch<zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1, DmabufFeedbackData, D>
135    for DmabufState
136where
137    D: Dispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufData>
138        + Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData>
139        + Dispatch<zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1, DmabufFeedbackData>
140        + 'static,
141{
142    fn request(
143        _state: &mut D,
144        _client: &Client,
145        resource: &zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1,
146        request: <zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1 as Resource>::Request,
147        data: &DmabufFeedbackData,
148        _dhandle: &DisplayHandle,
149        _data_init: &mut DataInit<'_, D>,
150    ) {
151        match request {
152            zwp_linux_dmabuf_feedback_v1::Request::Destroy => {
153                data.known_default_feedbacks
154                    .lock()
155                    .unwrap()
156                    .retain(|feedback| feedback != resource);
157
158                if let Some(surface) = data.surface.as_ref().and_then(|s| s.upgrade().ok()) {
159                    compositor::with_states(&surface, |states| {
160                        if let Some(surface_state) = states.data_map.get::<SurfaceDmabufFeedbackState>() {
161                            surface_state.remove_instance(resource);
162                        }
163                    })
164                }
165            }
166            _ => unreachable!(),
167        }
168    }
169}

171impl<D> GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData, D> for DmabufState
172where
173    D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
174        + Dispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufData>
175        + Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData>
176        + 'static,
177{
178    fn bind(
179        _state: &mut D,
180        _dh: &DisplayHandle,
181        _client: &Client,
182        resource: New<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1>,
183        global_data: &DmabufGlobalData,
184        data_init: &mut DataInit<'_, D>,
185    ) {
186        let data = DmabufData {
187            formats: global_data.formats.clone(),
188            id: global_data.id,
189            default_feedback: global_data.default_feedback.clone(),
190            known_default_feedbacks: global_data.known_default_feedbacks.clone(),
191        };
192
193        let zwp_dmabuf = data_init.init(resource, data);
194
195        // Immediately send format info to the client if we are the correct version.
196        //
197        // These events are deprecated in version 4 of the protocol.
198        if zwp_dmabuf.version() < zwp_linux_dmabuf_v1::REQ_GET_DEFAULT_FEEDBACK_SINCE {
199            for (fourcc, modifiers) in &*global_data.formats {
200                // Modifier support got added in version 3
201                if zwp_dmabuf.version() < zwp_linux_dmabuf_v1::EVT_MODIFIER_SINCE {
202                    if modifiers.contains(&Modifier::Invalid) || modifiers.contains(&Modifier::Linear) {
203                        zwp_dmabuf.format(*fourcc as u32);
204                    }
205                    continue;
206                }
207
208                for modifier in modifiers {
209                    let modifier_hi = (Into::<u64>::into(*modifier) >> 32) as u32;
210                    let modifier_lo = Into::<u64>::into(*modifier) as u32;
211                    zwp_dmabuf.modifier(*fourcc as u32, modifier_hi, modifier_lo);
212                }
213            }
214        }
215    }
216
217    fn can_view(client: Client, global_data: &DmabufGlobalData) -> bool {
218        (global_data.filter)(&client)
219    }
220}

222impl<D> Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData, D> for DmabufState
223where
224    D: Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData>
225        + Dispatch<wl_buffer::WlBuffer, Dmabuf>
226        + BufferHandler
227        + DmabufHandler,
228{
229    fn request(
230        state: &mut D,
231        _client: &Client,
232        params: &zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1,
233        request: zwp_linux_buffer_params_v1::Request,
234        data: &DmabufParamsData,
235        dh: &DisplayHandle,
236        data_init: &mut DataInit<'_, D>,
237    ) {
238        match request {
239            zwp_linux_buffer_params_v1::Request::Destroy => {}
240
241            zwp_linux_buffer_params_v1::Request::Add {
242                fd,
243                plane_idx,
244                offset,
245                stride,
246                modifier_hi,
247                modifier_lo,
248            } => {
249                if !data.ensure_unused(params) {
250                    return;
251                }
252
253                // Plane index should not be too large
254                if plane_idx as usize >= MAX_PLANES {
255                    params.post_error(
256                        zwp_linux_buffer_params_v1::Error::PlaneIdx,
257                        format!("Plane index {plane_idx} is out of bounds"),
258                    );
259                    return;
260                }
261
262                let mut planes = data.planes.lock().unwrap();
263
264                // Is the index already set?
265                if planes.iter().any(|plane| plane.plane_idx == plane_idx) {
266                    params.post_error(
267                        zwp_linux_buffer_params_v1::Error::PlaneSet,
268                        format!("Plane index {plane_idx} is already set."),
269                    );
270                    return;
271                }
272
273                planes.push(Plane {
274                    fd,
275                    plane_idx,
276                    offset,
277                    stride,
278                });
279
280                let modifier = Modifier::from(((modifier_hi as u64) << 32) + (modifier_lo as u64));
281                let mut data_modifier = data.modifier.lock().unwrap();
282                if let Some(data_modifier) = *data_modifier {
283                    if params.version() >= 5 && modifier != data_modifier {
284                        params.post_error(
285                            zwp_linux_buffer_params_v1::Error::InvalidFormat,
286                            format!("Planes have non-matching modifiers: {modifier:?} != {data_modifier:?}",),
287                        );
288                    }
289                } else {
290                    *data_modifier = Some(modifier);
291                }
292            }
293
294            zwp_linux_buffer_params_v1::Request::Create {
295                width,
296                height,
297                format,
298                flags,
299            } => {
300                // create_dmabuf performs an implicit ensure_unused function call.
301                if let Some(dmabuf) = data.create_dmabuf(params, width, height, format, flags, None) {
302                    if state.dmabuf_state().globals.contains_key(&data.id) {
303                        let notifier = ImportNotifier::new(
304                            params.clone(),
305                            dh.clone(),
306                            dmabuf.clone(),
307                            Import::Falliable,
308                        );
309                        state.dmabuf_imported(&DmabufGlobal { id: data.id }, dmabuf, notifier);
310                    } else {
311                        // If the dmabuf global was destroyed, we cannot import any buffers.
312                        params.failed();
313                    }
314                }
315            }
316
317            zwp_linux_buffer_params_v1::Request::CreateImmed {
318                buffer_id,
319                width,
320                height,
321                format,
322                flags,
323            } => {
324                // Client is killed if the if statement is not taken.
325                // create_dmabuf performs an implicit ensure_unused function call.
326                if let Some(dmabuf) = data.create_dmabuf(params, width, height, format, flags, None) {
327                    if state.dmabuf_state().globals.contains_key(&data.id) {
328                        // The buffer isn't technically valid during data_init, but the client is not allowed to use the buffer until ready.
329                        let buffer = data_init.init(buffer_id, dmabuf.clone());
330                        let notifier = ImportNotifier::new(
331                            params.clone(),
332                            dh.clone(),
333                            dmabuf.clone(),
334                            Import::Infallible(buffer),
335                        );
336                        state.dmabuf_imported(&DmabufGlobal { id: data.id }, dmabuf, notifier);
337                    } else {
338                        // Buffer import failed. The protocol documentation heavily implies killing the
339                        // client is the right thing to do here.
340                        params.post_error(
341                            zwp_linux_buffer_params_v1::Error::InvalidWlBuffer,
342                            "dmabuf global was destroyed on server",
343                        );
344                    }
345                }
346            }
347
348            _ => unreachable!(),
349        }
350    }
