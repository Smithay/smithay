1//! Linux DMABUF protocol
2//!
3//! This module provides helper to handle the linux-dmabuf protocol, which allows clients to submit their
4//! contents as dmabuf file descriptors. These handlers automate the aggregation of the metadata associated
5//! with a dma buffer, and do some basic checking of the sanity of what the client sends.
6//!
7//! ## How to use
8//!
9//! To setup the dmabuf global, you will need to provide 2 things:
10//!
11//! - the default [`DmabufFeedback`] containing the main device and the formats you wish to support when creating the `Global` through [`DmabufState::create_global_with_default_feedback`]
12//! - an implementation of [`DmabufHandler`] to test if a dmabuf buffer can be imported by your renderer and optionally override the initial surface feedback
13//!
14//! The list of supported formats is a `Vec<Format>`, where you will enter all the (code, modifier) pairs you
15//! support. You can typically receive a list of supported formats for one renderer by calling
16//! [`ImportDma::dmabuf_formats`](crate::backend::renderer::ImportDma::dmabuf_formats).
17//!
18//! ```no_run
19//! use smithay::{
20//!     delegate_dmabuf,
21//!     backend::allocator::dmabuf::{Dmabuf},
22//!     reexports::{
23//!         wayland_server::protocol::{
24//!             wl_buffer::WlBuffer,
25//!             wl_surface::WlSurface,
26//!         }
27//!     },
28//!     wayland::{
29//!         buffer::BufferHandler,
30//!         dmabuf::{DmabufFeedback, DmabufFeedbackBuilder, DmabufGlobal, DmabufHandler, DmabufState, ImportNotifier}
31//!     },
32//! };
33//!
34//! pub struct State {
35//!     dmabuf_state: DmabufState,
36//!     dmabuf_global: DmabufGlobal,
37//! }
38//!
39//! // Smithay's "DmabufHandler" also requires the buffer management utilities, you need to implement
40//! // "BufferHandler".
41//! impl BufferHandler for State {
42//!     fn buffer_destroyed(&mut self, buffer: &wayland_server::protocol::wl_buffer::WlBuffer) {
43//!         // All renderers can handle buffer destruction at this point. Some parts of window management may
44//!         // also use this function.
45//!         //
46//!         // If you need to mark a dmabuf elsewhere in your state as destroyed, you use the "get_dmabuf"
47//!         // function defined in this module to access the dmabuf associated the "Buffer".
48//!     }
49//! }
50//!
51//! impl DmabufHandler for State {
52//!     fn dmabuf_state(&mut self) -> &mut DmabufState {
53//!         &mut self.dmabuf_state
54//!     }
55//!
56//!     fn dmabuf_imported(&mut self, global: &DmabufGlobal, dmabuf: Dmabuf, notifier: ImportNotifier) {
57//!         // Here you should import the dmabuf into your renderer.
58//!         //
59//!         // The notifier is used to communicate whether import was successful. In this example we
60//!         // call successful to notify the client import was successful.
61//!         notifier.successful::<State>();
62//!     }
63//!
64//!     fn new_surface_feedback(
65//!         &mut self,
66//!         surface: &WlSurface,
67//!         global: &DmabufGlobal,
68//!     ) -> Option<DmabufFeedback> {
69//!         // Here you can override the initial feedback sent to a client requesting feedback for a specific
70//!         // surface. Returning `None` instructs the global to return the default feedback to the client which
71//!         // is also the default implementation for this function when not overridden
72//!         None
73//!     }
74//! }
75//!
76//! // Delegate dmabuf handling for State to DmabufState.
77//! delegate_dmabuf!(State);
78//!
79//! # let mut display = wayland_server::Display::<State>::new().unwrap();
80//! # let display_handle = display.handle();
81//! // First a DmabufState must be created. This type is used to create some "DmabufGlobal"s
82//! let mut dmabuf_state = DmabufState::new();
83//!
84//! // ...identify primary render node and load dmabuf formats supported for rendering...
85//! # let main_device = { todo!() };
86//! # let formats: Vec<_> = { todo!() };
87//!
88//! // Build the default feedback from the device node of the primary render node and
89//! // the supported dmabuf formats
90//! let default_feedback = DmabufFeedbackBuilder::new(main_device, formats).build().unwrap();
91//!
92//! // And create the dmabuf global.
93//! let dmabuf_global = dmabuf_state.create_global_with_default_feedback::<State>(
94//!     &display_handle,
95//!     &default_feedback,
96//! );
97//!
98//! let state = State {
99//!     dmabuf_state,
100//!     dmabuf_global,
101//! };
102//!
103//! // Rest of the compositor goes here...
104//! ```
105//!
106//! Accessing a [`Dmabuf`] associated with a [`WlBuffer`]
107//! may be achieved using [`get_dmabuf`].
108//!
109//! #### Notes on supporting per surface feedback
110//!
111//! If a client requests [`DmabufFeedback`] for a specific [`WlSurface`] it can be used to inform the client about
112//! sub-optimal buffer allocations. This is especially important to support direct scan-out over drm planes
113//! that typically only support a subset of supported formats from the rendering formats.
114//!
115//! The [`DmabufFeedback`] of a specific [`WlSurface`] can be updated by retrieving the [`SurfaceDmabufFeedbackState`]
116//! with [`SurfaceDmabufFeedbackState::from_states`] and setting the feedback with [`SurfaceDmabufFeedbackState::set_feedback`].
117//!
118//! [`DmabufFeedback`] uses preference tranches to inform the client about formats that could result on more optimal buffer placement.
119//! Preference tranches can be added to the feedback during initialization with [`DmabufFeedbackBuilder::add_preference_tranche`].
120//! Note that the order of formats within a tranche (`target_device` + `flags`) is undefined, if you want to communicate preference
121//! of a specific format you have to split the formats into multiple tranches. A tranche can additionally define [`TrancheFlags`](zwp_linux_dmabuf_feedback_v1::TrancheFlags)
122//! which can give clients additional context what the tranche represents. As an example formats gathered from drm planes
123//! should define [`TrancheFlags::Scanout`](`zwp_linux_dmabuf_feedback_v1::TrancheFlags::Scanout) to communicate that buffers should be allocated so that
124//! they support scan-out by the device specified as the `target device`.
125//!
126//! Note: Surface feedback only represents an optimization and the fallback path using compositing should always be supported, so
127//! typically you do not want to announce formats in a preference tranche that are not supported by the main device for rendering.
128//!
129//! #### Notes on clients binding version 3 or lower
130//!
131//! During instantiation the global will automatically build a format list from the provided [`DmabufFeedback`] consisting of all formats that are part of a tranche
132//! having the `target device` equal the `main device` and defining no special [`TrancheFlags`](zwp_linux_dmabuf_feedback_v1::TrancheFlags).
133//!
134//! ### Without feedback (v3)
135//!
136//! It is also possible to initialize the `Global` without support for [`DmabufFeedback`] by using [`DmabufState::create_global`] which
137//! will then only advertise version `3` to clients. This is mostly meant to guarantee an easy update path for compositors already
138//! supporting dmabuf version `3` without breakage.
139//!
140//! ```no_run
141//! # extern crate wayland_server;
142//! # use smithay::{
143//! #     delegate_dmabuf,
144//! #     backend::allocator::dmabuf::Dmabuf,
145//! #     reexports::{wayland_server::protocol::wl_buffer::WlBuffer},
146//! #     wayland::{
147//! #         buffer::BufferHandler,
148//! #         dmabuf::{DmabufGlobal, DmabufHandler, DmabufState, ImportNotifier}
149//! #     },
150//! # };
151//! # pub struct State {
152//! #     dmabuf_state: DmabufState,
153//! #     dmabuf_global: DmabufGlobal,
154//! # }
155//! # impl BufferHandler for State {
156//! #     fn buffer_destroyed(&mut self, buffer: &wayland_server::protocol::wl_buffer::WlBuffer) { }
157//! # }
158//! # impl DmabufHandler for State {
159//! #     fn dmabuf_state(&mut self) -> &mut DmabufState {
160//! #         &mut self.dmabuf_state
161//! #     }
162//! #     fn dmabuf_imported(&mut self, global: &DmabufGlobal, dmabuf: Dmabuf, notifier: ImportNotifier) {}
163//! # }
164//! # delegate_dmabuf!(State);
165//! # let mut display = wayland_server::Display::<State>::new().unwrap();
166//! # let display_handle = display.handle();
167//! # let mut dmabuf_state = DmabufState::new();
168//! #
169//! // define your supported formats
170//! let formats = vec![
171//!     /* ... */
172//! ];
173//!
174//! // And create the dmabuf global.
175//! let dmabuf_global = dmabuf_state.create_global::<State>(
176//!     &display_handle,
177//!     formats,
178//! );
179//!
180//! let state = State {
181//!     dmabuf_state,
182//!     dmabuf_global,
183//! };
184//!
185//! // Rest of the compositor goes here...
186//! ```

272impl DmabufFeedbackFormat {
273    fn to_ne_bytes(self) -> [u8; 16] {
274        let format: [u8; 4] = self.format.to_ne_bytes();
275        let reserved: [u8; 4] = self._reserved.to_ne_bytes();
276        let modifier: [u8; 8] = self.modifier.to_ne_bytes();
277
278        [
279            format[0],
280            format[1],
281            format[2],
282            format[3],
283            reserved[0],
284            reserved[1],
285            reserved[2],
286            reserved[3],
287            modifier[0],
288            modifier[1],
289            modifier[2],
290            modifier[3],
291            modifier[4],
292            modifier[5],
293            modifier[6],
294            modifier[7],
295        ]
296    }
297}

310impl DmabufFeedbackBuilder {
311    /// Create a new feedback builder with the specified device and formats as the main device
312    ///
313    /// Preference tranches can be added with [`DmabufFeedbackBuilder::add_preference_tranche`]
314    /// and the main tranche will be put after all preference tranches
315    pub fn new(main_device: libc::dev_t, formats: impl IntoIterator<Item = Format>) -> Self {
316        let feedback_formats: IndexSet<Format> = formats.into_iter().collect();
317        let format_indices: IndexSet<usize> = (0..feedback_formats.len()).collect();
318        let main_tranche = DmabufFeedbackTranche {
319            flags: zwp_linux_dmabuf_feedback_v1::TrancheFlags::empty(),
320            indices: format_indices,
321            target_device: main_device,
322        };
323
324        Self {
325            main_device,
326            formats: feedback_formats,
327            main_tranche,
328            preferred_tranches: Vec::new(),
329        }
330    }
331
332    /// Adds a preference tranche to the builder
333    ///
334    /// The tranches will be reported in the order they have been added with
335    /// this function.
336    ///
337    /// Note: Formats already present in a previously added preference tranche with the
338    /// same target device and flags will be skipped.
339    /// If all formats are already included in a previously added tranche
340    /// with the same target device and flags this tranche will be skipped.
341    pub fn add_preference_tranche(
342        mut self,
343        target_device: libc::dev_t,
344        flags: Option<zwp_linux_dmabuf_feedback_v1::TrancheFlags>,
345        formats: impl IntoIterator<Item = Format>,
346    ) -> Self {
347        let flags = flags.unwrap_or(zwp_linux_dmabuf_feedback_v1::TrancheFlags::empty());
348
349        let mut tranche = DmabufFeedbackTranche {
350            target_device,
351            flags,
352            indices: Default::default(),
353        };
354
355        for format in formats {
356            let (format_index, added) = self.formats.insert_full(format);
357
358            // Compositors must not send duplicate format + modifier pairs within
359            // the same tranche or across two different tranches with the same target device and flags.
360            //
361            // if the format has just been added there is no need to test if a previous tranche
362            // with the same target device and flags already contains the format
363            let duplicate_format = !added
364                && self.preferred_tranches.iter().any(|tranche| {
365                    tranche.target_device == target_device
366                        && tranche.flags == flags
367                        && tranche.indices.contains(&format_index)
368                });
369
370            if duplicate_format {
371                continue;
372            }
373
374            // ...Compositors must not send duplicate format + modifier pairs within the same tranche...
375            // this is handled by using a IndexSet which won't hold duplicates
376            tranche.indices.insert(format_index);
377        }
378
379        if !tranche.indices.is_empty() {
380            self.preferred_tranches.push(tranche);
381        }
382
383        self
384    }
385
386    /// Build the [`DmabufFeedback`]
387    ///
388    /// Returns an error if the format table shared memory file could
389    /// not be created.
390    pub fn build(mut self) -> Result<DmabufFeedback, std::io::Error> {
391        let formats = self
392            .formats
393            .iter()
394            .copied()
395            .map(DmabufFeedbackFormat::from)
396            .flat_map(DmabufFeedbackFormat::to_ne_bytes)
397            .collect::<Vec<_>>();
398
399        let name = c"smithay-dmabuffeedback-format-table";
400        let format_table_file = SealedFile::with_data(name, &formats)?;
401
402        // remove all formats from the main tranche that are already covered
403        // by a preference tranche
404        for duplicate_main_tranche in self.preferred_tranches.iter().filter(|tranche| {
405            tranche.target_device == self.main_tranche.target_device
406                && tranche.flags == self.main_tranche.flags
407        }) {
408            self.main_tranche.indices = self.main_tranche.indices.sub(&duplicate_main_tranche.indices);
409        }
410
411        if !self.main_tranche.indices.is_empty() {
412            self.preferred_tranches.push(self.main_tranche);
413        }
414
415        Ok(DmabufFeedback(Arc::new(DmabufFeedbackInner {
416            main_device: self.main_device,
417            format_table: DmabufFeedbackFormatTable {
418                file: format_table_file,
419                formats: self.formats,
420            },
421            tranches: self.preferred_tranches,
422        })))
423    }
424}

426/// Feedback for dmabuf allocation
427///
428/// Use the [`DmabufFeedbackBuilder`] to create a new instance.
429#[derive(Debug, Clone)]
430pub struct DmabufFeedback(Arc<DmabufFeedbackInner>);
431
432impl PartialEq for DmabufFeedback {
433    #[inline]
434    fn eq(&self, other: &Self) -> bool {
435        // Note: The dmabuf feedback can not change, it is initialized
436        // with the DmabufFeedbackBuilder, so if the arc ptr equal we
437        // can short-circuit the expensive equality test saving us
438        // a few cpu cycles
439        if Arc::ptr_eq(&self.0, &other.0) {
440            return true;
441        }
442
443        self.0 == other.0
444    }
445}
446
447impl DmabufFeedback {
448    /// Send this feedback to the provided [`ZwpLinuxDmabufFeedbackV1`](zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1)
449    pub fn send(&self, feedback: &zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1) {
450        feedback.main_device(self.0.main_device.to_ne_bytes().to_vec());
451        feedback.format_table(
452            self.0.format_table.file.as_fd(),
453            self.0.format_table.file.size() as u32,
454        );
455
456        for tranche in self.0.tranches.iter() {
457            feedback.tranche_target_device(tranche.target_device.to_ne_bytes().to_vec());
458            feedback.tranche_flags(tranche.flags);
459            feedback.tranche_formats(
460                tranche
461                    .indices
462                    .iter()
463                    .flat_map(|i| (*i as u16).to_ne_bytes())
464                    .collect::<Vec<_>>(),
465            );
466            feedback.tranche_done();
467        }
468
469        feedback.done();
470    }
471
472    fn main_formats(&self) -> Vec<Format> {
473        self.0
474            .tranches
475            .iter()
476            .filter(|tranche| tranche.target_device == self.0.main_device && tranche.flags.is_empty())
477            .map(|tranche| tranche.indices.clone())
478            .reduce(|mut acc, item| {
479                acc.extend(item);
480                acc
481            })
482            .unwrap_or_default()
483            .into_iter()
484            .map(|index| self.0.format_table.formats[index])
485            .collect()
486    }
487}

495/// Feedback state for a surface
496#[derive(Debug, Clone)]
497pub struct SurfaceDmabufFeedbackState {
498    inner: Arc<Mutex<SurfaceDmabufFeedbackStateInner>>,
499}
500
501impl SurfaceDmabufFeedbackState {
502    /// Get the surface dmabuf feedback stored in the surface states
503    ///
504    /// Returns `None` if no feedback has been requested
505    pub fn from_states(states: &compositor::SurfaceData) -> Option<&Self> {
506        states.data_map.get::<SurfaceDmabufFeedbackState>()
507    }
508
509    /// Set the feedback for this surface
510    ///
511    /// Note: If the feedback equals the current feedback this function does nothing
512    pub fn set_feedback(&self, feedback: &DmabufFeedback) {
513        let mut guard = self.inner.lock().unwrap();
514        if &guard.feedback == feedback {
515            return;
516        }
517
518        for instance in guard.known_instances.iter().filter_map(|i| i.upgrade().ok()) {
519            feedback.send(&instance);
520        }
521
522        guard.feedback = feedback.clone();
523    }
524
525    fn add_instance(
526        &self,
527        instance: &zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1,
528    ) -> DmabufFeedback {
529        let mut guard = self.inner.lock().unwrap();
530        guard.known_instances.push(instance.downgrade());
531        guard.feedback.clone()
532    }
533
534    fn remove_instance(&self, instance: &zwp_linux_dmabuf_feedback_v1::ZwpLinuxDmabufFeedbackV1) {
535        self.inner
536            .lock()
537            .unwrap()
538            .known_instances
539            .retain(|i| i != instance);
540    }
541}

570    /// Creates a dmabuf global with the specified supported formats.
571    ///
572    /// Note: This function will create a version 3 dmabuf global and thus not call [`DmabufHandler::new_surface_feedback`],
573    /// if you want to create a version 4 global you need to call [`DmabufState::create_global_with_default_feedback`].
574    pub fn create_global<D>(
575        &mut self,
576        display: &DisplayHandle,
577        formats: impl IntoIterator<Item = Format>,
578    ) -> DmabufGlobal
579    where
580        D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
581            + BufferHandler
582            + DmabufHandler
583            + 'static,
584    {
585        self.create_global_with_filter::<D, _>(display, formats, |_| true)
586    }

588    /// Creates a dmabuf global with the specified supported formats.
589    ///
590    /// This function unlike [`DmabufState::create_global`] also allows you to specify a filter function to
591    /// determine which clients may see this global. This functionality may be used on multi-gpu systems in
592    /// order to make a client choose the correct gpu.
593    ///
594    /// Note: This function will create a version 3 dmabuf global and thus not call [`DmabufHandler::new_surface_feedback`],
595    /// if you want to create a version 4 global you need to call [`DmabufState::create_global_with_filter_and_default_feedback`]
596    pub fn create_global_with_filter<D, F>(
597        &mut self,
598        display: &DisplayHandle,
599        formats: impl IntoIterator<Item = Format>,
600        filter: F,
601    ) -> DmabufGlobal
602    where
603        D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
604            + BufferHandler
605            + DmabufHandler
606            + 'static,
607        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
608    {
609        let formats = formats.into_iter().collect::<Vec<_>>();
610        self.create_global_with_filter_and_optional_default_feedback::<D, _>(
611            display,
612            Some(formats),
613            None,
614            filter,
615        )
616    }

618    /// Creates a dmabuf global with the specified default feedback.
619    ///
620    /// Clients binding to version 3 or lower will receive the formats from the main tranche.
621    pub fn create_global_with_default_feedback<D>(
622        &mut self,
623        display: &DisplayHandle,
624        default_feedback: &DmabufFeedback,
625    ) -> DmabufGlobal
626    where
627        D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
628            + BufferHandler
629            + DmabufHandler
630            + 'static,
631    {
632        self.create_global_with_filter_and_default_feedback::<D, _>(display, default_feedback, |_| true)
633    }
634
635    /// Creates a dmabuf global with the specified supported formats and default feedback
636    ///
637    /// This function unlike [`DmabufState::create_global_with_default_feedback`] also allows you to specify a filter function to
638    /// determine which clients may see this global. This functionality may be used on multi-gpu systems in
639    /// order to make a client choose the correct gpu.
640    ///
641    /// Clients binding to version 3 or lower will receive the formats from the main tranche.
642    pub fn create_global_with_filter_and_default_feedback<D, F>(
643        &mut self,
644        display: &DisplayHandle,
645        default_feedback: &DmabufFeedback,
646        filter: F,
647    ) -> DmabufGlobal
648    where
649        D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
650            + BufferHandler
651            + DmabufHandler
652            + 'static,
653        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
654    {
655        self.create_global_with_filter_and_optional_default_feedback::<D, _>(
656            display,
657            None,
658            Some(default_feedback),
659            filter,
660        )
661    }
662
663    fn create_global_with_filter_and_optional_default_feedback<D, F>(
664        &mut self,
665        display: &DisplayHandle,
666        formats: Option<Vec<Format>>,
667        default_feedback: Option<&DmabufFeedback>,
668        filter: F,
669    ) -> DmabufGlobal
670    where
671        D: GlobalDispatch<zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, DmabufGlobalData>
672            + BufferHandler
673            + DmabufHandler
674            + 'static,
675        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
676    {
677        let id = global_id::next();
678
679        let formats = formats
680            .or_else(|| default_feedback.map(|f| f.main_formats()))
681            .unwrap()
682            .into_iter()
683            .fold(
684                IndexMap::<Fourcc, IndexSet<Modifier>>::new(),
685                |mut formats, format| {
686                    if let Some(modifiers) = formats.get_mut(&format.code) {
687                        modifiers.insert(format.modifier);
688                    } else {
689                        formats.insert(format.code, IndexSet::from_iter(std::iter::once(format.modifier)));
690                    }
691                    formats
692                },
693            );
694
695        let formats = Arc::new(formats);
696        let version = if default_feedback.is_some() { 5 } else { 3 };
697
698        let known_default_feedbacks = Arc::new(Mutex::new(Vec::new()));
699        let default_feedback = default_feedback.map(|f| Arc::new(Mutex::new(f.clone())));
700
701        let data = DmabufGlobalData {
702            filter: Box::new(filter),
703            formats,
704            default_feedback: default_feedback.clone(),
705            known_default_feedbacks: known_default_feedbacks.clone(),
706            id,
707        };
708
709        let global = display.create_global::<D, zwp_linux_dmabuf_v1::ZwpLinuxDmabufV1, _>(version, data);
710        self.globals.insert(
711            id,
712            DmabufGlobalState {
713                id: global,
714                default_feedback,
715                known_default_feedbacks,
716            },
717        );
718
719        DmabufGlobal { id }
720    }
721
722    /// Set the default [`DmabufFeedback`] for the specified global and send it to
723    /// all known default feedbacks.
724    ///
725    /// Note: This will do nothing if the global is not found, the global has been
726    /// initialized without feedback or the feedback equals the current default feedback.
727    pub fn set_default_feedback(&self, global: &DmabufGlobal, default_feedback: &DmabufFeedback) {
728        let Some(global) = self.globals.get(&global.id) else {
729            return;
730        };
731
732        let Some(mut current_feedback) = global.default_feedback.as_ref().map(|f| f.lock().unwrap()) else {
733            return;
734        };
735
736        // No need to update if the feedback did not change
737        if &*current_feedback == default_feedback {
738            return;
739        }
740
741        let known_default_feedbacks = global.known_default_feedbacks.lock().unwrap();
742        for feedback in known_default_feedbacks.iter().filter_map(|f| f.upgrade().ok()) {
743            default_feedback.send(&feedback);
744        }
745
746        *current_feedback = default_feedback.clone();
747    }
748
749    /// Disables a dmabuf global.
750    ///
751    /// This operation is permanent and there is no way to re-enable a global.
752    pub fn disable_global<D: 'static>(&mut self, display: &DisplayHandle, global: &DmabufGlobal) {
753        if let Some(global_state) = self.globals.get(&global.id) {
754            display.disable_global::<D>(global_state.id.clone());
755        }
756    }
757
758    /// Destroys a dmabuf global.
759    ///
760    /// It is highly recommended you disable the global before destroying it and ensure all child objects have
761    /// been destroyed.
762    pub fn destroy_global<D: 'static>(&mut self, display: &DisplayHandle, global: DmabufGlobal) {
763        if global_id::remove(global.id) {
764            if let Some(global_state) = self.globals.remove(&global.id) {
765                display.remove_global::<D>(global_state.id);
766            }
767        }
768    }
769}

825/// An object to allow asynchronous creation of a [`Dmabuf`] backed [`WlBuffer`].
826///
827/// This object is [`Send`] to allow import of a [`Dmabuf`] to take place on another thread if desired.
828#[must_use = "This object must be used to notify the client whether dmabuf import succeeded"]
829#[derive(Debug)]
830pub struct ImportNotifier {
831    inner: ZwpLinuxBufferParamsV1,
832    display: DisplayHandle,
833    dmabuf: Dmabuf,
834    import: Import,
835    drop_ignore: bool,
836}
837
838/// Type of dmabuf import.
839#[derive(Debug)]
840enum Import {
841    /// The import can fail or create a WlBuffer.
842    Falliable,
843
844    /// A WlBuffer object has already been created. Failure causes client death.
845    Infallible(WlBuffer),
846}
847
848impl ImportNotifier {
849    /// Returns the client trying to import this dmabuf, if not dead.
850    pub fn client(&self) -> Option<Client> {
851        self.inner.client()
852    }
853
854    /// Dmabuf import was successful.
855    ///
856    /// This can return [`InvalidId`] if the client the buffer was imported from has died.
857    pub fn successful<D>(mut self) -> Result<WlBuffer, InvalidId>
858    where
859        D: Dispatch<zwp_linux_buffer_params_v1::ZwpLinuxBufferParamsV1, DmabufParamsData>
860            + Dispatch<wl_buffer::WlBuffer, Dmabuf>
861            + BufferHandler
862            + DmabufHandler
863            + 'static,
864    {
865        let client = self.inner.client();
866
867        let result = match self.import {
868            Import::Falliable => {
869                if let Some(client) = client {
870                    match client.create_resource::<wl_buffer::WlBuffer, Dmabuf, D>(
871                        &self.display,
872                        1,
873                        self.dmabuf.clone(),
874                    ) {
875                        Ok(buffer) => {
876                            self.inner.created(&buffer);
877                            Ok(buffer)
878                        }
879
880                        Err(err) => {
881                            tracing::error!("failed to create protocol object for \"create\" request");
882                            Err(err)
883                        }
884                    }
885                } else {
886                    tracing::error!("client was dead while creating wl_buffer resource");
887                    self.inner.post_error(
888                        zwp_linux_buffer_params_v1::Error::InvalidWlBuffer,
889                        "create_immed failed and produced an invalid wl_buffer",
890                    );
891                    Err(InvalidId)
892                }
893            }
894            Import::Infallible(ref buffer) => Ok(buffer.clone()),
895        };
896
897        self.drop_ignore = true;
898        result
899    }
900
901    /// The buffer being imported is incomplete.
902    ///
903    /// This may be the result of too few or too many planes being used when creating a buffer.
904    pub fn incomplete(mut self) {
905        self.inner.post_error(
906            zwp_linux_buffer_params_v1::Error::Incomplete,
907            "missing or too many planes to create a buffer",
908        );
909        self.drop_ignore = true;
910    }
911
912    /// The buffer being imported has an invalid width or height.
913    pub fn invalid_dimensions(mut self) {
914        self.inner.post_error(
915            zwp_linux_buffer_params_v1::Error::InvalidDimensions,
916            "width or height of dmabuf is invalid",
917        );
918        self.drop_ignore = true;
919    }
920
921    /// Import failed due to an invalid format and plane combination.
922    ///
923    /// This is always a client error and will result in the client being killed.
924    pub fn invalid_format(mut self) {
925        self.inner.post_error(
926            zwp_linux_buffer_params_v1::Error::InvalidFormat,
927            "format and plane combination are not valid",
928        );
929        self.drop_ignore = true;
930    }
931
932    /// Import failed for an implementation dependent reason.
933    pub fn failed(mut self) {
934        if matches!(self.import, Import::Falliable) {
935            self.inner.failed();
936        } else {
937            self.inner.post_error(
938                zwp_linux_buffer_params_v1::Error::InvalidWlBuffer,
939                "create_immed failed and produced an invalid wl_buffer",
940            );
941        }
942        self.drop_ignore = true;
943    }
944
945    fn new(params: ZwpLinuxBufferParamsV1, display: DisplayHandle, dmabuf: Dmabuf, import: Import) -> Self {
946        Self {
947            inner: params,
948            display,
949            dmabuf,
950            import,
951            drop_ignore: false,
952        }
953    }
954}
955
956impl Drop for ImportNotifier {
957    fn drop(&mut self) {
958        if !self.drop_ignore {
959            tracing::warn!(
960                "Compositor bug: Server ignored ImportNotifier for {:?}",
961                self.inner
962            );
963        }
964    }
965}

1056impl DmabufParamsData {
1057    /// Emits a protocol error if the params have already been used to create a dmabuf.
1058    ///
1059    /// This returns true if the protocol object has not been used.
1060    fn ensure_unused(&self, params: &ZwpLinuxBufferParamsV1) -> bool {
1061        if !self.used.load(Ordering::Relaxed) {
1062            return true;
1063        }
1064
1065        params.post_error(
1066            zwp_linux_buffer_params_v1::Error::AlreadyUsed,
1067            "This buffer_params has already been used to create a buffer.",
1068        );
1069
1070        false
1071    }
1072
1073    /// Attempt to create a Dmabuf from the parameters.
1074    ///
1075    /// This function will perform the necessary validation of all the parameters, emitting protocol errors as
1076    /// needed.
1077    ///
1078    /// A return value of [`None`] indicates buffer import has failed and the client has been killed.
1079    fn create_dmabuf(
1080        &self,
1081        params: &ZwpLinuxBufferParamsV1,
1082        width: i32,
1083        height: i32,
1084        format: u32,
1085        flags: WEnum<zwp_linux_buffer_params_v1::Flags>,
1086        _node: Option<libc::dev_t>,
1087    ) -> Option<Dmabuf> {
1088        // We cannot create a dmabuf if the parameters have already been used.
1089        if !self.ensure_unused(params) {
1090            return None;
1091        }
1092
1093        self.used.store(true, Ordering::Relaxed);
1094
1095        let format = match Fourcc::try_from(format) {
1096            Ok(format) => format,
1097            Err(_) => {
1098                params.post_error(
1099                    zwp_linux_buffer_params_v1::Error::InvalidFormat,
1100                    format!("Format {format:x} is not supported"),
1101                );
1102
1103                return None;
1104            }
1105        };
1106
1107        // Validate buffer parameters:
1108        // 1. Must have known format
1109        if !self.formats.contains_key(&format) {
1110            params.post_error(
1111                zwp_linux_buffer_params_v1::Error::InvalidFormat,
1112                format!("Format {format:?}/{:x} is not supported.", format as u32),
1113            );
1114            return None;
1115        }
1116
1117        // 2. Width and height must be positive
1118        if width < 1 {
1119            params.post_error(
1120                zwp_linux_buffer_params_v1::Error::InvalidDimensions,
1121                "invalid width",
1122            );
1123        }
1124
1125        if height < 1 {
1126            params.post_error(
1127                zwp_linux_buffer_params_v1::Error::InvalidDimensions,
1128                "invalid height",
1129            );
1130        }
1131
1132        // 3. Validate all the planes
1133        let mut planes = self.planes.lock().unwrap();
1134
1135        for plane in &*planes {
1136            // Must not overflow
1137            let end = match plane
1138                .stride
1139                .checked_mul(height as u32)
1140                .and_then(|o| o.checked_add(plane.offset))
1141            {
1142                Some(e) => e,
1143
1144                None => {
1145                    params.post_error(
1146                        zwp_linux_buffer_params_v1::Error::OutOfBounds,
1147                        format!("Size overflow for plane {}.", plane.plane_idx),
1148                    );
1149
1150                    return None;
1151                }
1152            };
1153
1154            if let Ok(size) = seek(&plane.fd, SeekFrom::End(0)) {
1155                // Reset seek point
1156                let _ = seek(&plane.fd, SeekFrom::Start(0));
1157
1158                if plane.offset as u64 > size {
1159                    params.post_error(
1160                        zwp_linux_buffer_params_v1::Error::OutOfBounds,
1161                        format!("Invalid offset {} for plane {}.", plane.offset, plane.plane_idx),
1162                    );
1163
1164                    return None;
1165                }
1166
1167                if (plane.offset + plane.stride) as u64 > size {
1168                    params.post_error(
1169                        zwp_linux_buffer_params_v1::Error::OutOfBounds,
1170                        format!("Invalid stride {} for plane {}.", plane.stride, plane.plane_idx),
1171                    );
1172
1173                    return None;
1174                }
1175
1176                // Planes > 0 can be subsampled, in which case 'size' will be smaller than expected.
1177                if plane.plane_idx == 0 && end as u64 > size {
1178                    params.post_error(
1179                        zwp_linux_buffer_params_v1::Error::OutOfBounds,
1180                        format!(
1181                            "Invalid stride ({}) or height ({}) for plane {}.",
1182                            plane.stride, height, plane.plane_idx
1183                        ),
1184                    );
1185
1186                    return None;
1187                }
1188            }
1189        }
1190
1191        let modifier = self.modifier.lock().unwrap().unwrap_or(Modifier::Invalid);
1192        let mut buf = Dmabuf::builder(
1193            (width, height),
1194            format,
1195            modifier,
1196            DmabufFlags::from_bits_truncate(flags.into()),
1197        );
1198
1199        for (i, plane) in planes.drain(..).enumerate() {
1200            let offset = plane.offset;
1201            let stride = plane.stride;
1202            buf.add_plane(plane.into(), i as u32, offset, stride);
1203        }
1204
1205        #[cfg(feature = "backend_drm")]
1206        if let Some(node) = _node.and_then(|node| DrmNode::from_dev_id(node).ok()) {
1207            buf.set_node(node);
1208        }
1209
1210        let dmabuf = match buf.build() {
1211            Some(buf) => buf,
1212
1213            None => {
1214                params.post_error(
1215                    zwp_linux_buffer_params_v1::Error::Incomplete as u32,
1216                    "Provided buffer is incomplete, it has zero planes",
1217                );
1218                return None;
1219            }
1220        };
1221
1222        Some(dmabuf)
1223    }
1224}
