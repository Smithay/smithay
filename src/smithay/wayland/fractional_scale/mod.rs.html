1//! Utilities for handling the `wp_fractional_scale` protocol
2//!
3//! ## How to use it
4//!
5//! ### Initialization
6//!
7//! To initialize this implementation create the [`FractionalScaleManagerState`], store it inside your `State` struct
8//! and implement the [`FractionalScaleHandler`], as shown in this example:
9//!
10//! ```
11//! use smithay::delegate_fractional_scale;
12//! use smithay::wayland::compositor;
13//! use smithay::reexports::wayland_server::protocol::wl_surface;
14//! use smithay::wayland::fractional_scale::{
15//!     self,
16//!     FractionalScaleManagerState,
17//!     FractionalScaleHandler,
18//! };
19//!
20//! # struct State { fractional_scale_manager_state: FractionalScaleManagerState }
21//! # let mut display = wayland_server::Display::<State>::new().unwrap();
22//! // Create the compositor state
23//! let fractional_scale_manager_state = FractionalScaleManagerState::new::<State>(
24//!     &display.handle(),
25//! );
26//!
27//! // insert the FractionalScaleManagerState into your state
28//! // ..
29//!
30//! // implement the necessary traits
31//! impl FractionalScaleHandler for State {
32//!    fn new_fractional_scale(&mut self, surface: wl_surface::WlSurface) {
33//!        // If you set preferred fractional scale for this surface before,
34//!        // then you don't need to do anything here.
35//!    }
36//! }
37//! delegate_fractional_scale!(State);
38//!
39//! // You're now ready to go!
40//! ```
41//!
42//! ### Use the fractional scale state
43//!
44//! Whenever the fractional scale for a surface changes set the preferred
45//! fractional scale like shown in the example:
46//!
47//! ```no_run
48//! # use wayland_server::{backend::ObjectId, protocol::wl_surface, Resource};
49//! use smithay::wayland::compositor;
50//! use smithay::wayland::fractional_scale;
51//! # struct State;
52//! # let mut display = wayland_server::Display::<State>::new().unwrap();
53//! # let dh = display.handle();
54//! # let surface = wl_surface::WlSurface::from_id(&dh, ObjectId::null()).unwrap();
55//! compositor::with_states(&surface, |states| {
56//!     fractional_scale::with_fractional_scale(states, |fractional_scale| {
57//!         // set the preferred scale for the surface
58//!         fractional_scale.set_preferred_scale(1.25);
59//!     });
60//! })
61//! ```

122impl<D> Dispatch<wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1, (), D>
123    for FractionalScaleManagerState
124where
125    D: GlobalDispatch<wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1, ()>
126        + Dispatch<wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1, ()>
127        + Dispatch<wp_fractional_scale_v1::WpFractionalScaleV1, Weak<wl_surface::WlSurface>>,
128    D: FractionalScaleHandler,
129{
130    fn request(
131        state: &mut D,
132        _client: &wayland_server::Client,
133        _resource: &wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1,
134        request: <wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1 as Resource>::Request,
135        _data: &(),
136        _dhandle: &DisplayHandle,
137        data_init: &mut wayland_server::DataInit<'_, D>,
138    ) {
139        match request {
140            wp_fractional_scale_manager_v1::Request::Destroy => {
141                // All is already handled by our destructor
142            }
143            wp_fractional_scale_manager_v1::Request::GetFractionalScale { id, surface } => {
144                let already_has_fractional_scale = with_states(&surface, |states| {
145                    states
146                        .data_map
147                        .get::<FractionalScaleStateUserData>()
148                        .map(|v| v.borrow().fractional_scale.is_some())
149                        .unwrap_or(false)
150                });
151
152                if already_has_fractional_scale {
153                    surface.post_error(
154                        wp_fractional_scale_manager_v1::Error::FractionalScaleExists as u32,
155                        "the surface already has a fractional_scale object associated".to_string(),
156                    );
157                    return;
158                }
159
160                let fractional_scale: wp_fractional_scale_v1::WpFractionalScaleV1 =
161                    data_init.init(id, surface.downgrade());
162
163                with_states(&surface, move |states| {
164                    with_fractional_scale(states, move |data| {
165                        // Send the scale that the user may have pre-filled for this surface.
166                        if let Some(scale) = data.preferred_scale {
167                            fractional_scale.preferred_scale(f64::round(scale * 120.0) as u32);
168                        }
169                        data.fractional_scale = Some(fractional_scale);
170                    });
171                });
172                state.new_fractional_scale(surface);
173            }
174            _ => unreachable!(),
175        }
176    }
177}

179impl<D> Dispatch<wp_fractional_scale_v1::WpFractionalScaleV1, Weak<wl_surface::WlSurface>, D>
180    for FractionalScaleManagerState
181where
182    D: GlobalDispatch<wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1, ()>
183        + Dispatch<wp_fractional_scale_manager_v1::WpFractionalScaleManagerV1, ()>
184        + Dispatch<wp_fractional_scale_v1::WpFractionalScaleV1, Weak<wl_surface::WlSurface>>,
185    D: FractionalScaleHandler,
186{
187    fn request(
188        _state: &mut D,
189        _client: &wayland_server::Client,
190        _resource: &wp_fractional_scale_v1::WpFractionalScaleV1,
191        request: <wp_fractional_scale_v1::WpFractionalScaleV1 as Resource>::Request,
192        data: &Weak<wl_surface::WlSurface>,
193        _dhandle: &DisplayHandle,
194        _data_init: &mut wayland_server::DataInit<'_, D>,
195    ) {
196        match request {
197            wp_fractional_scale_v1::Request::Destroy => {
198                if let Ok(surface) = data.upgrade() {
199                    with_states(&surface, |states| {
200                        states
201                            .data_map
202                            .get::<FractionalScaleStateUserData>()
203                            .and_then(|v| v.borrow_mut().fractional_scale.take());
204                    })
205                }
206            }
207            _ => unreachable!(),
208        }
209    }
210}

212/// Fractional scale handler type
213pub trait FractionalScaleHandler {
214    /// A new fractional scale was instantiated
215    fn new_fractional_scale(&mut self, _surface: wl_surface::WlSurface) {}
216}

218/// Type stored in WlSurface states data_map
219///
220/// ```rs
221/// compositor::with_states(surface, |states| {
222///     let data = states.data_map.get::<FractionalScaleStateUserData>();
223/// });
224/// ```
225pub type FractionalScaleStateUserData = RefCell<FractionalScaleState>;

227/// State for the fractional scale of a surface
228#[derive(Debug, Default)]
229pub struct FractionalScaleState {
230    /// The fractional scale object, if one exists for this surface.
231    fractional_scale: Option<wp_fractional_scale_v1::WpFractionalScaleV1>,
232    /// Preferred fractional scale for this surface.
233    preferred_scale: Option<f64>,
234}
235
236impl FractionalScaleState {
237    /// Set the preferred scale
238    pub fn set_preferred_scale(&mut self, scale: f64) {
239        if self.preferred_scale != Some(scale) {
240            if let Some(obj) = &self.fractional_scale {
241                obj.preferred_scale(f64::round(scale * 120.0) as u32);
242            }
243            self.preferred_scale = Some(scale);
244        }
245    }
246
247    /// Returns the current preferred scale
248    pub fn preferred_scale(&self) -> Option<f64> {
249        self.preferred_scale
250    }
251}

253/// Run a closure on the [`FractionalScaleState`] of a [`WlSurface`](wl_surface::WlSurface)
254pub fn with_fractional_scale<F, T>(states: &SurfaceData, f: F) -> T
255where
256    F: FnOnce(&mut FractionalScaleState) -> T,
257{
258    let mut fractional_scale = states
259        .data_map
260        .get_or_insert(FractionalScaleStateUserData::default)
261        .borrow_mut();
262
263    f(&mut fractional_scale)
264}
