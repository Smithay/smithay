1//! This interface allows clients to monitor user idle status.
2//!
3//! ```
4//! # extern crate wayland_server;
5//! # #[macro_use] extern crate smithay;
6//! use smithay::delegate_idle_notify;
7//! use smithay::wayland::idle_notify::{IdleNotifierState, IdleNotifierHandler};
8//! # use smithay::input::{Seat, SeatHandler, SeatState, pointer::CursorImageStatus};
9//! # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
10//!
11//! struct State { idle_notifier: IdleNotifierState<Self> }
12//! # let mut event_loop = smithay::reexports::calloop::EventLoop::<State>::try_new().unwrap();
13//! # let mut display = wayland_server::Display::<State>::new().unwrap();
14//! // Create the idle_notifier state
15//! let idle_notifier = IdleNotifierState::<State>::new(
16//!     &display.handle(),
17//!     event_loop.handle(),
18//! );
19//!
20//! let state = State { idle_notifier };
21//!
22//! // Implement the necessary trait
23//! # impl SeatHandler for State {
24//! #     type KeyboardFocus = WlSurface;
25//! #     type PointerFocus = WlSurface;
26//! #     type TouchFocus = WlSurface;
27//! #     fn seat_state(&mut self) -> &mut SeatState<Self> { unimplemented!() }
28//! #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) { unimplemented!() }
29//! #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) { unimplemented!() }
30//! # }
31//! impl IdleNotifierHandler for State {
32//!     fn idle_notifier_state(&mut self) -> &mut IdleNotifierState<Self> {
33//!         &mut self.idle_notifier
34//!     }
35//! }
36//! delegate_idle_notify!(State);
37//!
38//! // On input you should notify the idle_notifier
39//! // state.idle_notifier.notify_activity(&seat);
40//! ```

64/// Handler trait for ext-idle-notify
65pub trait IdleNotifierHandler: Sized {
66    /// [`IdleNotifierState`] getter
67    fn idle_notifier_state(&mut self) -> &mut IdleNotifierState<Self>;
68}

70/// User data of the [`ExtIdleNotificationV1`] resource
71#[derive(Debug)]
72pub struct IdleNotificationUserData {
73    seat: WlSeat,
74    is_idle: AtomicBool,
75    timeout: Duration,
76    timer_token: Mutex<Option<RegistrationToken>>,
77
78    /// If listener was created with `get_input_idle_notification`
79    ignore_inhibitor: bool,
80}
81
82impl IdleNotificationUserData {
83    fn take_timer_token(&self) -> Option<RegistrationToken> {
84        self.timer_token.lock().unwrap().take()
85    }
86
87    fn set_timer_token(&self, idle: Option<RegistrationToken>) {
88        *self.timer_token.lock().unwrap() = idle;
89    }
90
91    fn set_idle(&self, idle: bool) {
92        self.is_idle.store(idle, atomic::Ordering::Release);
93    }
94
95    fn is_idle(&self) -> bool {
96        self.is_idle.load(atomic::Ordering::Acquire)
97    }
98}

100/// State of ext-idle-notify module
101#[derive(Debug)]
102pub struct IdleNotifierState<D> {
103    global: GlobalId,
104    notifications: HashMap<WlSeat, Vec<ExtIdleNotificationV1>>,
105    loop_handle: LoopHandle<'static, D>,
106    is_inhibited: bool,
107}
108
109impl<D: IdleNotifierHandler> IdleNotifierState<D> {
110    /// Create new [`ExtIdleNotifierV1`] global.
111    pub fn new(display: &DisplayHandle, loop_handle: LoopHandle<'static, D>) -> Self
112    where
113        D: GlobalDispatch<ExtIdleNotifierV1, ()>,
114        D: Dispatch<ExtIdleNotifierV1, ()>,
115        D: Dispatch<ExtIdleNotificationV1, IdleNotificationUserData>,
116        D: IdleNotifierHandler,
117        D: 'static,
118    {
119        let global = display.create_global::<D, ExtIdleNotifierV1, _>(2, ());
120        Self {
121            global,
122            notifications: HashMap::new(),
123            loop_handle,
124            is_inhibited: false,
125        }
126    }
127
128    /// Inhibit entering idle state, eg. by the idle-inhibit protocol
129    pub fn set_is_inhibited(&mut self, is_inhibited: bool) {
130        if self.is_inhibited == is_inhibited {
131            return;
132        }
133
134        self.is_inhibited = is_inhibited;
135
136        for notification in self.notifications() {
137            let data = notification.data::<IdleNotificationUserData>().unwrap();
138
139            if data.ignore_inhibitor {
140                continue;
141            }
142
143            if is_inhibited {
144                if data.is_idle() {
145                    notification.resumed();
146                    data.set_idle(false);
147                }
148
149                if let Some(token) = data.take_timer_token() {
150                    self.loop_handle.remove(token);
151                }
152            } else {
153                self.reinsert_timer(notification);
154            }
155        }
156    }
157
158    /// Is idle state inhibited, eg. by the idle-inhibit protocol
159    pub fn is_inhibited(&mut self) -> bool {
160        self.is_inhibited
161    }
162
163    /// Should be called whenever activity occurs on a seat, eg. mouse/keyboard input.
164    ///
165    /// You may want to use [`Self::notify_activity`] instead which accepts a [`Seat`].
166    pub fn notify_activity_for_wl_seat(&mut self, seat: &WlSeat) {
167        let Some(notifications) = self.notifications.get(seat) else {
168            return;
169        };
170
171        for notification in notifications {
172            let data = notification.data::<IdleNotificationUserData>().unwrap();
173
174            if data.is_idle() {
175                notification.resumed();
176                data.set_idle(false);
177            }
178
179            self.reinsert_timer(notification);
180        }
181    }
182
183    /// Returns the [`ExtIdleNotifierV1`] global.
184    pub fn global(&self) -> GlobalId {
185        self.global.clone()
186    }
187
188    fn notifications(&self) -> impl Iterator<Item = &ExtIdleNotificationV1> {
189        self.notifications.values().flatten()
190    }
191
192    fn reinsert_timer(&self, notification: &ExtIdleNotificationV1) {
193        let data = notification.data::<IdleNotificationUserData>().unwrap();
194
195        if let Some(token) = data.take_timer_token() {
196            self.loop_handle.remove(token);
197        }
198
199        if !data.ignore_inhibitor && self.is_inhibited {
200            return;
201        }
202
203        let token = self
204            .loop_handle
205            .insert_source(calloop::timer::Timer::from_duration(data.timeout), {
206                let idle_notification = notification.clone();
207                move |_, _, state| {
208                    let data = idle_notification.data::<IdleNotificationUserData>().unwrap();
209
210                    let is_inhibited = !data.ignore_inhibitor && state.idle_notifier_state().is_inhibited;
211                    let is_idle_already = data.is_idle();
212
213                    if !is_inhibited && !is_idle_already {
214                        idle_notification.idled();
215                        data.set_idle(true);
216                    }
217
218                    data.set_timer_token(None);
219                    TimeoutAction::Drop
220                }
221            });
222
223        data.set_timer_token(token.ok());
224    }
225}

227impl<D: IdleNotifierHandler + SeatHandler> IdleNotifierState<D> {
228    /// Should be called whenever activity occurs on a seat, eg. mouse/keyboard input.
229    pub fn notify_activity(&mut self, seat: &Seat<D>) {
230        for seat in &seat.arc.inner.lock().unwrap().known_seats {
231            if let Ok(seat) = seat.upgrade() {
232                self.notify_activity_for_wl_seat(&seat);
233            }
234        }
235    }
236}

258impl<D> Dispatch<ExtIdleNotifierV1, (), D> for IdleNotifierState<D>
259where
260    D: GlobalDispatch<ExtIdleNotifierV1, ()>,
261    D: Dispatch<ExtIdleNotifierV1, ()>,
262    D: Dispatch<ExtIdleNotificationV1, IdleNotificationUserData>,
263    D: IdleNotifierHandler,
264    D: 'static,
265{
266    fn request(
267        state: &mut D,
268        _client: &Client,
269        _resource: &ExtIdleNotifierV1,
270        request: ext_idle_notifier_v1::Request,
271        _data: &(),
272        _dhandle: &DisplayHandle,
273        data_init: &mut DataInit<'_, D>,
274    ) {
275        match request {
276            ext_idle_notifier_v1::Request::GetIdleNotification { id, timeout, seat } => {
277                let timeout = Duration::from_millis(timeout as u64);
278
279                let idle_notifier_state = state.idle_notifier_state();
280
281                let idle_notification = data_init.init(
282                    id,
283                    IdleNotificationUserData {
284                        seat: seat.clone(),
285                        is_idle: AtomicBool::new(false),
286                        timeout,
287                        timer_token: Mutex::new(None),
288                        ignore_inhibitor: false,
289                    },
290                );
291
292                idle_notifier_state.reinsert_timer(&idle_notification);
293
294                state
295                    .idle_notifier_state()
296                    .notifications
297                    .entry(seat)
298                    .or_default()
299                    .push(idle_notification);
300            }
301            ext_idle_notifier_v1::Request::GetInputIdleNotification { id, timeout, seat } => {
302                let timeout = Duration::from_millis(timeout as u64);
303
304                let idle_notifier_state = state.idle_notifier_state();
305
306                let idle_notification = data_init.init(
307                    id,
308                    IdleNotificationUserData {
309                        seat: seat.clone(),
310                        is_idle: AtomicBool::new(false),
311                        timeout,
312                        timer_token: Mutex::new(None),
313                        ignore_inhibitor: true,
314                    },
315                );
316
317                idle_notifier_state.reinsert_timer(&idle_notification);
318
319                state
320                    .idle_notifier_state()
321                    .notifications
322                    .entry(seat)
323                    .or_default()
324                    .push(idle_notification);
325            }
326            ext_idle_notifier_v1::Request::Destroy => {}
327            _ => unimplemented!(),
328        }
329    }
330}

332impl<D> Dispatch<ExtIdleNotificationV1, IdleNotificationUserData, D> for IdleNotifierState<D>
333where
334    D: Dispatch<ExtIdleNotificationV1, IdleNotificationUserData>,
335    D: IdleNotifierHandler,
336{
337    fn request(
338        _state: &mut D,
339        _client: &Client,
340        _resource: &ExtIdleNotificationV1,
341        request: ext_idle_notification_v1::Request,
342        _data: &IdleNotificationUserData,
343        _dhandle: &DisplayHandle,
344        _data_init: &mut DataInit<'_, D>,
345    ) {
346        match request {
347            ext_idle_notification_v1::Request::Destroy => {}
348            _ => unimplemented!(),
349        }
350    }
351
352    fn destroyed(
353        state: &mut D,
354        _client: ClientId,
355        notification: &ExtIdleNotificationV1,
356        data: &IdleNotificationUserData,
357    ) {
358        let state = state.idle_notifier_state();
359        if let Some(notifications) = state.notifications.get_mut(&data.seat) {
360            notifications.retain(|x| x != notification);
361        }
362
363        state
364            .notifications
365            .retain(|seat, notifications| !notifications.is_empty() && seat.is_alive());
366    }
367}

369/// Macro to delegate implementation of the ext idle notify protocol
370#[macro_export]
371macro_rules! delegate_idle_notify {
372    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
373        const _: () = {
374            use $crate::{
375                reexports::{
376                    wayland_protocols::ext::idle_notify::v1::server::{
377                        ext_idle_notification_v1::ExtIdleNotificationV1,
378                        ext_idle_notifier_v1::ExtIdleNotifierV1,
379                    },
380                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
381                },
382                wayland::idle_notify::{IdleNotificationUserData, IdleNotifierState},
383            };
384
385            delegate_global_dispatch!(
386                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
387                $ty: [ExtIdleNotifierV1: ()] => IdleNotifierState<$ty>
388            );
389
390            delegate_dispatch!(
391                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
392                $ty: [ExtIdleNotifierV1: ()] => IdleNotifierState<$ty>
393            );
394
395            delegate_dispatch!(
396                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
397                $ty: [ExtIdleNotificationV1: IdleNotificationUserData] => IdleNotifierState<$ty>
398            );
399        };
400    };
