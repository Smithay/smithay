1//! Utilities for graphics tablet support
2//!
3//! This module provides helpers to handle graphics tablets.
4//!
5//! ```
6//! use smithay::{delegate_seat, delegate_tablet_manager};
7//! # use smithay::delegate_compositor;
8//! use smithay::backend::input::TabletToolDescriptor;
9//! use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus};
10//! # use smithay::wayland::compositor::{CompositorHandler, CompositorState, CompositorClientState};
11//! use smithay::wayland::tablet_manager::{TabletManagerState, TabletDescriptor, TabletSeatHandler};
12//! use smithay::reexports::wayland_server::{Display, protocol::wl_surface::WlSurface};
13//! # use smithay::reexports::wayland_server::Client;
14//!
15//! # struct State { seat_state: SeatState<Self> };
16//! # let mut display = Display::<State>::new().unwrap();
17//! # let display_handle = display.handle();
18//!
19//! let mut seat_state = SeatState::<State>::new();
20//! let tablet_state = TabletManagerState::new::<State>(&display_handle);
21//! // add the seat state to your state
22//! // ...
23//!
24//! // create the seat
25//! let seat = seat_state.new_wl_seat(
26//!     &display_handle,          // the display
27//!     "seat-0",                 // the name of the seat, will be advertized to clients
28//! );
29//!
30//! use smithay::wayland::tablet_manager::TabletSeatTrait;
31//!
32//! seat
33//!    .tablet_seat()                     // Get TabletSeat asosiated with this seat
34//!    .add_tablet::<State>(              // Add a new tablet to a seat
35//!      &display_handle,
36//!      &TabletDescriptor {    
37//!        name: "Test".into(),
38//!        usb_id: None,
39//!        syspath: None,
40//!      }
41//!    );
42//!
43//! // implement the required traits
44//! impl SeatHandler for State {
45//!     type KeyboardFocus = WlSurface;
46//!     type PointerFocus = WlSurface;
47//!     type TouchFocus = WlSurface;
48//!     fn seat_state(&mut self) -> &mut SeatState<Self> {
49//!         &mut self.seat_state
50//!     }
51//!     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&WlSurface>) {
52//!         // ...
53//!     }
54//!     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) {
55//!         // ...
56//!     }

62//!         // ...

65//! delegate_tablet_manager!(State);

67//! # impl CompositorHandler for State {

120/// Extends [Seat] with graphic tablet specific functionality
121pub trait TabletSeatTrait {
122    /// Get tablet seat associated with this seat
123    fn tablet_seat(&self) -> TabletSeatHandle;
124}
125
126impl<D: SeatHandler + 'static> TabletSeatTrait for Seat<D> {
127    fn tablet_seat(&self) -> TabletSeatHandle {
128        let user_data = self.user_data();
129        user_data.insert_if_missing(TabletSeatHandle::default);
130        user_data.get::<TabletSeatHandle>().unwrap().clone()
131    }
132}

134/// State of wp tablet protocol
135#[derive(Debug)]
136pub struct TabletManagerState {
137    global: GlobalId,
138}
139
140impl TabletManagerState {
141    /// Initialize a tablet manager global.
142    pub fn new<D>(display: &DisplayHandle) -> Self
143    where
144        D: GlobalDispatch<ZwpTabletManagerV2, ()>,
145        D: Dispatch<ZwpTabletManagerV2, ()>,
146        D: Dispatch<ZwpTabletSeatV2, TabletSeatUserData>,
147        D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
148        D: 'static,
149    {
150        let global = display.create_global::<D, ZwpTabletManagerV2, _>(MANAGER_VERSION, ());
151
152        Self { global }
153    }
154
155    /// Get the id of ZwpTabletManagerV2 global
156    pub fn global(&self) -> GlobalId {
157        self.global.clone()
158    }
159}

161impl<D> GlobalDispatch<ZwpTabletManagerV2, (), D> for TabletManagerState
162where
163    D: GlobalDispatch<ZwpTabletManagerV2, ()>,
164    D: Dispatch<ZwpTabletManagerV2, ()>,
165    D: Dispatch<ZwpTabletSeatV2, TabletSeatUserData>,
166    D: 'static,
167{
168    fn bind(
169        _: &mut D,
170        _: &DisplayHandle,
171        _: &Client,
172        resource: New<ZwpTabletManagerV2>,
173        _: &(),
174        data_init: &mut DataInit<'_, D>,
175    ) {
176        data_init.init(resource, ());
177    }
178}
179
180impl<D> Dispatch<ZwpTabletManagerV2, (), D> for TabletManagerState
181where
182    D: Dispatch<ZwpTabletManagerV2, ()>,
183    D: Dispatch<ZwpTabletSeatV2, TabletSeatUserData>,
184    D: Dispatch<ZwpTabletV2, TabletUserData>,
185    D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
186    D: SeatHandler + TabletSeatHandler + 'static,
187    D: CompositorHandler,
188{
189    fn request(
190        state: &mut D,
191        client: &Client,
192        _: &ZwpTabletManagerV2,
193        request: zwp_tablet_manager_v2::Request,
194        _: &(),
195        dh: &DisplayHandle,
196        data_init: &mut DataInit<'_, D>,
197    ) {
198        match request {
199            zwp_tablet_manager_v2::Request::GetTabletSeat { tablet_seat, seat } => {
200                let seat = Seat::<D>::from_resource(&seat).unwrap();
201
202                let user_data = seat.user_data();
203                user_data.insert_if_missing(TabletSeatHandle::default);
204
205                let handle = user_data.get::<TabletSeatHandle>().unwrap();
206                let instance = data_init.init(
207                    tablet_seat,
208                    TabletSeatUserData {
209                        handle: handle.clone(),
210                    },
211                );
212
213                handle.add_instance::<D>(state, dh, &instance, client);
214            }
215            zwp_tablet_manager_v2::Request::Destroy => {
216                // Nothing to do
217            }
218            _ => unreachable!(),
219        }
220    }
221}
