32    pending_pressure: Option<f64>,
33    pending_distance: Option<f64>,
34    pending_tilt: Option<(f64, f64)>,

40impl TabletTool {
41    fn proximity_in(
42        &mut self,
43        loc: Point<f64, Logical>,
44        (focus, sloc): (WlSurface, Point<f64, Logical>),
45        tablet: &TabletHandle,
46        serial: Serial,
47        time: u32,
48    ) {
49        let wl_tool = self.instances.iter().find(|i| i.id().same_client_as(&focus.id()));
50
51        if let Some(wl_tool) = wl_tool.and_then(|tool| tool.upgrade().ok()) {
52            tablet.with_focused_tablet(&focus, |wl_tablet| {
53                wl_tool.proximity_in(serial.into(), wl_tablet, &focus);
54                // proximity_in has to be followed by motion event (required by protocol)
55                let client_scale = wl_tool
56                    .data::<TabletToolUserData>()
57                    .unwrap()
58                    .client_scale
59                    .load(Ordering::Acquire);
60                let srel_loc = (loc - sloc).to_client(client_scale);
61                wl_tool.motion(srel_loc.x, srel_loc.y);
62                wl_tool.frame(time);
63            });
64        }
65
66        self.focus = Some(focus.clone());
67    }
68
69    fn proximity_out(&mut self, time: u32) {
70        if let Some(ref focus) = self.focus {
71            let wl_tool = self.instances.iter().find(|i| i.id().same_client_as(&focus.id()));
72
73            if let Some(wl_tool) = wl_tool.and_then(|tool| tool.upgrade().ok()) {
74                if self.is_down {
75                    wl_tool.pressure(0);
76                    wl_tool.up();
77                    self.is_down = false;
78                }
79                wl_tool.proximity_out();
80                wl_tool.frame(time);
81            }
82        }
83
84        self.focus = None;
85    }
86
87    fn tip_down(&mut self, serial: Serial, time: u32) {
88        if let Some(ref focus) = self.focus {
89            if let Some(wl_tool) = self
90                .instances
91                .iter()
92                .find(|i| i.id().same_client_as(&focus.id()))
93                .and_then(|tool| tool.upgrade().ok())
94            {
95                if !self.is_down {
96                    wl_tool.down(serial.into());
97                    wl_tool.frame(time);
98                }
99            }
100        }
101
102        self.is_down = true;
103    }
104
105    fn tip_up(&mut self, time: u32) {
106        if let Some(ref focus) = self.focus {
107            if let Some(wl_tool) = self
108                .instances
109                .iter()
110                .find(|i| i.id().same_client_as(&focus.id()))
111                .and_then(|tool| tool.upgrade().ok())
112            {
113                if self.is_down {
114                    wl_tool.pressure(0);
115                    wl_tool.up();
116                    wl_tool.frame(time);
117                }
118            }
119        }
120
121        self.is_down = false;
122    }
123
124    fn motion(
125        &mut self,
126        pos: Point<f64, Logical>,
127        focus: Option<(WlSurface, Point<f64, Logical>)>,
128        tablet: &TabletHandle,
129        serial: Serial,
130        time: u32,
131    ) {
132        match (focus, self.focus.as_ref()) {
133            (Some(focus), Some(prev_focus)) => {
134                if &focus.0 == prev_focus {
135                    if let Some(wl_tool) = self
136                        .instances
137                        .iter()
138                        .find(|i| i.id().same_client_as(&focus.0.id()))
139                        .and_then(|tool| tool.upgrade().ok())
140                    {
141                        let client_scale = wl_tool
142                            .data::<TabletToolUserData>()
143                            .unwrap()
144                            .client_scale
145                            .load(Ordering::Acquire);
146                        let srel_loc = (pos - focus.1).to_client(client_scale);
147                        wl_tool.motion(srel_loc.x, srel_loc.y);
148
149                        if let Some(pressure) = self.pending_pressure.take() {
150                            wl_tool.pressure((pressure * 65535.0).round() as u32);
151                        }
152
153                        if let Some(distance) = self.pending_distance.take() {
154                            wl_tool.distance((distance * 65535.0).round() as u32);
155                        }
156
157                        if let Some((x, y)) = self.pending_tilt.take() {
158                            wl_tool.tilt(x, y);
159                        }
160
161                        if let Some(slider) = self.pending_slider.take() {
162                            wl_tool.slider((slider * 65535.0).round() as i32);
163                        }
164
165                        if let Some(rotation) = self.pending_rotation.take() {
166                            wl_tool.rotation(rotation);
167                        }
168
169                        if let Some((degrees, clicks)) = self.pending_wheel.take() {
170                            wl_tool.wheel(degrees, clicks)
171                        }
172
173                        wl_tool.frame(time);
174                    }
175                } else {
176                    // If surface has changed
177
178                    // Unfocus previous surface
179                    self.proximity_out(time);
180                    // Focus a new one
181                    self.proximity_in(pos, focus, tablet, serial, time)
182                }
183            }
184            // New surface in focus
185            (Some(focus), None) => self.proximity_in(pos, focus, tablet, serial, time),
186            // No surface in focus
187            (None, _) => self.proximity_out(time),
188        }
189    }
190
191    fn pressure(&mut self, pressure: f64) {
192        self.pending_pressure = Some(pressure);
193    }
194
195    fn distance(&mut self, distance: f64) {
196        self.pending_distance = Some(distance);
197    }
198
199    fn tilt(&mut self, tilt: (f64, f64)) {
200        self.pending_tilt = Some(tilt);
201    }
202
203    fn rotation(&mut self, rotation: f64) {
204        self.pending_rotation = Some(rotation);
205    }
206
207    fn slider_position(&mut self, slider: f64) {
208        self.pending_slider = Some(slider);
209    }
210
211    fn wheel(&mut self, degrees: f64, clicks: i32) {
212        self.pending_wheel = Some((degrees, clicks));
213    }
214
215    /// Sent whenever a button on the tool is pressed or released.
216    fn button(&self, button: u32, state: ButtonState, serial: Serial, time: u32) {
217        if let Some(ref focus) = self.focus {
218            if let Some(wl_tool) = self
219                .instances
220                .iter()
221                .find(|i| i.id().same_client_as(&focus.id()))
222                .and_then(|tool| tool.upgrade().ok())
223            {
224                wl_tool.button(serial.into(), button, state.into());
225                wl_tool.frame(time);
226            }
227        }
228    }
229}

231impl Drop for TabletTool {
232    fn drop(&mut self) {
233        for instance in self.instances.iter().filter_map(|tool| tool.upgrade().ok()) {
234            // This event is sent when the tool is removed from the system and will send no further events.
235            instance.removed();
236        }
237    }

240/// Handle to a tablet tool device
241///
242/// TabletTool represents a physical tool that has been, or is currently in use with a tablet in seat.
243///
244/// A TabletTool relation to a physical tool depends on the tablet's ability to report serial numbers. If the tablet supports this capability, then the object represents a specific physical tool and can be identified even when used on multiple tablets.
245#[derive(Debug, Default, Clone)]
246pub struct TabletToolHandle {
247    pub(crate) inner: Arc<Mutex<TabletTool>>,
248}

250impl TabletToolHandle {
251    pub(super) fn new_instance<D>(
252        &mut self,
253        state: &mut D,
254        client: &Client,
255        dh: &DisplayHandle,
256        seat: &ZwpTabletSeatV2,
257        tool: &TabletToolDescriptor,
258    ) where
259        D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
260        D: TabletSeatHandler + 'static,
261        D: CompositorHandler,
262    {
263        let desc = tool.clone();
264
265        let client_scale = state.client_compositor_state(client).clone_client_scale();
266        let wl_tool = client
267            .create_resource::<ZwpTabletToolV2, _, D>(
268                dh,
269                seat.version(),
270                TabletToolUserData {
271                    handle: self.clone(),
272                    desc,
273                    client_scale,
274                },
275            )
276            .unwrap();
277
278        seat.tool_added(&wl_tool);
279
280        wl_tool._type(tool.tool_type.into());
281
282        let high: u32 = (tool.hardware_serial >> 16) as u32;
283        let low: u32 = tool.hardware_serial as u32;
284
285        wl_tool.hardware_serial(high, low);
286
287        let high: u32 = (tool.hardware_id_wacom >> 16) as u32;
288        let low: u32 = tool.hardware_id_wacom as u32;
289        wl_tool.hardware_id_wacom(high, low);
290
291        if tool.capabilities.contains(TabletToolCapabilities::PRESSURE) {
292            wl_tool.capability(zwp_tablet_tool_v2::Capability::Pressure);
293        }
294
295        if tool.capabilities.contains(TabletToolCapabilities::DISTANCE) {
296            wl_tool.capability(zwp_tablet_tool_v2::Capability::Distance);
297        }
298
299        if tool.capabilities.contains(TabletToolCapabilities::TILT) {
300            wl_tool.capability(zwp_tablet_tool_v2::Capability::Tilt);
301        }
302
303        if tool.capabilities.contains(TabletToolCapabilities::SLIDER) {
304            wl_tool.capability(zwp_tablet_tool_v2::Capability::Slider);
305        }
306
307        if tool.capabilities.contains(TabletToolCapabilities::ROTATION) {
308            wl_tool.capability(zwp_tablet_tool_v2::Capability::Rotation);
309        }
310
311        if tool.capabilities.contains(TabletToolCapabilities::WHEEL) {
312            wl_tool.capability(zwp_tablet_tool_v2::Capability::Wheel);
313        }
314
315        wl_tool.done();
316        self.inner.lock().unwrap().instances.push(wl_tool.downgrade());
317    }

471impl<D> Dispatch<ZwpTabletToolV2, TabletToolUserData, D> for TabletManagerState
472where
473    D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
474    D: TabletSeatHandler + 'static,
475{
476    fn request(
477        state: &mut D,
478        _client: &Client,
479        tool: &ZwpTabletToolV2,
480        request: zwp_tablet_tool_v2::Request,
481        data: &TabletToolUserData,
482        _dh: &DisplayHandle,
483        _data_init: &mut DataInit<'_, D>,
484    ) {
485        match request {
486            zwp_tablet_tool_v2::Request::SetCursor {
487                surface,
488                hotspot_x,
489                hotspot_y,
490                ..
491            } => {
492                let focus = data.handle.inner.lock().unwrap().focus.clone();
493
494                if let Some(focus) = focus {
495                    if focus.id().same_client_as(&tool.id()) {
496                        if let Some(surface) = surface {
497                            // tolerate re-using the same surface
498                            if compositor::give_role(&surface, CURSOR_IMAGE_ROLE).is_err()
499                                && compositor::get_role(&surface) != Some(CURSOR_IMAGE_ROLE)
500                            {
501                                tool.post_error(
502                                    zwp_tablet_tool_v2::Error::Role,
503                                    "Given wl_surface has another role.",
504                                );
505                                return;
506                            }
507
508                            compositor::with_states(&surface, |states| {
509                                states.data_map.insert_if_missing_threadsafe(|| {
510                                    Mutex::new(CursorImageAttributes {
511                                        hotspot: (0, 0).into(),
512                                    })
513                                });
514                                let client_scale = tool
515                                    .data::<TabletToolUserData>()
516                                    .unwrap()
517                                    .client_scale
518                                    .load(Ordering::Acquire);
519                                let hotspot = Point::<_, ClientCoords>::from((hotspot_x, hotspot_y))
520                                    .to_f64()
521                                    .to_logical(client_scale)
522                                    .to_i32_round();
523                                states
524                                    .data_map
525                                    .get::<Mutex<CursorImageAttributes>>()
526                                    .unwrap()
527                                    .lock()
528                                    .unwrap()
529                                    .hotspot = hotspot;
530                            });
531
532                            state.tablet_tool_image(&data.desc, CursorImageStatus::Surface(surface));
533                        } else {
534                            state.tablet_tool_image(&data.desc, CursorImageStatus::Hidden);
535                        };
536                    }
537                }
538            }
539            zwp_tablet_tool_v2::Request::Destroy => {
540                // Nothing to do
541            }
542            _ => unreachable!(),
543        }
544    }
545
546    fn destroyed(_state: &mut D, _client: ClientId, resource: &ZwpTabletToolV2, data: &TabletToolUserData) {
547        data.handle
548            .inner
549            .lock()
550            .unwrap()
551            .instances
552            .retain(|i| i.id() != resource.id());
553    }
