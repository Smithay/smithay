26pub(crate) struct TabletTool {

28    pub(crate) focus: Option<WlSurface>,

30    is_down: bool,
31
32    pending_pressure: Option<f64>,
33    pending_distance: Option<f64>,
34    pending_tilt: Option<(f64, f64)>,
35    pending_slider: Option<f64>,
36    pending_rotation: Option<f64>,

40impl TabletTool {
41    fn proximity_in(
42        &mut self,
43        loc: Point<f64, Logical>,
44        (focus, sloc): (WlSurface, Point<f64, Logical>),
45        tablet: &TabletHandle,
46        serial: Serial,
47        time: u32,
48    ) {
49        let wl_tool = self.instances.iter().find(|i| i.id().same_client_as(&focus.id()));
50
51        if let Some(wl_tool) = wl_tool.and_then(|tool| tool.upgrade().ok()) {
52            tablet.with_focused_tablet(&focus, |wl_tablet| {
53                wl_tool.proximity_in(serial.into(), wl_tablet, &focus);
54                // proximity_in has to be followed by motion event (required by protocol)
55                let client_scale = wl_tool
56                    .data::<TabletToolUserData>()
57                    .unwrap()
58                    .client_scale
59                    .load(Ordering::Acquire);
60                let srel_loc = (loc - sloc).to_client(client_scale);
61                wl_tool.motion(srel_loc.x, srel_loc.y);
62                wl_tool.frame(time);
63            });
64        }
65
66        self.focus = Some(focus.clone());
67    }
68
69    fn proximity_out(&mut self, time: u32) {
70        if let Some(ref focus) = self.focus {
71            let wl_tool = self.instances.iter().find(|i| i.id().same_client_as(&focus.id()));
72
73            if let Some(wl_tool) = wl_tool.and_then(|tool| tool.upgrade().ok()) {
74                if self.is_down {
75                    wl_tool.up();
76                    self.is_down = false;
77                }
78                wl_tool.proximity_out();
79                wl_tool.frame(time);
80            }
81        }
82
83        self.focus = None;
84    }
85
86    fn tip_down(&mut self, serial: Serial, time: u32) {
87        if let Some(ref focus) = self.focus {
88            if let Some(wl_tool) = self
89                .instances
90                .iter()
91                .find(|i| i.id().same_client_as(&focus.id()))
92                .and_then(|tool| tool.upgrade().ok())
93            {
94                if !self.is_down {
95                    wl_tool.down(serial.into());
96                    wl_tool.frame(time);
97                }
98            }
99        }
100
101        self.is_down = true;
102    }
103
104    fn tip_up(&mut self, time: u32) {
105        if let Some(ref focus) = self.focus {
106            if let Some(wl_tool) = self
107                .instances
108                .iter()
109                .find(|i| i.id().same_client_as(&focus.id()))
110                .and_then(|tool| tool.upgrade().ok())
111            {
112                if self.is_down {
113                    wl_tool.up();
114                    wl_tool.frame(time);
115                }
116            }
117        }
118
119        self.is_down = false;
120    }
121
122    fn motion(
123        &mut self,
124        pos: Point<f64, Logical>,
125        focus: Option<(WlSurface, Point<f64, Logical>)>,
126        tablet: &TabletHandle,
127        serial: Serial,
128        time: u32,
129    ) {
130        match (focus, self.focus.as_ref()) {
131            (Some(focus), Some(prev_focus)) => {
132                if &focus.0 == prev_focus {
133                    if let Some(wl_tool) = self
134                        .instances
135                        .iter()
136                        .find(|i| i.id().same_client_as(&focus.0.id()))
137                        .and_then(|tool| tool.upgrade().ok())
138                    {
139                        let client_scale = wl_tool
140                            .data::<TabletToolUserData>()
141                            .unwrap()
142                            .client_scale
143                            .load(Ordering::Acquire);
144                        let srel_loc = (pos - focus.1).to_client(client_scale);
145                        wl_tool.motion(srel_loc.x, srel_loc.y);
146
147                        if let Some(pressure) = self.pending_pressure.take() {
148                            wl_tool.pressure((pressure * 65535.0).round() as u32);
149                        }
150
151                        if let Some(distance) = self.pending_distance.take() {
152                            wl_tool.distance((distance * 65535.0).round() as u32);
153                        }
154
155                        if let Some((x, y)) = self.pending_tilt.take() {
156                            wl_tool.tilt(x, y);
157                        }
158
159                        if let Some(slider) = self.pending_slider.take() {
160                            wl_tool.slider((slider * 65535.0).round() as i32);
161                        }
162
163                        if let Some(rotation) = self.pending_rotation.take() {
164                            wl_tool.rotation(rotation);
165                        }
166
167                        if let Some((degrees, clicks)) = self.pending_wheel.take() {
168                            wl_tool.wheel(degrees, clicks)
169                        }
170
171                        wl_tool.frame(time);
172                    }
173                } else {
174                    // If surface has changed
175
176                    // Unfocus previous surface
177                    self.proximity_out(time);
178                    // Focuss a new one
179                    self.proximity_in(pos, focus, tablet, serial, time)
180                }
181            }
182            // New surface in focus
183            (Some(focus), None) => self.proximity_in(pos, focus, tablet, serial, time),
184            // No surface in focus
185            (None, _) => self.proximity_out(time),
186        }
187    }
188
189    fn pressure(&mut self, pressure: f64) {
190        self.pending_pressure = Some(pressure);
191    }
192
193    fn distance(&mut self, distance: f64) {
194        self.pending_distance = Some(distance);
195    }
196
197    fn tilt(&mut self, tilt: (f64, f64)) {
198        self.pending_tilt = Some(tilt);
199    }
200
201    fn rotation(&mut self, rotation: f64) {
202        self.pending_rotation = Some(rotation);
203    }
204
205    fn slider_position(&mut self, slider: f64) {
206        self.pending_slider = Some(slider);
207    }
208
209    fn wheel(&mut self, degrees: f64, clicks: i32) {
210        self.pending_wheel = Some((degrees, clicks));
211    }
212
213    /// Sent whenever a button on the tool is pressed or released.
214    fn button(&self, button: u32, state: ButtonState, serial: Serial, time: u32) {
215        if let Some(ref focus) = self.focus {
216            if let Some(wl_tool) = self
217                .instances
218                .iter()
219                .find(|i| i.id().same_client_as(&focus.id()))
220                .and_then(|tool| tool.upgrade().ok())
221            {
222                wl_tool.button(serial.into(), button, state.into());
223                wl_tool.frame(time);
224            }
225        }
226    }
227}

229impl Drop for TabletTool {
230    fn drop(&mut self) {
231        for instance in self.instances.iter().filter_map(|tool| tool.upgrade().ok()) {
232            // This event is sent when the tool is removed from the system and will send no further events.
233            instance.removed();
234        }
235    }

238/// Handle to a tablet tool device
239///
240/// TabletTool represents a physical tool that has been, or is currently in use with a tablet in seat.
241///
242/// A TabletTool relation to a physical tool depends on the tablet's ability to report serial numbers. If the tablet supports this capability, then the object represents a specific physical tool and can be identified even when used on multiple tablets.
243#[derive(Debug, Default, Clone)]
244pub struct TabletToolHandle {
245    pub(crate) inner: Arc<Mutex<TabletTool>>,
246}

248impl TabletToolHandle {
249    pub(super) fn new_instance<D>(
250        &mut self,
251        state: &mut D,
252        client: &Client,
253        dh: &DisplayHandle,
254        seat: &ZwpTabletSeatV2,
255        tool: &TabletToolDescriptor,
256    ) where
257        D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
258        D: TabletSeatHandler + 'static,
259        D: CompositorHandler,
260    {
261        let desc = tool.clone();
262
263        let client_scale = state.client_compositor_state(client).clone_client_scale();
264        let wl_tool = client
265            .create_resource::<ZwpTabletToolV2, _, D>(
266                dh,
267                seat.version(),
268                TabletToolUserData {
269                    handle: self.clone(),
270                    desc,
271                    client_scale,
272                },
273            )
274            .unwrap();
275
276        seat.tool_added(&wl_tool);
277
278        wl_tool._type(tool.tool_type.into());
279
280        let high: u32 = (tool.hardware_serial >> 16) as u32;
281        let low: u32 = tool.hardware_serial as u32;
282
283        wl_tool.hardware_serial(high, low);
284
285        let high: u32 = (tool.hardware_id_wacom >> 16) as u32;
286        let low: u32 = tool.hardware_id_wacom as u32;
287        wl_tool.hardware_id_wacom(high, low);
288
289        if tool.capabilities.contains(TabletToolCapabilities::PRESSURE) {
290            wl_tool.capability(zwp_tablet_tool_v2::Capability::Pressure);
291        }
292
293        if tool.capabilities.contains(TabletToolCapabilities::DISTANCE) {
294            wl_tool.capability(zwp_tablet_tool_v2::Capability::Distance);
295        }
296
297        if tool.capabilities.contains(TabletToolCapabilities::TILT) {
298            wl_tool.capability(zwp_tablet_tool_v2::Capability::Tilt);
299        }
300
301        if tool.capabilities.contains(TabletToolCapabilities::SLIDER) {
302            wl_tool.capability(zwp_tablet_tool_v2::Capability::Slider);
303        }
304
305        if tool.capabilities.contains(TabletToolCapabilities::ROTATION) {
306            wl_tool.capability(zwp_tablet_tool_v2::Capability::Rotation);
307        }
308
309        if tool.capabilities.contains(TabletToolCapabilities::WHEEL) {
310            wl_tool.capability(zwp_tablet_tool_v2::Capability::Wheel);
311        }
312
313        wl_tool.done();
314        self.inner.lock().unwrap().instances.push(wl_tool.downgrade());
315    }

469impl<D> Dispatch<ZwpTabletToolV2, TabletToolUserData, D> for TabletManagerState
470where
471    D: Dispatch<ZwpTabletToolV2, TabletToolUserData>,
472    D: TabletSeatHandler + 'static,
473{
474    fn request(
475        state: &mut D,
476        _client: &Client,
477        tool: &ZwpTabletToolV2,
478        request: zwp_tablet_tool_v2::Request,
479        data: &TabletToolUserData,
480        _dh: &DisplayHandle,
481        _data_init: &mut DataInit<'_, D>,
482    ) {
483        match request {
484            zwp_tablet_tool_v2::Request::SetCursor {
485                surface,
486                hotspot_x,
487                hotspot_y,
488                ..
489            } => {
490                let focus = data.handle.inner.lock().unwrap().focus.clone();
491
492                if let Some(focus) = focus {
493                    if focus.id().same_client_as(&tool.id()) {
494                        if let Some(surface) = surface {
495                            // tolerate re-using the same surface
496                            if compositor::give_role(&surface, CURSOR_IMAGE_ROLE).is_err()
497                                && compositor::get_role(&surface) != Some(CURSOR_IMAGE_ROLE)
498                            {
499                                tool.post_error(
500                                    zwp_tablet_tool_v2::Error::Role,
501                                    "Given wl_surface has another role.",
502                                );
503                                return;
504                            }
505
506                            compositor::with_states(&surface, |states| {
507                                states.data_map.insert_if_missing_threadsafe(|| {
508                                    Mutex::new(CursorImageAttributes {
509                                        hotspot: (0, 0).into(),
510                                    })
511                                });
512                                let client_scale = tool
513                                    .data::<TabletToolUserData>()
514                                    .unwrap()
515                                    .client_scale
516                                    .load(Ordering::Acquire);
517                                let hotspot = Point::<_, ClientCoords>::from((hotspot_x, hotspot_y))
518                                    .to_f64()
519                                    .to_logical(client_scale)
520                                    .to_i32_round();
521                                states
522                                    .data_map
523                                    .get::<Mutex<CursorImageAttributes>>()
524                                    .unwrap()
525                                    .lock()
526                                    .unwrap()
527                                    .hotspot = hotspot;
528                            });
529
530                            state.tablet_tool_image(&data.desc, CursorImageStatus::Surface(surface));
531                        } else {
532                            state.tablet_tool_image(&data.desc, CursorImageStatus::Hidden);
533                        };
534                    }
535                }
536            }
537            zwp_tablet_tool_v2::Request::Destroy => {
538                // Nothing to do
539            }
540            _ => unreachable!(),
541        }
542    }
543
544    fn destroyed(_state: &mut D, _client: ClientId, resource: &ZwpTabletToolV2, data: &TabletToolUserData) {
545        data.handle
546            .inner
547            .lock()
548            .unwrap()
549            .instances
550            .retain(|i| i.id() != resource.id());
551    }
