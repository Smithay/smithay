146/// The role for an xwayland-associated surface.
147pub const XWAYLAND_SHELL_ROLE: &str = "xwayland_shell";
148
149const VERSION: u32 = 1;

151/// The global for the xwayland shell protocol.
152#[derive(Debug, Clone)]
153pub struct XWaylandShellState {
154    global: GlobalId,
155    by_serial: HashMap<u64, WlSurface>,
156}
157
158impl XWaylandShellState {
159    /// Registers a new [XwaylandShellV1] global. Only XWayland clients will be
160    /// able to bind it.
161    pub fn new<D>(display: &DisplayHandle) -> Self
162    where
163        D: GlobalDispatch<XwaylandShellV1, ()>,
164        D: Dispatch<XwaylandShellV1, ()>,
165        D: Dispatch<XwaylandSurfaceV1, XWaylandSurfaceUserData>,
166        D: 'static,
167    {
168        let global = display.create_global::<D, XwaylandShellV1, _>(VERSION, ());
169        Self {
170            global,
171            by_serial: HashMap::new(),
172        }
173    }
174
175    /// Retrieve a handle for the [XwaylandShellV1] global.
176    pub fn global(&self) -> GlobalId {
177        self.global.clone()
178    }
179
180    /// Retrieves the surface for a given serial.
181    pub fn surface_for_serial(&self, serial: u64) -> Option<WlSurface> {
182        self.by_serial.get(&serial).cloned()
183    }
184}

192/// Handler for the xwayland shell protocol.
193pub trait XWaylandShellHandler {
194    /// Retrieves the global state.
195    fn xwayland_shell_state(&mut self) -> &mut XWaylandShellState;
196
197    /// An X11 window has been associated with a wayland surface. This doesn't
198    /// take effect until the wl_surface is committed.
199    fn surface_associated(&mut self, xwm: XwmId, wl_surface: wl_surface::WlSurface, surface: X11Surface) {
200        let _ = (xwm, wl_surface, surface);
201    }
202}

204/// Represents a pending X11 serial, used to associate X11 windows with wayland

244impl<D> Dispatch<XwaylandShellV1, (), D> for XWaylandShellState
245where
246    D: Dispatch<XwaylandShellV1, ()>,
247    D: Dispatch<XwaylandSurfaceV1, XWaylandSurfaceUserData>,
248    D: XWaylandShellHandler + XwmHandler + SeatHandler,
249    D: 'static,
250{
251    fn request(
252        _state: &mut D,
253        _client: &Client,
254        resource: &XwaylandShellV1,
255        request: <XwaylandShellV1 as Resource>::Request,
256        _data: &(),
257        _dhandle: &DisplayHandle,
258        data_init: &mut DataInit<'_, D>,
259    ) {
260        match request {
261            xwayland_shell_v1::Request::GetXwaylandSurface { id, surface } => {
262                if compositor::give_role(&surface, XWAYLAND_SHELL_ROLE).is_err() {
263                    resource.post_error(xwayland_shell_v1::Error::Role, "Surface already has a role.");
264                    return;
265                }
266
267                compositor::add_pre_commit_hook::<D, _>(&surface, serial_commit_hook);
268
269                data_init.init(id, XWaylandSurfaceUserData { wl_surface: surface });
270                // We call the handler callback once the serial is set.
271            }
272            xwayland_shell_v1::Request::Destroy => {
273                // The child objects created via this interface are unaffected.
274            }
275            _ => unreachable!(),
276        }
277    }
278}

280impl<D> Dispatch<XwaylandSurfaceV1, XWaylandSurfaceUserData, D> for XWaylandShellState
281where
282    D: Dispatch<XwaylandSurfaceV1, XWaylandSurfaceUserData>,
283    D: XWaylandShellHandler,
284    D: 'static,
285    D: XwmHandler,
286{
287    fn request(
288        _state: &mut D,
289        _client: &Client,
290        _resource: &XwaylandSurfaceV1,
291        request: <XwaylandSurfaceV1 as Resource>::Request,
292        data: &XWaylandSurfaceUserData,
293        _dhandle: &DisplayHandle,
294        _data_init: &mut DataInit<'_, D>,
295    ) {
296        match request {
297            // In order to match a WlSurface to an X11 window, we need to match a Serial sent both to the X11 window and the WlSurface.
298            // This can happen in any order so we need to store the Serial until we have both.
299            xwayland_surface_v1::Request::SetSerial { serial_lo, serial_hi } => {
300                let serial = u64::from(serial_lo) | (u64::from(serial_hi) << 32);
301
302                // Set the serial on the pending state of surface
303                compositor::with_states(&data.wl_surface, |states| {
304                    states
305                        .cached_state
306                        .get::<XWaylandShellCachedState>()
307                        .pending()
308                        .serial = Some(serial);
309                });
310            }
311            xwayland_surface_v1::Request::Destroy => {
312                // Any already existing associations are unaffected.
313            }
314            _ => unreachable!(),
315        }
316    }
317}
318
319fn serial_commit_hook<D: XWaylandShellHandler + XwmHandler + SeatHandler + 'static>(
320    state: &mut D,
321    _dh: &DisplayHandle,
322    surface: &WlSurface,
323) {
324    if let Some(serial) = compositor::with_states(surface, |states| {
325        states
326            .cached_state
327            .get::<XWaylandShellCachedState>()
328            .pending()
329            .serial
330    }) {
331        if let Some(client) = surface.client() {
332            // We only care about surfaces created by XWayland.
333            if let Some(xwm_id) = client
334                .get_data::<XWaylandClientData>()
335                .and_then(|data| data.user_data().get::<XwmId>())
336            {
337                let xwm = XwmHandler::xwm_state(state, *xwm_id);
338
339                // This handles the case that the serial was set on the X11
340                // window before surface. To handle the other case, we look for
341                // a matching surface when the WL_SURFACE_SERIAL atom is sent.
342                if let Some(window) = xwm.unpaired_surfaces.remove(&serial) {
343                    if let Some(xsurface) = xwm
344                        .windows
345                        .iter()
346                        .find(|x| x.window_id() == window || x.mapped_window_id() == Some(window))
347                        .cloned()
348                    {
349                        debug!(
350                            window = xsurface.window_id(),
351                            wl_surface = ?surface.id().protocol_id(),
352                            "associated X11 window to wl_surface in commit hook",
353                        );
354
355                        xsurface.set_wl_surface(state, Some(surface.clone()));
356
357                        XWaylandShellHandler::surface_associated(state, *xwm_id, surface.clone(), xsurface);
358                    } else {
359                        warn!(
360                            window,
361                            wl_surface = ?surface.id().protocol_id(),
362                            "Unknown X11 window associated to wl_surface in commit hook"
363                        )
364                    }
365                } else {
366                    // this is necessary for the atom-handler to look up the matching surface
367                    XWaylandShellHandler::xwayland_shell_state(state)
368                        .by_serial
369                        .insert(serial, surface.clone());
370                }
371            }
372        }
373    }
374}

376/// Macro to delegate implementation of the xwayland keyboard grab protocol
377#[macro_export]
378macro_rules! delegate_xwayland_shell {
379    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
380        const _: () = {
381            use $crate::{
382                reexports::{
383                    wayland_protocols::xwayland::shell::v1::server::{
384                        xwayland_shell_v1::XwaylandShellV1, xwayland_surface_v1::XwaylandSurfaceV1,
385                    },
386                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
387                },
388                wayland::xwayland_shell::{XWaylandShellState, XWaylandSurfaceUserData},
389            };
390
391            delegate_global_dispatch!(
392                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
393                $ty: [XwaylandShellV1: ()] => XWaylandShellState
394            );
395
396            delegate_dispatch!(
397                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
398                $ty: [XwaylandShellV1: ()] => XWaylandShellState
399            );
400
401            delegate_dispatch!(
402                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
403                $ty: [XwaylandSurfaceV1: XWaylandSurfaceUserData] => XWaylandShellState
404            );
405        };
406    };
