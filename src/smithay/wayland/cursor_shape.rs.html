1//! Utilities for compositor side cursor theming support
2//!
3//! This protocol allows clients to request compositor to draw a cursor for them, thus resulting
4//! in more consistent look and feel of the cursor across the applications.
5//!
6//! ## Initialization
7//!

122/// State of the cursor shape manager.
123#[derive(Debug)]
124pub struct CursorShapeManagerState {
125    global: GlobalId,
126}
127
128impl CursorShapeManagerState {
129    /// Register new [CursorShapeManager] global.
130    pub fn new<D>(display: &DisplayHandle) -> Self
131    where
132        D: GlobalDispatch<CursorShapeManager, ()>,
133        D: Dispatch<CursorShapeManager, ()>,
134        D: SeatHandler,
135        D: 'static,
136    {
137        let global = display.create_global::<D, CursorShapeManager, _>(2, ());
138        Self { global }
139    }
140
141    /// [CursorShapeManager] GlobalId getter.
142    pub fn global(&self) -> GlobalId {
143        self.global.clone()
144    }
145}

147impl<D> GlobalDispatch<CursorShapeManager, (), D> for CursorShapeManagerState
148where
149    D: GlobalDispatch<CursorShapeManager, ()>,
150    D: Dispatch<CursorShapeManager, ()>,
151    D: SeatHandler,
152    D: 'static,
153{
154    fn bind(

158        resource: wayland_server::New<CursorShapeManager>,

166impl<D> Dispatch<CursorShapeManager, (), D> for CursorShapeManagerState
167where
168    D: Dispatch<CursorShapeManager, ()>,
169    D: Dispatch<CursorShapeDevice, CursorShapeDeviceUserData<D>>,
170    D: SeatHandler,
171    D: 'static,
172{
173    fn request(
174        state: &mut D,
175        _client: &wayland_server::Client,
176        _resource: &CursorShapeManager,
177        request: <CursorShapeManager as wayland_server::Resource>::Request,
178        _data: &(),
179        _dhandle: &DisplayHandle,
180        data_init: &mut wayland_server::DataInit<'_, D>,
181    ) {
182        match request {
183            ManagerRequest::GetPointer {
184                cursor_shape_device,
185                pointer,
186            } => {
187                let pointer_data = pointer.data::<PointerUserData<D>>();
188                let handle = match pointer_data.and_then(|data| data.handle.as_ref()) {
189                    Some(handle) => handle,
190                    None => return,
191                };
192
193                let Some(seat) = state
194                    .seat_state()
195                    .seats
196                    .iter()
197                    .find(|seat| seat.get_pointer().map(|h| &h == handle).unwrap_or(false))
198                    .cloned()
199                else {
200                    return;
201                };
202
203                data_init.init(
204                    cursor_shape_device,
205                    CursorShapeDeviceUserData(CursorShapeDeviceUserDataInner::Pointer {
206                        seat: seat.downgrade(),
207                    }),
208                );
209            }
210            ManagerRequest::GetTabletToolV2 {
211                cursor_shape_device,
212                tablet_tool,
213            } => {
214                data_init.init(
215                    cursor_shape_device,
216                    CursorShapeDeviceUserData(CursorShapeDeviceUserDataInner::Tablet(
217                        tablet_tool.downgrade(),
218                    )),
219                );
220            }
221            ManagerRequest::Destroy => {}
222            _ => unreachable!(),
223        }
224    }
225}

229pub struct CursorShapeDeviceUserData<D: SeatHandler>(CursorShapeDeviceUserDataInner<D>);

231#[derive(Debug, Clone)]
232pub(crate) enum CursorShapeDeviceUserDataInner<D: SeatHandler> {
233    /// The device was created for the pointer.
234    Pointer { seat: WeakSeat<D> },
235    /// The device was created for the tablet tool.
236    Tablet(Weak<ZwpTabletToolV2>),
237}
238
239impl<D> Dispatch<CursorShapeDevice, CursorShapeDeviceUserData<D>, D> for CursorShapeManagerState
240where
241    D: Dispatch<CursorShapeManager, ()>,
242    D: Dispatch<CursorShapeDevice, CursorShapeDeviceUserData<D>>,
243    D: SeatHandler + TabletSeatHandler,
244    <D as SeatHandler>::PointerFocus: WaylandFocus,
245    D: 'static,
246{
247    fn request(
248        state: &mut D,
249        _client: &wayland_server::Client,
250        resource: &CursorShapeDevice,
251        request: <CursorShapeDevice as wayland_server::Resource>::Request,
252        data: &CursorShapeDeviceUserData<D>,
253        _dhandle: &DisplayHandle,
254        _data_init: &mut wayland_server::DataInit<'_, D>,
255    ) {
256        match request {
257            ShapeRequest::SetShape {
258                serial,
259                shape: WEnum::Value(shape),
260            } => {
261                match &data.0 {
262                    CursorShapeDeviceUserDataInner::Pointer { seat } => {
263                        let Some(seat) = seat.upgrade() else {
264                            return;
265                        };
266
267                        let Some(handle) = seat.get_pointer() else {
268                            // When the pointer capability is removed from the wl_seat,
269                            // the wp_cursor_shape_device_v1 object becomes inert.
270                            return;
271                        };
272
273                        if !allow_setting_cursor(&handle, Serial(serial), &resource.id()) {
274                            return;
275                        }
276
277                        let cursor_icon = shape_to_cursor_icon(shape);
278                        state.cursor_image(&seat, CursorImageStatus::Named(cursor_icon));
279                    }
280                    CursorShapeDeviceUserDataInner::Tablet(tablet) => {
281                        let Ok(tablet) = tablet.upgrade() else {
282                            // When the zwp_tablet_tool_v2 is removed, the wp_cursor_shape_device_v1 object becomes inert.
283                            return;
284                        };
285                        let tablet_data = match tablet.data::<TabletToolUserData>() {
286                            Some(data) => data,
287                            None => return,
288                        };
289
290                        // Check that tablet focus matches.
291                        if !tablet_data
292                            .handle
293                            .inner
294                            .lock()
295                            .unwrap()
296                            .focus
297                            .as_ref()
298                            .map(|focus| focus.same_client_as(&tablet.id()))
299                            .unwrap_or(false)
300                        {
301                            return;
302                        }
303
304                        let cursor_icon = shape_to_cursor_icon(shape);
305                        state.tablet_tool_image(&tablet_data.desc, CursorImageStatus::Named(cursor_icon));
306                    }
307                }
308            }
309            ShapeRequest::SetShape { .. } => {
310                // Ignore unknown shapes.
311            }
312            ShapeRequest::Destroy => {}
313            _ => unreachable!(),
314        }
315    }
316}

318fn shape_to_cursor_icon(shape: Shape) -> CursorIcon {
319    match shape {
320        Shape::Default => CursorIcon::Default,
321        Shape::ContextMenu => CursorIcon::ContextMenu,
322        Shape::Help => CursorIcon::Help,
323        Shape::Pointer => CursorIcon::Pointer,
324        Shape::Progress => CursorIcon::Progress,
325        Shape::Wait => CursorIcon::Wait,
326        Shape::Cell => CursorIcon::Cell,
327        Shape::Crosshair => CursorIcon::Crosshair,
328        Shape::Text => CursorIcon::Text,
329        Shape::VerticalText => CursorIcon::VerticalText,
330        Shape::Alias => CursorIcon::Alias,
331        Shape::Copy => CursorIcon::Copy,
332        Shape::Move => CursorIcon::Move,
333        Shape::NoDrop => CursorIcon::NoDrop,
334        Shape::NotAllowed => CursorIcon::NotAllowed,
335        Shape::Grab => CursorIcon::Grab,
336        Shape::Grabbing => CursorIcon::Grabbing,
337        Shape::EResize => CursorIcon::EResize,
338        Shape::NResize => CursorIcon::NResize,
339        Shape::NeResize => CursorIcon::NeResize,
340        Shape::NwResize => CursorIcon::NwResize,
341        Shape::SResize => CursorIcon::SResize,
342        Shape::SeResize => CursorIcon::SeResize,
343        Shape::SwResize => CursorIcon::SwResize,

356        _ => CursorIcon::Default,
357    }
