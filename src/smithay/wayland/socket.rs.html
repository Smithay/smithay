1//! Wayland listening socket.
2//!
3//! This module provides an [`EventSource`] that invokes a callback when a new client has connected to the
4//! socket. This is one of the ways a Wayland compositor allows clients to discover the compositor.
5//!
6//! The callback provides a [`UnixStream`] that represents the client connection. You need to create the
7//! client using this stream by calling [`DisplayHandle::insert_client`](wayland_server::DisplayHandle::insert_client).
8//!
9//! # Example usage
10//!
11//! ```no_run
12//! # use std::sync::Arc;
13//! use smithay::wayland::socket::ListeningSocketSource;
14//!
15//! // data passed into calloop
16//! struct Example {
17//!     display: wayland_server::Display<()>,
18//! }
19//!
20//! let event_loop = calloop::EventLoop::<Example>::try_new().unwrap();
21//! let mut display = wayland_server::Display::<()>::new().unwrap();
22//!
23//! // Create a socket for clients to discover the compositor.
24//! //
25//! // This function will select the next open name for a listening socket.
26//! let listening_socket = ListeningSocketSource::new_auto().unwrap();
27//!
28//! event_loop.handle().insert_source(listening_socket, |client_stream, _, state| {
29//!     // Inside the callback, you should insert the client into the display.
30//!     //
31//!     // You may also associate some data with the client when inserting the client.
32//!     state.display.handle().insert_client(client_stream, Arc::new(ExampleClientData));
33//! });
34//!
35//! # struct ExampleClientData;
36//! #
37//! # impl wayland_server::backend::ClientData for ExampleClientData {
38//! #     fn initialized(&self, _: wayland_server::backend::ClientId) {}
39//! #     fn disconnected(
40//! #         &self,
41//! #         _: wayland_server::backend::ClientId,
42//! #         _: wayland_server::backend::DisconnectReason
43//! #     ) {}
44//! # }
45//! ```
46
47use std::{ffi::OsStr, io, os::unix::net::UnixStream};
48
49use calloop::{
50    generic::Generic, EventSource, Interest, Mode, Poll, PostAction, Readiness, Token, TokenFactory,
51};
52use tracing::{debug, info};
53use wayland_server::{BindError, ListeningSocket};
54
55/// A Wayland listening socket event source.
56///
57/// This implements [`EventSource`] and may be inserted into an event loop.
58#[derive(Debug)]
59pub struct ListeningSocketSource {
60    socket: Generic<ListeningSocket>,
61}
62
63impl ListeningSocketSource {
64    /// Creates a new listening socket, automatically choosing the next available `wayland` socket name.
65    pub fn new_auto() -> Result<ListeningSocketSource, BindError> {
66        // Try socket numbers 1-32. Remember the upper bound of Range is exclusive.
67        //
68        // We don't try wayland-0 due since clients may connect to the wrong compositor. Clients these days
69        // should be connecting based off the WAYLAND_DISPLAY or WAYLAND_SOCKET environment variables.
70        let socket = ListeningSocket::bind_auto("wayland", 1..33)?;
71
72        info!(name = ?socket.socket_name(), "Created new socket");
73
74        Ok(ListeningSocketSource {
75            socket: Generic::new(socket, Interest::READ, Mode::Level),
76        })
77    }
78
79    /// Creates a new listening socket with the specified name.
80    pub fn with_name(name: &str) -> Result<ListeningSocketSource, BindError> {
81        let socket = ListeningSocket::bind(name)?;
82        info!(name = ?socket.socket_name(), "Created new socket");
83
84        Ok(ListeningSocketSource {
85            socket: Generic::new(socket, Interest::READ, Mode::Level),
86        })
87    }
88
89    /// Returns the name of the listening socket.
90    pub fn socket_name(&self) -> &OsStr {
91        self.socket.get_ref().socket_name().unwrap()
92    }
93}
94
95impl EventSource for ListeningSocketSource {
96    /// A stream to the new client.
97    ///
98    /// You must register the  client using the stream by calling
99    /// [`DisplayHandle::insert_client`](wayland_server::DisplayHandle::insert_client).
100    type Event = UnixStream;
101    type Metadata = ();
102    type Ret = ();
103    type Error = io::Error;
104
105    fn process_events<F>(
106        &mut self,
107        readiness: Readiness,
108        token: Token,
109        mut callback: F,
110    ) -> io::Result<PostAction>
111    where
112        F: FnMut(Self::Event, &mut Self::Metadata) -> Self::Ret,
113    {
114        self.socket.process_events(readiness, token, |_, socket| {
115            while let Some(client) = socket.accept()? {
116                debug!(socket = ?socket.socket_name(), client = ?client, "New client connected");
117                callback(client, &mut ());
118            }
119
120            Ok(PostAction::Continue)
121        })
122    }
123
124    fn register(&mut self, poll: &mut Poll, token_factory: &mut TokenFactory) -> calloop::Result<()> {
125        self.socket.register(poll, token_factory)
126    }
127
128    fn reregister(&mut self, poll: &mut Poll, token_factory: &mut TokenFactory) -> calloop::Result<()> {
129        self.socket.reregister(poll, token_factory)
130    }
131
132    fn unregister(&mut self, poll: &mut Poll) -> calloop::Result<()> {
133        self.socket.unregister(poll)
134    }
135}
