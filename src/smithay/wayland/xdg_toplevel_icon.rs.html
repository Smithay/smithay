1//! XDG Toplevel Icon
2//!
3//! This protocol allows clients to set icons for their toplevel surfaces either via the XDG icon stock (using an icon name), or from pixel data.
4//!
5//! In order to advertise toplevel icon global call [XdgToplevelIconManager::new] and delegate
6//! events to it with [`delegate_xdg_toplevel_icon`][crate::delegate_xdg_toplevel_icon].
7//! Currently attached icon is available in double-buffered [ToplevelIconCachedState]

37/// Handler trait for xdg toplevel icon events.
38pub trait XdgToplevelIconHandler:
39    GlobalDispatch<XdgToplevelIconManagerV1, XdgToplevelIconManagerUserData>
40    + Dispatch<XdgToplevelIconManagerV1, ()>
41    + Dispatch<XdgToplevelIconV1, XdgToplevelIconUserData>
42    + 'static
43{
44    /// Called when icon becomes pending, and awaits surface commit.
45    /// Icon is stored in wl_surface [ToplevelIconCachedState]
46    fn set_icon(&mut self, toplevel: XdgToplevel, wl_surface: WlSurface) {
47        let _ = toplevel;
48        let _ = wl_surface;
49    }
50}

52/// Data associated with WlSurface
53/// Represents the client pending state
54///
55/// ```no_run
56/// use smithay::wayland::compositor;
57/// use smithay::wayland::xdg_toplevel_icon::ToplevelIconCachedState;
58///
59/// # let wl_surface = todo!();
60/// compositor::with_states(&wl_surface, |states| {
61///     let mut data = states.cached_state.get::<ToplevelIconCachedState>();
62///     dbg!(data.current().icon_name());
63/// });
64/// ```
65#[derive(Debug, Clone, Default)]
66pub struct ToplevelIconCachedState {
67    icon: Option<XdgToplevelIconV1>,
68}
69
70impl ToplevelIconCachedState {
71    fn data(&self) -> Option<&IconData> {
72        let icon = self.icon.as_ref()?;
73        let data = icon
74            .data::<XdgToplevelIconUserData>()
75            .unwrap()
76            .constructed
77            .get()
78            .unwrap();
79        Some(data)
80    }
81
82    /// Icon name getter
83    pub fn icon_name(&self) -> Option<&str> {
84        self.data()?.icon_name.as_deref()
85    }
86
87    /// List of icon buffers and buffer scale
88    pub fn buffers(&self) -> &[(WlBuffer, i32)] {
89        let Some(data) = self.data() else {
90            return &[];
91        };
92        &data.buffers
93    }
94}
95
96impl Cacheable for ToplevelIconCachedState {
97    fn commit(&mut self, _dh: &DisplayHandle) -> Self {
98        self.clone()
99    }
100
101    fn merge_into(self, into: &mut Self, _dh: &DisplayHandle) {
102        *into = self;
103    }
104}

112/// User data of [XdgToplevelIconV1]
113#[derive(Debug, Default)]
114pub struct XdgToplevelIconUserData {
115    builder: Mutex<IconData>,
116    constructed: OnceLock<IconData>,
117    buffer_destruction_hooks: Mutex<Vec<(WlBuffer, HookId)>>,
118}
119
120impl XdgToplevelIconUserData {
121    fn freeze(&self) {
122        if self.constructed.get().is_none() {
123            let data = std::mem::take(&mut *self.builder.lock().unwrap());
124            self.constructed.set(data).unwrap();
125        }
126    }
127
128    fn is_immutable(&self) -> bool {
129        self.constructed.get().is_some()
130    }
131
132    fn set_icon_name(&self, icon_name: String) {
133        debug_assert!(!self.is_immutable());
134        self.builder.lock().unwrap().icon_name = Some(icon_name)
135    }
136
137    fn register_buffer_destruction_hook(
138        &self,
139        buffer: WlBuffer,
140        shm: &ShmBufferUserData,
141        hook: impl Fn() + Send + Sync + 'static,
142    ) {
143        let buffer_destruction_hook = shm.add_destruction_hook(move |_, _| hook());
144        self.buffer_destruction_hooks
145            .lock()
146            .unwrap()
147            .push((buffer, buffer_destruction_hook));
148    }
149
150    fn unregister_buffer_destruction_hook(&self, buffer: &WlBuffer) {
151        let mut guard = self.buffer_destruction_hooks.lock().unwrap();
152
153        if let Some((buffer, hook)) = guard
154            .iter()
155            .position(|(b, _)| b == buffer)
156            .map(|id| guard.remove(id))
157        {
158            if let Some(shm) = buffer.data::<ShmBufferUserData>() {
159                shm.remove_destruction_hook(hook);
160            }
161        }
162    }
163
164    fn unregister_all_hooks(&self) {
165        let mut guard = self.buffer_destruction_hooks.lock().unwrap();
166
167        for (buffer, hook) in guard.drain(..) {
168            if let Some(shm) = buffer.data::<ShmBufferUserData>() {
169                shm.remove_destruction_hook(hook);
170            }
171        }
172    }
173
174    fn add_buffer(&self, buffer: WlBuffer, scale: i32, shm: &ShmBufferUserData) {
175        debug_assert!(!self.is_immutable());
176
177        let mut guard = self.builder.lock().unwrap();
178        let list = &mut guard.buffers;
179
180        for (existing_buffer, existing_scale) in list.iter_mut() {
181            let existing_data = &existing_buffer.data::<ShmBufferUserData>().unwrap().data;
182
183            let same_width = existing_data.width == shm.data.width;
184            let same_height = existing_data.height == shm.data.height;
185            let same_scale = *existing_scale == scale;
186
187            // If this icon instance already has a buffer of the same size and scale from a previous 'add_buffer' request,
188            // data from the last request overrides the preexisting pixel data.
189            if same_width && same_height && same_scale {
190                // The existing buffer will not be used, so let's assume we don't care about it's
191                // destruction
192                self.unregister_buffer_destruction_hook(existing_buffer);
193                *existing_buffer = buffer;
194                return;
195            }
196        }
197
198        list.push((buffer, scale));
199    }
200}

211/// Delegate type for handling xdg toplevel icon events.
212#[derive(Debug)]
213pub struct XdgToplevelIconManager {
214    global: GlobalId,
215    data: Arc<Mutex<ManagerGlobalData>>,
216}
217
218impl XdgToplevelIconManager {
219    /// Creates a new delegate type for handling xdg toplevel icon events.
220    pub fn new<D: XdgToplevelIconHandler>(display: &DisplayHandle) -> Self {
221        let data = Arc::new(Mutex::new(ManagerGlobalData::default()));
222        let global = display
223            .create_global::<D, XdgToplevelIconManagerV1, _>(1, XdgToplevelIconManagerUserData(data.clone()));
224        XdgToplevelIconManager { global, data }
225    }
226
227    /// Request given icon size
228    ///
229    /// - `size` the edge size of the square icon in surface-local coordinates
230    pub fn add_icon_size(&mut self, size: i32) {
231        self.data.lock().unwrap().sizes.insert(size);
232    }
233
234    /// Returns the [XdgToplevelIconManagerV1] global id.
235    pub fn global(&self) -> GlobalId {
236        self.global.clone()
237    }
238}

240impl<D: XdgToplevelIconHandler> GlobalDispatch<XdgToplevelIconManagerV1, XdgToplevelIconManagerUserData, D>
241    for XdgToplevelIconManager
242{
243    fn bind(
244        _: &mut D,
245        _: &DisplayHandle,
246        _: &Client,
247        resource: New<XdgToplevelIconManagerV1>,
248        data: &XdgToplevelIconManagerUserData,
249        data_init: &mut DataInit<'_, D>,
250    ) {
251        let manager = data_init.init(resource, ());
252
253        for size in data.0.lock().unwrap().sizes.iter() {
254            manager.icon_size(*size);
255        }
256        manager.done();
257    }
258}

260impl<D: XdgToplevelIconHandler> Dispatch<XdgToplevelIconManagerV1, (), D> for XdgToplevelIconManager {
261    fn request(
262        state: &mut D,
263        _: &Client,
264        _resource: &XdgToplevelIconManagerV1,
265        request: xdg_toplevel_icon_manager_v1::Request,
266        _: &(),
267        _dh: &DisplayHandle,
268        data_init: &mut DataInit<'_, D>,
269    ) {
270        use xdg_toplevel_icon_manager_v1::Request;
271
272        match request {
273            Request::CreateIcon { id } => {
274                data_init.init(id, XdgToplevelIconUserData::default());
275            }
276            Request::SetIcon { toplevel, icon } => {
277                let data = toplevel.data::<XdgShellSurfaceUserData>().unwrap();
278                let wl_surface = data.wl_surface.clone();
279
280                compositor::with_states(&data.wl_surface, |states| {
281                    let mut state = states.cached_state.get::<ToplevelIconCachedState>();
282                    let pending = state.pending();
283                    match icon {
284                        Some(icon) => {
285                            let icon_data = icon.data::<XdgToplevelIconUserData>().unwrap();
286                            icon_data.freeze();
287                            pending.icon = Some(icon);
288                        }
289                        None => {
290                            *pending = Default::default();
291                        }
292                    }
293                });
294
295                state.set_icon(toplevel, wl_surface);
296            }
297            Request::Destroy => {}
298            _ => unreachable!(),
299        }
300    }
301}

303impl<D: XdgToplevelIconHandler> Dispatch<XdgToplevelIconV1, XdgToplevelIconUserData, D>
304    for XdgToplevelIconManager
305{
306    fn request(
307        _state: &mut D,
308        _: &Client,
309        icon: &XdgToplevelIconV1,
310        request: xdg_toplevel_icon_v1::Request,
311        data: &XdgToplevelIconUserData,
312        dh: &DisplayHandle,
313        _: &mut DataInit<'_, D>,
314    ) {
315        use xdg_toplevel_icon_v1::Request;
316
317        match request {
318            Request::SetName { icon_name } => {
319                if data.is_immutable() {
320                    dh.post_error(
321                        icon,
322                        xdg_toplevel_icon_v1::Error::Immutable as u32,
323                        "Request made after the icon has been assigned to a toplevel via 'set_icon'"
324                            .to_string(),
325                    );
326                }
327
328                data.set_icon_name(icon_name);
329            }
330            Request::AddBuffer { buffer, scale } => {
331                if data.is_immutable() {
332                    dh.post_error(
333                        icon,
334                        xdg_toplevel_icon_v1::Error::Immutable as u32,
335                        "Request made after the icon has been assigned to a toplevel via 'set_icon'"
336                            .to_string(),
337                    );
338                }
339
340                let Some(shm) = buffer.data::<ShmBufferUserData>() else {
341                    dh.post_error(
342                        icon,
343                        xdg_toplevel_icon_v1::Error::InvalidBuffer as u32,
344                        "The wl_buffer must be backed by wl_shm".to_string(),
345                    );
346                    return;
347                };
348
349                if shm.data.width != shm.data.height {
350                    dh.post_error(
351                        icon,
352                        xdg_toplevel_icon_v1::Error::InvalidBuffer as u32,
353                        "The wl_buffer must be a square".to_string(),
354                    );
355                    return;
356                };
357
358                // Let's listen for buffer destruction event to catch no_buffer protocol error
359                // This hook has to be unregistered once the icon is destroyed
360                data.register_buffer_destruction_hook(buffer.clone(), shm, {
361                    let icon = icon.clone();
362                    move || {
363                        icon.post_error(
364                            xdg_toplevel_icon_v1::Error::NoBuffer,
365                            "The provided buffer has been destroyed before the toplevel icon",
366                        )
367                    }
368                });
369                data.add_buffer(buffer.clone(), scale, shm);
370            }
371            Request::Destroy => {}
372            _ => unreachable!(),
373        }
374    }
375
376    fn destroyed(
377        _state: &mut D,
378        _client: ClientId,
379        _resource: &XdgToplevelIconV1,
380        data: &XdgToplevelIconUserData,
381    ) {
382        data.unregister_all_hooks();
383    }
384}

394        ] => $crate::wayland::xdg_toplevel_icon::XdgToplevelIconManager);
395
396        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
397            $crate::reexports::wayland_protocols::xdg::toplevel_icon::v1::server::xdg_toplevel_icon_manager_v1::XdgToplevelIconManagerV1: ()
398        ] => $crate::wayland::xdg_toplevel_icon::XdgToplevelIconManager);
399        $crate::reexports::wayland_server::delegate_dispatch!($(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)? $ty: [
400            $crate::reexports::wayland_protocols::xdg::toplevel_icon::v1::server::xdg_toplevel_icon_v1::XdgToplevelIconV1: $crate::wayland::xdg_toplevel_icon::XdgToplevelIconUserData
401        ] => $crate::wayland::xdg_toplevel_icon::XdgToplevelIconManager);
