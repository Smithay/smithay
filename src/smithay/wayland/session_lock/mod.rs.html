1//! Utilities for handling the `ext-session-lock` protocol
2//!
3//! ## How to use it
4//!
5//! ### Initialization
6//!
7//! To initialize this implementation create the [`SessionLockManagerState`] and
8//! implement the [`SessionLockHandler`], as shown in this example:
9//!
10//! ```
11//! use smithay::delegate_session_lock;
12//! use smithay::reexports::wayland_server::protocol::wl_output::WlOutput;
13//! use smithay::wayland::session_lock::{
14//!     LockSurface, SessionLockManagerState, SessionLockHandler, SessionLocker,
15//! };
16//!
17//! # struct State { session_lock_state: SessionLockManagerState }
18//! # let mut display = wayland_server::Display::<State>::new().unwrap();
19//! // Create the compositor state
20//! let session_lock_state = SessionLockManagerState::new::<State, _>(&display.handle(), |_| true);
21//!
22//! // Insert the SessionLockManagerState into your state.
23//!
24//! // Implement the necessary trait.
25//! impl SessionLockHandler for State {
26//!     fn lock_state(&mut self) -> &mut SessionLockManagerState {
27//!         &mut self.session_lock_state
28//!     }
29//!
30//!     fn lock(&mut self, _confirmation: SessionLocker) {
31//!         // Lock and clear the screen.
32//!
33//!         // Call `confirmation.lock()` after a cleared frame was presented on all outputs.
34//!
35//!         // Dropping `confirmation` will cancel the locking.
36//!     }
37//!
38//!     fn unlock(&mut self) {
39//!         // Remove session lock.
40//!     }
41//!
42//!     fn new_surface(&mut self, _surface: LockSurface, _output: WlOutput) {
43//!         // Display `LockSurface` on `WlOutput`.
44//!     }
45//! }
46//! delegate_session_lock!(State);
47//!
48//! // You're now ready to go!
49//! ```

70const MANAGER_VERSION: u32 = 1;

72/// State of the [`ExtSessionLockManagerV1`] Global.
73#[derive(Debug)]
74pub struct SessionLockManagerState {
75    pub(crate) locked_outputs: Vec<WlOutput>,
76}
77
78impl SessionLockManagerState {
79    /// Create new [`ExtSessionLockManagerV1`] global.
80    pub fn new<D, F>(display: &DisplayHandle, filter: F) -> Self
81    where
82        D: GlobalDispatch<ExtSessionLockManagerV1, SessionLockManagerGlobalData>,
83        D: Dispatch<ExtSessionLockManagerV1, ()>,
84        D: Dispatch<ExtSessionLockV1, SessionLockState>,
85        D: SessionLockHandler,
86        D: 'static,
87        F: for<'c> Fn(&'c Client) -> bool + Send + Sync + 'static,
88    {
89        let data = SessionLockManagerGlobalData {
90            filter: Box::new(filter),
91        };
92        display.create_global::<D, ExtSessionLockManagerV1, _>(MANAGER_VERSION, data);
93
94        Self {
95            locked_outputs: Vec::new(),
96        }
97    }
98}

100#[allow(missing_debug_implementations)]
101#[doc(hidden)]
102pub struct SessionLockManagerGlobalData {
103    /// Filter whether the clients can view global.
104    filter: Box<dyn for<'c> Fn(&'c Client) -> bool + Send + Sync>,
105}

107impl<D> GlobalDispatch<ExtSessionLockManagerV1, SessionLockManagerGlobalData, D> for SessionLockManagerState
108where
109    D: GlobalDispatch<ExtSessionLockManagerV1, SessionLockManagerGlobalData>,
110    D: Dispatch<ExtSessionLockManagerV1, ()>,
111    D: Dispatch<ExtSessionLockV1, SessionLockState>,
112    D: SessionLockHandler,
113    D: 'static,
114{
115    fn bind(
116        _state: &mut D,
117        _display: &DisplayHandle,
118        _client: &Client,
119        manager: New<ExtSessionLockManagerV1>,
120        _global_data: &SessionLockManagerGlobalData,
121        data_init: &mut DataInit<'_, D>,
122    ) {
123        data_init.init(manager, ());
124    }
125
126    fn can_view(client: Client, global_data: &SessionLockManagerGlobalData) -> bool {
127        (global_data.filter)(&client)
128    }
129}
130
131impl<D> Dispatch<ExtSessionLockManagerV1, (), D> for SessionLockManagerState
132where
133    D: GlobalDispatch<ExtSessionLockManagerV1, SessionLockManagerGlobalData>,
134    D: Dispatch<ExtSessionLockManagerV1, ()>,
135    D: Dispatch<ExtSessionLockV1, SessionLockState>,
136    D: SessionLockHandler,
137    D: 'static,
138{
139    fn request(
140        state: &mut D,
141        _client: &Client,
142        _manager: &ExtSessionLockManagerV1,
143        request: Request,
144        _data: &(),
145        _display: &DisplayHandle,
146        data_init: &mut DataInit<'_, D>,
147    ) {
148        match request {
149            Request::Lock { id } => {
150                let lock_state = SessionLockState::new();
151                let lock_status = lock_state.lock_status.clone();
152                let lock = data_init.init(id, lock_state);
153                state.lock(SessionLocker::new(lock, lock_status));
154            }
155            Request::Destroy => (),
156            _ => unreachable!(),
157        }
158    }
159}

161/// Handler trait for ext-session-lock.
162pub trait SessionLockHandler {
163    /// Session lock state.
164    fn lock_state(&mut self) -> &mut SessionLockManagerState;
165
166    /// Handle compositor locking requests.
167    ///
168    /// The [`SessionLocker`] parameter is used to confirm once the session was
169    /// locked and no more client data is accessible using the
170    /// [`SessionLocker::lock`] method.
171    ///
172    /// If locking was not possible, dropping the [`SessionLocker`] will
173    /// automatically notify the requesting client about the failure.
174    fn lock(&mut self, confirmation: SessionLocker);
175
176    /// Handle compositor lock removal.
177    fn unlock(&mut self);
178
179    /// Add a new lock surface for an output.
180    fn new_surface(&mut self, surface: LockSurface, output: WlOutput);
181
182    /// A surface has acknowledged a configure serial.
183    fn ack_configure(&mut self, _surface: WlSurface, _configure: LockSurfaceConfigure) {}
184}

186/// Manage session locking.
187///
188/// See [`SessionLockHandler::lock`] for more detail.
189#[derive(Debug)]
190pub struct SessionLocker {
191    lock: Option<ExtSessionLockV1>,
192    lock_status: Arc<AtomicBool>,
193}
194
195impl Drop for SessionLocker {
196    fn drop(&mut self) {
197        // If the session wasn't locked, we notify clients about the failure.
198        if let Some(lock) = self.lock.take() {
199            lock.finished();
200        }
201    }
202}
203
204impl SessionLocker {
205    fn new(lock: ExtSessionLockV1, lock_status: Arc<AtomicBool>) -> Self {
206        Self {
207            lock: Some(lock),
208            lock_status,
209        }
210    }
211
212    /// Get the underlying [`ExtSessionLockV1`]
213    pub fn ext_session_lock(&self) -> &ExtSessionLockV1 {
214        self.lock.as_ref().unwrap()
215    }
216
217    /// Notify the client that the session lock was successful.
218    pub fn lock(mut self) {
219        if let Some(lock) = self.lock.take() {
220            self.lock_status.store(true, Ordering::Relaxed);
221            lock.locked();
222        }
223    }
224}
