1//! Utilities for handling the `zwp_idle_inhibit` protocol
2//!
3//! ## How to use it
4//!
5//! ### Initialization
6//!
7//! To initialize this implementation create the [`IdleInhibitManagerState`] and
8//! implement the [`IdleInhibitHandler`], as shown in this example:
9//!
10//! ```
11//! use smithay::delegate_idle_inhibit;
12//! use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
13//! use smithay::wayland::idle_inhibit::{IdleInhibitManagerState, IdleInhibitHandler};
14//!
15//! # struct State;
16//! # let mut display = wayland_server::Display::<State>::new().unwrap();
17//! // Create the compositor state
18//! IdleInhibitManagerState::new::<State>(&display.handle());
19//!
20//! // implement the necessary trait
21//! impl IdleInhibitHandler for State {
22//!    fn inhibit(&mut self, surface: WlSurface) {
23//!        // …
24//!    }
25//!
26//!    fn uninhibit(&mut self, surface: WlSurface) {
27//!        // …
28//!    }
29//! }
30//! delegate_idle_inhibit!(State);
31//!
32//! // You're now ready to go!
33//! ```

48/// State of the zwp_idle_inhibit_manager_v1 Global
49#[derive(Debug)]
50pub struct IdleInhibitManagerState {
51    global: GlobalId,
52}
53
54impl IdleInhibitManagerState {
55    /// Create new [`zwp_idle_inhibit_manager`](ZwpIdleInhibitManagerV1) global.
56    pub fn new<D>(display: &DisplayHandle) -> Self
57    where
58        D: GlobalDispatch<ZwpIdleInhibitManagerV1, ()>,
59        D: Dispatch<ZwpIdleInhibitManagerV1, ()>,
60        D: Dispatch<ZwpIdleInhibitorV1, IdleInhibitorState>,
61        D: IdleInhibitHandler,
62        D: 'static,
63    {
64        let global = display.create_global::<D, ZwpIdleInhibitManagerV1, _>(MANAGER_VERSION, ());
65
66        Self { global }
67    }
68
69    /// Returns the fractional scale manager global.
70    pub fn global(&self) -> GlobalId {
71        self.global.clone()
72    }
73}

75impl<D> GlobalDispatch<ZwpIdleInhibitManagerV1, (), D> for IdleInhibitManagerState
76where
77    D: GlobalDispatch<ZwpIdleInhibitManagerV1, ()>,
78    D: Dispatch<ZwpIdleInhibitManagerV1, ()>,
79    D: Dispatch<ZwpIdleInhibitorV1, IdleInhibitorState>,
80    D: IdleInhibitHandler,
81    D: 'static,
82{
83    fn bind(
84        _state: &mut D,
85        _display: &DisplayHandle,
86        _client: &Client,
87        manager: New<ZwpIdleInhibitManagerV1>,
88        _manager_state: &(),
89        data_init: &mut DataInit<'_, D>,
90    ) {
91        data_init.init(manager, ());
92    }
93}
94
95impl<D> Dispatch<ZwpIdleInhibitManagerV1, (), D> for IdleInhibitManagerState
96where
97    D: GlobalDispatch<ZwpIdleInhibitManagerV1, ()>,
98    D: Dispatch<ZwpIdleInhibitManagerV1, ()>,
99    D: Dispatch<ZwpIdleInhibitorV1, IdleInhibitorState>,
100    D: IdleInhibitHandler,
101    D: 'static,
102{
103    fn request(
104        state: &mut D,
105        _client: &Client,
106        _manager: &ZwpIdleInhibitManagerV1,
107        request: Request,
108        _data: &(),
109        _display: &DisplayHandle,
110        data_init: &mut DataInit<'_, D>,
111    ) {
112        match request {
113            Request::CreateInhibitor { id, surface } => {
114                state.inhibit(surface.clone());
115                data_init.init(id, IdleInhibitorState::new(surface));
116            }
117            Request::Destroy => (),
118            _ => unreachable!(),
119        }
120    }
121}

123/// Handler trait for idle-inhibit.
124pub trait IdleInhibitHandler {
125    /// Enable idle inhibition for the output of the provided surface.
126    fn inhibit(&mut self, surface: WlSurface);
127
128    /// Stop inhibition for the provided surface.
129    ///
130    /// This function is only called when a client explicitly removes the session
131    /// inhibition. It is up to the compositor to ignore inhibiting surfaces which
132    /// are invisible or dead.
133    fn uninhibit(&mut self, surface: WlSurface);
134}

138macro_rules! delegate_idle_inhibit {
139    ($(@<$( $lt:tt $( : $clt:tt $(+ $dlt:tt )* )? ),+>)? $ty: ty) => {
140        const _: () = {
141            use $crate::{
142                reexports::{
143                    wayland_protocols::wp::idle_inhibit::zv1::server::{
144                        zwp_idle_inhibit_manager_v1::ZwpIdleInhibitManagerV1,
145                        zwp_idle_inhibitor_v1::ZwpIdleInhibitorV1,
146                    },
147                    wayland_server::{delegate_dispatch, delegate_global_dispatch},
148                },
149                wayland::idle_inhibit::{inhibitor::IdleInhibitorState, IdleInhibitManagerState},
150            };
151
152            delegate_global_dispatch!(
153                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
154                $ty: [ZwpIdleInhibitManagerV1: ()] => IdleInhibitManagerState
155            );
156
157            delegate_dispatch!(
158                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
159                $ty: [ZwpIdleInhibitManagerV1: ()] => IdleInhibitManagerState
160            );
161
162            delegate_dispatch!(
163                $(@< $( $lt $( : $clt $(+ $dlt )* )? ),+ >)?
164                $ty: [ZwpIdleInhibitorV1: IdleInhibitorState] => IdleInhibitManagerState
165            );
166        };
167    };
