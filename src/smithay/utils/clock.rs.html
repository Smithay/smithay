68impl Time<Monotonic> {
69    /// Returns the time in milliseconds
70    ///
71    /// This should match timestamps from libinput:
72    /// <https://wayland.freedesktop.org/libinput/doc/latest/timestamps.html>
73    pub fn as_millis(&self) -> u32 {
74        debug_assert!(self.tp.tv_sec >= 0);
75        debug_assert!(self.tp.tv_nsec >= 0);
76
77        // The monotonic clock does not fit as milliseconds in 32-bit after ~50 days of uptime. We
78        // do a modulo conversion which should match what happens in libinput.
79        (self.as_micros() / 1000) as u32
80    }
81
82    /// Returns the time in microseconds
83    ///
84    /// This should match timestamps from libinput:
85    /// <https://wayland.freedesktop.org/libinput/doc/latest/timestamps.html>
86    pub fn as_micros(&self) -> u64 {
87        // Assume monotonic clock (but not realitime) fits as microseconds in 64-bit
88        debug_assert!(self.tp.tv_sec >= 0);
89        debug_assert!(self.tp.tv_nsec >= 0);
90        self.tp.tv_sec as u64 * 1000000 + self.tp.tv_nsec as u64 / 1000
91    }
92}

180const NANOS_PER_SEC: rustix::time::Nsecs = 1_000_000_000;
181
182fn saturating_sub_timespec(lhs: Timespec, rhs: Timespec) -> Option<Duration> {
183    debug_assert!(!lhs.tv_sec.is_negative());
184    debug_assert!(!lhs.tv_nsec.is_negative());
185    debug_assert!(!rhs.tv_sec.is_negative());
186    debug_assert!(!rhs.tv_nsec.is_negative());
187    debug_assert!(lhs.tv_nsec < NANOS_PER_SEC);
188    debug_assert!(rhs.tv_nsec < NANOS_PER_SEC);
189
190    let lhs_tv_sec = lhs.tv_sec as u64;
191    let lhs_tv_nsec = lhs.tv_nsec as u64;
192    let rhs_tv_sec = rhs.tv_sec as u64;
193    let rhs_tv_nsec = rhs.tv_nsec as u64;
194
195    if let Some(mut secs) = lhs_tv_sec.checked_sub(rhs_tv_sec) {
196        let nanos = if lhs_tv_nsec >= rhs_tv_nsec {
197            lhs_tv_nsec - rhs_tv_nsec
198        } else if let Some(sub_secs) = secs.checked_sub(1) {
199            secs = sub_secs;
200            lhs_tv_nsec + (NANOS_PER_SEC as u64) - rhs_tv_nsec
201        } else {
202            return None;
203        };
204        debug_assert!(nanos < (NANOS_PER_SEC as u64));
205        Some(Duration::new(secs, nanos as u32))
206    } else {
207        None
208    }
209}

211impl<T, Kind> Add<T> for Time<Kind>
212where
213    T: Into<Time<Kind>>,
214{
215    type Output = Time<Kind>;
216
217    fn add(self, rhs: T) -> Self::Output {
218        let rhs = rhs.into();
219        let tv_nsec = (self.tp.tv_nsec + rhs.tp.tv_nsec) % NANOS_PER_SEC;
220        let tv_sec = (self.tp.tv_sec + rhs.tp.tv_sec) + ((self.tp.tv_nsec + rhs.tp.tv_nsec) / NANOS_PER_SEC);
221        Self::from(Timespec { tv_sec, tv_nsec })
222    }
223}
