1//! Helper utilities for using x11rb as an event source in calloop.

26/// Integration of an x11rb X11 connection with calloop.
27///
28/// This is a thin wrapper around `Channel`. It works by spawning an extra thread reads events from
29/// the X11 connection and then sends them across the channel.
30///
31/// See [1] for why this extra thread is necessary. The single-thread solution proposed on that
32/// page does not work with calloop, since it requires checking something on every main loop
33/// iteration. Calloop only allows "when an FD becomes readable".
34///
35/// [1]: https://docs.rs/x11rb/0.8.1/x11rb/event_loop_integration/index.html#threads-and-races
36#[derive(Debug)]
37pub struct X11Source {
38    connection: Arc<RustConnection>,
39    channel: Option<Channel<Event>>,
40    event_thread: Option<JoinHandle<()>>,
41    close_window: Window,
42    close_type: Atom,
43}
44
45impl X11Source {
46    /// Create a new X11 source.
47    ///
48    /// The returned instance will use `SendRequest` to cause a `ClientMessageEvent` to be sent to
49    /// the given window with the given type. The expectation is that this is a window that was
50    /// created by us. Thus, the event reading thread will wake up and check an internal exit flag,
51    /// then exit.
52    pub fn new(connection: Arc<RustConnection>, close_window: Window, close_type: Atom) -> Self {
53        let (sender, channel) = channel();
54        let conn = Arc::clone(&connection);
55        let event_thread = Some(spawn(move || {
56            run_event_thread(conn, sender);
57        }));
58
59        Self {
60            connection,
61            channel: Some(channel),
62            event_thread,
63            close_window,
64            close_type,
65        }
66    }
67}
68
69impl Drop for X11Source {
70    fn drop(&mut self) {
71        // Signal the worker thread to exit by dropping the read end of the channel.
72        self.channel.take();
73
74        // Send an event to wake up the worker so that it actually exits
75        let event = ClientMessageEvent {
76            response_type: CLIENT_MESSAGE_EVENT,
77            format: 8,
78            sequence: 0,
79            window: self.close_window,
80            type_: self.close_type,
81            data: [0; 20].into(),
82        };
83
84        let _ = self
85            .connection
86            .send_event(false, self.close_window, EventMask::NO_EVENT, event);
87        let _ = self.connection.flush();
88
89        // Wait for the worker thread to exit
90        self.event_thread.take().map(|handle| handle.join());
91    }
92}

94impl EventSource for X11Source {
95    type Event = ChannelEvent<Event>;
96    type Metadata = ();
97    type Ret = ();
98    type Error = ChannelError;
99
100    #[profiling::function]
101    fn process_events<C>(
102        &mut self,
103        readiness: Readiness,
104        token: Token,
105        mut callback: C,
106    ) -> Result<PostAction, ChannelError>
107    where
108        C: FnMut(Self::Event, &mut Self::Metadata) -> Self::Ret,
109    {
110        if let Some(channel) = &mut self.channel {
111            channel.process_events(readiness, token, move |event, meta| {
112                if matches!(event, ChannelEvent::Closed) {
113                    warn!("Event thread exited");
114                }
115                callback(event, meta)
116            })
117        } else {
118            Ok(PostAction::Remove)
119        }
120    }
121
122    fn register(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
123        if let Some(channel) = &mut self.channel {
124            channel.register(poll, factory)?;
125        }
126
127        Ok(())
128    }
129
130    fn reregister(&mut self, poll: &mut Poll, factory: &mut TokenFactory) -> calloop::Result<()> {
131        if let Some(channel) = &mut self.channel {
132            channel.reregister(poll, factory)?;
133        }
134
135        Ok(())
136    }
137
138    fn unregister(&mut self, poll: &mut Poll) -> calloop::Result<()> {
139        if let Some(channel) = &mut self.channel {
140            channel.unregister(poll)?;
141        }
142
143        Ok(())
144    }
145}

147/// This thread reads X11 events from the connection and sends them on the channel.
148///
149/// This is run in an extra thread since sending an X11 request or waiting for the reply to an X11
150/// request can both read X11 events from the underlying socket which are then saved in the
151/// RustConnection. Thus, readability of the underlying socket is not enough to guarantee we do not
152/// miss wakeups.
153///
154/// This thread will call wait_for_event(). RustConnection then ensures internally to wake us up
155/// when an event arrives. So far, this seems to be the only safe way to integrate x11rb with
156/// calloop.
157fn run_event_thread(connection: Arc<RustConnection>, sender: Sender<Event>) {
158    loop {
159        let event = match connection.wait_for_event() {
160            Ok(event) => event,
161            Err(err) => {
162                // Connection errors are most likely permanent. Thus, exit the thread.
163                error!("Event thread exiting due to connection error {}", err);
164                break;
165            }
166        };
167        match sender.send(event) {
168            Ok(()) => {}
169            Err(_) => {
170                // The only possible error is that the other end of the channel was dropped.
171                // This happens in X11Source's Drop impl.
172                break;
173            }
174        }
175    }
