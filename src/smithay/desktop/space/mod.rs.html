44/// Represents two dimensional plane to map windows and outputs upon.
45///
46/// Space is generic over the types of elements mapped onto it.
47/// The simplest usecase is a `Space<Window>`, but other types can be used
48/// by implementing [`SpaceElement`]. Multiple types might be quickly aggregated into
49/// an enum by using the [`space_elements!`]-macro.
50#[derive(Debug)]
51pub struct Space<E: SpaceElement> {
52    pub(super) id: usize,
53    // in z-order, back to front
54    elements: Vec<InnerElement<E>>,
55    outputs: Vec<Output>,
56    span: tracing::Span,
57}
58
59impl<E: SpaceElement> PartialEq for Space<E> {
60    #[inline]
61    fn eq(&self, other: &Self) -> bool {
62        self.id == other.id
63    }
64}

94    /// Map a [`SpaceElement`] and move it to top of the stack
95    ///
96    /// This can safely be called on an already mapped window
97    /// to update its location inside the space.
98    ///
99    /// If activate is true it will set the new windows state
100    /// to be activate and removes that state from every
101    /// other mapped window.
102    pub fn map_element<P>(&mut self, element: E, location: P, activate: bool)
103    where
104        P: Into<Point<i32, Logical>>,
105    {
106        #[allow(clippy::mutable_key_type)]
107        let outputs = if let Some(pos) = self.elements.iter().position(|inner| inner.element == element) {
108            self.elements.remove(pos).outputs
109        } else {
110            HashMap::new()
111        };
112
113        let inner = InnerElement {
114            element,
115            location: location.into(),
116            outputs,
117        };
118        self.insert_elem(inner, activate);
119    }
120
121    /// Moves an already mapped [`SpaceElement`] to top of the stack
122    ///
123    /// This function does nothing for unmapped windows.
124    ///
125    /// If activate is true it will set the new windows state
126    /// to be activate and removes that state from every
127    /// other mapped window.
128    pub fn raise_element(&mut self, element: &E, activate: bool) {
129        if let Some(pos) = self.elements.iter().position(|inner| &inner.element == element) {
130            let inner = self.elements.remove(pos);
131            self.insert_elem(inner, activate);
132        }
133    }
134
135    fn insert_elem(&mut self, elem: InnerElement<E>, activate: bool) {
136        if activate {
137            elem.element.set_activate(true);
138            for e in self.elements.iter() {
139                e.element.set_activate(false);
140            }
141        }
142
143        self.elements.push(elem);
144        self.elements
145            .sort_by(|e1, e2| e1.element.z_index().cmp(&e2.element.z_index()));
146    }
147
148    /// Unmap a [`SpaceElement`] from this space.
149    ///
150    /// This function does nothing for already unmapped windows
151    pub fn unmap_elem(&mut self, element: &E) {
152        if let Some(pos) = self.elements.iter().position(|inner| &inner.element == element) {
153            let elem = self.elements.remove(pos);
154            for output in elem.outputs.keys() {
155                elem.element.output_leave(output);
156            }
157        }
158    }

176    /// Finds the topmost element under this point if any and returns it
177    /// together with the location of this element relative to this space.
178    ///
179    /// This is equivalent to iterating the elements in the space from
180    /// top to bottom and testing if the point is within the elements
181    /// input region and returning the first matching one.
182    ///
183    /// Note that [`SpaceElement::is_in_input_region`] expects the point
184    /// to be relative to the elements origin.
185    pub fn element_under<P: Into<Point<f64, Logical>>>(&self, point: P) -> Option<(&E, Point<i32, Logical>)> {
186        let point = point.into();
187        self.elements
188            .iter()
189            .rev()
190            .filter(|e| e.bbox().to_f64().contains(point))
191            .find_map(|e| {
192                // we need to offset the point to the location where the surface is actually drawn
193                let render_location = e.render_location();
194                if e.element.is_in_input_region(&(point - render_location.to_f64())) {
195                    Some((&e.element, render_location))
196                } else {
197                    None
198                }
199            })
200    }
201
202    /// Get a reference to the outputs under a given point
203    pub fn output_under<P: Into<Point<f64, Logical>>>(&self, point: P) -> impl Iterator<Item = &Output> {
204        let point = point.into();
205        self.outputs.iter().rev().filter(move |o| {
206            let bbox = self.output_geometry(o);
207            bbox.map(|bbox| bbox.to_f64().contains(point)).unwrap_or(false)
208        })
209    }

252    /// Maps an [`Output`] inside the space.
253    ///
254    /// Can be safely called on an already mapped
255    /// [`Output`] to update its location.
256    ///
257    /// *Note:* Remapping an output does reset it's damage memory.
258    pub fn map_output<P: Into<Point<i32, Logical>>>(&mut self, output: &Output, location: P) {
259        let location = location.into();
260        set_output_location(self.id, output, location);
261        if !self.outputs.contains(output) {
262            debug!(parent: &self.span, output = output.name(), "Mapping output at {:?}", location);
263            self.outputs.push(output.clone());
264        }
265    }

267    /// Iterate over all mapped [`Output`]s of this space.
268    pub fn outputs(&self) -> impl Iterator<Item = &Output> {
269        self.outputs.iter()
270    }

272    /// Unmap an [`Output`] from this space.
273    ///
274    /// Does nothing if the output was not previously mapped.
275    pub fn unmap_output(&mut self, output: &Output) {
276        if !self.outputs.contains(output) {
277            return;
278        }
279        debug!(parent: &self.span, output = output.name(), "Unmapping output");
280        set_output_location(self.id, output, None);
281        self.outputs.retain(|o| o != output);
282    }

284    /// Returns the geometry of the output including it's relative position inside the space.
285    ///
286    /// The size is matching the amount of logical pixels of the space visible on the output
287    /// given its current mode and scale.
288    pub fn output_geometry(&self, o: &Output) -> Option<Rectangle<i32, Logical>> {
289        if !self.outputs.contains(o) {
290            return None;
291        }
292
293        let transform: Transform = o.current_transform();
294        let location = output_location(self.id, o);
295        o.current_mode().map(|mode| {
296            Rectangle::new(
297                location,
298                transform
299                    .transform_size(mode.size)
300                    .to_f64()
301                    .to_logical(o.current_scale().fractional_scale())
302                    .to_i32_ceil(),
303            )
304        })
305    }

307    /// Returns all [`Output`]s a [`SpaceElement`] overlaps with.
308    pub fn outputs_for_element(&self, elem: &E) -> Vec<Output> {
309        if !self.elements.iter().any(|e| &e.element == elem) {
310            return Vec::new();
311        }
312
313        self.elements
314            .iter()
315            .find(|e| &e.element == elem)
316            .into_iter()
317            .flat_map(|e| &e.outputs)
318            .map(|(o, _)| o)
319            .cloned()
320            .collect()
321    }

323    /// Refresh some internal values and update client state,
324    /// meaning this will handle output enter and leave events
325    /// for mapped outputs and windows based on their position.
326    ///
327    /// Needs to be called periodically, at best before every
328    /// wayland socket flush.
329    #[profiling::function]
330    pub fn refresh(&mut self) {
331        self.elements.retain(|e| e.alive());
332
333        let outputs = self
334            .outputs
335            .iter()
336            .cloned()
337            .map(|o| {
338                let geo = self.output_geometry(&o).unwrap_or_else(Rectangle::zero);
339                (o, geo)
340            })
341            .collect::<Vec<_>>();
342        for e in &mut self.elements {
343            let bbox = e.bbox();
344
345            for (output, output_geometry) in &outputs {
346                // Check if the bounding box of the toplevel intersects with the output
347                if let Some(mut overlap) = output_geometry.intersection(bbox) {
348                    // output_enter expects the overlap to be relative to the element
349                    overlap.loc -= bbox.loc;
350                    let old = e.outputs.insert(output.clone(), overlap);
351                    if old.is_none() || matches!(old, Some(old_overlap) if old_overlap != overlap) {
352                        e.element.output_enter(output, overlap);
353                    }
354                } else if e.outputs.remove(output).is_some() {
355                    e.element.output_leave(output);
356                }
357            }
358            e.outputs.retain(|output, _| {
359                if !outputs.iter().any(|(o, _)| o == output) {
360                    e.element.output_leave(output);
361                    false
362                } else {
363                    true
364                }
365            });
366        }
367
368        self.elements.iter().for_each(|e| e.element.refresh());
369        for (output, _) in outputs {
370            output.cleanup();
371        }
372    }

374    /// Retrieve the render elements for a given region of the space.
375    ///
376    /// *Note:* Because this is not rendering a specific output,
377    /// this will not contain layer surfaces.
378    /// Use [`Space::render_elements_for_output`], if you care about this.
379    #[instrument(level = "trace", skip(self, renderer, scale), parent = &self.span)]
380    #[profiling::function]
381    pub fn render_elements_for_region<'a, R: Renderer, S: Into<Scale<f64>>>(
382        &'a self,
383        renderer: &mut R,
384        region: &Rectangle<i32, Logical>,
385        scale: S,
386        alpha: f32,
387    ) -> Vec<<E as AsRenderElements<R>>::RenderElement>
388    where
389        R::TextureId: Texture + 'static,
390        E: AsRenderElements<R>,
391        <E as AsRenderElements<R>>::RenderElement: 'a,
392    {
393        let scale = scale.into();
394
395        self.elements
396            .iter()
397            .rev()
398            .filter(|e| {
399                let geometry = e.bbox();
400                region.overlaps(geometry)
401            })
402            .flat_map(|e| {
403                let location = e.render_location() - region.loc;
404                e.element
405                    .render_elements::<<E as AsRenderElements<R>>::RenderElement>(
406                        renderer,
407                        location.to_physical_precise_round(scale),
408                        scale,
409                        alpha,
410                    )
411            })
412            .collect::<Vec<_>>()
413    }

415    /// Retrieve the render elements for an output
416    #[instrument(level = "trace", skip(self, renderer), parent = &self.span)]
417    #[profiling::function]
418    pub fn render_elements_for_output<
419        'a,
420        #[cfg(feature = "wayland_frontend")] R: Renderer + ImportAll,
421        #[cfg(not(feature = "wayland_frontend"))] R: Renderer,
422    >(
423        &'a self,
424        renderer: &mut R,
425        output: &Output,
426        alpha: f32,
427    ) -> Result<Vec<SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>>, OutputError>
428    where
429        R::TextureId: Clone + Texture + 'static,
430        E: AsRenderElements<R>,
431        <E as AsRenderElements<R>>::RenderElement: 'a,
432        SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>:
433            From<Wrap<<E as AsRenderElements<R>>::RenderElement>>,
434    {
435        if !self.outputs.contains(output) {
436            return Err(OutputError::Unmapped);
437        }
438
439        let output_scale = output.current_scale().fractional_scale();
440        // The unwrap is safe or we would have returned OutputError::Unmapped already
441        let output_geo = self.output_geometry(output).unwrap();
442
443        let mut space_elements: Vec<SpaceElements<'a, E>> =
444            self.elements.iter().rev().map(SpaceElements::Element).collect();
445
446        #[cfg(feature = "wayland_frontend")]
447        {
448            let layer_map = layer_map_for_output(output);
449            space_elements.extend(layer_map.layers().rev().cloned().map(|l| SpaceElements::Layer {
450                surface: l,
451                output_location: output_geo.loc,
452            }));
453        }
454
455        space_elements.sort_by_key(|e| std::cmp::Reverse(e.z_index()));
456
457        Ok(space_elements
458            .into_iter()
459            .filter(|e| {
460                let geometry = e.bbox();
461                output_geo.overlaps(geometry)
462            })
463            .flat_map(|e| {
464                let location = e.render_location() - output_geo.loc;
465                e.render_elements::<SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>>(
466                    renderer,
467                    location.to_physical_precise_round(output_scale),
468                    Scale::from(output_scale),
469                    alpha,
470                )
471            })
472            .collect::<Vec<_>>())
473    }

494impl<E: SpaceElement> InnerElement<E> {
495    // the inner geometry of the element in space coordinates
496    fn geometry(&self) -> Rectangle<i32, Logical> {
497        let mut geo = self.element.geometry();
498        geo.loc = self.location;
499        geo
500    }
501
502    // the bounding box of the element in space coordinates
503    fn bbox(&self) -> Rectangle<i32, Logical> {
504        let mut bbox = self.element.bbox();
505        bbox.loc += self.location - self.element.geometry().loc;
506        bbox
507    }
508
509    fn render_location(&self) -> Point<i32, Logical> {
510        self.location - self.element.geometry().loc
511    }
512}

568/// Get the render elements for a specific output
569///
570/// If multiple spaces are given their elements will be stacked
571/// the same way.
572///
573/// *Note*: If the `wayland_frontend`-feature is enabled
574/// this will include layer-shell surfaces added to this
575/// outputs [`LayerMap`](crate::desktop::LayerMap).
576#[instrument(level = "trace", skip(spaces, renderer))]
577#[profiling::function]
578pub fn space_render_elements<
579    'a,
580    #[cfg(feature = "wayland_frontend")] R: Renderer + ImportAll,
581    #[cfg(not(feature = "wayland_frontend"))] R: Renderer,
582    E: SpaceElement + PartialEq + AsRenderElements<R> + 'a,
583    S: IntoIterator<Item = &'a Space<E>>,
584>(
585    renderer: &mut R,
586    spaces: S,
587    output: &Output,
588    alpha: f32,
589) -> Result<Vec<SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>>, OutputNoMode>
590where
591    R::TextureId: Clone + Texture + 'static,
592    <E as AsRenderElements<R>>::RenderElement: 'a,
593    SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>:
594        From<Wrap<<E as AsRenderElements<R>>::RenderElement>>,
595{
596    let mut render_elements = Vec::new();
597    let output_scale = output.current_scale().fractional_scale();
598
599    #[cfg(feature = "wayland_frontend")]
600    let layer_map = layer_map_for_output(output);
601    #[cfg(feature = "wayland_frontend")]
602    let lower = {
603        let (lower, upper): (Vec<&LayerSurface>, Vec<&LayerSurface>) = layer_map
604            .layers()
605            .rev()
606            .partition(|s| matches!(s.layer(), Layer::Background | Layer::Bottom));
607
608        render_elements.extend(
609            upper
610                .into_iter()
611                .filter_map(|surface| layer_map.layer_geometry(surface).map(|geo| (geo.loc, surface)))
612                .flat_map(|(loc, surface)| {
613                    AsRenderElements::<R>::render_elements::<WaylandSurfaceRenderElement<R>>(
614                        surface,
615                        renderer,
616                        loc.to_physical_precise_round(output_scale),
617                        Scale::from(output_scale),
618                        alpha,
619                    )
620                    .into_iter()
621                    .map(SpaceRenderElements::Surface)
622                }),
623        );
624
625        lower
626    };
627
628    for space in spaces {
629        let _guard = space.span.enter();
630        if let Some(output_geo) = space.output_geometry(output) {
631            render_elements.extend(
632                space
633                    .render_elements_for_region(renderer, &output_geo, output_scale, alpha)
634                    .into_iter()
635                    .map(|e| SpaceRenderElements::Element(Wrap::from(e))),
636            );
637        }
638    }
639
640    #[cfg(feature = "wayland_frontend")]
641    render_elements.extend(
642        lower
643            .into_iter()
644            .filter_map(|surface| layer_map.layer_geometry(surface).map(|geo| (geo.loc, surface)))
645            .flat_map(|(loc, surface)| {
646                AsRenderElements::<R>::render_elements::<WaylandSurfaceRenderElement<R>>(
647                    surface,
648                    renderer,
649                    loc.to_physical_precise_round(output_scale),
650                    Scale::from(output_scale),
651                    alpha,
652                )
653                .into_iter()
654                .map(SpaceRenderElements::Surface)
655            }),
656    );
657
658    Ok(render_elements)
659}

677    renderer: &mut R,
678    framebuffer: &mut R::Framebuffer<'_>,
679    alpha: f32,
680    age: usize,
681    spaces: S,
682    custom_elements: &'a [C],
683    damage_tracker: &'d mut OutputDamageTracker,
684    clear_color: impl Into<Color32F>,
685) -> Result<RenderOutputResult<'d>, OutputDamageTrackerError<R::Error>>
686where
687    R::TextureId: Clone + Texture + 'static,
688    <E as AsRenderElements<R>>::RenderElement: 'a,
689    SpaceRenderElements<R, <E as AsRenderElements<R>>::RenderElement>:
690        From<Wrap<<E as AsRenderElements<R>>::RenderElement>>,
691{
692    if let OutputModeSource::Auto(renderer_output) = damage_tracker.mode() {
693        assert!(renderer_output == output);
694    }
695
696    let space_render_elements = space_render_elements(renderer, spaces, output, alpha)?;
697
698    let mut render_elements: Vec<OutputRenderElements<'a, R, <E as AsRenderElements<R>>::RenderElement, C>> =
699        Vec::with_capacity(custom_elements.len() + space_render_elements.len());
700
701    render_elements.extend(custom_elements.iter().map(OutputRenderElements::Custom));
702    render_elements.extend(space_render_elements.into_iter().map(OutputRenderElements::Space));
703
704    damage_tracker.render_output(renderer, framebuffer, age, &render_elements, clear_color)
705}
