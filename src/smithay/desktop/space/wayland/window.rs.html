17impl SpaceElement for Window {
18    fn geometry(&self) -> Rectangle<i32, Logical> {
19        self.geometry()
20    }
21
22    fn bbox(&self) -> Rectangle<i32, Logical> {
23        self.bbox_with_popups()
24    }
25
26    fn is_in_input_region(&self, point: &Point<f64, Logical>) -> bool {
27        self.surface_under(*point, WindowSurfaceType::ALL).is_some()
28    }
29
30    fn z_index(&self) -> u8 {
31        self.0.z_index.load(std::sync::atomic::Ordering::SeqCst)
32    }
33
34    fn set_activate(&self, activated: bool) {
35        self.set_activated(activated);
36    }
37
38    #[profiling::function]
39    fn output_enter(&self, output: &Output, overlap: Rectangle<i32, Logical>) {
40        self.user_data().insert_if_missing(WindowOutputUserData::default);
41        {
42            let mut state = self
43                .user_data()
44                .get::<WindowOutputUserData>()
45                .unwrap()
46                .borrow_mut();
47            state.output_overlap.insert(output.downgrade(), overlap);
48            state.output_overlap.retain(|weak, _| weak.is_alive());
49        }
50        self.refresh()
51    }
52
53    #[profiling::function]
54    fn output_leave(&self, output: &Output) {
55        if let Some(state) = self.user_data().get::<WindowOutputUserData>() {
56            state.borrow_mut().output_overlap.retain(|weak, _| weak != output);
57        }
58
59        if let Some(surface) = self.wl_surface() {
60            output_update(output, None, &surface);
61            for (popup, _) in PopupManager::popups_for_surface(&surface) {
62                output_update(output, None, popup.wl_surface());
63            }
64        }
65    }
66
67    #[profiling::function]
68    fn refresh(&self) {
69        self.user_data().insert_if_missing(WindowOutputUserData::default);
70        let state = self.user_data().get::<WindowOutputUserData>().unwrap().borrow();
71
72        if let Some(surface) = self.wl_surface() {
73            for (weak, overlap) in state.output_overlap.iter() {
74                if let Some(output) = weak.upgrade() {
75                    output_update(&output, Some(*overlap), &surface);
76                    for (popup, location) in PopupManager::popups_for_surface(&surface) {
77                        let mut overlap = *overlap;
78                        overlap.loc -= location;
79                        output_update(&output, Some(overlap), popup.wl_surface());
80                    }
81                }
82            }
83        }
84    }
85}

87impl<R> AsRenderElements<R> for Window
88where
89    R: Renderer + ImportAll,
90    R::TextureId: Clone + 'static,
91{
92    type RenderElement = WaylandSurfaceRenderElement<R>;
93
94    #[profiling::function]
95    fn render_elements<C: From<WaylandSurfaceRenderElement<R>>>(
96        &self,
97        renderer: &mut R,
98        location: Point<i32, Physical>,
99        scale: Scale<f64>,
100        alpha: f32,
101    ) -> Vec<C> {
102        match self.underlying_surface() {
103            WindowSurface::Wayland(s) => {
104                let mut render_elements: Vec<C> = Vec::new();
105                let surface = s.wl_surface();
106                let popup_render_elements =
107                    PopupManager::popups_for_surface(surface).flat_map(|(popup, popup_offset)| {
108                        let offset = (self.geometry().loc + popup_offset - popup.geometry().loc)
109                            .to_physical_precise_round(scale);
110
111                        render_elements_from_surface_tree(
112                            renderer,
113                            popup.wl_surface(),
114                            location + offset,
115                            scale,
116                            alpha,
117                            Kind::Unspecified,
118                        )
119                    });
120
121                render_elements.extend(popup_render_elements);
122
123                render_elements.extend(render_elements_from_surface_tree(
124                    renderer,
125                    surface,
126                    location,
127                    scale,
128                    alpha,
129                    Kind::Unspecified,
130                ));
131
132                render_elements
133            }
134            #[cfg(feature = "xwayland")]
135            WindowSurface::X11(s) => AsRenderElements::render_elements(s, renderer, location, scale, alpha),
136        }
137    }
