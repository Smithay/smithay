25impl RendererSurfaceState {
26    fn contains_point<P: Into<Point<f64, Logical>>>(&self, attrs: &SurfaceAttributes, point: P) -> bool {
27        let point = point.into();
28        let size = match self.surface_view.map(|view| view.dst) {
29            None => return false, // If the surface has no size, it can't have an input region.
30            Some(size) => size,
31        };
32
33        let rect = Rectangle::from_size(size).to_f64();
34
35        // The input region is always within the surface itself, so if the surface itself doesn't contain the
36        // point we can return false.
37        if !rect.contains(point) {
38            return false;
39        }
40
41        // If there's no input region, we're done.
42        if attrs.input_region.is_none() {
43            return true;
44        }
45
46        attrs
47            .input_region
48            .as_ref()
49            .unwrap()
50            .contains(point.to_i32_floor())
51    }
52}

54/// Returns the bounding box of a given surface and all its subsurfaces.
55///
56/// - `location` can be set to offset the returned bounding box.
57pub fn bbox_from_surface_tree<P>(surface: &wl_surface::WlSurface, location: P) -> Rectangle<i32, Logical>
58where
59    P: Into<Point<i32, Logical>>,
60{
61    let location = location.into();
62    let mut bounding_box = Rectangle::new(location, (0, 0).into());
63    with_surface_tree_downward(
64        surface,
65        location,
66        |_, states, loc: &Point<i32, Logical>| {
67            let mut loc = *loc;
68            let data = states.data_map.get::<RendererSurfaceStateUserData>();
69
70            if let Some(surface_view) = data.and_then(|d| d.lock().unwrap().surface_view) {
71                loc += surface_view.offset;
72                // Update the bounding box.
73                bounding_box = bounding_box.merge(Rectangle::new(loc, surface_view.dst));
74
75                TraversalAction::DoChildren(loc)
76            } else {
77                // If the parent surface is unmapped, then the child surfaces are hidden as
78                // well, no need to consider them here.
79                TraversalAction::SkipChildren
80            }
81        },
82        |_, _, _| {},
83        |_, _, _| true,
84    );
85    bounding_box
86}

88/// Returns the topmost (sub-)surface under a given position matching the input regions of the surface.
89///
90/// In case no surface input region matches the point [`None`] is returned.
91///
92/// - `point` has to be the position to query, relative to (0, 0) of the given surface + `location`.
93/// - `location` can be used to offset the returned point.
94pub fn under_from_surface_tree<P>(
95    surface: &wl_surface::WlSurface,
96    point: Point<f64, Logical>,
97    location: P,
98    surface_type: WindowSurfaceType,
99) -> Option<(wl_surface::WlSurface, Point<i32, Logical>)>
100where
101    P: Into<Point<i32, Logical>>,
102{
103    let found = RefCell::new(None);
104    with_surface_tree_downward(
105        surface,
106        location.into(),
107        |_, states, location: &Point<i32, Logical>| {
108            let mut location = *location;
109            let data = states.data_map.get::<RendererSurfaceStateUserData>();
110
111            if let Some(surface_view) = data.and_then(|d| d.lock().unwrap().surface_view) {
112                location += surface_view.offset;
113                if surface_type.contains(WindowSurfaceType::SUBSURFACE) {
114                    TraversalAction::DoChildren(location)
115                } else {
116                    TraversalAction::SkipChildren
117                }
118            } else {
119                // We are completely hidden, no point in traversing our children
120                TraversalAction::SkipChildren
121            }
122        },
123        |wl_surface, states, location: &Point<i32, Logical>| {
124            let mut location = *location;
125            let data = states.data_map.get::<RendererSurfaceStateUserData>();
126
127            if let Some(surface_view) = data.and_then(|d| d.lock().unwrap().surface_view) {
128                location += surface_view.offset;
129
130                let contains_the_point = data
131                    .map(|data| {
132                        data.lock()
133                            .unwrap()
134                            .contains_point(&*states.cached_state.get().current(), point - location.to_f64())
135                    })
136                    .unwrap_or(false);
137                if contains_the_point {
138                    *found.borrow_mut() = Some((wl_surface.clone(), location));
139                }
140            }
141        },
142        |_, _, _| {
143            // only continue if the point is not found
144            found.borrow().is_none()
145        },
146    );
147    found.into_inner()
148}

152/// Run a closure on all surfaces of a surface tree
153pub fn with_surfaces_surface_tree<F>(surface: &wl_surface::WlSurface, mut processor: F)
154where
155    F: FnMut(&wl_surface::WlSurface, &SurfaceData),
156{
157    with_surface_tree_downward(
158        surface,
159        (),
160        |_, _, _| TraversalAction::DoChildren(()),
161        |surface, states, _| processor(surface, states),
162        |_, _, _| true,
163    )

166/// Retrieve a previously stored primary scan-out output from a surface
167///
168/// This will always return `None` if [`update_surface_primary_scanout_output`] is not used.
169pub fn surface_primary_scanout_output(
170    _surface: &wl_surface::WlSurface,
171    states: &SurfaceData,
172) -> Option<Output> {
173    states
174        .data_map
175        .insert_if_missing_threadsafe(SurfacePrimaryScanoutOutput::default);
176    let surface_primary_scanout_output = states.data_map.get::<SurfacePrimaryScanoutOutput>().unwrap();
177    surface_primary_scanout_output.lock().unwrap().current_output()
178}

180/// Update the surface primary scan-out output from a output render report
181///
182/// The compare function can be used to alter the behavior for selecting the primary scan-out
183/// output. See [`update_from_render_element_states`](crate::backend::renderer::element::PrimaryScanoutOutput::update_from_render_element_states) for more information about primary scan-out
184/// output selection.
185pub fn update_surface_primary_scanout_output<F>(
186    surface: &wl_surface::WlSurface,
187    output: &Output,
188    surface_data: &SurfaceData,
189    states: &RenderElementStates,
190    compare: F,
191) -> Option<Output>
192where
193    F: for<'a> Fn(&'a Output, &'a RenderElementState, &'a Output, &'a RenderElementState) -> &'a Output,
194{
195    surface_data
196        .data_map
197        .insert_if_missing_threadsafe(SurfacePrimaryScanoutOutput::default);
198    let surface_primary_scanout_output = surface_data
199        .data_map
200        .get::<SurfacePrimaryScanoutOutput>()
201        .unwrap();
202    surface_primary_scanout_output
203        .lock()
204        .unwrap()
205        .update_from_render_element_states(surface, output, states, compare)
206}

208/// Sends frame callbacks for a surface and its subsurfaces with the given `time`.
209///
210/// The frame callbacks for a [`WlSurface`](wl_surface::WlSurface) will only be sent if the
211/// primary scan-out output equals the provided output or if the surface has no primary
212/// scan-out output and the frame callback is overdue. A frame callback is considered
213/// overdue if the last time a frame callback has been sent is greater than the provided
214/// throttle threshold. If the threshold is `None` this will never send frame callbacks
215/// for a surface that is not visible. Specifying [`Duration::ZERO`] as the throttle threshold
216/// will always send frame callbacks for non visible surfaces.
217pub fn send_frames_surface_tree<T, F>(
218    surface: &wl_surface::WlSurface,
219    output: &Output,
220    time: T,
221    throttle: Option<Duration>,
222    mut primary_scan_out_output: F,
223) where
224    T: Into<Duration>,
225    F: FnMut(&wl_surface::WlSurface, &SurfaceData) -> Option<Output>,
226{
227    let time = time.into();
228
229    with_surface_tree_downward(
230        surface,
231        (),
232        |_, _, &()| TraversalAction::DoChildren(()),
233        |surface, states, &()| {
234            states
235                .data_map
236                .insert_if_missing_threadsafe(SurfaceFrameThrottlingState::default);
237            let surface_frame_throttling_state =
238                states.data_map.get::<SurfaceFrameThrottlingState>().unwrap();
239
240            let on_primary_scanout_output = primary_scan_out_output(surface, states)
241                .map(|preferred_output| preferred_output == *output)
242                .unwrap_or(false);
243
244            let frame_overdue = surface_frame_throttling_state.update(time, throttle);
245
246            // We only want to send frame callbacks on the primary scan-out output
247            // or if we have no output and the frame is overdue, this can only
248            // happen if the surface is completely occluded on all outputs
249            let send_frame_callback = on_primary_scanout_output || frame_overdue;
250
251            if send_frame_callback {
252                // the surface may not have any user_data if it is a subsurface and has not
253                // yet been commited
254                for callback in states
255                    .cached_state
256                    .get::<SurfaceAttributes>()
257                    .current()
258                    .frame_callbacks
259                    .drain(..)
260                {
261                    callback.done(time.as_millis() as u32);
262                }
263            }
264        },
265        |_, _, &()| true,
266    );
267}

269/// Sends dmabuf feedback for a surface and its subsurfaces with the given select function.
270///
271/// The dmabuf feedback for a [`WlSurface`](wl_surface::WlSurface) will only be sent if the
272/// primary scan-out output equals the provided output and the surface has requested dmabuf
273/// feedback.
274pub fn send_dmabuf_feedback_surface_tree<'a, P, F>(
275    surface: &wl_surface::WlSurface,
276    output: &Output,
277    mut primary_scan_out_output: P,
278    select_dmabuf_feedback: F,
279) where
280    P: FnMut(&wl_surface::WlSurface, &SurfaceData) -> Option<Output>,
281    F: Fn(&wl_surface::WlSurface, &SurfaceData) -> &'a DmabufFeedback,
282{
283    with_surface_tree_downward(
284        surface,
285        (),
286        |_, _, &()| TraversalAction::DoChildren(()),
287        |surface, states, &()| {
288            let on_primary_scanout_output = primary_scan_out_output(surface, states)
289                .map(|preferred_output| preferred_output == *output)
290                .unwrap_or(false);
291
292            if !on_primary_scanout_output {
293                return;
294            }
295
296            let Some(surface_feedback) = SurfaceDmabufFeedbackState::from_states(states) else {
297                return;
298            };
299
300            let feedback = select_dmabuf_feedback(surface, states);
301            surface_feedback.set_feedback(feedback);
302        },
303        |_, _, &()| true,
304    );
305}

307/// Holds the presentation feedback for a surface
308#[derive(Debug)]
309pub struct SurfacePresentationFeedback {
310    callbacks: Vec<PresentationFeedbackCallback>,
311    flags: wp_presentation_feedback::Kind,
312}
313
314impl SurfacePresentationFeedback {
315    /// Create a [`SurfacePresentationFeedback`] from the surface states.
316    ///
317    /// Returns `None` if the surface has no stored presentation feedback
318    pub fn from_states(states: &SurfaceData, flags: wp_presentation_feedback::Kind) -> Option<Self> {
319        let mut guard = states.cached_state.get::<PresentationFeedbackCachedState>();
320        let presentation_feedback_state = guard.current();
321        if presentation_feedback_state.callbacks.is_empty() {
322            return None;
323        }
324
325        let callbacks = std::mem::take(&mut presentation_feedback_state.callbacks);
326        Some(SurfacePresentationFeedback { callbacks, flags })
327    }
328
329    /// Mark the presentation feedbacks for this surface as presented
330    ///
331    /// If the passed in clk_id does not match the clk_id of a stored
332    /// presentation feedback the feedback will be discarded.
333    pub fn presented(
334        &mut self,
335        output: &Output,
336        clk_id: u32,
337        time: impl Into<Duration>,
338        refresh: Refresh,
339        seq: u64,
340        flags: wp_presentation_feedback::Kind,
341    ) {
342        let time = time.into();
343        for callback in self.callbacks.drain(..) {
344            if callback.clk_id() == clk_id {
345                callback.presented(output, time, refresh, seq, flags | self.flags)
346            } else {
347                callback.discarded()
348            }
349        }
350    }
351
352    /// Mark the presentation feedbacks for this surface as discarded
353    pub fn discarded(&mut self) {
354        for callback in self.callbacks.drain(..) {
355            callback.discarded()
356        }
357    }
358}
359
360impl Drop for SurfacePresentationFeedback {
361    fn drop(&mut self) {
362        self.discarded()
363    }
364}

366/// Stores the [`SurfacePresentationFeedback`] for a specific output
367///
368/// This is intended to be used in combination with [`take_presentation_feedback_surface_tree`].
369#[derive(Debug)]
370pub struct OutputPresentationFeedback {
371    output: WeakOutput,
372    callbacks: Vec<SurfacePresentationFeedback>,
373}
374
375impl OutputPresentationFeedback {
376    /// Create a new [`OutputPresentationFeedback`] for a specific [`Output`]
377    pub fn new(output: &Output) -> Self {
378        OutputPresentationFeedback {
379            output: output.downgrade(),
380            callbacks: Vec::new(),
381        }
382    }
383
384    /// Returns the associated output
385    ///
386    /// Returns `None` if the output has been destroyed.
387    pub fn output(&self) -> Option<Output> {
388        self.output.upgrade()
389    }
390
391    /// Mark all stored [`SurfacePresentationFeedback`]s as presented
392    ///
393    /// The flags passed to this function will be combined with the stored
394    /// per surface flags.
395    pub fn presented<T, Kind>(
396        &mut self,
397        time: T,
398        refresh: Refresh,
399        seq: u64,
400        flags: wp_presentation_feedback::Kind,
401    ) where
402        T: Into<Time<Kind>>,
403        Kind: crate::utils::NonNegativeClockSource,
404    {
405        let time = time.into();
406        let clk_id = Kind::ID as u32;
407        if let Some(output) = self.output.upgrade() {
408            for mut callback in self.callbacks.drain(..) {
409                callback.presented(&output, clk_id, time, refresh, seq, flags);
410            }
411        } else {
412            self.discarded();
413        }
414    }
415
416    /// Mark all stored [`SurfacePresentationFeedback`]s as discarded
417    pub fn discarded(&mut self) {
418        for mut callback in self.callbacks.drain(..) {
419            callback.discarded();
420        }
421    }
422}

424/// Takes the [`PresentationFeedbackCallback`]s from the surface tree
425///
426/// This moves the [`PresentationFeedbackCallback`]s from the surfaces
427/// where the primary scan-out matches the output of the [`OutputPresentationFeedback`]
428/// to the [`OutputPresentationFeedback`]
429///
430/// The flags closure can be used to set special flags per surface like [`wp_presentation_feedback::Kind::ZeroCopy`]
431pub fn take_presentation_feedback_surface_tree<F1, F2>(
432    surface: &wl_surface::WlSurface,
433    output_feedback: &mut OutputPresentationFeedback,
434    mut primary_scan_out_output: F1,
435    mut presentation_feedback_flags: F2,
436) where
437    F1: FnMut(&wl_surface::WlSurface, &SurfaceData) -> Option<Output>,
438    F2: FnMut(&wl_surface::WlSurface, &SurfaceData) -> wp_presentation_feedback::Kind,
439{
440    with_surface_tree_downward(
441        surface,
442        (),
443        |_, _, &()| TraversalAction::DoChildren(()),
444        |surface, states, &()| {
445            let on_primary_scanout_output = primary_scan_out_output(surface, states)
446                .map(|preferred_output| preferred_output == output_feedback.output)
447                .unwrap_or(false);
448
449            if !on_primary_scanout_output {
450                return;
451            }
452
453            let flags = presentation_feedback_flags(surface, states);
454            if let Some(feedback) = SurfacePresentationFeedback::from_states(states, flags) {
455                output_feedback.callbacks.push(feedback);
456            }
457        },
458        |_, _, &()| true,
459    );
460}

462/// Retrieves the per surface [`wp_presentation_feedback::Kind`] flags
463///
464/// This will return [`wp_presentation_feedback::Kind::ZeroCopy`] if the surface
465/// has been presented using zero-copy according to the [`RenderElementState`]
466/// in the provided [`RenderElementStates`]
467pub fn surface_presentation_feedback_flags_from_states(
468    surface: &wl_surface::WlSurface,
469    states: &RenderElementStates,
470) -> wp_presentation_feedback::Kind {
471    let zero_copy = states
472        .element_render_state(surface)
473        .map(|state| state.presentation_state == RenderElementPresentationState::ZeroCopy)
474        .unwrap_or(false);
475
476    if zero_copy {
477        wp_presentation_feedback::Kind::ZeroCopy
478    } else {
479        wp_presentation_feedback::Kind::empty()
480    }
481}

483#[derive(Debug, Default)]
484struct SurfaceFrameThrottlingState(Mutex<Option<Duration>>);
485
486impl SurfaceFrameThrottlingState {
487    pub fn update(&self, time: Duration, throttle: Option<Duration>) -> bool {
488        if let Some(throttle) = throttle {
489            let mut guard = self.0.lock().unwrap();
490            let send_throttled_frame = guard
491                .map(|last| time.saturating_sub(last) > throttle)
492                .unwrap_or(true);
493            if send_throttled_frame {
494                *guard = Some(time);
495            }
496            send_throttled_frame
497        } else {
498            false
499        }
500    }
