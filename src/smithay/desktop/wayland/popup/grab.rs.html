31/// Defines the possible errors that
32/// can be returned from [`PopupManager::grab_popup`]
33#[derive(Debug, Error)]
34pub enum PopupGrabError {
35    /// This resource has been destroyed and can no longer be used.
36    #[error(transparent)]
37    DeadResource(#[from] DeadResource),
38    /// The client tried to grab a popup after it's parent has been dismissed
39    #[error("the parent of the popup has been already dismissed")]
40    ParentDismissed,
41    /// The client tried to grab a popup after it has been mapped
42    #[error("tried to grab after being mapped")]
43    InvalidGrab,
44    /// The client tried to grab a popup which is not the topmost
45    #[error("popup was not created on the topmost popup")]
46    NotTheTopmostPopup,
47}

49/// Defines the possibly strategies
50/// for the [`PopupGrab::ungrab`] operation
51#[derive(Debug, Copy, Clone)]
52pub enum PopupUngrabStrategy {
53    /// Only ungrab the topmost popup
54    Topmost,
55    /// Ungrab all active popups
56    All,
57}

59#[derive(Debug, Default)]
60struct PopupGrabInternal {
61    serial: Option<Serial>,
62    active_grabs: Vec<PopupKind>,
63    dismissed_grabs: Vec<PopupKind>,
64}
65
66impl PopupGrabInternal {
67    fn has_any_grabs(&self) -> bool {
68        !self.active_grabs.is_empty() || !self.dismissed_grabs.is_empty()
69    }
70
71    fn has_active_grabs(&self) -> bool {
72        !self.active_grabs.is_empty()
73    }
74
75    fn current_grab(&self) -> Option<&WlSurface> {
76        self.active_grabs
77            .iter()
78            .rev()
79            .find(|p| p.alive())
80            .map(|p| p.wl_surface())
81    }
82
83    fn is_dismissed(&self, surface: &WlSurface) -> bool {
84        self.dismissed_grabs.iter().any(|p| p.wl_surface() == surface)
85    }
86
87    fn append_grab(&mut self, popup: &PopupKind) {
88        self.active_grabs.push(popup.clone());
89    }
90
91    fn cleanup(&mut self) {
92        let mut i = 0;
93        while i < self.active_grabs.len() {
94            if !self.active_grabs[i].alive() {
95                let grab = self.active_grabs.remove(i);
96                self.dismissed_grabs.push(grab);
97            } else {
98                i += 1;
99            }
100        }
101
102        self.dismissed_grabs.retain(|p| p.alive());
103    }
104}

107pub(super) struct PopupGrabInner {
108    internal: Arc<Mutex<PopupGrabInternal>>,
109}

111impl PopupGrabInner {
112    pub(super) fn has_any_grabs(&self) -> bool {
113        let guard = self.internal.lock().unwrap();
114        guard.has_any_grabs()
115    }
116
117    pub(super) fn has_active_grabs(&self) -> bool {
118        let guard = self.internal.lock().unwrap();
119        guard.has_active_grabs()
120    }
121
122    fn current_grab(&self) -> Option<PopupKind> {
123        let guard = self.internal.lock().unwrap();
124        guard.active_grabs.iter().rev().find(|p| p.alive()).cloned()
125    }
126
127    pub(super) fn cleanup(&self) {
128        let mut guard = self.internal.lock().unwrap();
129        guard.cleanup();
130    }
131
132    pub(super) fn grab(&self, popup: &PopupKind, serial: Serial) -> Result<Option<Serial>, PopupGrabError> {
133        let parent = popup.parent().ok_or(DeadResource)?;
134
135        self.cleanup();
136
137        let mut guard = self.internal.lock().unwrap();
138
139        if let Some(grab) = guard.current_grab() {
140            if grab != &parent {
141                // If the parent is a grabbing popup which has already been dismissed, this popup will be immediately dismissed.
142                if guard.is_dismissed(&parent) {
143                    return Err(PopupGrabError::ParentDismissed);
144                }
145
146                // If the parent is a popup that did not take an explicit grab, an error will be raised.
147                return Err(PopupGrabError::NotTheTopmostPopup);
148            }
149        }
150
151        guard.append_grab(popup);
152
153        Ok(guard.serial.replace(serial))
154    }
155
156    fn ungrab(&self, root: &WlSurface, strategy: PopupUngrabStrategy) -> Option<WlSurface> {
157        let mut guard = self.internal.lock().unwrap();
158        let dismissed = match strategy {
159            PopupUngrabStrategy::Topmost => {
160                if let Some(grab) = guard.active_grabs.pop() {
161                    let dismissed = PopupManager::dismiss_popup(root, &grab);
162
163                    if dismissed.is_ok() {
164                        guard.dismissed_grabs.push(grab);
165                    }
166
167                    dismissed
168                } else {
169                    Ok(())
170                }
171            }
172            PopupUngrabStrategy::All => {
173                let grabs = guard.active_grabs.drain(..).collect::<Vec<_>>();
174
175                if let Some(grab) = grabs.first() {
176                    let dismissed = PopupManager::dismiss_popup(root, grab);
177
178                    if dismissed.is_ok() {
179                        guard.dismissed_grabs.push(grab.clone());
180                        guard.dismissed_grabs.extend(grabs);
181                    }
182
183                    dismissed
184                } else {
185                    Ok(())
186                }
187            }
188        };
189
190        if dismissed.is_err() {
191            // If dismiss_popup returns Err(DeadResource) there is not much what
192            // can do about it here, we just remove all our grabs as they are dead now
193            // anyway. The pointer/keyboard grab will be unset automatically so we
194            // should be fine.
195            guard.active_grabs.drain(..);
196        }
197
198        guard.current_grab().cloned()
199    }
200}

202/// Represents the explicit grab a client requested for a popup
203///
204/// An explicit grab can be used by a client to redirect all keyboard
205/// input to a single popup. The focus of the keyboard will stay on
206/// the popup for as long as the grab is valid, that is as long as the
207/// compositor did not call [`ungrab`](PopupGrab::ungrab) or the client
208/// did not destroy the popup. A grab can be nested by requesting a grab
209/// on a popup who's parent is the currently grabbed popup. The grab will
210/// be returned to the parent after the popup has been dismissed.
211///
212/// This module also provides default implementations for [`KeyboardGrab`] and
213/// [`PointerGrab`] that implement the behavior described in the [`xdg-shell`](https://wayland.app/protocols/xdg-shell#xdg_popup:request:grab)
214/// specification. See [`PopupKeyboardGrab`] and [`PopupPointerGrab`] for more
215/// information on the default implementations.
216///
217/// In case the implemented behavior is not suited for your use-case the grab can be
218/// either decorated or a custom [`KeyboardGrab`]/[`PointerGrab`] can use the methods
219/// on the [`PopupGrab`] to implement a custom behavior.
220///
221/// One example would be to use a timer to automatically dismiss the popup after some
222/// timeout.
223///
224/// The grab is obtained by calling [`PopupManager::grap_popup`](super::PopupManager::grab_popup).
225pub struct PopupGrab<D>
226where
227    D: SeatHandler + 'static,
228    <D as SeatHandler>::KeyboardFocus: WaylandFocus,
229    <D as SeatHandler>::PointerFocus: From<<D as SeatHandler>::KeyboardFocus> + WaylandFocus,
230{
231    root: <D as SeatHandler>::KeyboardFocus,
232    serial: Serial,
233    previous_serial: Option<Serial>,
234    toplevel_grab: PopupGrabInner,
235    keyboard_handle: Option<KeyboardHandle<D>>,
236    keyboard_grab_start_data: KeyboardGrabStartData<D>,
237    pointer_grab_start_data: PointerGrabStartData<D>,
238}

246    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {

380    fn unset_keyboard_grab(&self, data: &mut D, serial: Serial) {
381        if let Some(keyboard) = self.keyboard_handle.as_ref() {
382            if keyboard.is_grabbed()
383                && (keyboard.has_grab(self.serial)
384                    || keyboard.has_grab(self.previous_serial.unwrap_or(self.serial)))

435impl<D> KeyboardGrab<D> for PopupKeyboardGrab<D>
436where
437    D: SeatHandler + 'static,
438    <D as SeatHandler>::KeyboardFocus: WaylandFocus + From<PopupKind>,
439    <D as SeatHandler>::PointerFocus: From<<D as SeatHandler>::KeyboardFocus> + WaylandFocus,
440{
441    fn input(
442        &mut self,
443        data: &mut D,
444        handle: &mut KeyboardInnerHandle<'_, D>,
445        keycode: Keycode,
446        state: KeyState,
447        modifiers: Option<ModifiersState>,
448        serial: Serial,
449        time: u32,
450    ) {
451        // Check if the grab changed and update the focus
452        // If the grab has ended this will return the root
453        // surface to restore the client expected focus.
454        if let Some(focus) = self.popup_grab.current_grab() {
455            handle.set_focus(data, Some(focus), serial);
456        }
457
458        if self.popup_grab.has_ended() {
459            handle.unset_grab(self, data, serial, false);
460        }
461
462        handle.input(data, keycode, state, modifiers, serial, time)
463    }
464
465    fn set_focus(
466        &mut self,
467        data: &mut D,
468        handle: &mut KeyboardInnerHandle<'_, D>,
469        focus: Option<<D as SeatHandler>::KeyboardFocus>,
470        serial: Serial,
471    ) {
472        // Ignore focus changes unless the grab has ended
473        if self.popup_grab.has_ended() {
474            handle.set_focus(data, focus, serial);
475            handle.unset_grab(self, data, serial, false);
476            return;
477        }
478
479        // Allow to set the focus to the current grab, this can
480        // happen if the user initially sets the focus to
481        // popup instead of relying on the grab behavior
482        if self.popup_grab.current_grab() == focus {
483            handle.set_focus(data, focus, serial);
484        }
485    }
486
487    fn start_data(&self) -> &KeyboardGrabStartData<D> {
488        self.popup_grab.keyboard_grab_start_data()
489    }
490
491    fn unset(&mut self, _data: &mut D) {}
492}

543impl<D> PointerGrab<D> for PopupPointerGrab<D>
544where
545    D: SeatHandler + 'static,
546    <D as SeatHandler>::KeyboardFocus: WaylandFocus + From<PopupKind>,
547    <D as SeatHandler>::PointerFocus: From<<D as SeatHandler>::KeyboardFocus> + WaylandFocus,
548{
549    fn motion(
550        &mut self,
551        data: &mut D,
552        handle: &mut PointerInnerHandle<'_, D>,
553        focus: Option<(<D as SeatHandler>::PointerFocus, Point<f64, Logical>)>,
554        event: &MotionEvent,
555    ) {
556        if self.popup_grab.has_ended() {
557            handle.unset_grab(self, data, event.serial, event.time, true);
558            return;
559        }
560
561        // Check that the focus is of the same client as the grab
562        // If yes allow it, if not unset the focus.
563        if focus
564            .as_ref()
565            .and_then(|f1| {
566                self.popup_grab
567                    .current_grab()
568                    .as_ref()
569                    .and_then(|f2| f2.wl_surface())
570                    .map(|s| f1.0.same_client_as(&s.id()))
571            })
572            .unwrap_or(false)
573        {
574            handle.motion(data, focus, event);
575        } else {
576            handle.motion(data, None, event);
577        }
578    }

585        event: &RelativeMotionEvent,

590    fn button(&mut self, data: &mut D, handle: &mut PointerInnerHandle<'_, D>, event: &ButtonEvent) {
591        let serial = event.serial;
592        let time = event.time;
593        let state = event.state;
594
595        if self.popup_grab.has_ended() {
596            handle.unset_grab(self, data, serial, time, true);
597            handle.button(data, event);
598            return;
599        }
600
601        // Check if the the client of the focused surface is still equal to the grabbed surface client
602        // if not the popup will be dismissed
603        if state == ButtonState::Pressed
604            && !handle
605                .current_focus()
606                .and_then(|f| {
607                    self.popup_grab
608                        .current_grab()
609                        .and_then(|f2| f.0.wl_surface().map(|s| f2.same_client_as(&s.id())))
610                })
611                .unwrap_or(false)
612        {
613            let _ = self.popup_grab.ungrab(PopupUngrabStrategy::All);
614            handle.unset_grab(self, data, serial, time, true);
615            handle.button(data, event);
616            return;
617        }
618
619        handle.button(data, event);
620    }
