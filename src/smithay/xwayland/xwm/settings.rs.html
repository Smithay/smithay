26#[derive(Debug)]
27struct Setting {
28    val: Value,

32#[derive(Debug)]
33/// XSETTINGS value
34pub enum Value {
35    /// String value
36    String(String),
37    /// Integer value
38    Integer(i32),
39    /// Color value
40    Color([u16; 4]),
41}
42
43impl From<String> for Value {
44    fn from(value: String) -> Self {
45        Value::String(value)
46    }
47}
48impl From<i32> for Value {
49    fn from(value: i32) -> Self {
50        Value::Integer(value)
51    }
52}
53impl From<[u16; 4]> for Value {
54    fn from(value: [u16; 4]) -> Self {
55        Value::Color(value)
56    }
57}

59const fn pad(len: usize) -> usize {
60    (4 - (len % 4)) % 4
61}

63/// XSETTINGS Name validation error
64#[derive(Debug, thiserror::Error)]
65pub enum NameError {
66    /// Name contained an invalid character. Only a-z, A-Z, 0-9, /, _ are allowed.
67    #[error("Name contained an invalid character. Only a-z, A-Z, 0-9, /, _ are allowed.")]
68    DisallowedCharacter,
69    /// Name started or ended on a slash.
70    #[error("Name started or ended on a slash.")]
71    SlashAtStartOrEnd,
72    /// Name contained two consecutive slashes.
73    #[error("Name contained two consecutive slashes.")]
74    DoubleSlash,
75    /// Name contained a number at the start or immediately following a slash.
76    #[error("Name contained a number at the start or immediately following a slash.")]
77    LeadingNumber,
78    /// Name was empty.
79    #[error("Name was empty.")]
80    EmptyName,
81}

83impl XSettings {
84    pub(super) fn new(
85        conn: &Arc<RustConnection>,
86        depth: u8,
87        root: Window,
88        atoms: &super::Atoms,
89    ) -> Result<XSettings, ReplyOrIdError> {
90        let win = conn.generate_id()?;
91        conn.create_window(
92            depth,
93            win,
94            root,
95            // x, y, width, height, border width
96            0,
97            0,
98            1,
99            1,
100            0,
101            WindowClass::INPUT_OUTPUT,
102            x11rb::COPY_FROM_PARENT,
103            &Default::default(),
104        )?;
105        conn.set_selection_owner(win, atoms._XSETTINGS_S0, x11rb::CURRENT_TIME)?;
106        debug!(window = win, "Created XSettings window");
107
108        let mut this = XSettings {
109            window: OwnedX11Window::new(win, conn),
110            atoms: *atoms,
111            values: HashMap::default(),
112            current_serial: 0,
113        };
114        this.update(conn)?;
115        Ok(this)
116    }
117
118    pub fn set(&mut self, name: impl Into<String>, value: impl Into<Value>) -> Result<(), NameError> {
119        let name = name.into();
120        if name
121            .find(|c: char| !(c.is_ascii_alphanumeric() || c == '/' || c == '_'))
122            .is_some()
123        {
124            return Err(NameError::DisallowedCharacter);
125        }
126        if name.is_empty() {
127            return Err(NameError::EmptyName);
128        }
129        if name.starts_with('/') || name.ends_with('/') {
130            return Err(NameError::SlashAtStartOrEnd);
131        }
132        if name.split('/').any(str::is_empty) {
133            return Err(NameError::DoubleSlash);
134        }
135        if name
136            .split('/')
137            .any(|s| str::starts_with(s, |c: char| c.is_ascii_digit()))
138        {
139            return Err(NameError::LeadingNumber);
140        }
141
142        self.values.insert(
143            name,
144            Setting {
145                val: value.into(),
146                last_changed_serial: self.current_serial,
147            },
148        );
149
150        Ok(())
151    }
152
153    #[allow(dead_code)]
154    pub fn get<Q>(&self, name: &Q) -> Option<&Value>
155    where
156        Q: Hash + Eq + ?Sized,
157        String: Borrow<Q>,
158    {
159        self.values.get(name).map(|setting| &setting.val)
160    }
161
162    #[allow(dead_code)]
163    pub fn remove<Q>(&mut self, name: &Q) -> Option<Value>
164    where
165        Q: Hash + Eq + ?Sized,
166        String: Borrow<Q>,
167    {
168        self.values.remove(name).map(|setting| setting.val)
169    }
170
171    fn serialize(&mut self) -> Vec<u8> {
172        let mut data = Vec::new();
173
174        if cfg!(target_endian = "little") {
175            data.push(0);
176        } else {
177            data.push(1);
178        }
179        data.extend_from_slice(&[0u8; 3]);
180        data.extend_from_slice(&self.current_serial.to_ne_bytes());
181        data.extend_from_slice(&(self.values.len() as i32).to_ne_bytes());
182
183        for (name, setting) in self.values.iter() {
184            debug_assert!(name.is_ascii());
185            let name = name.as_bytes();
186
187            data.extend(&setting.val._type().to_ne_bytes());
188            data.push(0u8);
189            data.extend(&(name.len() as u16).to_ne_bytes());
190            data.extend(name);
191            data.extend(std::iter::repeat_n(0u8, pad(name.len())));
192            data.extend(&setting.last_changed_serial.to_ne_bytes());
193
194            setting.val.serialize(&mut data);
195        }
196
197        self.current_serial = self.current_serial.wrapping_add(1);
198
199        data
200    }
201
202    pub(super) fn update(&mut self, conn: &RustConnection) -> Result<(), ConnectionError> {
203        conn.change_property8(
204            PropMode::REPLACE,
205            *self.window,
206            self.atoms._XSETTINGS_SETTINGS,
207            self.atoms._XSETTINGS_SETTINGS,
208            &self.serialize(),
209        )
210        .map(|_| ())
211    }
212}
213
214impl Value {
215    fn _type(&self) -> i8 {
216        match self {
217            Value::Integer(_) => 0,
218            Value::String(_) => 1,
219            Value::Color(_) => 2,
220        }
221    }
222
223    fn serialize(&self, data: &mut Vec<u8>) {
224        match self {
225            Value::Integer(val) => {
226                data.extend(&val.to_ne_bytes());
227            }
228            Value::String(val) => {
229                let val = val.as_bytes();
230                data.extend(&(val.len() as u32).to_ne_bytes());
231                data.extend(val);
232                data.extend(std::iter::repeat_n(0u8, pad(val.len())));
233            }
234            Value::Color(val) => {
235                for component in val {
236                    data.extend(&component.to_ne_bytes());
237                }
238            }
239        }
240    }
