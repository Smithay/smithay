32/// XSETTINGS value
33pub enum Value {
34    /// String value
35    String(String),
36    /// Integer value
37    Integer(i32),
38    /// Color value
39    Color([u16; 4]),
40}
41
42impl From<String> for Value {
43    fn from(value: String) -> Self {
44        Value::String(value)
45    }
46}
47impl From<i32> for Value {
48    fn from(value: i32) -> Self {
49        Value::Integer(value)
50    }
51}
52impl From<[u16; 4]> for Value {
53    fn from(value: [u16; 4]) -> Self {
54        Value::Color(value)
55    }
56}

58const fn pad(len: usize) -> usize {
59    (4 - (len % 4)) % 4
60}

62/// XSETTINGS Name validation error
63#[derive(Debug, thiserror::Error)]
64pub enum NameError {
65    /// Name contained an invalid character. Only a-z, A-Z, 0-9, /, _ are allowed.
66    #[error("Name contained an invalid character. Only a-z, A-Z, 0-9, /, _ are allowed.")]
67    DisallowedCharacter,
68    /// Name started or ended on a slash.
69    #[error("Name started or ended on a slash.")]
70    SlashAtStartOrEnd,
71    /// Name contained two consecutive slashes.
72    #[error("Name contained two consecutive slashes.")]
73    DoubleSlash,
74    /// Name contained a number at the start or immediately following a slash.
75    #[error("Name contained a number at the start or immediately following a slash.")]
76    LeadingNumber,
77    /// Name was empty.
78    #[error("Name was empty.")]
79    EmptyName,
80}

82impl XSettings {
83    pub(super) fn new(
84        conn: &RustConnection,
85        depth: u8,
86        root: Window,
87        atoms: &super::Atoms,
88    ) -> Result<XSettings, ReplyOrIdError> {
89        let win = conn.generate_id()?;
90        conn.create_window(
91            depth,
92            win,
93            root,
94            // x, y, width, height, border width
95            0,
96            0,
97            1,
98            1,
99            0,
100            WindowClass::INPUT_OUTPUT,
101            x11rb::COPY_FROM_PARENT,
102            &Default::default(),
103        )?;
104        conn.set_selection_owner(win, atoms._XSETTINGS_S0, x11rb::CURRENT_TIME)?;
105        debug!(window = win, "Created XSettings window");
106
107        let mut this = XSettings {
108            window: win,
109            atoms: *atoms,
110            values: HashMap::default(),
111            current_serial: 0,
112        };
113        this.update(conn)?;
114        Ok(this)
115    }
116
117    pub fn set(&mut self, name: impl Into<String>, value: impl Into<Value>) -> Result<(), NameError> {
118        let name = name.into();
119        if name
120            .find(|c: char| !(c.is_ascii_alphanumeric() || c == '/' || c == '_'))
121            .is_some()
122        {
123            return Err(NameError::DisallowedCharacter);
124        }
125        if name.is_empty() {
126            return Err(NameError::EmptyName);
127        }
128        if name.starts_with('/') || name.ends_with('/') {
129            return Err(NameError::SlashAtStartOrEnd);
130        }
131        if name.split('/').any(str::is_empty) {
132            return Err(NameError::DoubleSlash);
133        }
134        if name
135            .split('/')
136            .any(|s| str::starts_with(s, |c: char| c.is_ascii_digit()))
137        {
138            return Err(NameError::LeadingNumber);
139        }
140
141        self.values.insert(
142            name,
143            Setting {
144                val: value.into(),
145                last_changed_serial: self.current_serial,
146            },
147        );
148
149        Ok(())
150    }
151
152    #[allow(dead_code)]
153    pub fn get<Q>(&self, name: &Q) -> Option<&Value>
154    where
155        Q: Hash + Eq + ?Sized,
156        String: Borrow<Q>,
157    {
158        self.values.get(name).map(|setting| &setting.val)
159    }
160
161    #[allow(dead_code)]
162    pub fn remove<Q>(&mut self, name: &Q) -> Option<Value>
163    where
164        Q: Hash + Eq + ?Sized,
165        String: Borrow<Q>,
166    {
167        self.values.remove(name).map(|setting| setting.val)
168    }
169
170    fn serialize(&mut self) -> Vec<u8> {
171        let mut data = Vec::new();
172
173        if cfg!(target_endian = "little") {
174            data.push(0);
175        } else {
176            data.push(1);
177        }
178        data.extend_from_slice(&[0u8; 3]);
179        data.extend_from_slice(&self.current_serial.to_ne_bytes());
180        data.extend_from_slice(&(self.values.len() as i32).to_ne_bytes());
181
182        for (name, setting) in self.values.iter() {
183            debug_assert!(name.is_ascii());
184            let name = name.as_bytes();
185
186            data.extend(&setting.val._type().to_ne_bytes());
187            data.push(0u8);
188            data.extend(&(name.len() as u16).to_ne_bytes());
189            data.extend(name);
190            data.extend(std::iter::repeat(0u8).take(pad(name.len())));
191            data.extend(&setting.last_changed_serial.to_ne_bytes());
192
193            setting.val.serialize(&mut data);
194        }
195
196        self.current_serial = self.current_serial.wrapping_add(1);
197
198        data
199    }
200
201    pub(super) fn update(&mut self, conn: &RustConnection) -> Result<(), ConnectionError> {
202        conn.change_property8(
203            PropMode::REPLACE,
204            self.window,
205            self.atoms._XSETTINGS_SETTINGS,
206            self.atoms._XSETTINGS_SETTINGS,
207            &self.serialize(),
208        )
209        .map(|_| ())
210    }
211}
212
213impl Value {
214    fn _type(&self) -> i8 {
215        match self {
216            Value::Integer(_) => 0,
217            Value::String(_) => 1,
218            Value::Color(_) => 2,
219        }
220    }
221
222    fn serialize(&self, data: &mut Vec<u8>) {
223        match self {
224            Value::Integer(val) => {
225                data.extend(&val.to_ne_bytes());
226            }
227            Value::String(val) => {
228                let val = val.as_bytes();
229                data.extend(&(val.len() as u32).to_ne_bytes());
230                data.extend(val);
231                data.extend(std::iter::repeat(0u8).take(pad(val.len())));
232            }
233            Value::Color(val) => {
234                for component in val {
235                    data.extend(&component.to_ne_bytes());
236                }
237            }
238        }
239    }
