1//!
2//! Xwayland Window Manager module
3//!
4//! Provides an [`X11Wm`] type, which will register itself as a window manager
5//! for a previously spawned Xwayland instances, allowing
6//! backwards-compatibility by seamlessly integrating X11 windows into a wayland
7//! compositor.
8//!
9//! To use this functionality you must first spawn an
10//! [`XWayland`](super::XWayland) instance to attach a [`X11Wm`] to.
11//!
12//! # Associating an [`X11Surface`] with a [`WlSurface`][wayland_server::protocol::wl_surface::WlSurface]
13//!
14//! When an X11 window is created, XWayland will associate it with a
15//! `wl_surface` that it creates via the wayland connection. This happens in two
16//! steps:
17//!
18//!  - Via the [xwayland shell](crate::wayland::xwayland_shell) protocol, a
19//!    serial number is set on the surface, and it is given the
20//!    "xwayland_shell" role.
21//!  - Via the X11 connection, a matching `WL_SURFACE_SERIAL` atom is set on the
22//!    X11 window, which can be queried with
23//!    [`X11Surface::wl_surface_serial()`].
24//!
25//! Note that these two steps can happen in any order.
26//!
27//! # Example
28//!
29//! ```no_run
30//! #  use smithay::wayland::xwayland_shell::{XWaylandShellHandler, XWaylandShellState};
31//! #  use smithay::wayland::selection::{SelectionTarget, SelectionHandler, data_device::{DataDeviceHandler, DataDeviceState, WaylandDndGrabHandler}};
32//! #  use smithay::xwayland::{XWayland, XWaylandEvent, X11Wm, X11Surface, XwmHandler, xwm::{XwmId, ResizeEdge, Reorder}};
33//! #  use smithay::utils::{Rectangle, Logical};
34//! #  use std::os::unix::io::OwnedFd;
35//! #  use std::process::Stdio;
36//! #
37//! struct State { /* ... */ }
38//! # impl XWaylandShellHandler for State {
39//! #     fn xwayland_shell_state(&mut self) -> &mut XWaylandShellState {
40//! #         unreachable!()
41//! #     }
42//! # }
43//! # use smithay::reexports::wayland_server::protocol::wl_surface::WlSurface;
44//! # use smithay::input::{Seat, SeatState, SeatHandler, pointer::CursorImageStatus, dnd::DndGrabHandler};
45//! # use smithay::backend::input::KeyState;
46//! # use smithay::input::{
47//! #   pointer::{PointerTarget, AxisFrame, MotionEvent, ButtonEvent, RelativeMotionEvent,
48//! #             GestureSwipeBeginEvent, GestureSwipeUpdateEvent, GestureSwipeEndEvent,
49//! #             GesturePinchBeginEvent, GesturePinchUpdateEvent, GesturePinchEndEvent,
50//! #             GestureHoldBeginEvent, GestureHoldEndEvent},
51//! #   keyboard::{KeyboardTarget, KeysymHandle, ModifiersState},
52//! #   touch::{DownEvent, UpEvent, MotionEvent as TouchMotionEvent, ShapeEvent, OrientationEvent, TouchTarget},
53//! # };
54//! # use smithay::utils::{IsAlive, Serial};
55//! #
56//! # type Target = WlSurface;
57//! # impl SeatHandler for State {
58//! #     type KeyboardFocus = Target;
59//! #     type PointerFocus = Target;
60//! #     type TouchFocus = Target;
61//! #     fn seat_state(&mut self) -> &mut SeatState<Self> { unreachable!() }
62//! #     fn focus_changed(&mut self, seat: &Seat<Self>, focused: Option<&Target>) {}
63//! #     fn cursor_image(&mut self, seat: &Seat<Self>, image: CursorImageStatus) {}
64//! # }
65//! # impl DndGrabHandler for State {}
66//! # impl DataDeviceHandler for State {
67//! #     fn data_device_state(&mut self) -> &mut DataDeviceState { unreachable!() }
68//! # }
69//! # impl SelectionHandler for State {
70//! #     type SelectionUserData = ();
71//! # }
72//! # impl WaylandDndGrabHandler for State {}
73//!
74//! impl XwmHandler for State {
75//!     fn xwm_state(&mut self, xwm: XwmId) -> &mut X11Wm {
76//!         // ...
77//! #       unreachable!()
78//!     }
79//!     fn new_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
80//!     fn new_override_redirect_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
81//!     fn map_window_request(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
82//!     fn mapped_override_redirect_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
83//!     fn unmapped_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
84//!     fn destroyed_window(&mut self, xwm: XwmId, window: X11Surface) { /* ... */ }
85//!     fn configure_request(&mut self, xwm: XwmId, window: X11Surface, x: Option<i32>, y: Option<i32>, w: Option<u32>, h: Option<u32>, reorder: Option<Reorder>) { /* ... */ }
86//!     fn configure_notify(&mut self, xwm: XwmId, window: X11Surface, geometry: Rectangle<i32, Logical>, above: Option<u32>) { /* ... */ }
87//!     fn resize_request(&mut self, xwm: XwmId, window: X11Surface, button: u32, resize_edge: ResizeEdge) { /* ... */ }
88//!     fn move_request(&mut self, xwm: XwmId, window: X11Surface, button: u32) { /* ... */ }
89//!     fn send_selection(&mut self, xwm: XwmId, selection: SelectionTarget, mime_type: String, fd: OwnedFd) { /* ... */ }
90//! }
91//! #
92//! # let dh = unreachable!();
93//! # let handle: smithay::reexports::calloop::LoopHandle<'static, State> = unreachable!();
94//!
95//! let (xwayland, client) = XWayland::spawn(
96//!     &dh,
97//!     None,
98//!     std::iter::empty::<(String, String)>(),
99//!     true,
100//!     Stdio::null(),
101//!     Stdio::null(),
102//!     |_| (),
103//! )
104//! .expect("failed to start XWayland");
105
106//! let ret = handle.insert_source(xwayland, move |event, _, data| match event {
107//!     XWaylandEvent::Ready {
108//!         x11_socket,
109//!         display_number: _,
110//!     } => {
111//!         let wm = X11Wm::start_wm(
112//!             handle.clone(),
113//!             &dh,
114//!             x11_socket,
115//!             client.clone(),
116//!         )
117//!         .expect("Failed to attach X11 Window Manager");
118//!         
119//!         // store the WM somewhere
120//!     }
121//!     XWaylandEvent::Error => eprintln!("XWayland failed to start!"),
122//! }); if let Err(e) = ret { tracing::error!( "Failed to insert the
123//! XWaylandSource into the event loop: {}", e ); }
124//! ```

310/// Window asks to be re-stacked
311#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord, Hash)]
312pub enum Reorder {
313    /// to the top of the stack
314    Top,
315    /// directly above the given window id
316    Above(X11Window),
317    /// directly below the given window id
318    Below(X11Window),
319    /// to the bottom of the stack
320    Bottom,
321}
322
323enum StackingDirection {
324    Downwards,
325    Upwards,
326}
327
328impl StackingDirection {
329    fn pos_comparator(&self, pos: usize, last_pos: usize) -> bool {
330        match self {
331            Self::Downwards => last_pos < pos,
332            Self::Upwards => last_pos > pos,
333        }
334    }
335
336    fn stack_mode(&self) -> StackMode {
337        match self {
338            Self::Downwards => StackMode::BELOW,
339            Self::Upwards => StackMode::ABOVE,
340        }
341    }
342}

603impl Drop for X11Wm {
604    fn drop(&mut self) {
605        xwm_id::remove(self.id.0);
606
607        // Break reference cycle caused by deferred_sync hook
608        for window in std::mem::take(&mut self.windows) {
609            window.handle_destroyed();
610        }
611    }

616    pending: Arc<AtomicBool>,
617    signal: ping::Ping,
618    conn: Weak<RustConnection>,
619}

621impl FocusReleaseHandle {
622    fn new(conn: &Arc<RustConnection>) -> std::io::Result<(Self, ping::PingSource)> {
623        let (signal, source) = ping::make_ping()?;
624        Ok((
625            Self {
626                pending: Arc::new(AtomicBool::new(false)),
627                signal,
628                conn: Arc::downgrade(conn),
629            },
630            source,
631        ))
632    }
633
634    fn schedule(&self) {
635        self.pending.store(true, Ordering::Release);
636        self.signal.ping();
637    }
638
639    fn cancel(&self) {
640        self.pending.store(false, Ordering::Release);
641    }
642
643    fn dispatch(&self) {
644        if !self.pending.swap(false, Ordering::AcqRel) {
645            return;
646        }
647        let Some(conn) = self.conn.upgrade() else { return };
648        if let Err(err) = conn.set_input_focus(InputFocus::NONE, x11rb::NONE, x11rb::CURRENT_TIME) {
649            warn!("Unable to release X11 keyboard focus: {}", err);
650        }
651        let _ = conn.flush();
652    }

1055    /// Raises a window in the internal X11 state
1056    ///
1057    /// Needs to be called to match raising of windows inside the compositor to keep the stacking order
1058    /// in sync with the compositor to avoid erroneous behavior.
1059    pub fn raise_window<'a, W: X11Relatable + 'a>(&mut self, window: &'a W) -> Result<(), ConnectionError> {
1060        if let Some(elem) = self.windows.iter().find(|s| window.is_window(s)) {
1061            if self.client_list_stacking.last() == Some(&elem.window_id()) {
1062                return Ok(());
1063            }
1064
1065            let _guard = scopeguard::guard((), |_| {
1066                let _ = self.conn.ungrab_server();
1067                let _ = self.conn.flush();
1068            });
1069            self.conn.grab_server()?;
1070            self.conn.configure_window(
1071                elem.mapped_window_id().unwrap_or_else(|| elem.window_id()),
1072                &ConfigureWindowAux::new().stack_mode(StackMode::ABOVE),
1073            )?;
1074            self.client_list_stacking.retain(|e| *e != elem.window_id());
1075            self.client_list_stacking.push(elem.window_id());
1076            self.conn.change_property32(
1077                PropMode::REPLACE,
1078                self.screen.root,
1079                self.atoms._NET_CLIENT_LIST_STACKING,
1080                AtomEnum::WINDOW,
1081                &self.client_list_stacking,
1082            )?;
1083        }
1084        Ok(())
1085    }

1087    fn update_stacking_order_impl<'a, W: X11Relatable + 'a>(
1088        &mut self,
1089        order: impl Iterator<Item = &'a W>,
1090        direction: StackingDirection,
1091    ) -> Result<(), ConnectionError> {
1092        let _guard = scopeguard::guard((), |_| {
1093            let _ = self.conn.ungrab_server();
1094            let _ = self.conn.flush();
1095        });
1096        self.conn.grab_server()?;
1097
1098        let mut last_pos = None;
1099        let mut changed = false;
1100
1101        for relatable in order {
1102            let stacking_ordered_elems: Vec<&X11Surface> = self
1103                .client_list_stacking
1104                .iter()
1105                .filter_map(|w| self.windows.iter().find(|s| s.window_id() == *w))
1106                .collect();
1107            let pos = stacking_ordered_elems.iter().position(|w| relatable.is_window(w));
1108            if let (Some(pos), Some(last_pos)) = (pos, last_pos) {
1109                if direction.pos_comparator(pos, last_pos) {
1110                    let sibling = stacking_ordered_elems[last_pos];
1111                    let sibling_id = self.client_list_stacking[last_pos];
1112                    let elem = stacking_ordered_elems[pos];
1113                    let elem_id = self.client_list_stacking.remove(pos);
1114                    self.conn.configure_window(
1115                        elem.mapped_window_id().unwrap_or(elem_id),
1116                        &ConfigureWindowAux::new()
1117                            .sibling(sibling.mapped_window_id().unwrap_or(sibling_id))
1118                            .stack_mode(direction.stack_mode()),
1119                    )?;
1120                    self.client_list_stacking.insert(last_pos, elem_id);
1121                    changed = true;
1122                    continue;
1123                }
1124            }
1125
1126            if pos.is_some() {
1127                last_pos = pos;
1128            }
1129        }
1130
1131        if changed {
1132            self.conn.change_property32(
1133                PropMode::REPLACE,
1134                self.screen.root,
1135                self.atoms._NET_CLIENT_LIST_STACKING,
1136                AtomEnum::WINDOW,
1137                &self.client_list_stacking,
1138            )?;
1139        }
1140        Ok(())
1141    }

1143    /// Updates the stacking order by matching provided windows downwards.
1144    ///
1145    /// This function reorders provided x11 windows in such a way,
1146    /// that windows inside the internal X11 stack follow the provided `order`
1147    /// without moving other windows around as much as possible. The internal stack
1148    /// stores windows bottom -> top, so order here is also bottom -> top.
1149    ///
1150    /// Window IDs unknown to this XWM will be ignored.
1151    /// The first window in `order` found will not be moved.
1152    ///
1153    /// If a window is encountered in `order` that is stacked below the first window,
1154    /// it will be moved to be directly above the previous window in `order`.
1155    /// E.g. Windows `C -> A -> B -> E` given in order with an internal stack of `D -> A -> B -> C`,
1156    /// will be reordered as `D -> C -> A -> B`. First `A` is moved to be directly
1157    /// above `C`, then `B`is moved to be directly above `A`.
1158    ///
1159    /// Windows in the internal stack, that are not present in `order`
1160    /// will be skipped over in the process.
1161    ///
1162    /// So if windows `A -> C` are given in order and the internal stack is `A -> B -> C`,
1163    /// no reordering will occur.
1164    ///
1165    /// See [`X11Wm::update_stacking_order_upwards`] for a variant of this algorithm,
1166    /// which works from the bottom up or [`X11Wm::raise_window`] for an easier but
1167    /// much more limited way to reorder.
1168    pub fn update_stacking_order_downwards<'a, W: X11Relatable + 'a>(
1169        &mut self,
1170        order: impl Iterator<Item = &'a W>,
1171    ) -> Result<(), ConnectionError> {
1172        self.update_stacking_order_impl(order, StackingDirection::Downwards)
1173    }

1394            .randr_get_output_primary(self.screen.root)?
1395            .reply()?

1476
1477fn handle_event<D>(
1478    loop_handle: &LoopHandle<'_, D>,
1479    dh: &DisplayHandle,
1480    state: &mut D,
1481    xwm_id: XwmId,
1482    event: Event,
1483) -> Result<(), Box<dyn std::error::Error>>
1484where
1485    D: XwmHandler,
1486    D: xwayland_shell::XWaylandShellHandler,
1487    D: SeatHandler,
1488    D: DndGrabHandler,
1489    <D as SeatHandler>::PointerFocus: DndFocus<D>,
1490    <D as SeatHandler>::TouchFocus: DndFocus<D>,
1491    D: 'static,
1492{
1493    let xwm = state.xwm_state(xwm_id);
1494    let _guard = xwm.span.enter();
1495    let conn = xwm.conn.clone();
1496
1497    let mut should_ignore = false;
1498    if let Some(seqno) = event.wire_sequence_number() {
1499        // Check sequences_to_ignore and remove entries with old (=smaller) numbers.
1500        while let Some(&Reverse(to_ignore)) = xwm.sequences_to_ignore.peek() {
1501            // Sequence numbers can wrap around, so we cannot simply check for
1502            // "to_ignore <= seqno". This is equivalent to "to_ignore - seqno <= 0", which is what we
1503            // check instead. Since sequence numbers are unsigned, we need a trick: We decide
1504            // that values from [MAX/2, MAX] count as "<= 0" and the rest doesn't.
1505            if to_ignore.wrapping_sub(seqno) <= u16::MAX / 2 {
1506                // If the two sequence numbers are equal, this event should be ignored.
1507                should_ignore = to_ignore == seqno;
1508                break;
1509            }
1510            xwm.sequences_to_ignore.pop();
1511        }
1512    }
1513
1514    trace!(
1515        event = ?event,
1516        should_ignore = should_ignore,
1517        "Got X11 event",
1518    );
1519    if should_ignore {
1520        return Ok(());
1521    }
1522
1523    match event {
1524        Event::CreateNotify(n) => {
1525            if n.window == *xwm.wm_window
1526                || n.window == *xwm.xsettings.window
1527                || xwm.clipboard.has_window(&n.window)
1528                || xwm.primary.has_window(&n.window)
1529                || xwm.dnd.has_window(&n.window)
1530            {
1531                return Ok(());
1532            }
1533
1534            xwm.conn.change_window_attributes(
1535                n.window,
1536                &ChangeWindowAttributesAux::new()
1537                    .event_mask(EventMask::PROPERTY_CHANGE | EventMask::FOCUS_CHANGE),
1538            )?;
1539            xwm.conn.flush()?;
1540
1541            if xwm.windows.iter().any(|s| s.mapped_window_id() == Some(n.window)) {
1542                return Ok(());
1543            }
1544
1545            let geo = conn.get_geometry(n.window)?.reply()?;
1546            let geometry = Rectangle::<i32, Client>::new(
1547                (geo.x as i32, geo.y as i32).into(),
1548                (geo.width as i32, geo.height as i32).into(),
1549            )
1550            .to_f64()
1551            .to_logical(xwm.client_scale.load(Ordering::Acquire))
1552            .to_i32_round();
1553
1554            let surface = X11Surface::new(
1555                Some(xwm),
1556                n.window,
1557                n.override_redirect,
1558                Arc::downgrade(&conn),
1559                xwm.atoms,
1560                xwm.servertime_counter,
1561                geometry,
1562                xwm.dnd.xdnd_active.clone(),
1563            );
1564            surface.update_properties()?;
1565            xwm.windows.push(surface.clone());
1566
1567            drop(_guard);
1568            if n.override_redirect {
1569                state.new_override_redirect_window(xwm_id, surface);
1570            } else {
1571                state.new_window(xwm_id, surface);
1572            }
1573        }
1574        Event::MapRequest(r) => {
1575            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == r.window).cloned() {
1576                if surface.state.lock().unwrap().mapped_onto.is_none() {
1577                    // we reparent windows, because a lot of stuff expects, that we do
1578                    let geo_cookie = conn.get_geometry(r.window)?;
1579                    let attrs_cookie = conn.get_window_attributes(r.window)?;
1580                    let geo = geo_cookie.reply()?;
1581                    let attrs = attrs_cookie.reply()?;
1582                    let colormap = xwm.colormap_for_visual(attrs.visual)?;
1583
1584                    let win = r.window;
1585                    let frame_win = conn.generate_id()?;
1586                    let win_aux = CreateWindowAux::new()
1587                        .event_mask(
1588                            EventMask::SUBSTRUCTURE_NOTIFY
1589                                | EventMask::SUBSTRUCTURE_REDIRECT
1590                                | EventMask::PROPERTY_CHANGE,
1591                        )
1592                        .colormap(colormap)
1593                        .background_pixmap(x11rb::NONE)
1594                        .border_pixel(0);
1595
1596                    {
1597                        let _guard = scopeguard::guard((), |_| {
1598                            let _ = conn.ungrab_server();
1599                        });
1600
1601                        conn.grab_server()?;
1602                        let cookie1 = conn.create_window(
1603                            geo.depth,
1604                            frame_win,
1605                            xwm.screen.root,
1606                            geo.x,
1607                            geo.y,
1608                            geo.width,
1609                            geo.height,
1610                            0,
1611                            WindowClass::INPUT_OUTPUT,
1612                            attrs.visual,
1613                            &win_aux,
1614                        )?;
1615                        let cookie2 = conn.reparent_window(win, frame_win, 0, 0)?;
1616                        conn.map_window(win)?;
1617
1618                        // Ignore all events caused by reparent_window(). All those events have the sequence number
1619                        // of the reparent_window() request, thus remember its sequence number. The
1620                        // grab_server()/ungrab_server() is done so that the server does not handle other clients
1621                        // in-between, which could cause other events to get the same sequence number.
1622                        xwm.sequences_to_ignore
1623                            .push(Reverse(cookie1.sequence_number() as u16));
1624                        xwm.sequences_to_ignore
1625                            .push(Reverse(cookie2.sequence_number() as u16));
1626                    }
1627
1628                    surface.state.lock().unwrap().mapped_onto = Some(frame_win);
1629
1630                    // A MapRequest can only happen when an X11 window does not have the
1631                    // override-redirect flag set. It's possible for a window to be created as
1632                    // override-redirect and later set the flag to false before mapping.
1633                    // In that case, we set the X11Surface's override-redirect state to false here
1634                    // to prevent `set_mapped` and `configure` from failing.
1635                    surface.state.lock().unwrap().override_redirect = false;
1636
1637                    // Read initial _NET_WM_STATE set by the client before mapping.
1638                    // Per EWMH spec, clients may set _NET_WM_STATE prior to mapping
1639                    // and the window manager must respect it.
1640                    if let Ok(reply) = conn
1641                        .get_property(false, win, xwm.atoms._NET_WM_STATE, AtomEnum::ATOM, 0, 1024)?
1642                        .reply()
1643                    {
1644                        if let Some(states) = reply.value32() {
1645                            let mut state_lock = surface.state.lock().unwrap();
1646                            for atom in states {
1647                                state_lock.net_state.insert(atom);
1648                            }
1649                        }
1650                    }
1651
1652                    drop(_guard);
1653                    state.map_window_request(xwm_id, surface);
1654                }
1655            }
1656        }
1657        Event::MapNotify(n) => {
1658            trace!(window = ?n.window, "mapped X11 Window");
1659            if let Some(surface) = xwm
1660                .windows
1661                .iter()
1662                .find(|x|
1663                      // don't include the reparenting windows
1664                      (x.window_id() == n.window && n.override_redirect) ||
1665                      x.mapped_window_id() == Some(n.window))
1666                .cloned()
1667            {
1668                // Client may have changed override-redirect flag since window creation
1669                surface.state.lock().unwrap().override_redirect = n.override_redirect;
1670
1671                if n.override_redirect {
1672                    drop(_guard);
1673                    state.mapped_override_redirect_window(xwm_id, surface);
1674                } else {
1675                    xwm.client_list.push(surface.window_id());
1676                    xwm.client_list_stacking.push(surface.window_id());
1677                    conn.change_property32(
1678                        PropMode::APPEND,
1679                        xwm.screen.root,
1680                        xwm.atoms._NET_CLIENT_LIST,
1681                        AtomEnum::WINDOW,
1682                        &[surface.window_id()],
1683                    )?;
1684                    conn.change_property32(
1685                        PropMode::APPEND,
1686                        xwm.screen.root,
1687                        xwm.atoms._NET_CLIENT_LIST_STACKING,
1688                        AtomEnum::WINDOW,
1689                        &[surface.window_id()],
1690                    )?;
1691                    drop(_guard);
1692                    state.map_window_notify(xwm_id, surface);
1693                }
1694            }
1695        }
1696        Event::ConfigureRequest(r) => {
1697            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == r.window).cloned() {
1698                drop(_guard);
1699
1700                let client_scale = xwm.client_scale.load(Ordering::Acquire);
1701
1702                // Pass the request to downstream to decide
1703                state.configure_request(
1704                    xwm_id,
1705                    surface.clone(),
1706                    if u16::from(r.value_mask) & u16::from(ConfigWindow::X) != 0 {
1707                        Some(((r.x as f64) / client_scale).round() as i32)
1708                    } else {
1709                        None
1710                    },
1711                    if u16::from(r.value_mask) & u16::from(ConfigWindow::Y) != 0 {
1712                        Some(((r.y as f64) / client_scale).round() as i32)
1713                    } else {
1714                        None
1715                    },
1716                    if u16::from(r.value_mask) & u16::from(ConfigWindow::WIDTH) != 0 {
1717                        Some(((r.width as f64) / client_scale).round() as u32)
1718                    } else {
1719                        None
1720                    },
1721                    if u16::from(r.value_mask) & u16::from(ConfigWindow::HEIGHT) != 0 {
1722                        Some(((r.height as f64) / client_scale).round() as u32)
1723                    } else {
1724                        None
1725                    },
1726                    if u16::from(r.value_mask) & u16::from(ConfigWindow::STACK_MODE) != 0 {
1727                        match r.stack_mode {
1728                            StackMode::ABOVE => {
1729                                if u16::from(r.value_mask) & u16::from(ConfigWindow::SIBLING) != 0 {
1730                                    Some(Reorder::Above(r.sibling))
1731                                } else {
1732                                    Some(Reorder::Top)
1733                                }
1734                            }
1735                            StackMode::BELOW => {
1736                                if u16::from(r.value_mask) & u16::from(ConfigWindow::SIBLING) != 0 {
1737                                    Some(Reorder::Below(r.sibling))
1738                                } else {
1739                                    Some(Reorder::Bottom)
1740                                }
1741                            }
1742                            _ => None,
1743                        }
1744                    } else {
1745                        None
1746                    },
1747                );
1748                // Synthetic event
1749                surface.configure(None).map_err(|err| match err {
1750                    X11SurfaceError::UnsupportedForOverrideRedirect => unreachable!(),
1751                    err => err,
1752                })?;
1753            }
1754        }
1755        Event::ConfigureNotify(n) => {
1756            trace!(window = ?n, "configured X11 Window");
1757
1758            let client_scale = xwm.client_scale.load(Ordering::Acquire);
1759            let geometry = Rectangle::<i32, Client>::new(
1760                (n.x as i32, n.y as i32).into(),
1761                (n.width as i32, n.height as i32).into(),
1762            )
1763            .to_f64()
1764            .to_logical(client_scale)
1765            .to_i32_round();
1766
1767            if let Some(surface) = xwm
1768                .windows
1769                .iter()
1770                .find(|x| x.mapped_window_id() == Some(n.window))
1771                .cloned()
1772            {
1773                drop(_guard);
1774                state.configure_notify(
1775                    xwm_id,
1776                    surface,
1777                    geometry,
1778                    if n.above_sibling == x11rb::NONE {
1779                        None
1780                    } else {
1781                        Some(n.above_sibling)
1782                    },
1783                );
1784            } else if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == n.window).cloned() {
1785                if surface.is_override_redirect() {
1786                    surface.state.lock().unwrap().geometry = geometry;
1787                    drop(_guard);
1788                    state.configure_notify(
1789                        xwm_id,
1790                        surface,
1791                        geometry,
1792                        if n.above_sibling == x11rb::NONE {
1793                            None
1794                        } else {
1795                            Some(n.above_sibling)
1796                        },
1797                    );
1798                }
1799            }
1800        }
1801        Event::UnmapNotify(n) => {
1802            if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == n.window).cloned() {
1803                xwm.client_list.retain(|w| *w != surface.window_id());
1804                xwm.client_list_stacking.retain(|w| *w != surface.window_id());
1805                {
1806                    let _guard = scopeguard::guard((), |_| {
1807                        let _ = conn.ungrab_server();
1808                    });
1809                    conn.grab_server()?;
1810                    conn.change_property32(
1811                        PropMode::REPLACE,
1812                        xwm.screen.root,
1813                        xwm.atoms._NET_CLIENT_LIST,
1814                        AtomEnum::WINDOW,
1815                        &xwm.client_list,
1816                    )?;
1817                    conn.change_property32(
1818                        PropMode::REPLACE,
1819                        xwm.screen.root,
1820                        xwm.atoms._NET_CLIENT_LIST_STACKING,
1821                        AtomEnum::WINDOW,
1822                        &xwm.client_list_stacking,
1823                    )?;
1824                    if !surface.is_override_redirect() {
1825                        let mut state = surface.state.lock().unwrap();
1826                        conn.reparent_window(
1827                            n.window,
1828                            xwm.screen.root,
1829                            state.geometry.loc.x as i16,
1830                            state.geometry.loc.y as i16,
1831                        )?;
1832                        if let Some(frame) = state.mapped_onto.take() {
1833                            conn.destroy_window(frame)?;
1834                        }
1835                    }
1836                }
1837                drop(_guard);
1838                state.unmapped_window(xwm_id, surface.clone());
1839                {
1840                    surface.set_wl_surface(state, None);
1841                }
1842            }
1843        }
1844        Event::DestroyNotify(n) => {
1845            xwm.clipboard.window_destroyed(&n.window, loop_handle);
1846            xwm.primary.window_destroyed(&n.window, loop_handle);
1847            xwm.dnd.window_destroyed(&n.window, loop_handle);
1848
1849            if let Some(pos) = xwm.windows.iter().position(|x| x.window_id() == n.window) {
1850                let surface = xwm.windows.remove(pos);
1851                surface.handle_destroyed();
1852                drop(_guard);
1853                state.destroyed_window(xwm_id, surface);
1854            }
1855        }
1856        Event::XfixesSelectionNotify(n) => {
1857            let selection = match n.selection {
1858                x if x == xwm.atoms.CLIPBOARD => &mut xwm.clipboard,

2474                                    && y == xwm.atoms._NET_WM_STATE_MAXIMIZED_VERT)
2475                                    || (x == xwm.atoms._NET_WM_STATE_MAXIMIZED_VERT
2476                                        && y == xwm.atoms._NET_WM_STATE_MAXIMIZED_HORZ) =>
2477                            {
2478                                match data[0] {
2479                                    0 => {
2480                                        if surface.is_maximized() {
2481                                            state.unmaximize_request(xwm_id, surface)
2482                                        }
2483                                    }
2484                                    1 => {
2485                                        if !surface.is_maximized() {
2486                                            state.maximize_request(xwm_id, surface)
2487                                        }
2488                                    }
2489                                    2 => {
2490                                        if surface.is_maximized() {
2491                                            state.unmaximize_request(xwm_id, surface)
2492                                        } else {
2493                                            state.maximize_request(xwm_id, surface)
2494                                        }
2495                                    }
2496                                    _ => {}
2497                                }
2498                            }
2499                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_FULLSCREEN) => {
2500                                match data[0] {
2501                                    0 => state.unfullscreen_request(xwm_id, surface),
2502                                    1 => state.fullscreen_request(xwm_id, surface),
2503                                    2 => {
2504                                        if surface.is_fullscreen() {
2505                                            state.unfullscreen_request(xwm_id, surface)
2506                                        } else {
2507                                            state.fullscreen_request(xwm_id, surface)
2508                                        }
2509                                    }
2510                                    _ => {}
2511                                }
2512                            }
2513                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_ABOVE) => match data[0] {
2514                                0 => state.unabove_request(xwm_id, surface),
2515                                1 => state.above_request(xwm_id, surface),
2516                                2 => {
2517                                    if surface.is_above() {
2518                                        state.unabove_request(xwm_id, surface)
2519                                    } else {
2520                                        state.above_request(xwm_id, surface)
2521                                    }
2522                                }
2523                                _ => {}
2524                            },
2525                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_BELOW) => match data[0] {
2526                                0 => state.unbelow_request(xwm_id, surface),
2527                                1 => state.below_request(xwm_id, surface),
2528                                2 => {
2529                                    if surface.is_below() {
2530                                        state.unbelow_request(xwm_id, surface)
2531                                    } else {
2532                                        state.below_request(xwm_id, surface)
2533                                    }
2534                                }
2535                                _ => {}
2536                            },
2537                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_STICKY) => match data[0] {
2538                                0 => state.unstick_request(xwm_id, surface),
2539                                1 => state.stick_request(xwm_id, surface),
2540                                2 => {
2541                                    if surface.is_sticky() {
2542                                        state.unstick_request(xwm_id, surface)
2543                                    } else {
2544                                        state.stick_request(xwm_id, surface)
2545                                    }
2546                                }
2547                                _ => {}
2548                            },
2549                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_SHADED) => match data[0] {
2550                                0 => state.unshade_request(xwm_id, surface),
2551                                1 => state.shade_request(xwm_id, surface),
2552                                2 => {
2553                                    if surface.is_shaded() {
2554                                        state.unshade_request(xwm_id, surface)
2555                                    } else {
2556                                        state.shade_request(xwm_id, surface)
2557                                    }
2558                                }
2559                                _ => {}
2560                            },
2561                            actions if actions.contains(&xwm.atoms._NET_WM_STATE_DEMANDS_ATTENTION) => {
2562                                match data[0] {
2563                                    0 => state.undemands_attention_request(xwm_id, surface),
2564                                    1 => state.demands_attention_request(xwm_id, surface),
2565                                    2 => {
2566                                        if surface.demands_attention() {
2567                                            state.undemands_attention_request(xwm_id, surface)
2568                                        } else {
2569                                            state.demands_attention_request(xwm_id, surface)
2570                                        }
2571                                    }
2572                                    _ => {}
2573                                }
2574                            }
2575                            _ => {}
2576                        }
2577                    }
2578                }
2579                x if x == xwm.atoms._NET_WM_MOVERESIZE => {
2580                    if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == msg.window).cloned() {
2581                        drop(_guard);
2582                        let data = msg.data.as_data32();
2583                        match data[2] {
2584                            x @ 0..=7 => {
2585                                let resize_edge = match x {
2586                                    0 => ResizeEdge::TopLeft,
2587                                    1 => ResizeEdge::Top,
2588                                    2 => ResizeEdge::TopRight,
2589                                    3 => ResizeEdge::Right,
2590                                    4 => ResizeEdge::BottomRight,
2591                                    5 => ResizeEdge::Bottom,
2592                                    6 => ResizeEdge::BottomLeft,
2593                                    7 => ResizeEdge::Left,
2594                                    _ => unreachable!(),
2595                                };
2596                                state.resize_request(xwm_id, surface, data[3], resize_edge);
2597                            }
2598                            8 => state.move_request(xwm_id, surface, data[3]),
2599                            _ => {} // ignore keyboard moves/resizes for now
2600                        }
2601                    }
2602                }
2603                x if x == xwm.atoms._NET_ACTIVE_WINDOW => {
2604                    let data = msg.data.as_data32();
2605                    let timestamp = data[1];
2606                    let currently_active_window =
2607                        xwm.windows.iter().find(|x| x.window_id() == data[2]).cloned();
2608                    if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == msg.window).cloned() {
2609                        drop(_guard);
2610                        state.active_window_request(xwm_id, surface, timestamp, currently_active_window);
2611                    }
2612                }
2613                x if x == xwm.atoms.WM_PROTOCOLS => {
2614                    let data = msg.data.as_data32();
2615
2616                    match data[0] {
2617                        x if x == xwm.atoms._NET_WM_PING => {
2618                            let timestamp = data[1];
2619                            // Some older clients will not faithfully copy the data back to us, and
2620                            // will only set the timestamp, so `data[2]` may be zero or unreliable.
2621                            let window_id = data[2];
2622
2623                            let surface = (window_id != x11rb::NONE)
2624                                .then(|| xwm.windows.iter().find(|x| x.window_id() == window_id))
2625                                .flatten()
2626                                .filter(|surface| {
2627                                    surface.state.lock().unwrap().pending_ping_timestamp == Some(timestamp)
2628                                })
2629                                .or_else(|| {
2630                                    xwm.windows.iter().find(|x| {
2631                                        x.state.lock().unwrap().pending_ping_timestamp == Some(timestamp)
2632                                    })
2633                                })
2634                                .cloned();
2635
2636                            if let Some(surface) = surface {
2637                                surface.state.lock().unwrap().pending_ping_timestamp = None;
2638                                drop(_guard);
2639                                state.ping_acked(xwm_id, surface.clone(), timestamp);
2640                            }
2641                        }
2642
2643                        _ => {
2644                            debug!(
2645                                "Unhandled WM_PROTOCOLS client msg of type {:?}",
2646                                String::from_utf8(
2647                                    conn.get_atom_name(data[0])?.reply_unchecked()?.unwrap().name
2648                                )
2649                                .ok()
2650                            )
2651                        }
2652                    }
2653                }
2654                x if x == xwm.atoms._NET_SHOWING_DESKTOP => {
2655                    let data = msg.data.as_data32();
2656                    let show_desktop = data[0];
2657                    if show_desktop == 1 && !xwm.is_showing_desktop {
2658                        drop(_guard);
2659                        state.show_desktop_request(xwm_id);
2660                    } else if show_desktop == 0 && xwm.is_showing_desktop {
2661                        drop(_guard);
2662                        state.unshow_desktop_request(xwm_id);
2663                    }
2664                }
2665                // Dnd events incoming
2666                x if x == xwm.atoms.XdndStatus => {
2667                    let window_id = msg.data.as_data32()[0];
2668                    if let Some(surface) = xwm.windows.iter().find(|x| x.window_id() == window_id) {
2669                        xwm.dnd
2670                            .handle_status(surface, msg.data, &xwm.atoms, xwm.client_scale.clone());
2671                    }
2672                }
2673                x if x == xwm.atoms.XdndFinished => {
2674                    xwm.dnd.handle_finished(msg.data);
2675                }
2676                // Dnd events incoming
2677                x if x == xwm.atoms.XdndEnter => {
2678                    xwm.dnd.handle_enter(msg.data)?;
2679                }
2680                x if x == xwm.atoms.XdndPosition => {
2681                    xwm.dnd.handle_position(msg.data)?;
2682                }
2683                x if x == xwm.atoms.XdndLeave => {
2684                    xwm.dnd.handle_leave(msg.data)?;
2685                }
2686                x if x == xwm.atoms.XdndDrop => {
2687                    xwm.dnd.handle_drop(msg.data)?;
2688                }
2689                x => {
2690                    debug!(
2691                        "Unhandled client msg of type {:?}",
2692                        String::from_utf8(conn.get_atom_name(x)?.reply_unchecked()?.unwrap().name).ok()
2693                    )
2694                }
2695            }
2696        }
2697        Event::RandrNotify(n) => {
2698            if n.sub_code == Notify::OUTPUT_CHANGE {
2699                let current_primary = xwm
2700                    .conn
2701                    .randr_get_output_primary(xwm.screen.root)?
2702                    .reply()?
2703                    .output;
2704                if current_primary != xwm.randr_primary {
2705                    xwm.randr_primary = current_primary;
2706                    let primary_name = xwm.get_randr_primary_output().unwrap();
2707                    drop(_guard);
2708
2709                    state.randr_primary_output_change(xwm_id, primary_name);
2710                }
2711            }
2712        }
2713        Event::RandrScreenChangeNotify(n) => {
2714            xwm.screen.width_in_pixels = n.width;
2715            xwm.screen.height_in_pixels = n.height;
2716            xwm.screen.width_in_millimeters = n.mwidth;
2717            xwm.screen.height_in_millimeters = n.mheight;
2718
2719            xwm.dnd.update_screen(&xwm.screen)?;
2720        }
2721        Event::SyncAlarmNotify(n) => {
2722            if let Some(surface) = xwm
2723                .windows
2724                .iter()
2725                .find(|x| {
2726                    let state = x.state.lock().unwrap();
2727                    state.sync_alarm == Some(n.alarm) || state.sync_timeout_alarm == Some(n.alarm)
2728                })
2729                .cloned()
2730            {
2731                let is_sync_ack = surface.state.lock().unwrap().sync_alarm == Some(n.alarm);
2732
2733                if is_sync_ack {
2734                    if surface.handle_sync_alarm(n.counter_value) {
2735                        drop(_guard);
2736                        state.sync_request_acked(xwm_id, surface);
2737                    }
2738                } else {
2739                    surface.handle_sync_timeout();
2740                    drop(_guard);
2741                    state.sync_request_timeout(xwm_id, surface);
2742                }
2743            }
2744        }
2745        Event::Error(err) => {
2746            info!(?err, "Got X11 Error");
2747        }
2748        _ => {}
2749    }
2750    conn.flush()?;
2751    Ok(())
2752}
