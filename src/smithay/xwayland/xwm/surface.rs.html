234    /// Set the X11 windows as mapped/unmapped affecting its visibility.
235    ///
236    /// It is an error to call this function on override redirect windows
237    pub fn set_mapped(&self, mapped: bool) -> Result<(), X11SurfaceError> {
238        if self.is_override_redirect() {
239            return Err(X11SurfaceError::UnsupportedForOverrideRedirect);
240        }
241
242        if let Some(conn) = self.conn.upgrade() {
243            if let Some(frame) = self.state.lock().unwrap().mapped_onto {
244                if mapped {
245                    let property = [1u32 /*NormalState*/, 0 /*WINDOW_NONE*/];
246                    conn.change_property32(
247                        PropMode::REPLACE,
248                        self.window,
249                        self.atoms.WM_STATE,
250                        self.atoms.WM_STATE,
251                        &property,
252                    )?;
253                    conn.map_window(frame)?;
254                } else {
255                    let property = [3u32 /*IconicState*/, 0 /*WINDOW_NONE*/];
256                    conn.change_property32(
257                        PropMode::REPLACE,
258                        self.window,
259                        self.atoms.WM_STATE,
260                        self.atoms.WM_STATE,
261                        &property,
262                    )?;
263                    conn.unmap_window(frame)?;
264                }
265                conn.flush()?;
266            }
267        }
268        Ok(())
269    }

288    /// Send a configure to this window.
289    ///
290    /// If `rect` is provided the new state will be send to the window.
291    /// If `rect` is `None` a synthetic configure event with the existing state will be send.
292    pub fn configure(&self, rect: impl Into<Option<Rectangle<i32, Logical>>>) -> Result<(), X11SurfaceError> {
293        let rect = rect.into();
294        if self.is_override_redirect() && rect.is_some() {
295            return Err(X11SurfaceError::UnsupportedForOverrideRedirect);
296        }
297
298        if let Some(conn) = self.conn.upgrade() {
299            let mut state = self.state.lock().unwrap();
300            let client_scale = self
301                .client_scale
302                .as_ref()
303                .map(|s| s.load(Ordering::Acquire))
304                .unwrap_or(1.);
305            let logical_rect = rect.unwrap_or(state.geometry);
306            let rect = logical_rect.to_client_precise_round(client_scale);
307            let aux = ConfigureWindowAux::default()
308                .x(rect.loc.x)
309                .y(rect.loc.y)
310                .width(rect.size.w as u32)
311                .height(rect.size.h as u32)
312                .border_width(0);
313            if let Some(frame) = state.mapped_onto {
314                let win_aux = ConfigureWindowAux::default()
315                    .width(rect.size.w as u32)
316                    .height(rect.size.h as u32)
317                    .border_width(0);
318                conn.configure_window(frame, &aux)?;
319                conn.configure_window(self.window, &win_aux)?;
320                send_configure_notify(&conn, &self.window, rect, false)?;
321            } else {
322                conn.configure_window(self.window, &aux)?;
323            }
324            conn.flush()?;
325
326            state.geometry = logical_rect;
327        }
328        Ok(())
329    }

364    pub(crate) fn set_wl_surface<D: SeatHandler + 'static>(&self, data: &mut D, surface: Option<WlSurface>) {
365        let mut state = self.state.lock().unwrap();
366        if let (Some(surface), None) = (surface.as_ref(), state.wl_surface.as_ref()) {
367            if let Some((seat, keys, mods, serial)) = state.pending_enter.take() {
368                if let Ok(seat) = seat.downcast::<Seat<D>>() {
369                    enter_internal(surface, &seat, data, keys.into_iter(), serial);
370                    if let Some(modifiers) = mods {
371                        KeyboardTarget::modifiers(surface, &seat, data, modifiers, serial);
372                    }
373                }
374            }
375        }
376        state.wl_surface = surface;
377    }

439    /// Returns the suggested minimum size of the underlying X11 window
440    pub fn min_size(&self) -> Option<Size<i32, Logical>> {
441        let client_scale = self
442            .client_scale
443            .as_ref()
444            .map(|s| s.load(Ordering::Acquire))
445            .unwrap_or(1.);
446        let state = self.state.lock().unwrap();

449            .as_ref()

452            .map(|s| s.to_f64().to_logical(client_scale).to_i32_round())
453    }

617    fn change_net_state(&self, added: &[Atom], removed: &[Atom]) -> Result<(), ConnectionError> {
618        let mut state = self.state.lock().unwrap();
619
620        let mut changed = false;
621        for atom in removed {
622            changed |= state.net_state.remove(atom);
623        }
624        for atom in added {
625            changed |= state.net_state.insert(*atom);
626        }
627
628        if changed {
629            let new_props = Vec::from_iter(state.net_state.iter().copied());
630
631            let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
632            conn.grab_server()?;
633            let _guard = scopeguard::guard((), |_| {
634                let _ = conn.ungrab_server();
635                let _ = conn.flush();
636            });
637
638            conn.change_property32(
639                PropMode::REPLACE,
640                self.window,
641                self.atoms._NET_WM_STATE,
642                AtomEnum::ATOM,

650    /// Input handling model requested by the underlying X11 window.
651    ///
652    /// See ICCCM §4.1.7 for details.
653    pub fn input_model(&self) -> WmInputModel {
654        let state = self.state.lock().unwrap();
655        match (
656            state.hints.as_ref().and_then(|hints| hints.input).unwrap_or(true),
657            state.protocols.contains(&WMProtocol::TakeFocus),
658        ) {
659            (false, false) => WmInputModel::None,
660            (true, false) => WmInputModel::Passive, // the default
661            (true, true) => WmInputModel::LocallyActive,
662            (false, true) => WmInputModel::GloballyActive,
663        }
664    }

682    pub(super) fn update_property(&self, atom: Atom) -> Result<Option<WmWindowProperty>, ConnectionError> {
683        match atom {
684            atom if atom == self.atoms._NET_WM_NAME || atom == AtomEnum::WM_NAME.into() => {
685                self.update_title()?;
686                Ok(Some(WmWindowProperty::Title))
687            }
688            atom if atom == AtomEnum::WM_CLASS.into() => {
689                self.update_class()?;
690                Ok(Some(WmWindowProperty::Class))
691            }
692            atom if atom == self.atoms.WM_PROTOCOLS => {
693                self.update_protocols()?;
694                Ok(Some(WmWindowProperty::Protocols))
695            }
696            atom if atom == self.atoms.WM_HINTS => {
697                self.update_hints()?;
698                Ok(Some(WmWindowProperty::Hints))
699            }
700            atom if atom == AtomEnum::WM_NORMAL_HINTS.into() => {
701                self.update_normal_hints()?;
702                Ok(Some(WmWindowProperty::NormalHints))
703            }
704            atom if atom == AtomEnum::WM_TRANSIENT_FOR.into() => {
705                self.update_transient_for()?;
706                Ok(Some(WmWindowProperty::TransientFor))
707            }
708            atom if atom == self.atoms._NET_WM_WINDOW_TYPE => {
709                self.update_net_window_type()?;
710                Ok(Some(WmWindowProperty::WindowType))
711            }
712            atom if atom == self.atoms._MOTIF_WM_HINTS => {
713                self.update_motif_hints()?;
714                Ok(Some(WmWindowProperty::MotifHints))
715            }
716            atom if atom == self.atoms._NET_STARTUP_ID => {
717                self.update_startup_id()?;
718                Ok(Some(WmWindowProperty::StartupId))
719            }
720            atom if atom == self.atoms._NET_WM_PID => {
721                self.update_pid()?;
722                Ok(Some(WmWindowProperty::Pid))
723            }
724            atom if atom == self.atoms._NET_WM_WINDOW_OPACITY => {
725                self.update_opacity()?;
726                Ok(Some(WmWindowProperty::Opacity))
727            }
728
729            _ => Ok(None), // unknown
730        }
731    }

733    fn update_class(&self) -> Result<(), ConnectionError> {
734        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
735        let (class, instance) = match WmClass::get(&*conn, self.window)?.reply_unchecked() {
736            Ok(Some(wm_class)) => (
737                WINDOWS_1252.decode(wm_class.class()).0.to_string(),
738                WINDOWS_1252.decode(wm_class.instance()).0.to_string(),
739            ),
740            Ok(None) | Err(ConnectionError::ParseError(_)) => (Default::default(), Default::default()), // Getting the property failed
741            Err(err) => return Err(err),
742        };
743
744        let mut state = self.state.lock().unwrap();
745        state.class = class;
746        state.instance = instance;
747
748        Ok(())
749    }

826    fn update_protocols(&self) -> Result<(), ConnectionError> {
827        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
828        let Some(protocols) = (match conn
829            .get_property(
830                false,
831                self.window,
832                self.atoms.WM_PROTOCOLS,
833                AtomEnum::ATOM,
834                0,
835                2048,
836            )?
837            .reply_unchecked()
838        {
839            Ok(Some(reply)) => reply.value32().map(|vals| vals.collect::<Vec<_>>()),
840            Ok(None) | Err(ConnectionError::ParseError(_)) => return Ok(()),
841            Err(err) => return Err(err),
842        }) else {
843            return Ok(());
844        };
845
846        let mut state = self.state.lock().unwrap();
847        state.protocols = protocols
848            .into_iter()
849            .filter_map(|atom| match atom {
850                x if x == self.atoms.WM_TAKE_FOCUS => Some(WMProtocol::TakeFocus),
851                x if x == self.atoms.WM_DELETE_WINDOW => Some(WMProtocol::DeleteWindow),
852                _ => None,
853            })

855        Ok(())

858    fn update_transient_for(&self) -> Result<(), ConnectionError> {
859        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
860        let reply = match conn
861            .get_property(
862                false,
863                self.window,
864                AtomEnum::WM_TRANSIENT_FOR,
865                AtomEnum::WINDOW,
866                0,
867                2048,
868            )?
869            .reply_unchecked()
870        {
871            Ok(Some(reply)) => reply,
872            Ok(None) | Err(ConnectionError::ParseError(_)) => return Ok(()),
873            Err(err) => return Err(err),
874        };
875        let window = reply
876            .value32()
877            .and_then(|mut iter| iter.next())
878            .filter(|w| *w != 0);
879
880        let mut state = self.state.lock().unwrap();
881        state.transient_for = window;
882        Ok(())
883    }

921    fn read_window_property_string(&self, atom: impl Into<Atom>) -> Result<Option<String>, ConnectionError> {
922        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
923        let reply = match conn
924            .get_property(false, self.window, atom, AtomEnum::ANY, 0, 2048)?
925            .reply_unchecked()
926        {
927            Ok(Some(reply)) => reply,
928            Ok(None) | Err(ConnectionError::ParseError(_)) => return Ok(None),
929            Err(err) => return Err(err),
930        };
931        let Some(bytes) = reply.value8() else {
932            return Ok(None);
933        };
934        let bytes = bytes.collect::<Vec<u8>>();
935
936        match reply.type_ {
937            x if x == AtomEnum::STRING.into() => Ok(Some(WINDOWS_1252.decode(&bytes).0.to_string())),
938            x if x == self.atoms.UTF8_STRING => Ok(String::from_utf8(bytes).ok()),
939            _ => Ok(None),
940        }
941    }

943    fn read_window_property_u32(&self, atom: impl Into<Atom>) -> Result<Option<u32>, ConnectionError> {
944        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
945        let reply = match conn
946            .get_property(false, self.window, atom, AtomEnum::CARDINAL, 0, 1)?
947            .reply_unchecked()
948        {
949            Ok(Some(reply)) => reply,
950            Ok(None) | Err(ConnectionError::ParseError(_)) => return Ok(None),
951            Err(err) => return Err(err),
952        };
953
954        if let Some(mut value32) = reply.value32() {
955            if let Some(value) = value32.next() {
956                return Ok(Some(value));
957            }
958        }
959
960        Ok(None)
961    }

968    /// Send a close request to this window.
969    ///
970    /// Will outright destroy windows that don't support the `NET_DELETE_WINDOW` protocol.
971    pub fn close(&self) -> Result<(), ConnectionError> {
972        let conn = self.conn.upgrade().ok_or(ConnectionError::UnknownError)?;
973        let state = self.state.lock().unwrap();
974        if state.protocols.contains(&WMProtocol::DeleteWindow) {
975            let event = ClientMessageEvent::new(
976                32,
977                self.window,
978                self.atoms.WM_PROTOCOLS,
979                [self.atoms.WM_DELETE_WINDOW, 0, 0, 0, 0],
980            );
981            conn.send_event(false, self.window, EventMask::NO_EVENT, event)?;
982        } else {
983            conn.destroy_window(self.window)?;
984        }
985        conn.flush()
986    }

988    /// Get the client PID associated with the X11 window.
989    pub fn get_client_pid(&self) -> Result<u32, Box<dyn std::error::Error>> {
990        if let Some(connection) = self.conn.upgrade() {
991            let window = self.window;
992
993            match query_client_ids(
994                &connection,
995                &[ClientIdSpec {
996                    client: window,
997                    mask: x11rb::protocol::res::ClientIdMask::LOCAL_CLIENT_PID,
998                }],
999            ) {
1000                Ok(cookie) => {
1001                    let reply = cookie.reply()?;
1002
1003                    if let Some(id_value) = reply.ids.first() {
1004                        if let Some(pid) = id_value.value.first().copied() {
1005                            return Ok(pid);
1006                        } else {
1007                            return Err(Box::new(std::io::Error::new(
1008                                std::io::ErrorKind::NotFound,
1009                                "No matching client ID found",
1010                            )));
1011                        }
1012                    }
1013                }
1014                Err(_) => {
1015                    return Ok(0);
1016                }
1017            }
1018        }
1019        Ok(0)
1020    }

1057impl<D: SeatHandler + 'static> KeyboardTarget<D> for X11Surface {
1058    fn enter(&self, seat: &Seat<D>, data: &mut D, keys: Vec<KeysymHandle<'_>>, serial: Serial) {
1059        let (set_input_focus, send_take_focus) = match self.input_model() {
1060            WmInputModel::None => return,
1061            WmInputModel::Passive => (true, false),
1062            WmInputModel::LocallyActive => (true, true),
1063            WmInputModel::GloballyActive => (false, true),
1064        };
1065
1066        if let Some(conn) = self.conn.upgrade() {
1067            if set_input_focus {
1068                if let Err(err) = conn.set_input_focus(InputFocus::NONE, self.window, x11rb::CURRENT_TIME) {
1069                    warn!("Unable to set focus for X11Surface ({:?}): {}", self.window, err);
1070                }
1071            }
1072
1073            if send_take_focus {
1074                let event = ClientMessageEvent::new(
1075                    32,
1076                    self.window,
1077                    self.atoms.WM_PROTOCOLS,
1078                    [self.atoms.WM_TAKE_FOCUS, x11rb::CURRENT_TIME, 0, 0, 0],
1079                );
1080                if let Err(err) = conn.send_event(false, self.window, EventMask::NO_EVENT, event) {
1081                    warn!(
1082                        "Unable to send take focus event for X11Surface ({:?}): {}",
1083                        self.window, err
1084                    );
1085                }
1086                let _ = conn.flush();
1087            }
1088
1089            let _ = conn.flush();
1090        }
1091
1092        let mut state = self.state.lock().unwrap();
1093        if let Some(surface) = state.wl_surface.as_ref() {
1094            KeyboardTarget::enter(surface, seat, data, keys, serial);
1095        } else {
1096            state.pending_enter = Some((
1097                Box::new(seat.clone()) as Box<dyn std::any::Any + Send>,
1098                keys.iter().map(|x| x.raw_code()).collect(),
1099                None,
1100                serial,
1101            ));
1102        }
1103    }
1104
1105    fn leave(&self, seat: &Seat<D>, data: &mut D, serial: Serial) {
1106        if self.input_model() == WmInputModel::None {
1107            return;
1108        } else if let Some(conn) = self.conn.upgrade() {
1109            if let Err(err) = conn.set_input_focus(InputFocus::NONE, x11rb::NONE, x11rb::CURRENT_TIME) {
1110                warn!("Unable to unfocus X11Surface ({:?}): {}", self.window, err);
1111            }
1112            let _ = conn.flush();
1113        }
1114
1115        let mut state = self.state.lock().unwrap();
1116        let _ = state.pending_enter.take();
1117        if let Some(surface) = state.wl_surface.as_ref() {
1118            KeyboardTarget::leave(surface, seat, data, serial);
1119        }
1120    }
1121
1122    fn key(
1123        &self,
1124        seat: &Seat<D>,
1125        data: &mut D,
1126        key: KeysymHandle<'_>,
1127        state: KeyState,
1128        serial: Serial,
1129        time: u32,
1130    ) {
1131        if self.input_model() == WmInputModel::None {
1132            return;
1133        }
1134
1135        let mut xstate = self.state.lock().unwrap();
1136        if let Some(surface) = xstate.wl_surface.as_ref() {
1137            KeyboardTarget::key(surface, seat, data, key, state, serial, time)
1138        } else if let Some((_, keys, _, pending_serial)) = xstate.pending_enter.as_mut() {
1139            let raw = key.raw_code();
1140            if state == KeyState::Released {
1141                keys.retain(|c| *c != raw);
1142            } else {
1143                keys.push(raw);
1144            }
1145            *pending_serial = serial
1146        }
1147    }
1148
1149    fn modifiers(&self, seat: &Seat<D>, data: &mut D, modifiers: ModifiersState, serial: Serial) {
1150        if self.input_model() == WmInputModel::None {
1151            return;
1152        }
1153
1154        let mut state = self.state.lock().unwrap();
1155        if let Some(surface) = state.wl_surface.as_ref() {
1156            KeyboardTarget::modifiers(surface, seat, data, modifiers, serial);
1157        } else if let Some((_, _, pending_modifiers, pending_serial)) = state.pending_enter.as_mut() {
1158            *pending_modifiers = Some(modifiers);
1159            *pending_serial = serial;
1160        }
1161    }
1162}
